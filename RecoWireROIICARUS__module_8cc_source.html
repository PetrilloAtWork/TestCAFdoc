129void RecoWireROIICARUS::reconfigure(fhicl::ParameterSet const& pset)
130{
131 // Recover the vector of fhicl parameters for the ROI tools
132 const fhicl::ParameterSet& roiFinderTools = pset.get<fhicl::ParameterSet>("ROIFinderToolVec");
133
134 fROIFinderVec.resize(roiFinderTools.get_pset_names().size());
135
136 for(const std::string& roiFinderTool : roiFinderTools.get_pset_names())
137 {
138 const fhicl::ParameterSet& roiFinderToolParamSet = roiFinderTools.get<fhicl::ParameterSet>(roiFinderTool);
139 size_t planeIdx = roiFinderToolParamSet.get<size_t>("Plane");
140
141 fROIFinderVec.at(planeIdx) = art::make_tool<icarus_tool::IROIFinder>(roiFinderToolParamSet);
142 }
143
144 std::sort(fROIFinderVec.begin(),fROIFinderVec.end(),[](const auto& left,const auto& right){return left->plane() < right->plane();});
145
146 fDeconvolution = art::make_tool<icarus_tool::IDeconvolution>(pset.get<fhicl::ParameterSet>("Deconvolution"));
147
148 fDigitModuleLabel = pset.get< std::string > ("DigitModuleLabel", "daq");
149 fNoiseSource = pset.get< unsigned short >("NoiseSource", 3);
150 fSaveWireWF = pset.get< int > ("SaveWireWF" );
151 fMinAllowedChanStatus = pset.get< int > ("MinAllowedChannelStatus");
152 fTruncRMSThreshold = pset.get< float > ("TruncRMSThreshold", 6.);
153 fTruncRMSMinFraction = pset.get< float > ("TruncRMSMinFraction", 0.6);
154 fOutputHistograms = pset.get< bool > ("OutputHistograms", true);
155
156 fSpillName.clear();
157
158 size_t pos = fDigitModuleLabel.find(":");
159 if( pos!=std::string::npos )
160 {
161 fSpillName = fDigitModuleLabel.substr( pos+1 );
162 fDigitModuleLabel = fDigitModuleLabel.substr( 0, pos );
163 }
164
165 if (fOutputHistograms)
166 {
167 // Access ART's TFileService, which will handle creating and writing
168 // histograms and n-tuples for us.
169 art::ServiceHandle<art::TFileService> tfs;
170
171 fPedestalOffsetVec.resize(3);
172 fFullRMSVec.resize(3);
173 fTruncRMSVec.resize(3);
174 fNumTruncBinsVec.resize(3);
175 fPedByChanVec.resize(3);
176 fTruncRMSByChanVec.resize(3);
177 fNumROIsHistVec.resize(3);
178 fROILenHistVec.resize(3);
179
180 constexpr geo::TPCID tpcid{0, 0};
181 for(size_t planeIdx = 0; planeIdx < 3; planeIdx++)
182 {
183 geo::PlaneID const planeID(tpcid, planeIdx);
184 fPedestalOffsetVec[planeIdx] = tfs->make<TH1F>( Form("PedPlane_%02zu",planeIdx), ";Pedestal Offset (ADC);", 100, -5., 5.);
185 fFullRMSVec[planeIdx] = tfs->make<TH1F>( Form("RMSFPlane_%02zu",planeIdx), "Full RMS;RMS (ADC);", 100, 0., 10.);
186 fTruncRMSVec[planeIdx] = tfs->make<TH1F>( Form("RMSTPlane_%02zu",planeIdx), "Truncated RMS;RMS (ADC);", 100, 0., 10.);
187 fNumTruncBinsVec[planeIdx] = tfs->make<TH1F>( Form("NTruncBins_%02zu",planeIdx), ";# bins", 640, 0., 6400.);
188 fPedByChanVec[planeIdx] = tfs->make<TProfile>(Form("PedByWirePlane_%02zu",planeIdx), ";Wire#", fGeometry->Nwires(planeID), 0., fGeometry->Nwires(planeID), -5., 5.);
189 fTruncRMSByChanVec[planeIdx] = tfs->make<TProfile>(Form("TruncRMSByWirePlane_%02zu",planeIdx), ";Wire#", fGeometry->Nwires(planeID), 0., fGeometry->Nwires(planeID), 0., 10.);
190 fNumROIsHistVec[planeIdx] = tfs->make<TH1F>( Form("NROISplane_%02zu",planeIdx), ";# ROIs;", 100, 0, 100);
191 fROILenHistVec[planeIdx] = tfs->make<TH1F>( Form("ROISIZEplane_%02zu",planeIdx), ";ROI size;", 500, 0, 500);
192 }
193 }
194
195 return;
196}

210void RecoWireROIICARUS::produce(art::Event& evt)
211{
212 //get pedestal conditions
213 const lariov::DetPedestalProvider& pedestalRetrievalAlg = art::ServiceHandle<lariov::DetPedestalService>()->GetPedestalProvider();
214
215 // make a collection of Wires
216 std::unique_ptr<std::vector<recob::Wire> > wirecol(new std::vector<recob::Wire>);
217 // ... and an association set
218 std::unique_ptr<art::Assns<raw::RawDigit,recob::Wire> > WireDigitAssn(new art::Assns<raw::RawDigit,recob::Wire>);
219
220 // Read in the digit List object(s).
221 art::Handle< std::vector<raw::RawDigit> > digitVecHandle;
222
223 if(fSpillName.size()>0) evt.getByLabel(fDigitModuleLabel, fSpillName, digitVecHandle);
224 else evt.getByLabel(fDigitModuleLabel, digitVecHandle);
225
226 if (!digitVecHandle->size())
227 {
228 evt.put(std::move(wirecol), fSpillName);
229 evt.put(std::move(WireDigitAssn), fSpillName);
230 return;
231 }
232
233 raw::ChannelID_t channel = raw::InvalidChannelID; // channel number
234
235 const lariov::ChannelStatusProvider& chanFilt = art::ServiceHandle<lariov::ChannelStatusService>()->GetProvider();
236
237 auto const clockData = art::ServiceHandle<detinfo::DetectorClocksService const>()->DataFor(evt);
238 double const samplingRate = sampling_rate(clockData);
239 // loop over all wires
240 wirecol->reserve(digitVecHandle->size());
241 for(size_t rdIter = 0; rdIter < digitVecHandle->size(); ++rdIter)
242 {
243 // vector that will be moved into the Wire object
244 recob::Wire::RegionsOfInterest_t ROIVec;
245
246 // get the reference to the current raw::RawDigit
247 art::Ptr<raw::RawDigit> digitVec(digitVecHandle, rdIter);
248 channel = digitVec->Channel();
249
250 // The following test is meant to be temporary until the "correct" solution is implemented
251 if (!chanFilt.IsPresent(channel)) continue;
252
253 // Testing an idea about rejecting channels
254 if (digitVec->GetPedestal() < 0.) continue;
255
256 float pedestal = 0.;
257
258 // skip bad channels
259 if( chanFilt.Status(channel) >= fMinAllowedChanStatus)
260 {
261 size_t dataSize = digitVec->Samples();
262
263 // Recover the plane info
264 std::vector<geo::WireID> wids = fGeometry->ChannelToWire(channel);
265 const geo::PlaneID& planeID = wids[0].planeID();
266
267 // vector holding uncompressed adc values
268 std::vector<short> rawadc(dataSize);
269
270 // uncompress the data
271 raw::Uncompress(digitVec->ADCs(), rawadc, digitVec->Compression());
272
273 // loop over all adc values and subtract the pedestal
274 // When we have a pedestal database, can provide the digit timestamp as the third argument of GetPedestalMean
275 pedestal = pedestalRetrievalAlg.PedMean(channel);
276
277 // Get the pedestal subtracted data, centered in the deconvolution vector
278 std::vector<float> rawAdcLessPedVec(dataSize);
279
280 std::transform(rawadc.begin(),rawadc.end(),rawAdcLessPedVec.begin(),std::bind(std::minus<short>(),std::placeholders::_1,pedestal));
281
282 // It seems there are deviations from the pedestal when using wirecell for noise filtering
283 float raw_noise = fixTheFreakingWaveform(rawAdcLessPedVec, channel, rawAdcLessPedVec);
284
285 // Recover a measure of the noise on the channel for use in the ROI finder
286 //float raw_noise = getTruncatedRMS(rawAdcLessPedVec);
287
288 // Try smoothing the input waveform
289// std::vector<float> rawAdcSmoothVec;
290// fWaveformTool->medianSmooth(rawAdcLessPedVec,rawAdcSmoothVec);
291
292 // vector of candidate ROI begin and end bins
293 icarus_tool::IROIFinder::CandidateROIVec candRoiVec;
294
295 // Now find the candidate ROI's
296 fROIFinderVec.at(planeID.Plane)->FindROIs(rawAdcLessPedVec, channel, fEventCount, raw_noise, candRoiVec);
297
298 // Do the deconvolution
299 fDeconvolution->Deconvolve(rawAdcLessPedVec, samplingRate, channel, candRoiVec, ROIVec);
300
301 // Make some histograms?
302 if (fOutputHistograms)
303 {
304 // First up, determine what kind of wire we have
305 std::vector<geo::WireID> wids = fGeometry->ChannelToWire(channel);
306 const geo::PlaneID& planeID = wids[0].planeID();
307
308 fNumROIsHistVec.at(planeID.Plane)->Fill(candRoiVec.size(), 1.);
309
310 for(const auto& pair : candRoiVec)
311 fROILenHistVec.at(planeID.Plane)->Fill(pair.second-pair.first, 1.);
312 }
313 } // end if not a bad channel
314
315 // create the new wire directly in wirecol
316 wirecol->push_back(recob::WireCreator(std::move(ROIVec),*digitVec).move());
317
318 // add an association between the last object in wirecol
319 // (that we just inserted) and digitVec
320 if (!util::CreateAssn(*this, evt, *wirecol, digitVec, *WireDigitAssn, fSpillName))
321 {
322 throw art::Exception(art::errors::ProductRegistrationFailure)
323 << "Can't associate wire #" << (wirecol->size() - 1)
324 << " with raw digit #" << digitVec.key();
325 } // if failed to add association
326 // DumpWire(wirecol->back()); // for debugging
327 }
328
329 if(wirecol->size() == 0)
330 mf::LogWarning("RecoWireROIICARUS") << "No wires made for this event.";
331
332 //Make Histogram of recob::wire objects from Signal() vector
333 // get access to the TFile service
334 if ( fSaveWireWF ){
335 art::ServiceHandle<art::TFileService> tfs;
336 for (size_t wireN = 0; wireN < wirecol->size(); wireN++){
337 std::vector<float> sigTMP = wirecol->at(wireN).Signal();
338 TH1D* fWire = tfs->make<TH1D>(Form("Noise_Evt%04zu_N%04zu",fEventCount,wireN), ";Noise (ADC);",
339 sigTMP.size(),-0.5,sigTMP.size()-0.5);
340 for (size_t tick = 0; tick < sigTMP.size(); tick++){
341 fWire->SetBinContent(tick+1, sigTMP.at(tick) );
342 }
343 }
344 }
345
346 evt.put(std::move(wirecol), fSpillName);
347 evt.put(std::move(WireDigitAssn), fSpillName);
348
349 fEventCount++;
350
351 return;
352} // produce

354float RecoWireROIICARUS::getTruncatedRMS(const std::vector<float>& waveform) const
355{
356 // do rms calculation - the old fashioned way and over all adc values
357 std::vector<float> locWaveform = waveform;
358
359 // sort in ascending order so we can truncate the sume
360 std::sort(locWaveform.begin(), locWaveform.end(),[](const auto& left, const auto& right){return std::fabs(left) < std::fabs(right);});
361
362 float threshold = fTruncRMSThreshold;
363
364 std::vector<float>::iterator threshItr = std::find_if(locWaveform.begin(),locWaveform.end(),[threshold](const auto& val){return std::fabs(val) > threshold;});
365
366 int minNumBins = std::max(int(fTruncRMSMinFraction * locWaveform.size()),int(std::distance(locWaveform.begin(),threshItr)));
367
368 // Get the truncated sum
369 float truncRms = std::inner_product(locWaveform.begin(), locWaveform.begin() + minNumBins, locWaveform.begin(), 0.);
370
371 truncRms = std::sqrt(std::max(0.,truncRms / double(minNumBins)));
372
373 return truncRms;
374}

376float RecoWireROIICARUS::fixTheFreakingWaveform(const std::vector<float>& waveform, raw::ChannelID_t channel, std::vector<float>& fixedWaveform)
377{
378 // Get the truncated mean and rms
379 float fullRMS;
380 float truncRMS;
381 float truncMean;
382 float nSig(2.0); // make tight constraint
383 int nTrunc;
384 int range;
385
386 fixedWaveform.resize(waveform.size());
387
388 fWaveformTool.getPedestalCorrectedWaveform(waveform, fixedWaveform, nSig, truncMean, fullRMS, truncRMS, nTrunc, range);
389
390 // Fill histograms
391 if (fOutputHistograms)
392 {
393 std::vector<geo::WireID> wids = fGeometry->ChannelToWire(channel);
394
395 // Recover plane and wire in the plane
396 size_t plane = wids[0].Plane;
397 size_t wire = wids[0].Wire;
398
399 fPedestalOffsetVec[plane]->Fill(truncMean,1.);
400 fFullRMSVec[plane]->Fill(fullRMS, 1.);
401 fTruncRMSVec[plane]->Fill(truncRMS, 1.);
402 fNumTruncBinsVec[plane]->Fill(nTrunc, 1.);
403 fPedByChanVec[plane]->Fill(wire, truncMean, 1.);
404 fTruncRMSByChanVec[plane]->Fill(wire, truncRMS, 1.);
405 }
406
407 return truncRMS;
408}
