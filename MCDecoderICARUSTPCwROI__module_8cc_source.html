214MCDecoderICARUSTPCwROI::MCDecoderICARUSTPCwROI(fhicl::ParameterSet const & pset, art::ProcessingFrame const& frame) :
215 art::ReplicatedProducer(pset, frame),
216 fLogCategory("MCDecoderICARUSTPCwROI"),fNumEvent(0), fNumROPs(0)
217{
218 fGeometry = art::ServiceHandle<geo::Geometry const>{}.get();
219 fChannelMap = art::ServiceHandle<icarusDB::IICARUSChannelMap const>{}.get();
220
221 configure(pset);
222
223 // Check the concurrency
224 int max_concurrency = art::Globals::instance()->nthreads();
225
226 mf::LogDebug("MCDecoderICARUSTPCwROI") << " ==> concurrency: " << max_concurrency << std::endl;
227
228 // Recover the vector of fhicl parameters for the ROI tools
229 const fhicl::ParameterSet& decoderToolParams = pset.get<fhicl::ParameterSet>("DecoderTool");
230
231 fDecoderToolVec.resize(max_concurrency);
232
233 for(auto& decoderTool : fDecoderToolVec)
234 {
235 // Get instance of tool
236 decoderTool = art::make_tool<INoiseFilter>(decoderToolParams);
237 }
238
239 // Set up our "produces"
240 // Note that we can have multiple instances input to the module
241 // Our convention will be to create a similar number of outputs with the same instance names
242 for(const auto& instanceLabel : fOutInstanceLabelVec)
243 {
244 produces<std::vector<raw::RawDigit>>(instanceLabel);
245 produces<std::vector<recob::Wire>>(instanceLabel);
246
247 if (fOutputRawWaveform)
248 produces<std::vector<raw::RawDigit>>(instanceLabel + fOutputRawWavePath);
249
250 if (fOutputCorrection)
251 produces<std::vector<raw::RawDigit>>(instanceLabel + fOutputCoherentPath);
252 }
253
254 // Set up a WireID to ROP plane number table
255 PlaneToWireOffsetMap planeToLastWireOffsetMap;
256
257 for(size_t cryoIdx = 0; cryoIdx < 2; cryoIdx++)
258 {
259 for(size_t logicalTPCIdx = 0; logicalTPCIdx < 4; logicalTPCIdx++)
260 {
261 for(size_t planeIdx = 0; planeIdx < 3; planeIdx++)
262 {
263 geo::PlaneID planeID(cryoIdx,logicalTPCIdx,planeIdx);
264
265 raw::ChannelID_t channel = fGeometry->PlaneWireToChannel(geo::WireID(planeID, 0));
266
267 readout::ROPID ropID = fGeometry->ChannelToROP(channel);
268
269 fPlaneToROPPlaneMap[planeID] = ropID.ROP;
270 fPlaneToWireOffsetMap[planeID] = channel;
271 planeToLastWireOffsetMap[planeID] = fGeometry->PlaneWireToChannel(geo::WireID(planeID, fGeometry->Nwires(planeID)));
272 fROPToNumWiresMap[ropID.ROP] = fGeometry->Nwires(planeID);
273
274 // Special case handling
275// if (ropID.ROP > 1) fROPToNumWiresMap[ropID.ROP] *= 2;
276
277 if (ropID.ROP > fNumROPs) fNumROPs = ropID.ROP;
278
279 // Watch for the middle induction and collection plane logical TPC split
280 if (ropID.ROP > 1 && (logicalTPCIdx == 1 || logicalTPCIdx == 3))
281 {
282 geo::PlaneID tempID(cryoIdx,logicalTPCIdx-1,planeIdx);
283
284 fPlaneToWireOffsetMap[planeID] = fPlaneToWireOffsetMap[tempID];
285 fROPToNumWiresMap[ropID.ROP] = planeToLastWireOffsetMap[planeID] - fPlaneToWireOffsetMap[planeID];
286 }
287
288 // Diagnostic output if requested
289 mf::LogDebug(fLogCategory) << "Initializing C/T/P: " << planeID.Cryostat << "/" << planeID.TPC << "/" << planeID.Plane << ", base channel: " << fPlaneToWireOffsetMap[planeID] << ", ROP: " << ropID << ", index: " << ropID.ROP;
290
291 }
292 }
293 }
294
295 fNumROPs++;
296
297 // We need to build a mapping fronm channel to a readout board/wire pair
298 // Get the board ids for this fragment
299 const icarusDB::TPCReadoutBoardToChannelMap& readoutBoardToChannelMap = fChannelMap->getReadoutBoardToChannelMap();
300
301 for(const auto& boardPair : readoutBoardToChannelMap)
302 {
303 // The board pair will give us the readout board and a vector of "wires"
304 unsigned int readoutBoardID = boardPair.first;
305
306 // Loop through the vector of wires on this board
307 for(unsigned int wireIdx = 0; wireIdx < boardPair.second.second.size(); wireIdx++)
308 {
309 unsigned int channelID = boardPair.second.second[wireIdx].first;
310 unsigned int planeID = boardPair.second.second[wireIdx].second;
311
312 fChannelToBoardWirePlaneMap[channelID] = BoardWirePlanePair(readoutBoardID,WirePlanePair(wireIdx,planeID));
313 }
314 }
315
316
317 // Report.
318 mf::LogInfo("MCDecoderICARUSTPCwROI") << "MCDecoderICARUSTPCwROI configured\n";
319}

354//----------------------------------------------------------------------------
363void MCDecoderICARUSTPCwROI::produce(art::Event & event, art::ProcessingFrame const&)
364{
365 ++fNumEvent;
366
367 mf::LogDebug("MCDecoderICARUSTPCwROI") << "**** Processing raw data fragments ****" << std::endl;
368
369 // Check the concurrency
370 int max_concurrency = tbb::this_task_arena::max_concurrency();
371
372 mf::LogDebug("MCDecoderICARUSTPCwROI") << " ==> concurrency: " << max_concurrency << std::endl;
373
374 cet::cpu_timer theClockTotal;
375
376 theClockTotal.start();
377
378 // Loop through the list of input daq fragment collections one by one
379 // We are not trying to multi thread at this stage because we are trying to control
380 // overall memory usage at this level. We'll multi thread internally...
381 size_t instanceIdx(0);
382
383 for(const auto& rawDigitLabel : fRawDigitLabelVec)
384 {
385 art::Handle<artdaq::Fragments> daq_handle;
386 event.getByLabel(rawDigitLabel, daq_handle);
387 //std::cout << "\nLabel=" << rawDigitLabel << std::endl;
388
389 ConcurrentRawDigitCol concurrentRawDigits;
390 ConcurrentRawDigitCol concurrentRawRawDigits;
391 ConcurrentRawDigitCol coherentRawDigits;
392 ConcurrentWireCol concurrentROIs;
393
394 PlaneIdxToImageMap planeIdxToImageMap;
395 PlaneIdxToChannelMap planeIdxToChannelMap;
396
397 ChannelArrayPairVec channelArrayPairVec(fNumROPs);
398
399 // Because the arrays can be variable size we need to loop to initialize
400 for(size_t ropIdx = 0; ropIdx < fNumROPs; ropIdx++)
401 {
402 ChannelArrayPair& channelArrayPair = channelArrayPairVec[ropIdx];
403
404 channelArrayPair.first.resize(fROPToNumWiresMap[ropIdx]);
405 channelArrayPair.second.resize(fROPToNumWiresMap[ropIdx],icarus_signal_processing::VectorFloat(4096));
406
407 mf::LogDebug("MCDecoderICARUSTPCwROI") << "**> Initializing ropIdx: " << ropIdx << " channelPairVec to " << channelArrayPair.first.size() << " channels with " << channelArrayPair.second[0].size() << " ticks" << std::endl;
408 }
409
410 mf::LogDebug("MCDecoderICARUSTPCwROI") << "****> Let's get ready to rumble!" << std::endl;
411
412 // Now let's process the resulting images
413 auto const clockData = art::ServiceHandle<detinfo::DetectorClocksService>()->DataFor(event);
414
415 // ... repackage the input MC data to format suitable for noise processing
416 processSingleLabel(event, rawDigitLabel, clockData, channelArrayPairVec, fCoherentNoiseGrouping, concurrentRawDigits, concurrentRawRawDigits, coherentRawDigits, concurrentROIs);
417
418// multiThreadImageProcessing imageProcessing(*this, clockData, channelArrayPairVec, fCoherentNoiseGrouping, concurrentRawDigits, concurrentRawRawDigits, coherentRawDigits, concurrentROIs);
419//
420// tbb::parallel_for(tbb::blocked_range<size_t>(0, fNumROPs), imageProcessing);
421
422 // Copy the raw digits from the concurrent vector to our output vector
423 RawDigitCollectionPtr rawDigitCollection = std::make_unique<std::vector<raw::RawDigit>>(std::move_iterator(concurrentRawDigits.begin()),
424 std::move_iterator(concurrentRawDigits.end()));
425
426 // Want the RawDigits to be sorted in channel order... has to be done somewhere so why not now?
427 std::sort(rawDigitCollection->begin(),rawDigitCollection->end(),[](const auto& left,const auto&right){return left.Channel() < right.Channel();});
428
429 // What did we get back?
430 mf::LogDebug("MCDecoderICARUSTPCwROI") << "****> Total size of map: " << planeIdxToImageMap.size() << std::endl;
431 for(const auto& planeImagePair : planeIdxToImageMap)
432 {
433 mf::LogDebug("MCDecoderICARUSTPCwROI") << " - plane: " << planeImagePair.first << " has " << planeImagePair.second.size() << " wires" << std::endl;
434 }
435
436 // Now transfer ownership to the event store
437 event.put(std::move(rawDigitCollection), fOutInstanceLabelVec[instanceIdx]);
438
439 // Do the same to output the candidate ROIs
440 WireCollectionPtr wireCollection = std::make_unique<std::vector<recob::Wire>>(std::move_iterator(concurrentROIs.begin()),
441 std::move_iterator(concurrentROIs.end()));
442
443 std::sort(wireCollection->begin(),wireCollection->end(),[](const auto& left, const auto& right){return left.Channel() < right.Channel();});
444
445 event.put(std::move(wireCollection), fOutInstanceLabelVec[instanceIdx]);
446
447 if (fOutputRawWaveform)
448 {
449 // Copy the raw digits from the concurrent vector to our output vector
450 RawDigitCollectionPtr rawRawDigitCollection = std::make_unique<std::vector<raw::RawDigit>>(std::move_iterator(concurrentRawRawDigits.begin()),
451 std::move_iterator(concurrentRawRawDigits.end()));
452
453 // Want the RawDigits to be sorted in channel order... has to be done somewhere so why not now?
454 std::sort(rawRawDigitCollection->begin(),rawRawDigitCollection->end(),[](const auto& left,const auto&right){return left.Channel() < right.Channel();});
455
456 // Now transfer ownership to the event store
457 event.put(std::move(rawRawDigitCollection),fOutInstanceLabelVec[instanceIdx] + fOutputRawWavePath);
458 }
459
460 if (fOutputCorrection)
461 {
462 // Copy the raw digits from the concurrent vector to our output vector
463 RawDigitCollectionPtr coherentCollection = std::make_unique<std::vector<raw::RawDigit>>(std::move_iterator(coherentRawDigits.begin()),
464 std::move_iterator(coherentRawDigits.end()));
465
466 // Want the RawDigits to be sorted in channel order... has to be done somewhere so why not now?
467 std::sort(coherentCollection->begin(),coherentCollection->end(),[](const auto& left,const auto&right){return left.Channel() < right.Channel();});
468
469 // Now transfer ownership to the event store
470 event.put(std::move(coherentCollection),fOutInstanceLabelVec[instanceIdx] + fOutputCoherentPath);
471 }
472
473 instanceIdx++;
474 }
475
476 theClockTotal.stop();
477
478 double totalTime = theClockTotal.accumulated_real_time();
479
480 mf::LogInfo(fLogCategory) << "==> MCDecoderICARUSTPCwROI total time: " << totalTime << std::endl;
481
482 return;
483}

485void MCDecoderICARUSTPCwROI::processSingleLabel(art::Event& event,
486 const art::InputTag& inputLabel,
487 detinfo::DetectorClocksData const& clockData,
488 ChannelArrayPairVec const& channelArrayPairVec,
489 size_t const& coherentNoiseGrouping,
490 ConcurrentRawDigitCol& concurrentRawDigits,
491 ConcurrentRawDigitCol& concurrentRawRawDigits,
492 ConcurrentRawDigitCol& coherentRawDigits,
493 ConcurrentWireCol& concurrentROIs) const
494{
495 cet::cpu_timer theClockProcess;
496
497 theClockProcess.start();
498
499 // Read in the digit List object(s).
500 art::Handle< std::vector<raw::RawDigit> > digitVecHandle;
501 event.getByLabel(inputLabel, digitVecHandle);
502 const raw::RawDigit invalidRD;
503
504 // Require a valid handle
505 if (digitVecHandle.isValid() && digitVecHandle->size()>0 )
506 {
507 const unsigned int dataSize = digitVecHandle->front().Samples(); //size of raw data vectors
508
509 std::map<unsigned int, std::vector<const raw::RawDigit*> > boardToRawDigitMap;
510
511 for(const raw::RawDigit& rawDigit: *digitVecHandle)
512 {
513 raw::ChannelID_t channel = rawDigit.Channel();
514 ChannelToBoardWirePlaneMap::const_iterator channelToBoardItr = fChannelToBoardWirePlaneMap.find(channel);
515
516 if (channelToBoardItr == fChannelToBoardWirePlaneMap.end())
517 {
518 std::cout << "********************************************************************************" << std::endl;
519 std::cout << "********* We did not find channel " << channel << "*****************************" << std::endl;
520 std::cout << "********************************************************************************" << std::endl;
521 continue;
522 }
523
524 unsigned int board = channelToBoardItr->second.first;
525 auto mapIter = boardToRawDigitMap.find(board);
526 unsigned int wireIdx = channelToBoardItr->second.second.first;
527
528 if (mapIter == boardToRawDigitMap.end())
529 {
530 boardToRawDigitMap.insert({board,std::vector<const raw::RawDigit*>(64,&invalidRD)});
531 mapIter = boardToRawDigitMap.find(board);
532 }
533
534 mapIter->second[wireIdx] = &rawDigit;
535 }
536
537 //std::cout << "boardToRawDigitMap.size()=" << boardToRawDigitMap.size() << std::endl;
538 //for (auto elem : boardToRawDigitMap) {
539 tbb::parallel_for (static_cast<std::size_t>(0),boardToRawDigitMap.size(),[&](size_t& r)
540 {
541 auto elem = std::next(boardToRawDigitMap.begin(),r);
542
543 //std::cout << "board=" << elem->first << " nch=" << elem->second.size() << " first=" << elem->second[0]->Channel() << " last=" << elem->second[elem->second.size()-1]->Channel() << std::endl;
544
545 std::vector<const raw::RawDigit*> rawDigitVec;
546
547 for (auto e : elem->second) if (raw::isValidChannelID(e->Channel())) rawDigitVec.push_back(e);
548
549 ChannelArrayPair chanArr;
550 for (const auto rawDigit : rawDigitVec)
551 {
552 // Declare a temporary digit holder and resize it if downsizing the waveform
553 raw::RawDigit::ADCvector_t rawDataVec(dataSize);
554
555 // Decompress data into local holder
556 raw::Uncompress(rawDigit->ADCs(), rawDataVec, rawDigit->Compression());
557
558 // Fill into the data structure
559 raw::ChannelID_t channel = rawDigit->Channel();
560 unsigned int planeIdx = fChannelToBoardWirePlaneMap.find(channel)->second.second.second;
561
562 icarus_signal_processing::VectorFloat boardDataVec(rawDataVec.cbegin(),rawDataVec.cend());
563 chanArr.first.push_back(daq::INoiseFilter::ChannelPlanePair(channel,planeIdx));
564 chanArr.second.push_back(boardDataVec);
565 }
566
567 if (chanArr.second.size() < 64) processSingleImage(clockData, chanArr, chanArr.second.size(), concurrentRawDigits, concurrentRawRawDigits, coherentRawDigits, concurrentROIs);
568 else processSingleImage(clockData, chanArr, coherentNoiseGrouping, concurrentRawDigits, concurrentRawRawDigits, coherentRawDigits, concurrentROIs);
569 });
570 }
571
572 theClockProcess.stop();
573
574 double totalTime = theClockProcess.accumulated_real_time();
575
576 mf::LogDebug(fLogCategory) << "--> Exiting fragment processing for thread: " << tbb::this_task_arena::current_thread_index() << ", time: " << totalTime << std::endl;
577
578 return;
579}

581void MCDecoderICARUSTPCwROI::processSingleImage(const detinfo::DetectorClocksData& clockData,
582 const ChannelArrayPair& channelArrayPair,
583 size_t coherentNoiseGrouping,
584 ConcurrentRawDigitCol& concurrentRawDigitCol,
585 ConcurrentRawDigitCol& concurrentRawRawDigitCol,
586 ConcurrentRawDigitCol& coherentRawDigitCol,
587 ConcurrentWireCol& concurrentROIs) const
588{
589 // Let's go through and fill the output vector
590 const daq::INoiseFilter::ChannelPlaneVec& channelVec = channelArrayPair.first;
591 const icarus_signal_processing::ArrayFloat& dataArray = channelArrayPair.second;
592
593 unsigned int numChannels = dataArray.size();
594 unsigned int numTicks = dataArray[0].size();
595
596 // Recover pointer to the decoder needed here
597 INoiseFilter* decoderTool = fDecoderToolVec[tbb::this_task_arena::current_thread_index()].get();
598
599 //process_fragment(event, rawfrag, product_collection, header_collection);
600 decoderTool->process_fragment(clockData, channelVec, dataArray, coherentNoiseGrouping);
601
602 // Now set up for output, we need to convert back from float to short int so use this
603 raw::RawDigit::ADCvector_t wvfm(numTicks);
604
605 // We need to recalculate pedestals for the noise corrected waveforms
606 icarus_signal_processing::WaveformTools<float> waveformTools;
607
608 // Local storage for recomputing the the pedestals for the noise corrected data
609 float localPedestal(0.);
610 float localFullRMS(0.);
611 float localTruncRMS(0.);
612 int localNumTruncBins(0);
613 int localRangeBins(0);
614
615 float sigmaCut(fSigmaForTruncation);
616
617 // Loop over the channels to recover the RawDigits after filtering
618 for(size_t chanIdx = 0; chanIdx < numChannels; chanIdx++)
619 {
620 // Skip if no channel data (plane is wrong)
621 if (channelVec[chanIdx].second > 2) continue;
622
623 raw::ChannelID_t channel = channelVec[chanIdx].first;
624
625 if (fOutputRawWaveform)
626 {
627 //const icarus_signal_processing::VectorFloat& waveform = decoderTool->getPedCorWaveforms()[chanIdx];
628 const icarus_signal_processing::VectorFloat& waveform = decoderTool->getRawWaveforms()[chanIdx];
629
630 // Need to convert from float to short int
631 std::transform(waveform.begin(),waveform.end(),wvfm.begin(),[](const auto& val){return short(std::round(val));});
632
633 ConcurrentRawDigitCol::iterator newRawObjItr = concurrentRawRawDigitCol.emplace_back(channel,wvfm.size(),wvfm);
634
635 newRawObjItr->SetPedestal(decoderTool->getPedestalVals()[chanIdx],decoderTool->getFullRMSVals()[chanIdx]);
636 }
637
638 if (fOutputCorrection)
639 {
640 const icarus_signal_processing::VectorFloat& corrections = decoderTool->getCorrectedMedians()[chanIdx];
641
642 // Need to convert from float to short int
643 std::transform(corrections.begin(),corrections.end(),wvfm.begin(),[](const auto& val){return short(std::round(val));});
644
645 //ConcurrentRawDigitCol::iterator newRawObjItr = coherentRawDigitCol.emplace_back(channel,wvfm.size(),wvfm);
646 ConcurrentRawDigitCol::iterator newRawObjItr = coherentRawDigitCol.push_back(raw::RawDigit(channel,wvfm.size(),wvfm));
647
648 newRawObjItr->SetPedestal(0.,0.);
649 }
650
651 // Recover the denoised waveform
652 const icarus_signal_processing::VectorFloat& waveLessCoherentVec = decoderTool->getWaveLessCoherent()[chanIdx];
653 icarus_signal_processing::VectorFloat denoised(waveLessCoherentVec.size());
654
655 // Now determine the pedestal and correct for it
656 waveformTools.getPedestalCorrectedWaveform(waveLessCoherentVec,
657 denoised,
658 sigmaCut,
659 localPedestal,
660 localFullRMS,
661 localTruncRMS,
662 localNumTruncBins,
663 localRangeBins);
664
665 // Need to convert from float to short int
666 std::transform(denoised.begin(),denoised.end(),wvfm.begin(),[](const auto& val){return short(std::round(val));});
667
668 ConcurrentRawDigitCol::iterator newObjItr = concurrentRawDigitCol.emplace_back(channel,wvfm.size(),wvfm);
669
670 newObjItr->SetPedestal(localPedestal,localTruncRMS);
671
672 // And, finally, the ROIs
673 const icarus_signal_processing::VectorBool& chanROIs = decoderTool->getROIVals()[chanIdx];
674 recob::Wire::RegionsOfInterest_t ROIVec;
675
676 // Go through candidate ROIs and create Wire ROIs
677 size_t roiIdx = 0;
678
679 while(roiIdx < chanROIs.size())
680 {
681 size_t roiStartIdx = roiIdx;
682
683 while(roiIdx < chanROIs.size() && chanROIs[roiIdx]) roiIdx++;
684
685 if (roiIdx > roiStartIdx)
686 {
687 std::vector<float> holder(roiIdx - roiStartIdx, 10.);
688
689 ROIVec.add_range(roiStartIdx, std::move(holder));
690 }
691
692 roiIdx++;
693 }
694
695 concurrentROIs.push_back(recob::WireCreator(std::move(ROIVec),channel,fGeometry->View(channel)).move());
696 }//loop over channel indices
697
698 return;
699}
