554 template <typename Coll, typename=std::enable_if_t<std::is_rvalue_reference_v<Coll&&>>>
555 std::vector<typename std::decay_t<Coll>::value_type::value_type> flatten
556 (Coll&& src)
557 {
558
559 using SrcColl_t = typename std::decay_t<Coll>::value_type;
560 using value_type = typename SrcColl_t::value_type;
561 using DestColl_t = std::vector<value_type>;
562
563 DestColl_t dest;
564 for (SrcColl_t& coll: src) {
565 for (value_type& value: coll) dest.push_back(std::move(value));
566 } // for src

638void icarus::opdet::SimPMTreadout::PrimitiveManager::refreshCaches
639 (bool force /* = false */) const
640{
641 /*
642 * All in all, this class is still not thread-safe, because it would need
643 * explicit protection in the non-const methods.
644 * I am cheap and am only guaranteeing thread-safety for const ones.
645 */
646 if (!force && !fModified) return;
647
648 std::lock_guard guard{ fRefreshMutex };
649
650 if (!force && !fModified) return; // again. just in case something happened
651
652 // for correct multithreading, this would need a lock
653 using std::begin, std::end;
654 for (std::vector<electronics_time>& primitives: fPrimitives)
655 std::sort(begin(primitives), end(primitives));
656
657 fModified = false;
658
659} // icarus::opdet::SimPMTreadout::PrimitiveManager::refreshCaches()

831// -----------------------------------------------------------------------------
832void icarus::opdet::SimPMTreadout::produce(art::Event& event) {
833
834 using detinfo::timescales::electronics_time;
835
836 detinfo::DetectorTimings const detTimings
837 = icarus::ns::util::makeDetTimings(event);
838
839 //
840 // determine the primitives and the channels they affect
841 //
842 PrimitiveManager primitives{ fGeom };
843
844 // add the trigger primitives
845 for (art::InputTag const& tag: fPrimitiveTags) {
846 addLocalPrimitives
847 (primitives, event.getProduct<std::vector<raw::Trigger>>(tag));
848 }
849
850 // add global triggers
851 for (art::InputTag const& tag: fTriggerTags) {
852 addGlobalPrimitives
853 (primitives, event.getProduct<std::vector<raw::Trigger>>(tag));
854 }
855
856 // add artificial primitives
857 electronics_time const beamGateTime = detTimings.BeamGateTime();
858 for (microseconds const time: fFixedTriggerTimes)
859 primitives.addPrimitiveToAllChannels(beamGateTime + time);
860
861 //
862 // read the input waveforms
863 //
864 WaveformManager simWaveforms{ fNOpChannels };
865 for (art::InputTag const& tag: fWaveformTags)
866 simWaveforms.add(event.getProduct<std::vector<raw::OpDetWaveform>>(tag));
867
868
869 //
870 // create the new waveforms
871 //
872
873 // transforms trigger primitives into readout windows
874 // (we might do that once per cryostat rather than per channel...):
875 PMTReadoutWindowMaker<electronics_time> const makeReadoutWindows{
876 fReadoutParams.bufferSize * fOpticalTick,
877 fReadoutParams.preTriggerSize() * fOpticalTick,
878 fLogCategory + "_PMTReadoutWindowMaker"
879 };
880
881 std::uint64_t const beamGateTimestamp = event.time().value();
882 std::vector<std::vector<raw::OpDetWaveform>> readoutWaveforms{ fNOpChannels };
883 for (auto const channel: util::counter<raw::Channel_t>(fNOpChannels)) {
884
885 std::vector<Window_t> const readoutWindows
886 = makeReadoutWindows(primitives.forChannel(channel));
887 std::vector<raw::OpDetWaveform const*> const waveforms
888 = simWaveforms[channel];
889
890 readoutWaveforms[channel]
891 = makeWaveforms(channel, beamGateTimestamp, waveforms, readoutWindows);
892
893 } // for channels
894
895
896 //
897 // sort the output and put it into the event
898 //
899 std::vector<raw::OpDetWaveform> allReadoutWaveforms
900 = flatten(std::move(readoutWaveforms));
901
902 auto [ allReadoutWaveformsMeta, waveToMeta ]
903 = makeWaveformMetadata(allReadoutWaveforms, detTimings, event);
904
905 event.put(moveToUniquePtr(std::move(allReadoutWaveforms)));
906 event.put(moveToUniquePtr(std::move(allReadoutWaveformsMeta)));
907 event.put(moveToUniquePtr(std::move(waveToMeta)));
908
909} // icarus::opdet::SimPMTreadout::produce()

913std::vector<geo::CryostatID> icarus::opdet::SimPMTreadout::cryostatsOf
914 (raw::Trigger const& trigger) const
915{
916 TriggerBits_t const bits = trigger.TriggerBits();
917
918 std::vector<geo::CryostatID> cryostats;
919 TriggerBits_t cryoMask = fCryostatZeroMask;
920 for (auto const c: util::counter<unsigned int>(fNCryostats)) {
921
922 if (bits & cryoMask) cryostats.emplace_back(c);
923 cryoMask <<= 1;
924
925 } // for
926
927 return cryostats;
928} // geo::CryostatID icarus::opdet::SimPMTreadout::cryostatOf()

979std::vector<raw::OpDetWaveform> icarus::opdet::SimPMTreadout::makeWaveforms(
980 raw::Channel_t channel, std::uint64_t beamGateTimestamp,
981 std::vector<raw::OpDetWaveform const*> const& source,
982 std::vector<Window_t> const& readoutWindows
983) const {
984
985 // assumption: source waveforms are sorted by timestamp
986 // assumption: readoutWindows sorted by timestamp and non-contiguous
987
988 using microsecond = util::quantities::points::microsecond;
989
990 auto const toTicks = [this](nanoseconds interval)
991 { return static_cast<int>(std::round(interval / fOpticalTick)); };
992 auto const toTimestamp = [this,beamGateTimestamp](electronics_time t)
993 {
994 return beamGateTimestamp + static_cast<std::int64_t>
995 (std::round(nanoseconds(t - fDetTimings.BeamGateTime()).value()));
996 };
997
998 std::vector<raw::OpDetWaveform> waveforms;
999 waveforms.reserve(readoutWindows.size());
1000
1001 // --- BEGIN DEBUG -----------------------------------------------------------
1002 {
1003 mf::LogTrace log{ "SimPMTreadout" };
1004 log << "makeWaveforms(channel=" << channel << ", { " << source.size()
1005 << " waveforms }, { " << readoutWindows.size() << " windows })";
1006 for (auto const& [ i, w ]: util::enumerate(source)) {
1007 double const length = w->size() * fOpticalTick.value() / 1000.;
1008 log << "\n [S" << i << "](" << ((void*) w)
1009 << ") ch=" << w->ChannelNumber() << " t=" << w->TimeStamp()
1010 << " -- " << (w->TimeStamp() + length) << " l=" << length
1011 << " s=" << w->size();
1012 } // for
1013 for (auto const& [ i, r ]: util::enumerate(readoutWindows)) {
1014 log << "\n [R" << i << "]"
1015 << " " << r.start() << " -- " << r.stop() << " (" << r.width() << ")";
1016 } // for
1017 }
1018 // --- END DEBUG -----------------------------------------------------------
1019
1020 auto nextSource = source.begin();
1021 auto const sourceEnd = source.end();
1022
1023 for (Window_t const& window: readoutWindows) {
1024 // --- BEGIN DEBUG ---------------------------------------------------------
1025 MF_LOG_TRACE("SimPMTreadout") << "Window: " << window.start() << " -- "
1026 << window.stop() << " (" << window.width() << ")";
1027 // --- END DEBUG ---------------------------------------------------------
1028
1029 if (window.empty()) continue;
1030
1031 std::size_t const nSamples = toTicks(window.width());
1032 std::vector<raw::ADC_Count_t> samples(nSamples); // allocated, uninitialized
1033 auto itSample = samples.begin();
1034 auto const send [[maybe_unused]] = samples.end();
1035
1036 electronics_time neededTime = window.start();
1037
1038 // --- BEGIN DEBUG ---------------------------------------------------------
1039 auto sampleIndex [[maybe_unused]]
1040 = [&itSample,b=samples.begin()](){ return itSample - b; };
1041 MF_LOG_TRACE("SimPMTreadout")
1042 << "Expected waveform with " << samples.size() << " samples;"
1043 << " neededTime=" << neededTime << " itSample=#" << sampleIndex();
1044 // --- END DEBUG ---------------------------------------------------------
1045 while (neededTime < window.stop()) {
1046
1047 // find the first waveform that might include the time we need
1048 while (nextSource != sourceEnd) {
1049 assert(*nextSource); // no null pointers please
1050 electronics_time const srcStart
1051 { microsecond{ (*nextSource)->TimeStamp() } };
1052 electronics_time const srcStop
1053 { srcStart + (*nextSource)->size() * fOpticalTick };
1054
1055 // --- BEGIN DEBUG -----------------------------------------------------
1056 MF_LOG_TRACE("SimPMTreadout")
1057 << "Next source: " << ((void*) *nextSource) << " t=" << srcStart
1058 << " -- " << srcStop << " s=" << (*nextSource)->size();
1059 // --- END DEBUG -----------------------------------------------------
1060
1061 if (srcStop > neededTime) break;
1062 ++nextSource;
1063 } // while source available
1064
1065 // start of the data of the next useful source waveform (may be never)
1066 electronics_time const srcStart{
1067 (nextSource == sourceEnd)
1068 ? std::numeric_limits<microsecond>::max()
1069 : microsecond{ (*nextSource)->TimeStamp() }
1070 };
1071
1072 // --- BEGIN DEBUG -------------------------------------------------------
1073 if (nextSource == sourceEnd) {
1074 MF_LOG_TRACE("SimPMTreadout")
1075 << "No source can help with neededTime=" << neededTime;
1076 }
1077 else {
1078 MF_LOG_TRACE("SimPMTreadout")
1079 << "Source might help with neededTime=" << neededTime
1080 << " (srcStart=" << srcStart << ")";
1081 }
1082 // --- END DEBUG -------------------------------------------------------
1083
1084 if (neededTime < srcStart) {
1085 // the data at needed time is not available: fill with noise
1086 electronics_time const noiseEnd = std::min(srcStart, window.stop());
1087 std::size_t const nNoiseSamples = toTicks(noiseEnd - neededTime);
1088 // --- BEGIN DEBUG -----------------------------------------------------
1089 MF_LOG_TRACE("SimPMTreadout") << "neededTime=" << neededTime
1090 << " < srcStart=" << srcStart << ": noise time";
1091 // --- END DEBUG -----------------------------------------------------
1092
1093 // while the pedestal generator is working in floating point,
1094 // our source and destination is in rounded ADC counts;
1095 // we need to convert (emitting directly in ADC counts is only good if
1096 // the pedestal is already integer, otherwise we stack yet another
1097 // rounding)
1098 std::vector<util::quantities::counts_f> extendedNoise(nNoiseSamples);
1099 fPedestalGenerator->fill
1100 (channel, toTimestamp(neededTime), extendedNoise);
1101 for (util::quantities::counts_f const noise: extendedNoise) {
1102 *(itSample++)
1103 = static_cast<raw::ADC_Count_t>(std::round(noise.value()));
1104 }
1105 assert(itSample <= send);
1106
1107 neededTime = noiseEnd;
1108 // --- BEGIN DEBUG -----------------------------------------------------
1109 MF_LOG_TRACE("SimPMTreadout") << "Added " << nNoiseSamples
1110 << " samples of noise, now neededTime=" << neededTime
1111 << " itSample=#" << sampleIndex()
1112 << " window: " << window.start() << " -- " << window.stop();
1113 // --- END DEBUG -----------------------------------------------------
1114 } // if add noise
1115
1116 if (neededTime < window.stop()) {
1117 // adding noise (if any) was not enough: there must be some data to copy
1118 // --- BEGIN DEBUG -----------------------------------------------------
1119 MF_LOG_TRACE("SimPMTreadout")
1120 << "neededTime=" << neededTime << " window: "
1121 << window.start() << "--" << window.stop() << ": copy!";
1122 // --- END DEBUG -----------------------------------------------------
1123 assert(nextSource != sourceEnd);
1124 electronics_time const srcStart
1125 { microsecond{ (*nextSource)->TimeStamp() } };
1126 electronics_time const srcStop
1127 { srcStart + (*nextSource)->size() * fOpticalTick };
1128 assert(neededTime < srcStop);
1129 assert(neededTime >= srcStart);
1130
1131 electronics_time const dataEndTime = std::min(srcStop, window.stop());
1132
1133 auto const dbegin
1134 = (*nextSource)->cbegin() + toTicks(neededTime - srcStart);
1135 auto const dend = dbegin + toTicks(dataEndTime - neededTime);
1136
1137 // --- BEGIN DEBUG -----------------------------------------------------
1138 MF_LOG_TRACE("SimPMTreadout") << "Copy data from " << srcStart << " (#"
1139 << (dbegin - (*nextSource)->cbegin()) << ") to " << dataEndTime
1140 << " (#" << (dend - (*nextSource)->cbegin()) << ")"
1141 << " into itSample=#" << sampleIndex();
1142 // --- END DEBUG -----------------------------------------------------
1143
1144 itSample = std::copy(dbegin, dend, itSample);
1145 neededTime = dataEndTime;
1146
1147 // if this source is over, point to next one for the next iteration
1148 if (dend >= (*nextSource)->cend()) ++nextSource;
1149
1150 // --- BEGIN DEBUG -----------------------------------------------------
1151 MF_LOG_TRACE("SimPMTreadout") << "Copy is over, neededTime="
1152 << neededTime << " itSample=#" << sampleIndex();
1153 // --- END DEBUG -----------------------------------------------------
1154
1155 } // if copy more data
1156
1157 // --- BEGIN DEBUG -------------------------------------------------------
1158 MF_LOG_TRACE("SimPMTreadout")
1159 << "Done with this source+window, neededTime=" << neededTime
1160 << " window.stop()=" << window.stop();
1161 // --- END DEBUG -------------------------------------------------------
1162
1163 } // while needed time
1164
1165 // --- BEGIN DEBUG ---------------------------------------------------------
1166 MF_LOG_TRACE("SimPMTreadout") << "Now neededTime=" << neededTime
1167 << " window.stop()=" << window.stop() << " => done";
1168 // --- END DEBUG ---------------------------------------------------------
1169
1170 assert(itSample == send);
1171
1172 // add the new waveform (with a weird way dictated by a weird interface)
1173 waveforms.emplace_back(window.start().value(), channel);
1174 waveforms.back().Waveform() = std::move(samples);
1175
1176 // --- BEGIN DEBUG ---------------------------------------------------------
1177 {
1178 raw::OpDetWaveform const& ow [[maybe_unused]] = waveforms.back();
1179 double const length [[maybe_unused]]
1180 = ow.size() * fOpticalTick.value() / 1000.;
1181 MF_LOG_TRACE("SimPMTreadout")
1182 << "For window " << window.start() << " -- " << window.stop()
1183 << " added ch=" << ow.ChannelNumber() << " t=" << ow.TimeStamp()
1184 << " -- " << (ow.TimeStamp() + length) << " l=" << length
1185 << " s=" << ow.size() << std::endl;
1186 }
1187 // --- END DEBUG ---------------------------------------------------------
1188
1189 } // for
1190
1191 return waveforms;
1192
1193} // icarus::opdet::SimPMTreadout::makeWaveforms()
