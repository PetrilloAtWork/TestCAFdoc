238template <typename EnumType, typename... OtherBits>
239constexpr auto sbn::bits::mask(EnumType bit, OtherBits... otherBits)
240 -> mask_t<EnumType>
241{
242 unsigned int m { 1U << value(bit) };
243 if constexpr(sizeof...(OtherBits) > 0U) m |= mask(otherBits...);
244 return { m };
245} // sbn::mask()

260// -----------------------------------------------------------------------------
261template <typename EnumType>
262std::vector<std::string> sbn::bits::names(mask_t<EnumType> mask) {
263 static_assert(value(EnumType::NBits) >= 0);
264
265 using namespace std::string_literals;
266
267 constexpr std::size_t MaxBits = sizeof(mask) * 8U;
268 constexpr std::size_t NSupportedBits = value(EnumType::NBits);
269
270 std::vector<std::string> names;
271 for (std::size_t bit = 0U; bit < MaxBits; ++bit) {
272 auto const typedBit = static_cast<EnumType>(bit);
273 if (!hasBitSet(mask, typedBit)) continue;
274 names.push_back((bit > NSupportedBits)
275 ? "<unsupported ["s + std::to_string(bit) + "]>"s: name(typedBit)
276 );
277 } // for
278 return names;
279
280} // sbn::bits::names()
