466void SnippetHit3DBuilderICARUS::Hit3DBuilder(art::Event& evt, reco::HitPairList& hitPairList, RecobHitToPtrMap& clusterHitToArtPtrMap)
467{
468 // Clear the internal data structures
469 m_clusterHit2DMasterList.clear();
470 m_planeToSnippetHitMap.clear();
471 m_planeToWireToHitSetMap.clear();
472
473 // Do the one time initialization of the tick offsets.
474 if (m_PlaneToT0OffsetMap.empty())
475 {
476 // Need the detector properties which needs the clocks
477 auto const clock_data = art::ServiceHandle<detinfo::DetectorClocksService const>()->DataFor(evt);
478 auto const det_prop = art::ServiceHandle<detinfo::DetectorPropertiesService const>()->DataFor(evt, clock_data);
479
480 // Initialize the plane to hit vector map
481 for(size_t cryoIdx = 0; cryoIdx < m_geometry->Ncryostats(); cryoIdx++)
482 {
483 for(size_t tpcIdx = 0; tpcIdx < m_geometry->NTPC(); tpcIdx++)
484 {
485 for(size_t planeIdx = 0; planeIdx < m_geometry->Nplanes(); planeIdx++)
486 {
487 geo::PlaneID planeID(cryoIdx,tpcIdx,planeIdx);
488
489 if (m_useT0Offsets) m_PlaneToT0OffsetMap[planeID] = det_prop.GetXTicksOffset(planeID) - det_prop.GetXTicksOffset(geo::PlaneID(cryoIdx,tpcIdx,0));
490 else m_PlaneToT0OffsetMap[planeID] = 0.;
491 }
492 }
493 }
494 }
495
496 m_timeVector.resize(NUMTIMEVALUES, 0.);
497
498 // Get a hit refiner
499 std::unique_ptr<std::vector<recob::Hit>> outputHitPtrVec(new std::vector<recob::Hit>);
500
501 // Recover the 2D hits and then organize them into data structures which will be used in the
502 // DBscan algorithm for building the 3D clusters
503 this->CollectArtHits(evt);
504
505 // If there are no hits in our view/wire data structure then do not proceed with the full analysis
506 if (!m_planeToWireToHitSetMap.empty())
507 {
508 // Call the algorithm that builds 3D hits
509 this->BuildHit3D(hitPairList);
510
511 // If we built 3D points then attempt to output a new hit list as well
512 if (!hitPairList.empty())
513 CreateNewRecobHitCollection(evt, hitPairList, *outputHitPtrVec, clusterHitToArtPtrMap);
514 }
515
516 // Set up to make the associations (if desired)
518 std::unique_ptr<art::Assns<recob::Wire, recob::Hit>> wireAssns(new art::Assns<recob::Wire, recob::Hit>);
519
520 if (m_makeAssociations) makeWireAssns(evt, *wireAssns, clusterHitToArtPtrMap);
521
523 std::unique_ptr<art::Assns<raw::RawDigit, recob::Hit>> rawDigitAssns(new art::Assns<raw::RawDigit, recob::Hit>);
524
525 if (m_makeAssociations) makeRawDigitAssns(evt, *rawDigitAssns, clusterHitToArtPtrMap);
526
527 // Move everything into the event
528 evt.put(std::move(outputHitPtrVec));
529 evt.put(std::move(wireAssns));
530 evt.put(std::move(rawDigitAssns));
531
532 // Handle tree output too
533 if (m_outputHistograms)
534 {
535 m_tupleTree->Fill();
536
537 clear();
538 }
539
540 return;
541}

653size_t SnippetHit3DBuilderICARUS::BuildHitPairMapByTPC(PlaneSnippetHitMapItrPairVec& snippetHitMapItrVec, reco::HitPairList& hitPairList) const
654{
665 // Define functions to set start/end iterators in the loop below
666 auto SetStartIterator = [](SnippetHitMap::iterator startItr, SnippetHitMap::iterator endItr, float startTime)
667 {
668 while(startItr != endItr)
669 {
670 if (startItr->first.second < startTime) startItr++;
671 else break;
672 }
673 return startItr;
674 };
675
676 auto SetEndIterator = [](SnippetHitMap::iterator lastItr, SnippetHitMap::iterator endItr, float endTime)
677 {
678 while(lastItr != endItr)
679 {
680 if (lastItr->first.first < endTime) lastItr++;
681 else break;
682 }
683 return lastItr;
684 };
685
686 size_t nTriplets(0);
687 size_t nOrphanPairs(0);
688
689 //*********************************************************************************
690 // Basically, we try to loop until done...
691 while(1)
692 {
693 // Sort so that the earliest hit time will be the first element, etc.
694 std::sort(snippetHitMapItrVec.begin(),snippetHitMapItrVec.end(),SetStartTimeOrder());
695
696 // Make sure there are still hits on at least
697 int nPlanesWithHits(0);
698
699 for(auto& pair : snippetHitMapItrVec)
700 if (pair.first != pair.second) nPlanesWithHits++;
701
702 if (nPlanesWithHits < 2) break;
703
704 // End condition: no more hit snippets
705// if (snippetHitMapItrVec.front().first == snippetHitMapItrVec.front().second) break;
706
707 // This loop iteration's snippet iterator
708 SnippetHitMap::iterator firstSnippetItr = snippetHitMapItrVec.front().first;
709
710 // Set iterators to insure we'll be in the overlap ranges
711 SnippetHitMap::iterator snippetHitMapItr1Start = SetStartIterator(snippetHitMapItrVec[1].first, snippetHitMapItrVec[1].second, firstSnippetItr->first.first);
712 SnippetHitMap::iterator snippetHitMapItr1End = SetEndIterator( snippetHitMapItr1Start, snippetHitMapItrVec[1].second, firstSnippetItr->first.second);
713 SnippetHitMap::iterator snippetHitMapItr2Start = SetStartIterator(snippetHitMapItrVec[2].first, snippetHitMapItrVec[2].second, firstSnippetItr->first.first);
714 SnippetHitMap::iterator snippetHitMapItr2End = SetEndIterator( snippetHitMapItr2Start, snippetHitMapItrVec[2].second, firstSnippetItr->first.second);
715
716 // Since we'll use these many times in the internal loops, pre make the pairs for the second set of hits
717 size_t curHitListSize(hitPairList.size());
718 HitMatchTripletVecMap pair12Map;
719 HitMatchTripletVecMap pair13Map;
720
721 size_t n12Pairs = findGoodHitPairs(firstSnippetItr, snippetHitMapItr1Start, snippetHitMapItr1End, pair12Map);
722 size_t n13Pairs = findGoodHitPairs(firstSnippetItr, snippetHitMapItr2Start, snippetHitMapItr2End, pair13Map);
723
724 curHitListSize = hitPairList.size();
725
726 if (n12Pairs > n13Pairs) findGoodTriplets(pair12Map, pair13Map, hitPairList);
727 else findGoodTriplets(pair13Map, pair12Map, hitPairList);
728
729 if (m_saveMythicalPoints)
730 {
731 nOrphanPairs += saveOrphanPairs(pair12Map, hitPairList);
732 nOrphanPairs += saveOrphanPairs(pair13Map, hitPairList);
733 }
734
735 nTriplets += hitPairList.size() - curHitListSize;
736
737 snippetHitMapItrVec.front().first++;
738 }
739
740 mf::LogDebug("SnippetHit3D") << "--> Created " << nTriplets << " triplets of which " << nOrphanPairs << " are orphans" << std::endl;
741
742 return hitPairList.size();
743}

753 float firstPHCut = firstMaxItr != firstSnippetItr->second.end() ? m_pulseHeightFrac * (*firstMaxItr)->getHit()->PeakAmplitude() : 4096.;
754

757 {
758 // Let's focus on the largest hit in the chain
759 if (hit1->getHit()->DegreesOfFreedom() > 1 && hit1->getHit()->PeakAmplitude() < firstPHCut && hit1->getHit()->PeakAmplitude() < m_PHLowSelection) continue;

761 // Inside loop iterator

768 float secondPHCut = secondMaxItr != secondHitItr->second.end() ? m_pulseHeightFrac * (*secondMaxItr)->getHit()->PeakAmplitude() : 0.;
769
770 for(const auto& hit2 : secondHitItr->second)
771 {
772 // Again, focus on the large hits
773 if (hit2->getHit()->DegreesOfFreedom() > 1 && hit2->getHit()->PeakAmplitude() < secondPHCut && hit2->getHit()->PeakAmplitude() < m_PHLowSelection) continue;

775 reco::ClusterHit3D pair;

778 if (!makeHitPair(pair, hit1, hit2, m_hitWidthSclFctr)) continue;

782 recobHitVec[hit1->WireID().Plane] = hit1->getHit();

786
787 hitMatchMap[wireID].emplace_back(hit1,hit2,pair);

790 }

792 secondHitItr++;
793 }
794 }

862 // Temporarily deactivate following loop so we can do compiler upgrade

908 }
909 }
910 }

968bool SnippetHit3DBuilderICARUS::makeHitPair(reco::ClusterHit3D& hitPair,
969 const reco::ClusterHit2D* hit1,
970 const reco::ClusterHit2D* hit2,
971 float hitWidthSclFctr,
972 size_t hitPairCntr) const
973{
974 // Assume failure
975 bool result(false);
976
977 // Start by checking time consistency since this is fastest
978 // Wires intersect so now we can check the timing
979 float hit1Peak = hit1->getTimeTicks();
980 float hit1Sigma = hit1->getHit()->RMS();
981
982 float hit2Peak = hit2->getTimeTicks();
983 float hit2Sigma = hit2->getHit()->RMS();
984
985 // "Long hits" are an issue... so we deal with these differently
986 int hit1NDF = hit1->getHit()->DegreesOfFreedom();
987 int hit2NDF = hit2->getHit()->DegreesOfFreedom();
988
989 // Basically, allow the range to extend to the nearest end of the snippet
990 if (hit1NDF < 2) hit1Sigma *= m_LongHitStretchFctr; // This sets the range to the width of the pulse
991 if (hit2NDF < 2) hit2Sigma *= m_LongHitStretchFctr;
992
993 // The "hit sigma" is the gaussian fit sigma of the hit, we need to expand a bit to allow hit overlap efficiency
994 float hit1Width = hitWidthSclFctr * hit1Sigma;
995 float hit2Width = hitWidthSclFctr * hit2Sigma;
996
997 // Coarse check hit times are "in range"
998 if (fabs(hit1Peak - hit2Peak) <= (hit1Width + hit2Width))
999 {
1000 // Check to see that hit peak times are consistent with each other
1001 float hit1SigSq = std::pow(hit1Sigma,2);
1002 float hit2SigSq = std::pow(hit2Sigma,2);
1003 float deltaPeakTime = std::fabs(hit1Peak - hit2Peak);
1004 float sigmaPeakTime = std::sqrt(hit1SigSq + hit2SigSq);
1005
1006 if (m_outputHistograms)
1007 {
1008 // brute force... sigh...
1009 int plane1 = hit1->WireID().Plane;
1010 int plane2 = hit2->WireID().Plane;
1011 int planeIdx = (plane1 + plane2) - 1; // should be 0 for 0-1, 1 for 0-2 and 2 for 1-2
1012
1013 float deltaTicks = hit2Peak - hit1Peak;
1014
1015 if (plane1 > plane2) deltaTicks = -deltaTicks;
1016
1017 if (planeIdx == 0)
1018 {
1019 m_deltaPeakTimePlane0Vec.emplace_back(deltaTicks);
1020 m_deltaPeakSigmaPlane0Vec.emplace_back(sigmaPeakTime);
1021 }
1022 else if (planeIdx == 1)
1023 {
1024 m_deltaPeakTimePlane1Vec.emplace_back(deltaTicks);
1025 m_deltaPeakSigmaPlane1Vec.emplace_back(sigmaPeakTime);
1026 }
1027 else
1028 {
1029 m_deltaPeakTimePlane2Vec.emplace_back(deltaTicks);
1030 m_deltaPeakSigmaPlane2Vec.emplace_back(sigmaPeakTime);
1031 }
1032 }
1033
1034 // delta peak time consistency check here
1035 if (deltaPeakTime < m_deltaPeakTimeSig * sigmaPeakTime) // 2 sigma consistency? (do this way to avoid divide)
1036 {
1037 // We assume in this routine that we are looking at hits in different views
1038 // The first mission is to check that the wires intersect
1039 const geo::WireID& hit1WireID = hit1->WireID();
1040 const geo::WireID& hit2WireID = hit2->WireID();
1041
1042 geo::WireIDIntersection widIntersect;
1043
1044 if (WireIDsIntersect(hit1WireID, hit2WireID, widIntersect))
1045 {
1046 float oneOverWghts = hit1SigSq * hit2SigSq / (hit1SigSq + hit2SigSq);
1047 float avePeakTime = (hit1Peak / hit1SigSq + hit2Peak / hit2SigSq) * oneOverWghts;
1048 float averageCharge = 0.5 * (hit1->getHit()->Integral() + hit2->getHit()->Integral());
1049 float hitChiSquare = std::pow((hit1Peak - avePeakTime),2) / hit1SigSq
1050 + std::pow((hit2Peak - avePeakTime),2) / hit2SigSq;
1051
1052 float xPositionHit1(hit1->getXPosition());
1053 float xPositionHit2(hit2->getXPosition());
1054 float xPosition = (xPositionHit1 / hit1SigSq + xPositionHit2 / hit2SigSq) * hit1SigSq * hit2SigSq / (hit1SigSq + hit2SigSq);
1055
1056 Eigen::Vector3f position(xPosition, float(widIntersect.y), float(widIntersect.z)-m_zPosOffset);
1057
1058 // If to here then we need to sort out the hit pair code telling what views are used
1059 unsigned statusBits = 1 << hit1->WireID().Plane | 1 << hit2->WireID().Plane;
1060
1061 // handle status bits for the 2D hits
1062 if (hit1->getStatusBits() & reco::ClusterHit2D::USEDINPAIR) hit1->setStatusBit(reco::ClusterHit2D::SHAREDINPAIR);
1063 if (hit2->getStatusBits() & reco::ClusterHit2D::USEDINPAIR) hit2->setStatusBit(reco::ClusterHit2D::SHAREDINPAIR);
1064
1065 hit1->setStatusBit(reco::ClusterHit2D::USEDINPAIR);
1066 hit2->setStatusBit(reco::ClusterHit2D::USEDINPAIR);
1067
1068 reco::ClusterHit2DVec hitVector;
1069
1070 hitVector.resize(3, NULL);
1071
1072 hitVector[hit1->WireID().Plane] = hit1;
1073 hitVector[hit2->WireID().Plane] = hit2;
1074
1075 unsigned int cryostatIdx = hit1->WireID().Cryostat;
1076 unsigned int tpcIdx = hit1->WireID().TPC;
1077
1078 // Initialize the wireIdVec
1079 std::vector<geo::WireID> wireIDVec = {geo::WireID(cryostatIdx,tpcIdx,0,0),
1080 geo::WireID(cryostatIdx,tpcIdx,1,0),
1081 geo::WireID(cryostatIdx,tpcIdx,2,0)};
1082
1083 wireIDVec[hit1->WireID().Plane] = hit1->WireID();
1084 wireIDVec[hit2->WireID().Plane] = hit2->WireID();
1085
1086 // For compiling at the moment
1087 std::vector<float> hitDelTSigVec = {0.,0.,0.};
1088
1089 hitDelTSigVec[hit1->WireID().Plane] = deltaPeakTime / sigmaPeakTime;
1090 hitDelTSigVec[hit2->WireID().Plane] = deltaPeakTime / sigmaPeakTime;
1091
1092 // Create the 3D cluster hit
1093 hitPair.initialize(hitPairCntr,
1094 statusBits,
1095 position,
1096 averageCharge,
1097 avePeakTime,
1098 deltaPeakTime,
1099 sigmaPeakTime,
1100 hitChiSquare,
1101 0.,
1102 0.,
1103 0.,
1104 0.,
1105 hitVector,
1106 hitDelTSigVec,
1107 wireIDVec);
1108
1109 result = true;
1110 }
1111 }
1112 }
1113
1114 // Send it back
1115 return result;
1116}

1178 float xPosition(0.);

1190 float hitRMS = hit2D->getHit()->RMS();

1452float SnippetHit3DBuilderICARUS::closestApproach(const Eigen::Vector3f& P0,
1453 const Eigen::Vector3f& u0,
1454 const Eigen::Vector3f& P1,
1455 const Eigen::Vector3f& u1,
1456 float& arcLen0,
1457 float& arcLen1) const
1458{
1459 // Technique is to compute the arclength to each point of closest approach
1460 Eigen::Vector3f w0 = P0 - P1;
1461 float a(1.);
1462 float b(u0.dot(u1));
1463 float c(1.);
1464 float d(u0.dot(w0));
1465 float e(u1.dot(w0));
1466 float den(a * c - b * b);
1467
1468 arcLen0 = (b * e - c * d) / den;
1469 arcLen1 = (a * e - b * d) / den;
1470
1471 Eigen::Vector3f poca0 = P0 + arcLen0 * u0;
1472 Eigen::Vector3f poca1 = P1 + arcLen1 * u1;
1473
1474 return (poca0 - poca1).norm();
1475}

1871void SnippetHit3DBuilderICARUS::CreateNewRecobHitCollection(art::Event& event,
1872 reco::HitPairList& hitPairList,
1873 std::vector<recob::Hit>& hitPtrVec,
1874 RecobHitToPtrMap& recobHitToPtrMap)

1878
1879 if (m_enableMonitoring) theClockBuildNewHits.start();
1880
1881 // We want to build a unique list of hits from the input 3D points which, we know, reuse 2D points frequently
1882 // At the same time we need to create a new 2D hit with the "correct" WireID and replace the old 2D hit in the
1883 // ClusterHit2D object with this new one... while keeping track of the use of the old ones. My head is spinning...
1884 // Declare a set which will allow us to keep track of those CusterHit2D objects we have seen already
1885 std::set<const reco::ClusterHit2D*> visitedHit2DSet;

1889
1890 // Reserve enough memory to replace every recob::Hit we have considered (this is upper limit)
1891 hitPtrVec.reserve(m_clusterHit2DMasterList.size());
1892
1893 // Scheme is to loop through all 3D hits, then through each associated ClusterHit2D object
1894 for(reco::ClusterHit3D& hit3D : hitPairList)
1895 {
1896 reco::ClusterHit2DVec& hit2DVec = hit3D.getHits();
1897
1898 // The loop is over the index so we can recover the correct WireID to associate to the new hit when made
1899 for(size_t idx = 0; idx < hit3D.getHits().size(); idx++)
1900 {
1901 const reco::ClusterHit2D* hit2D = hit2DVec[idx];
1902
1903 if (!hit2D) continue;
1904
1905 // Have we seen this 2D hit already?
1906 if (visitedHit2DSet.find(hit2D) == visitedHit2DSet.end())
1907 {
1908 visitedHit2DSet.insert(hit2D);
1909
1910 // Create and save the new recob::Hit with the correct WireID
1911 hitPtrVec.emplace_back(recob::HitCreator(*hit2D->getHit(), hit3D.getWireIDs()[idx]).copy());
1912
1913 // Recover a pointer to it...
1914 recob::Hit* newHit = &hitPtrVec.back();
1915
1916 // Create a mapping from this hit to an art Ptr representing it
1917 recobHitToPtrMap[newHit] = ptrMaker(hitPtrVec.size()-1);
1918
1919 // And set the pointer to this hit in the ClusterHit2D object
1920 const_cast<reco::ClusterHit2D*>(hit2D)->setHit(newHit);
1921 }
1922 }
1923 }
1924
1925 size_t numNewHits = hitPtrVec.size();
1926
1927 if (m_enableMonitoring)
1928 {
1929 theClockBuildNewHits.stop();
1930
1931 m_timeVector[BUILDNEWHITS] = theClockBuildNewHits.accumulated_real_time();
1932 }
1933
1934 mf::LogDebug("SnippetHit3D") << ">>>>> New output recob::Hit size: " << numNewHits << " (vs " << m_clusterHit2DMasterList.size() << " input)" << std::endl;
