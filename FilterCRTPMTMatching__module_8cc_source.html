187bool icarus::crt::FilterCRTPMTMatching::filter(art::Event& e) {
188 mf::LogDebug("FilterCRTPMTMatching: ") << "beginning analyis";
189
190 fEvent = e.id().event();
191 fRun = e.run();
192 fSubRun = e.subRun();
193 ClearVecs();
194
195 auto const& triggerInfo = e.getProduct<sbn::ExtraTriggerInfo>(fTriggerLabel);
196 sbn::triggerSource bit = triggerInfo.sourceType;
197 m_gate_type = (unsigned int)bit;
198 m_gate_name = bitName(bit);
199 m_trigger_timestamp = triggerInfo.triggerTimestamp;
200 m_gate_start_timestamp = triggerInfo.beamGateTimestamp;
201 m_trigger_gate_diff = triggerInfo.triggerTimestamp - triggerInfo.beamGateTimestamp;
202 m_gate_width = fTriggerConfiguration.getGateWidth(m_gate_type);
203
204 auto const& crtpmtMatches = e.getProduct<std::vector<CRTPMTMatching>>(fCrtPmtModuleLabel);
205
206 if ((fFilterLevel != "loose") && (fFilterLevel != "medium") && (fFilterLevel != "tight"))
207 throw art::Exception{ art::errors::Configuration } << "Invalid CRT/PMT filter level: '" << fFilterLevel << "'\n";
208
209 mf::LogInfo("FilterCRTPMTMatching::FilteringLevel ") << fFilterLevel;
210 std::vector<CRTPMTMatching> EventFlashes;
211 for(auto const & crtpmt : crtpmtMatches) {
212 if(fSpillOnly==true){
213 if(crtpmt.flashInBeam==1) EventFlashes.push_back(crtpmt);
214 }
215 else {
216 if(crtpmt.flashInGate==1) EventFlashes.push_back(crtpmt);
217 }
218 }
219
220 bool hasOnlyCosmics = false;
221 if (fFilterLevel == "loose") {
222 hasOnlyCosmics = false; // By default, with loose Filtering, everything is
223 // "intersting", nothing tagged as clear cosmic
224 if(fSpillOnly==true && EventFlashes.empty()) hasOnlyCosmics = true;
225 }
226 else if (fFilterLevel == "medium") {
227 hasOnlyCosmics = true;
228 for (const auto& f : EventFlashes) {
229 bool isCosmic = false;
230 if (f.flashClassification == sbn::crt::MatchType::enTop || f.flashClassification == sbn::crt::MatchType::enTop_exSide ||
231 f.flashClassification == sbn::crt::MatchType::enTop_mult ||
232 f.flashClassification == sbn::crt::MatchType::enTop_exSide_mult) {
233 isCosmic = true;
234 }
235 // With Medium filter, everything (inTime) which is associated with Top
236 // CRT Hit before the Flash is filtered as clear cosmic
237 else
238 isCosmic = false;
239 hasOnlyCosmics = hasOnlyCosmics && isCosmic;
240 }
241 }
242 else if (fFilterLevel == "tight") {
243 hasOnlyCosmics = true;
244 for (const auto& f : EventFlashes) {
245 bool isCosmic = false;
246 if ((f.flashClassification != sbn::crt::MatchType::noMatch) && (f.flashClassification != sbn::crt::MatchType::exTop) &&
247 (f.flashClassification != sbn::crt::MatchType::exSide))
248 isCosmic = true;
249 // With Medium filter, everything (inTime) which is associated with Top
250 // CRT Hit before the Flash is filtered as clear cosmic
251 else
252 isCosmic = false;
253 hasOnlyCosmics = hasOnlyCosmics && isCosmic;
254 }
255 }
256 fFiltered = hasOnlyCosmics;
257 if (fMatchTree) {
258 for(const auto& f : EventFlashes){
259 fClassification = (int) f.flashClassification;
260 fOpFlashPos_X = f.flashPosition.X();
261 fOpFlashPos_Y = f.flashPosition.Y();
262 fOpFlashPos_Z = f.flashPosition.Z();
263 fOpFlashTime_us = f.flashTime;
264 finGate = f.flashInGate;
265 finBeam = f.flashInBeam;
266 fFlashBeamTime_ns = f.flashGateTime*1e3;
267 for (const auto& crt : f.matchedCRTHits){
268 fCRTHitPos_X.push_back(crt.position.X());
269 fCRTHitPos_Y.push_back(crt.position.Y());
270 fCRTHitPos_Z.push_back(crt.position.Z());
271 fCRTHitTime_us.push_back(crt.time);
272 fCRTFlashTime_ns.push_back(crt.PMTTimeDiff*1e3);
273 fCRTHitRegion.push_back(crt.region);
274 fCRTHitSystem.push_back(crt.sys);
275 }
276 fMatchTree->Fill();
277 ClearVecs();
278 }
279 if (EventFlashes.empty()) {
280 fClassification = 9;
281 fOpFlashPos_X = 0;
282 fOpFlashPos_Y = 0;
283 fOpFlashPos_Z = 0;
284 fOpFlashTime_us = 0;
285 finGate = false;
286 finBeam = false;
287 fFlashBeamTime_ns = 0;
288 fMatchTree->Fill();
289 }
290 }
291 return !hasOnlyCosmics;
292}
