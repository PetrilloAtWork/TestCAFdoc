69 void ValidateCompression::event(art::Event const& evt)
70 {
71 std::vector<artdaq::Fragment> originalFragments = evt.getProduct<std::vector<artdaq::Fragment>>(fFragmentsLabel);
72
73 // if we want to make a histogram for the waveform
74 art::ServiceHandle<art::TFileService> tfs;
75
76 // this is useful
77 size_t fragNumber = 0;
78 for (auto const& frag : originalFragments)
79 {
80
81 // put the fragment in a compression compliant overlay
82 icarus::PhysCrateFragment fragOverlay(frag);
83 std::string fragCrateName = fChannelMap->getCrateName(frag.fragmentID());
84
85 if (true)
86 {
87 // iterate through the fragment which wasn't compressed and check which boards are or aren't working
88 MF_LOG_VERBATIM("ValidateCompression")
89 << "****************************" << '\n'
90 << "Crate " << fragCrateName;
91 size_t cumulativePrevBlockSize = 0;
92 icarus::A2795DataBlock::data_t const* boardData = new icarus::A2795DataBlock::data_t[frag.dataSize()];
93 for (size_t board = 0; board < fragOverlay.nBoards(); ++board)
94 {
95 bool boardCompressed = true;
96 size_t word = 0;
97 boardData = reinterpret_cast<icarus::A2795DataBlock::data_t const*>
98 ( frag.dataBeginBytes()
99 + (1+board)*sizeof(icarus::PhysCrateDataTileHeader)
100 + (1+board)*sizeof(icarus::A2795DataBlock::Header)
101 + 4*board*sizeof(uint16_t)
102 + cumulativePrevBlockSize );
103
104 for (size_t sample = 0; sample < fragOverlay.nSamplesPerChannel(); ++sample)
105 {
106 bool oddCompressions = false;
107 std::vector<icarus::A2795DataBlock::data_t> prevSample(fragOverlay.nChannelsPerBoard(), 0);
108 for (size_t cSet = 0; cSet < fragOverlay.nChannelsPerBoard() / 4; ++cSet)
109 {
110 // get word
111 icarus::A2795DataBlock::data_t dataWord = boardData[word];
112
113 // check if the word is interpretable as a compressed set of 4 ADC differences
114 bool wordNotTagged = ((dataWord & 0xF000) != 0x8000);
115 if (not wordNotTagged)
116 {
117 // if the word is tagged, then it's not compressed, but should exist in a compressed fragment
118 // check the next three words are also tagged, if they are then we are good (might need to check the delta sizes)
119 // otherwise something has gone wrong
120 bool wordTagged_plus1 = (boardData[word + 1] & 0xF000) == 0x8000;
121 bool wordTagged_plus2 = (boardData[word + 2] & 0xF000) == 0x8000;
122 bool wordTagged_plus3 = (boardData[word + 3] & 0xF000) == 0x8000;
123 if (wordTagged_plus1 && wordTagged_plus2 && wordTagged_plus3)
124 {
125 if (not boardCompressed)
126 {
127 MF_LOG_VERBATIM("ValidateCompression")
128 << "SOMEHOW FOUND COMPRESSED SEQUENCE IN UNCOMPRESSED BOARD";
129 }
130 // it's good
131 word += 4;
132 cumulativePrevBlockSize += 8;
133 for (size_t cInSet = 0; cInSet < 4; ++cInSet)
134 {
135 icarus::A2795DataBlock::data_t twelveBitDiff = (boardData[word + 1] & 0x0FFF);
136 bool isNeg = (sample != 0) && (twelveBitDiff >> 11);
137 prevSample[4*cSet + cInSet] += isNeg*0xF000 + twelveBitDiff;
138 }
139 } else {
140 // something has gone wrong
141 MF_LOG_VERBATIM("ValidateCompression")
142 << "TAGGED BUT NOT FROM COMPRESSED FRAGMENT";
143 boardCompressed = false;
144 word += 4;
145 cumulativePrevBlockSize += 8;
146 for (size_t cInSet = 0; cInSet < 4; ++cInSet)
147 {
148 prevSample[4*cSet + cInSet] = (boardData[word + cInSet] & (~(1<<(fragOverlay.metadata()->num_adc_bits()+1))));
149 }
150 }
151 } else {
152 if (sample == 0)
153 {
154 boardCompressed = false;
155 }
156 if (boardCompressed)
157 {
158 oddCompressions = (not oddCompressions);
159 // if we don't have the tag check if it makes more sense as one value or 4 differences
160 // first try adding to previous samples and see if the results are reasonable
161 for (size_t cInSet = 0; cInSet < 4; ++cInSet)
162 {
163 icarus::A2795DataBlock::data_t fourBitDiff = (dataWord >> (4*cInSet)) & 0x000F;
164 bool isNeg = (fourBitDiff >> 3);
165 prevSample[4*cSet + cInSet] += (isNeg*0xFFF0 + fourBitDiff);
166 }
167 ++word;
168 cumulativePrevBlockSize += 2;
169 } else {
170 word += 4;
171 cumulativePrevBlockSize += 8;
172 for (size_t cInSet = 0; cInSet < 4; ++cInSet)
173 {
174 prevSample[4*cSet + cInSet] = (boardData[word + cInSet] & (~(1<<(fragOverlay.metadata()->num_adc_bits()+1))));
175 }
176 }
177 }
178 }
179 if (oddCompressions)
180 {
181 ++word;
182 cumulativePrevBlockSize += 2;
183 }
184 }
185 std::string boardCompStr = (boardCompressed) ? " is compressed" : " is UNCOMPRESSED";
186 MF_LOG_VERBATIM("ValidateCompression")
187 << " Board " << board << boardCompStr;
188 }
189 uint32_t totalBytes = fragOverlay.nBoards()*sizeof(icarus::PhysCrateDataTileHeader)
190 + fragOverlay.nBoards()*sizeof(icarus::A2795DataBlock::Header)
191 + 4*fragOverlay.nBoards()*sizeof(uint16_t)
192 + cumulativePrevBlockSize;
193 MF_LOG_DEBUG("ValidateCompression")
194 << "****************************" << '\n'
195 << "Crate " << fragCrateName << '\n'
196 << " Estimated data bytes " << totalBytes << '\n'
197 << " should be " << frag.dataSizeBytes() << '\n'
198 << "****************************";
199 }
200
201 ++fragNumber;
202 }
