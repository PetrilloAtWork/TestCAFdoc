1150template <typename SampleType>
1151raw::OpDetWaveform icarus::opdet::OpDetWaveformMakerClass<SampleType>::create
1152 (raw::Channel_t opChannel, BufferRange_t const& bufferRange) const
1153{
1154 std::size_t const start
1155 = std::min(std::size_t(bufferRange.first.value()), fWaveform.size());
1156 std::size_t const end
1157 = std::min(std::size_t(bufferRange.second.value()), fWaveform.size());
1158 assert(start <= end);
1159
1160 // start of the waveform (tick #0) in the full optical reading
1161 raw::TimeStamp_t const timeStamp { fPMTstartTime + start * fSamplingPeriod };
1162
1163 // create a new waveform preallocating enough room for the full buffer
1164 raw::OpDetWaveform outputWaveform(timeStamp, opChannel, end - start);
1165
1166 // copy the buffer (need to unwrap the ADCcount value)
1167 std::transform(
1168 fWaveform.begin() + start,
1169 fWaveform.begin() + end,
1170 std::back_inserter(outputWaveform),
1171 [](auto sample){ return sample.value(); }
1172 );
1173
1174 return outputWaveform;
1175} // icarus::opdet::OpDetWaveformMakerClass<>::create()
