169template <typename Time, typename Interval>
170auto icarus::opdet::PMTReadoutWindowMaker<Time, Interval>::makeWindows
171 (std::vector<TimePoint_t> const& seeds) const
172 -> std::vector<Window_t>
173{
174 std::vector<Window_t> windows;
175 if (seeds.empty()) return windows;
176
177 auto itSeed = seeds.begin();
178 auto const send = seeds.end();
179
180 windows.emplace_back(*itSeed - fPreBuffer, *itSeed + fPostBuffer);
181 Window_t* lastWindow = &(windows.back());
182
183 mfLogTrace() << "Processing " << seeds.size() << " readout seeds."
184 << "\nFirst window from t=" << *itSeed << ": " << *lastWindow;
185
186 while (++itSeed != send) {
187
188 TimePoint_t const time = *itSeed;
189
190 TimePoint_t const start = time - fPreBuffer;
191 TimePoint_t const stop = time + fPostBuffer;
192
193 mfLogTrace() << "Seed at: " << time << " (starts at " << start << ")";
194 if (start <= lastWindow->stop()) { // merge
195 assert(lastWindow->start() <= start);
196 lastWindow->setStop(stop);
197
198 mfLogTrace()
199 << " starts within last window: extended, now " << *lastWindow;
200
201 }
202 else { // new
203 windows.emplace_back(start, stop);
204 lastWindow = &(windows.back());
205 mfLogTrace() << " new window: " << *lastWindow;
206 }
207
208 } // while
209
210 return windows;
