328//------------------------------------------------------------------------------
329void icarus::trigger::BeamGateInfoFromTracks::produce
330 (art::Event& event, art::ProcessingFrame const&)
331{
332
333 //
334 // fetch input
335 //
336 //auto const& particles
337 //= event.getProduct<std::vector<art::Ptr<recob::PFParticle>>>(fT0selProducer);
338
339 auto const& tracks
340 = event.getProduct<std::vector<art::Ptr<recob::Track>>>(fT0selProducer);
341
342 mf::LogDebug(fLogCategory)
343 << "Writing gates for " << tracks.size() << " tracks.";
344
345 art::FindOneP<anab::T0> const trackT0(tracks, event, fT0Producer);
346
347 detinfo::DetectorTimings const detTimings {
348 art::ServiceHandle<detinfo::DetectorClocksService const>()->DataFor(event)
349 };
350
351 // add this offset to a time vs. trigger to make it vs. beam gate
352 nanoseconds const triggerToBeamGate
353 = detTimings.TriggerTime() - detTimings.BeamGateTime();
354
355 //
356 // create the content
357 //
358 std::vector<sim::BeamGateInfo> gates;
359 art::Assns<sim::BeamGateInfo, recob::Track> gateToParticle;
360 art::Assns<sim::BeamGateInfo, anab::T0> gateToTime;
361
362 art::PtrMaker<sim::BeamGateInfo> const makeGatePtr { event };
363
364 for (auto const& [ iTracks, trackPtr ]: util::enumerate(tracks)) {
365
366 art::Ptr<anab::T0> const t0Ptr = trackT0.at(iTracks);
367 if (t0Ptr.isNull()) {
368 art::Exception e { art::errors::NotFound };
369 e << "Selected track #" << iTracks << " (";
370 if (trackPtr) e << "ID=" << trackPtr->ID() << ", ";
371 e << "#" << trackPtr.key()
372 << " in its collection) has no associated time."
373 ;
374 throw e << '\n';
375 } // if no T0
376
377 // t0 is stored in trigger time
378 detinfo::timescales::trigger_time const t0
379 { util::quantities::points::nanosecond(t0Ptr->Time()) };
380
381 // 1DetectorTimings1 does not handle the beam gate time when converting to
382 // simulation time, so I need to explicitly add the difference
383 detinfo::timescales::simulation_time const gateStart
384 = detTimings.toSimulationTime(t0 + triggerToBeamGate + fGateStartOffset);
385
386 // time conversion is currently redundant here, left as documentation
387 sim::BeamGateInfo gate {
388 gateStart.value() // start
389 , fGateDuration.convertInto<nanoseconds>().value() // width
390 , fBeamGateType // type
391 };
392
393 {
394 mf::LogTrace log{ fLogCategory };
395 log << "Gate for selected track #" << iTracks << " (";
396 if (trackPtr) log << "ID=" << trackPtr->ID() << ", ";
397 log << "#" << trackPtr.key()
398 << " in its collection): time = " << t0 << " => "
399 << gate.Start() << " -- " << (gate.Start() + gate.Width())
400 << " ns (" << gate.Width() << " ns)"
401 ;
402 }
403
404 gates.push_back(std::move(gate));
405
406 art::Ptr<sim::BeamGateInfo> gatePtr { makeGatePtr(gates.size() - 1) };
407 gateToParticle.addSingle(gatePtr, trackPtr);
408 gateToTime.addSingle(gatePtr, t0Ptr);
409
410 } // for
411
412 //
413 // store output
414 //
415 event.put(moveToUniquePtr(gates));
416 event.put(moveToUniquePtr(gateToParticle));
417 event.put(moveToUniquePtr(gateToTime));
418
419} // icarus::trigger::BeamGateInfoFromTracks::produce()
