393 template <typename ConfigMap>
394 icarus::TriggerConfiguration extractTriggerReadoutConfigurationImpl
395 (ConfigMap const& configMap, icarus::TriggerConfigurationExtractor extractor)
396 {
397
398 /*
399 * Requirements: ConfigMap is a mapping type supporting iteration and whose
400 * elements support structured assignment into a pair, the first element
401 * being an identifier (process name, parameter set ID...) and the second
402 * being a `fhicl::ParameterSet` with the configuration (or something
403 * offering the same interface).
404 *
405 * The plan is to look in all the FHiCL configuration fragments we can find
406 * in the input config, and find all the useful configuration therein.
407 * Given that there may be multiple input files, there may also be multiple
408 * configurations for the same detector components.
409 * In that case, we will extract parameters from each and every one of the
410 * configurations, and throw an exception if they are not all consistent.
411 *
412 * Consistency is tested only for the extracted parameters, not for the whole
413 * FHiCL configuration fragment.
414 */
415
416 using Key_t = std::tuple_element_t<0U, typename ConfigMap::value_type>;
417
418 std::optional<std::pair<Key_t, icarus::TriggerConfiguration>> config;
419
420 // look in the global configuration for all parameter sets which contain
421 // `configuration_documents` as a (direct) name;
422 for (auto const& [ id, pset ]: configMap) {
423 if (!extractor.mayHaveConfiguration(pset)) continue;
424
425 fhicl::ParameterSet const configDocs
426 = extractor.convertConfigurationDocuments
427 (pset, "configuration_documents", { std::regex{ "icarustrigger.*" } })
428 ;
429
430 icarus::TriggerConfiguration candidateConfig = extractor.extract(configDocs);
431 if (config) {
432 if (config->second == candidateConfig) continue;
433 mf::LogError log("extractTriggerReadoutConfiguration");
434 log << "Found two candidate configurations differring:"
435 "\nFirst [" << config->first << "]:\n" << config->second
436 << "\nSecond [" << id << "]:\n" << candidateConfig
437 ;
438 throw cet::exception("extractTriggerReadoutConfiguration")
439 << "extractTriggerReadoutConfiguration() found inconsistent configurations.\n";
440 } // if incompatible configurations
441
442 config.emplace(std::move(id), std::move(candidateConfig));
443 } // for all configuration documents
444
445 if (!config) {
446 throw cet::exception("extractTriggerReadoutConfiguration")
447 << "extractTriggerReadoutConfiguration() could not find a suitable configuration.\n";
448 }
449
450 return std::move(config->second);
451 } // extractPMTreadoutConfigurationImpl(ConfigMap)
