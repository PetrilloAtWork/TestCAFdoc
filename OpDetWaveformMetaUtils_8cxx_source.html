66 (raw::OpDetWaveform const& waveform) const
67{
68
69 using detinfo::timescales::electronics_time;
70
71 raw::Channel_t const channel = waveform.ChannelNumber();
72 std::size_t const nSamples = waveform.Waveform().size();
73 electronics_time const startTime { waveform.TimeStamp() };
74 electronics_time const endTime
75 = startTime + waveform.Waveform().size() * fOpDetTickPeriod;
76
77 sbn::OpDetWaveformMeta info {
78 channel // channel
79 , nSamples // nSamples
80 , startTime.value() // startTime
81 , endTime.value() // endTime
82 /* the following are left default:
83 // flags
84 */
85 };
86
87 auto const setFlag = [&info]
88 (sbn::OpDetWaveformMeta::Flags_t::Flag_t flag, bool value)
89 { if (value) info.flags.set(flag); else info.flags.unset(flag); };
90
91 auto const isInWaveform = [startTime,endTime](electronics_time t)
92 { return (t >= startTime) && (t < endTime); };
93
94 if (fTriggerTime) {
95 setFlag
96 (sbn::OpDetWaveformMeta::bits::WithTrigger, isInWaveform(*fTriggerTime));
97 }
98
99 if (fBeamGateTime) {
100 setFlag
101 (sbn::OpDetWaveformMeta::bits::WithBeamGate, isInWaveform(*fBeamGateTime));
102 }
103
104 return info;
