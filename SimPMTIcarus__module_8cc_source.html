361 // ---------------------------------------------------------------------------
362 void SimPMTIcarus::produce(art::Event & e)
363 {
364 mf::LogDebug("SimPMTIcarus") << e.id();
365
366 //
367 // fetch the input
368 //
369 auto pulseVecPtr = std::make_unique< std::vector< raw::OpDetWaveform > > ();
370
371 std::unique_ptr<std::vector<sim::SimPhotons>> simphVecPtr;
372 if (fWritePhotons)
373 simphVecPtr = std::make_unique< std::vector< sim::SimPhotons > > ();
374
375 //
376 // prepare the output
377 //
378 art::Handle<std::vector<sim::SimPhotons> > pmtVector;
379 art::Handle<std::vector<sim::SimPhotonsLite> > pmtLiteVector;
380 pmtVector = e.getHandle< std::vector<sim::SimPhotons> >(fInputModuleName);
381 if(!pmtVector.isValid())
382 pmtLiteVector = e.getHandle< std::vector<sim::SimPhotonsLite> >(fInputModuleName);
383
384 auto const clockData =
385 art::ServiceHandle<detinfo::DetectorClocksService const>()->DataFor(e);
386 //
387 // prepare the algorithm
388 //
389 auto PMTsimulator = makePMTsimulator(
390 e.time().value(), // using the event generation time as beam time stamp
391 *(lar::providerFrom<detinfo::LArPropertiesService>()),
392 clockData,
393 *fSinglePhotonResponseFunc,
394 *fPedestalGen,
395 fEfficiencyEngine,
396 fDarkNoiseEngine,
397 fElectronicsNoiseEngine,
398 fWritePhotons
399 );
400
401 if (firstTime()) {
402 mf::LogInfo log { "SimPMTIcarus" };
403 log << "PMT simulation configuration (first event):\n";
404 PMTsimulator->printConfiguration(log);
405 } // if first time
406
407 //
408 // run the algorithm
409 //
410 unsigned int nopch = 0;
411 if(pmtVector.isValid()) {
412 nopch = pmtVector->size();
413 for(auto const& photons : *pmtVector) {
414
415 // Make an empty SimPhotonsLite with the same channel number.
416
417 sim::SimPhotonsLite lite_photons(photons.OpChannel());
418
419 auto const& [ channelWaveforms, photons_used ]
420 = PMTsimulator->simulate(photons, lite_photons);
421 std::move(
422 channelWaveforms.cbegin(), channelWaveforms.cend(),
423 std::back_inserter(*pulseVecPtr)
424 );
425 if (simphVecPtr && photons_used)
426 simphVecPtr->emplace_back(std::move(photons_used.value()));
427
428 } // for
429 }
430 else if(pmtLiteVector.isValid()) {
431 nopch = pmtLiteVector->size();
432 for(auto const& lite_photons : *pmtLiteVector) {
433
434 // Make an empty SimPhotons with the same channel number.
435
436 sim::SimPhotons photons(lite_photons.OpChannel);
437
438 auto const& [ channelWaveforms, photons_used ]
439 = PMTsimulator->simulate(photons, lite_photons);
440 std::move(
441 channelWaveforms.cbegin(), channelWaveforms.cend(),
442 std::back_inserter(*pulseVecPtr)
443 );
444
445 } // for
446 }
447
448 mf::LogInfo("SimPMTIcarus") << "Generated " << pulseVecPtr->size()
449 << " waveforms out of " << nopch << " optical channels.";
450
451 // waveform metadata
452 std::unique_ptr<std::vector<sbn::OpDetWaveformMeta>> metadataVec;
453 std::unique_ptr<art::Assns<raw::OpDetWaveform, sbn::OpDetWaveformMeta>>
454 metadataAssns;
455 if (fMakeMetadata) {
456 metadataVec = std::make_unique<std::vector<sbn::OpDetWaveformMeta>>();
457 metadataAssns =
458 std::make_unique<art::Assns<raw::OpDetWaveform, sbn::OpDetWaveformMeta>>
459 ();
460
461 auto const detTimings = detinfo::makeDetectorTimings
462 (art::ServiceHandle<detinfo::DetectorClocksService const>()->DataFor(e))
463 ;
464 std::tie(*metadataVec, *metadataAssns)
465 = makeMetadata(e, *pulseVecPtr, detTimings);
466 }
467
468 //
469 // save the result
470 //
471 e.put(std::move(pulseVecPtr));
472 if (fMakeMetadata) {
473 e.put(std::move(metadataVec));
474 e.put(std::move(metadataAssns));
475 }
476 if (simphVecPtr) e.put(std::move(simphVecPtr));
477 } // SimPMTIcarus::produce()

481 std::pair<
482 std::vector<sbn::OpDetWaveformMeta>,
483 art::Assns<raw::OpDetWaveform, sbn::OpDetWaveformMeta>
484 >
485 SimPMTIcarus::makeMetadata(
486 art::Event const& event,
487 std::vector<raw::OpDetWaveform> const& waveforms,
488 detinfo::DetectorTimings const& detTimings
489 ) const {
490
491 std::vector<sbn::OpDetWaveformMeta> meta;
492 art::Assns<raw::OpDetWaveform, sbn::OpDetWaveformMeta> assns;
493
494 art::PtrMaker<raw::OpDetWaveform> const makeWaveformPtr{ event };
495 art::PtrMaker<sbn::OpDetWaveformMeta> const makeMetaPtr{ event };
496
497 sbn::OpDetWaveformMetaMaker const makeOpDetWaveformMeta{ detTimings };
498
499 for (auto const& [iWaveform, waveform ]: util::enumerate(waveforms)) {
500 meta.push_back(makeOpDetWaveformMeta(waveform));
501 assns.addSingle(makeWaveformPtr(iWaveform), makeMetaPtr(iWaveform));
502 } // for waveforms
503
504 return { std::move(meta), std::move(assns) };
505
506 } // SimPMTIcarus::makeMetadata()
