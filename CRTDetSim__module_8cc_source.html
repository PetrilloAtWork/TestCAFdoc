148//module producer
149void icarus::crt::CRTDetSim::produce(art::Event& event) {
150
151 //pointer to vector of CRT data products to be pushed to event
152 std::unique_ptr< vector<CRTData> > triggeredCRTHits (
153 new vector<CRTData>);
154 //pointer to vector of AuxDetIDE products to be pushed to event
155 std::unique_ptr< vector<sim::AuxDetIDE> > ides (
156 new vector<sim::AuxDetIDE>);
157 //pointer associations between CRT data products and AuxDetIDEs to be pushed to event
158 std::unique_ptr< art::Assns<CRTData,sim::AuxDetIDE> > dataAssn (
159 new art::Assns<CRTData,sim::AuxDetIDE> );
160 art::PtrMaker<CRTData> makeDataPtr(event);
161 art::PtrMaker<sim::AuxDetIDE> makeIDEPtr(event);
162
163 //clear detAlg member data
164 detAlg.ClearTaggers();
165
166 // Handle for (truth) AuxDetSimChannels
167 art::Handle< vector<sim::AuxDetSimChannel> > adChanHandle;;
168 vector< art::Ptr<sim::AuxDetSimChannel> > adChanList;
169 if (event.getByLabel(fG4ModuleLabel, adChanHandle) )
170 art::fill_ptr_vector(adChanList, adChanHandle);
171
172 mf::LogInfo("CRTDetSimProducer")
173 <<"Number of AuxDetChannels = " << adChanList.size();
174
175 int nide=0;
176 auto const clockData = art::ServiceHandle<detinfo::DetectorClocksService>()->DataFor(event);
177 for(auto const& adsc : adChanList) {
178
179 auto const& auxDetIDEs = adsc->AuxDetIDEs();
180 if(auxDetIDEs.size()>0) {
181 nide++;
182 detAlg.FillTaggers(clockData, adsc->AuxDetID(), adsc->AuxDetSensitiveID(), auxDetIDEs);
183 }
184
185 }//loop over AuxDetSimChannels
186
187 //generate CRTData products, associates from filled detAlg member data
188 int nData = 0;
189 if(nide>0) {
190 vector<std::pair<CRTData,vector<sim::AuxDetIDE>>> data = detAlg.CreateData();
191 //time sort CRT data (ascending)
192 std::sort(data.begin(),data.end(),
193 [](const std::pair<CRTData,vector<sim::AuxDetIDE>>& d1,
194 const std::pair<CRTData,vector<sim::AuxDetIDE>>& d2) {
195 return d1.first.fTs0 < d2.first.fTs0;
196 });
197
198 for(auto const& dataPair : data){
199
200 triggeredCRTHits->push_back(dataPair.first);
201 art::Ptr<CRTData> dataPtr = makeDataPtr(triggeredCRTHits->size()-1);
202 nData++;
203
204 for(auto const& ide : dataPair.second){
205 ides->push_back(ide);
206 art::Ptr<sim::AuxDetIDE> idePtr = makeIDEPtr(ides->size()-1);
207 dataAssn->addSingle(dataPtr, idePtr);
208 }
209 }
210
211 }
212
213 if(nData==0)
214 mf::LogWarning("CRTDetSimProducer")
215 << "0 CRTData produced (expected for most neutrino events, never for cosmics)";
216
217 //push products to event
218 event.put(std::move(triggeredCRTHits));
219 event.put(std::move(ides));
220 event.put(std::move(dataAssn));
221
222 mf::LogInfo("CRTDetSimProducer")
223 << "Number of CRT data produced = "<< nData << '\n'
224 << " from " << nide << " AuxDetIDEs";
225}
