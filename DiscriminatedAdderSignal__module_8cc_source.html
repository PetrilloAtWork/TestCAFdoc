629icarus::trigger::DiscriminatedAdderSignal::DiscriminatedAdderSignal
630 (Parameters const& config, art::ProcessingFrame const& frame)
631 : art::SharedProducer{ config }
632 // configuration
633 , fWaveformTag { config().WaveformTag() }
634 , fBaselineTag { config().BaselineTag() }
635 , fWaveformMetaTag { config().WaveformMetaTag().value_or(fWaveformTag) }
636 , fAmplitudeScale { config().AmplitudeScale() }
637 , fTimeInterval
638 {
639 icarus::ns::fhicl::makeTimeInterval(config().TimeInterval())
640 .value_or(RelTimeInterval_t{})
641 }
642 , fMissingChannels { sorted(config().MissingChannels()) }
643 , fSaveWaveforms { config().SaveWaveforms() }
644 , fSavePMTcoverage { config().SavePMTcoverage() }
645 , fSaveMetaAssns { config().SaveMetaAssns() }
646 , fAdderChannelOffset{ config().ChannelOffset() }
647 , fChannelsPerAdder { config().ChannelsPerAdder() }
648 , fLogCategory { config().LogCategory() }
649 , fTriggerGateBuilder{
650 art::make_tool<icarus::trigger::TriggerGateBuilder>
651 (config().TriggerGateBuilder_.get<fhicl::ParameterSet>())
652 }
653 // service cache
654 , fGeom { *lar::providerFrom<geo::Geometry>() }
655 , fOpticalTick {
656 detinfo::makeDetectorClocksWithUnits
657 (art::ServiceHandle<detinfo::DetectorClocksService const>()->DataForJob())
658 .OpticalClockPeriod()
659 }
660{
661 async<art::InEvent>();
662
663 //
664 // optional configuration parameters
665 //
666 if (fTimeInterval.empty()) {
667 throw art::Exception{ art::errors::Configuration }
668 << "The '" << config().TimeInterval.name()
669 << "' parameters configures an empty time interval [ "
670 << fTimeInterval.start << " -- " << fTimeInterval.stop << " ]\n";
671 }
672
673 //
674 // configuration post-processing
675 //
676 std::vector<raw::ADC_Count_t> selectedThresholds;
677 if (!config().SelectThresholds(selectedThresholds)) {
678 std::vector<ADCCounts_t> const& allThresholds
679 = fTriggerGateBuilder->channelThresholds();
680
681 if (allThresholds.empty()) {
682 throw art::Exception{ art::errors::Configuration }
683 << "Trigger building algorithm reports no threshold!\n"
684 << "Check the configuration of `TriggerGateBuilder` tool.\n";
685 }
686 selectedThresholds = transformColl
687 (allThresholds, std::mem_fn(&icarus::trigger::ADCCounts_t::value));
688 }
689 else if (selectedThresholds.empty()) {
690 throw art::Exception{ art::errors::Configuration }
691 << "Configured to save no thresholds!\n"
692 << "Add values to the `SelectThresholds` configuration parameter.\n";
693 }
694
695 for (auto const& threshold: selectedThresholds) {
696 fSelectedThresholds[icarus::trigger::ADCCounts_t{threshold}]
697 = util::to_string(threshold);
698 }
699
700 {
701 mf::LogInfo log(fLogCategory);
702 log << "Configuration:"
703 << "\n * time interval: [ "
704 << fTimeInterval.start << " - " << fTimeInterval.stop << " ]"
705 << "\n * selected thresholds:"
706 ;
707 for (auto const& [ threshold, instance ]: fSelectedThresholds) {
708 log << " " << threshold << " ADC = " << to_mV(threshold)
709 << " (\"" << instance << "\")";
710 }
711 } // nameless block
712
713 //
714 // declaration of input
715 //
716 consumes<std::vector<raw::OpDetWaveform>>(fWaveformTag);
717 consumes<art::Assns<raw::OpDetWaveform, icarus::WaveformBaseline>>
718 (fBaselineTag);
719 if (fSaveMetaAssns) {
720 consumes<art::Assns<raw::OpDetWaveform, sbn::OpDetWaveformMeta>>
721 (fWaveformMetaTag);
722 }
723
724 //
725 // declaration of output
726 //
727
728 if (fSaveWaveforms) {
729 produces<std::vector<raw::OpDetWaveform>>();
730 produces<std::vector<icarus::WaveformBaseline>>();
731 produces<art::Assns<raw::OpDetWaveform, icarus::WaveformBaseline>>();
732 }
733 if (fSavePMTcoverage) {
734 produces<std::vector<sbn::OpDetWaveformMeta>>();
735 if (fSaveWaveforms)
736 produces<art::Assns<raw::OpDetWaveform, sbn::OpDetWaveformMeta>>();
737 }
738 for (std::string const& instanceName: util::const_values(fSelectedThresholds))
739 {
740 produces<std::vector<TriggerGateData_t>>(instanceName);
741 produces<art::Assns<TriggerGateData_t, raw::OpDetWaveform>>
742 (instanceName + PMTDataProductSuffix);
743 if (fSaveMetaAssns) {
744 produces<art::Assns<TriggerGateData_t, sbn::OpDetWaveformMeta>>
745 (instanceName + PMTDataProductSuffix);
746 }
747 if (fSaveWaveforms) {
748 produces<art::Assns<TriggerGateData_t, raw::OpDetWaveform>>
749 (instanceName + AdderDataProductSuffix);
750 } // if saving waveforms
751 if (fSavePMTcoverage) {
752 produces<art::Assns<TriggerGateData_t, sbn::OpDetWaveformMeta>>
753 (instanceName + AdderDataProductSuffix);
754 } // if saving waveform metadata
755 } // for thresholds

760//------------------------------------------------------------------------------
761void icarus::trigger::DiscriminatedAdderSignal::produce
762 (art::Event& event, art::ProcessingFrame const&)
763{
764
765 //
766 // set up the algorithm to create the trigger gates
767 //
768 detinfo::DetectorTimings const detTimings {
769 art::ServiceHandle<detinfo::DetectorClocksService const>()->DataFor(event)
770 };
771 fTriggerGateBuilder->resetup(detTimings);
772 mf::LogDebug{ fLogCategory }
773 << "Trigger at " << detTimings.TriggerTime() << ", "
774 << (detTimings.TriggerTime() - detTimings.BeamGateTime())
775 << " after beam gate opened at " << detTimings.BeamGateTime();
776
777 //
778 // prepare input
779 //
780 auto const& waveformHandle
781 = event.getValidHandle<std::vector<raw::OpDetWaveform>>(fWaveformTag);
782 auto const& waveforms = *waveformHandle;
783
784 // retrieve the baseline information
785 art::FindOne<icarus::WaveformBaseline> const waveformBaselines
786 { waveformHandle, event, fBaselineTag };
787
788 // attach to each waveform additional information: baseline
789 std::vector<WaveformWithBaseline> waveformInfo;
790 waveformInfo.reserve(waveforms.size());
791 for (auto const& [ iWaveform, waveform ]: util::enumerate(waveforms)) {
792 cet::maybe_ref<icarus::WaveformBaseline const> const baseline
793 { waveformBaselines.at(iWaveform) };
794 if (!baseline) {
795 throw art::Exception{ art::errors::NotFound }
796 << "No baseline ('" << fBaselineTag.encode()
797 << "') found for waveform #" << iWaveform << " (channel="
798 << waveform.ChannelNumber() << " time=" << waveform.TimeStamp()
799 << ").\n";
800 }
801 waveformInfo.emplace_back(&waveform, &(baseline.ref()));
802 }
803
804 // returns a art pointer to the waveform of `wi`
805 auto PMTwaveformInfoPtr
806 = [handle=waveformHandle, firstWF=&(waveformHandle->front())]
807 (WaveformWithBaseline const* wi)
808 {
809 return art::Ptr<raw::OpDetWaveform>(handle, wi->waveformPtr() - firstWF);
810 };
811
812 mf::LogDebug{ fLogCategory }
813 << "Addition of " << waveforms.size() << " PMT waveforms from '"
814 << fWaveformTag.encode() << "' with baselines from '"
815 << fBaselineTag.encode() << "'";
816
817 //
818 // define the time interval
819 //
820 TimeInterval_t const timeInterval{
821 detTimings.BeamGateTime() + fTimeInterval.start,
822 detTimings.BeamGateTime() + fTimeInterval.stop
823 };
824
825 //
826 // define the adder windows in terms of contributing channels;
827 // this implicitly defines the channel number
828 //
829 icarus::trigger::TriggerWindowDefs_t const adderChannels
830 = createAdderWindows(fGeom);
831
832 //
833 // create adder waveforms
834 //
835 // We need to create it in a format that `TriggerGateBuilder::build()`
836 // is happy with, that is a pointer to waveform plus pointer to baseline;
837 // even if the baselines are all `0`, we need them in a stable data object
838
839 WaveformsByChannel_t const waveformInfoByChannel
840 = groupByChannel(waveformInfo);
841
842 auto const isMissing = [&missing=fMissingChannels](raw::Channel_t channel)
843 { return std::binary_search(missing.cbegin(), missing.cend(), channel); };
844 std::vector<raw::OpDetWaveform> adderWaveforms;
845 std::vector<icarus::WaveformBaseline> adderWaveformBaselines; // quite a dummy
846 std::vector<WaveformWithBaseline> adderWaveformInfo;
847 std::vector<std::vector<art::Ptr<raw::OpDetWaveform>>>
848 adderWaveformContribPtrs;
849 // preallocate all the space we need
850 adderWaveforms.reserve(adderChannels.size());
851 adderWaveformBaselines.reserve(adderChannels.size());
852 for (auto const& [ window, windowChannels ]: util::enumerate(adderChannels)) {
853
854 raw::Channel_t const adderChannel = fAdderChannelOffset + window;
855
856 mf::LogTrace log{ fLogCategory };
857 log << "Collecting waveforms for adder channel " << adderChannel << " from "
858 << windowChannels.size() << " channels on window " << window << ":";
859 std::vector<std::vector<WaveformWithBaseline const*>> inputWaveforms;
860 for (raw::Channel_t const ch: windowChannels) {
861 if (isMissing(ch)) {
862 // these should have been already filtered out by `createAdderWindows()`
863 log << " (!" << ch << ")";
864 continue;
865 }
866 inputWaveforms.push_back(waveformInfoByChannel[ch]);
867 log << " " << ch;
868 if (auto nCh = waveformInfoByChannel[ch].size(); nCh != 1)
869 log << " [x" << nCh << "]";
870 }
871
872 auto [ waveform, waveformContribs ] = makeAdderWaveform
873 (timeInterval, inputWaveforms, adderChannel);
874
875 adderWaveforms.push_back(std::move(waveform));
876 adderWaveformBaselines.emplace_back(0);
877 // the preallocation of the two vectors guarantees that the pointers
878 // are not going to be invalidated by a vector resize
879 adderWaveformInfo.emplace_back
880 (&adderWaveforms.back(), &adderWaveformBaselines.back());
881
882 std::vector<art::Ptr<raw::OpDetWaveform>> waveformContribPtrs
883 = transformColl(waveformContribs, PMTwaveformInfoPtr);
884
885 adderWaveformContribPtrs.push_back(std::move(waveformContribPtrs));
886
887 } // for adder windows
888
889 //
890 // discrimination (all thresholds at once)
891 //
892
893 // this is a collection where each entry (of type `TriggerGates`) contains
894 // the complete set of trigger gates for an event for a given threshold.
895 std::vector<icarus::trigger::TriggerGateBuilder::TriggerGates>
896 triggerGatesByThreshold = fTriggerGateBuilder->build(adderWaveformInfo);
897
898 using TriggerGates_t = icarus::trigger::TriggerGateBuilder::TriggerGates;
899
900 { // nameless block
901 mf::LogTrace log(fLogCategory);
902 log << "Trigger gates from " << triggerGatesByThreshold.size()
903 << " thresholds:\n";
904 for (TriggerGates_t const& triggerGates: triggerGatesByThreshold)
905 triggerGates.dump(log);
906
907 } // nameless block
908
909 // fix the channels in the gates: they contain the adder waveform channel,
910 // we want the contributing PMT channels
911 // (also because that will be needed when simulating a trigger)
912 std::vector<std::pair<ADCCounts_t, std::vector<TriggerGateData_t>>>
913 adderGatesAndThresholds;
914 adderGatesAndThresholds.reserve(triggerGatesByThreshold.size());
915 for (TriggerGates_t& triggerGates: triggerGatesByThreshold) {
916 std::vector<TriggerGateData_t> thresholdAdders;
917 for(
918 auto const& [ adderWaveform, contribs ]
919 : util::zip(adderWaveforms, adderWaveformContribPtrs)
920 ) {
921 // must already exist:
922 assert(triggerGates.getGateFor(adderWaveform.ChannelNumber()));
923
924 // tracking is used to keep a record of contributing waveforms from the
925 // same channel; this does not apply here because only the adder "channel"
926 // is contributing, and only with a single waveform (if we wished to
927 // emulate trigger primitives from adders, it would be a different story)
928 // ... so we drop tracking without looking back
929
930 TriggerGateData_t gate; // no channel list yet // ... so we drop it
931 // this assigns only the gate data ("levels"), not affecting its channels
932 gate = std::move(triggerGates.gateFor(adderWaveform).gate().gateLevels());
933 // we want to set the list of channels in the trigger gate
934 // to the one of the contributing PMT channels (not the adder "channel")
935 for (art::Ptr<raw::OpDetWaveform> const& waveform: contribs)
936 gate.addChannel(waveform->ChannelNumber());
937
938 thresholdAdders.push_back(std::move(gate));

1143//------------------------------------------------------------------------------
1144auto icarus::trigger::DiscriminatedAdderSignal::computeTimeInterval(
1145 std::vector<std::vector<WaveformWithBaseline const*>> const& waveforms,
1146 TimeInterval_t const& timeInterval
1147) const -> TimeInterval_t {
1148 /*
1149 * If we are willing to make up data (e.g. assigning samples of value `0` to
1150 * the ticks that are not covered by any waveform in a channel), we can then
1151 * exactly will the required `timeInterval`, with the caveat that we need
1152 * still to supply the limits that the `timeInterval` does not specify
1153 * (e.g. if `timeInterval.end` is "infinite", we may replace it with the last
1154 * waveform time end).
1155 *
1156 * If we are not willing to make up data, then we need to trim the allowed
1157 * interval so that all channels have data in it. This is a more complicate
1158 * choice in that it requires an "and" of all the coverage in each channel,
1159 * in conjunction with the required `timeInterval`, and this and may also end
1160 * up being not contiguous, in which case a choice must be made since only a
1161 * single contiguous interval is accepted (with the premise above that we
1162 * don't want to fill the gaps with made up data).
1163 */
1164 using TimeCoverage_t = icarus::trigger::TriggerGateData
1165 <electronics_time, detinfo::timescales::time_interval>;
1166
1167 //
1168 // 1. determine all the time sub-intervals within `timeInterval` covered by
1169 // all channels
1170 //
1171 TimeCoverage_t coverage;
1172 coverage.setOpeningAt(TimeCoverage_t::MinTick, 1U); // start always open
1173
1174 mf::LogTrace{ fLogCategory } << "Coverage within " << timeInterval
1175 << " (only channels narrowing it are listed):";
1176 for (std::vector<WaveformWithBaseline const*> const& chWaveforms: waveforms) {
1177
1178 TimeCoverage_t chCoverage;
1179
1180 for (WaveformWithBaseline const* wi: chWaveforms) {
1181 TimeInterval_t wInterval = waveformTimeCoverage(wi->waveform());
1182 assert(!wInterval.empty());
1183
1184 // trim to the allowed` timeInterval`; we don't assume `wi` time ordering
1185 wInterval.intersect(timeInterval);
1186 if (wInterval.empty()) continue;
1187
1188 chCoverage.openBetween(wInterval.start, wInterval.stop);
1189 } // for waveforms in channel
1190
1191 if (chWaveforms.empty()) {
1192 mf::LogTrace{ fLogCategory } << " - no waveform in channel";
1193 }
1194 else if (chCoverage != coverage) {
1195 mf::LogTrace{ fLogCategory } << " - channel "
1196 << chWaveforms.front()->waveform().ChannelNumber()
1197 << " with " << chWaveforms.size() << " waveforms => " << chCoverage;
1198 }
1199 coverage = icarus::trigger::GateOps::Mul(coverage, chCoverage);
1200
1201 } // for channels
1202 mf::LogTrace{ fLogCategory } << "Total coverage from " << waveforms.size()
1203 << " channels: " << coverage;
1204
1205 //
1206 // 2. if there are multiple non-contiguous sub-intervals, pick the widest one
1207 //
1208 auto nextWindow
1209 = [](TimeCoverage_t const& coverage, TimeInterval_t const& prev)
1210 {
1211 TimeInterval_t next;
1212 next.start = coverage.findOpen(1, prev.stop);
1213 if (next.start == TimeCoverage_t::MaxTick) return TimeInterval_t{};
1214 next.stop = coverage.findClose(1, next.start);
1215 return next;
1216 };
1217
1218 TimeInterval_t interval{ timeInterval.start };
1219 auto duration = interval.duration();
1220 TimeInterval_t runningWindow{ TimeCoverage_t::MinTick };
1221 while (true) {
1222 runningWindow = nextWindow(coverage, runningWindow);
1223 if (runningWindow.empty()) break; // no more openings left
1224 assert(runningWindow.stop != TimeCoverage_t::MaxTick);
1225
1226 auto const windowDuration = runningWindow.duration();
1227 if (duration > windowDuration) continue; // smaller window, discarded
1228 duration = windowDuration;
1229 interval = runningWindow;
1230 } // while
1231
1232 return interval;
1233} // icarus::trigger::DiscriminatedAdderSignal::computeTimeInterval()

1236//------------------------------------------------------------------------------
1237auto icarus::trigger::DiscriminatedAdderSignal::addWaveformsInInterval(
1238 TimeInterval_t const& timeInterval,
1239 std::vector<std::vector<WaveformWithBaseline const*>> const& waveforms,
1240 raw::Channel_t channel /* = std::numeric_limits<raw::Channel_t>::max() */
1241) const
1242 -> std::pair<std::valarray<float> , std::vector<WaveformWithBaseline const*>>
1243{
1244
1245 std::size_t const nSamples = timeInterval.empty()
1246 ? 0: tickDistance(timeInterval.start, timeInterval.stop);
1247
1248 std::valarray<float> added(0.0f, nSamples);
1249 std::vector<WaveformWithBaseline const*> used;
1250
1251 mf::LogTrace{ fLogCategory } << "Building adder waveform CH=" << channel
1252 << ", " << nSamples << " sample long, at interval " << timeInterval
1253 << " from " << waveforms.size() << " channels";
1254
1255 for (std::vector<WaveformWithBaseline const*> const& chWaveforms: waveforms) {
1256
1257 for (WaveformWithBaseline const* wi: chWaveforms) {
1258
1259 TimeInterval_t const wfCoverage = waveformTimeCoverage(wi->waveform());
1260
1261 if (!wfCoverage.contains(timeInterval.start)) continue;
1262 // in this approach, we assume that one of the selected waveforms covers
1263 // the whole interval (if it were more, there would be a gap, and this
1264 // algorithm should have carved an interval where no channel has gaps)
1265 assert(wfCoverage.contains(timeInterval.stop));
1266
1267 // add (minus baseline)
1268 std::ptrdiff_t const startSample
1269 = tickDistance(wfCoverage.start, timeInterval.start);
1270
1271 auto itSample = std::next(wi->waveform().cbegin(), startSample);
1272 for (float& sample: added) sample += *(itSample++);
1273
1274 if (wi->hasBaseline()) added -= wi->baseline().baseline();
1275
1276 { // --- BEGIN -- DEBUG --------------------------------------------------
1277 auto const [ startBaseline, endBaseline ]
1278 = computeSimpleBaselines(wi->waveform(), 500_ns);
1279 auto const itLowest
1280 = std::min_element(wi->waveform().begin(), wi->waveform().end());
1281 raw::ADC_Count_t const lowest = *itLowest;
1282 electronics_time const lowestTime = wfCoverage.start
1283 + fOpticalTick * std::distance(wi->waveform().begin(), itLowest);
1284 mf::LogTrace log{ fLogCategory };
1285 log << " - from ch=" << std::setw(3) << wi->waveform().ChannelNumber()
1286 << ", " << wfCoverage
1287 << ", sample=" << std::setw(5) << startSample << ", baseline: ";
1288 if (wi->hasBaseline()) {
1289 double const baseline = wi->baseline().baseline();
1290 log << std::setw(7) << std::setprecision(6) << std::left << baseline
1291 << " ADC, diff of local start="
1292 << std::showpos << std::setprecision(2) << std::setw(7) << std::left
1293 << (startBaseline - baseline) << " ADC, end="
1294 << std::showpos << std::setprecision(2) << std::setw(7) << std::left
1295 << (endBaseline - baseline) << " ADC, peak amplitude: "
1296 << std::defaultfloat << std::setprecision(6) << std::setw(8)
1297 << (baseline - lowest) << " ADC at time " << lowestTime;
1298 }
1299 else {
1300 log << "n/a, local start=" << std::setw(7) << std::setprecision(1)
1301 << startBaseline << " ADC, end="
1302 << std::setw(7) << std::setprecision(1) << endBaseline << " ADC"
1303 << " ADC, lowest point: " << lowest << " ADC at time "
1304 << lowestTime;
1305 }
1306 } // --- END ---- DEBUG --------------------------------------------------
1307
1308 used.push_back(wi); // track
1309 break; // this waveform has full coverage as per above, no more needed
1310 } // for waveforms in channel
1311
1312 if (chWaveforms.empty()) {
1313 mf::LogTrace{ fLogCategory } << " - empty input collection!";
1314 }
1315 else if (used.empty()
1316 || (
1317 chWaveforms.front()->waveform().ChannelNumber()
1318 != used.back()->waveform().ChannelNumber()
1319 )
1320 ) {
1321 mf::LogTrace{ fLogCategory } << " - none of the " << chWaveforms.size()
1322 << " waveforms of channel "
1323 << chWaveforms.front()->waveform().ChannelNumber()
1324 << " falls in the requested interval";
1325 }
1326
1327 } // for channels
1328
1329 added *= fAmplitudeScale;
1330
1331 return { std::move(added), std::move(used) };
1332
1333} // icarus::trigger::DiscriminatedAdderSignal::addWaveformsInInterval()
