117icarus::crt::MatchedCRT CRTHitmatched(
118 const double& flashTime, geo::Point_t const& flashpos,
119 std::vector<art::Ptr<sbn::crt::CRTHit>>& crtHits, const double& interval) {
120 std::vector<icarus::crt::CRTPMT> enteringCRTHits;
121 std::vector<icarus::crt::CRTPMT> exitingCRTHits;
122
123 hasCRTHit MatchType;
124 int topen = 0, topex = 0, sideen = 0, sideex = 0;
125 for (auto const crtHit : crtHits) {
126 double tof = crtHit->ts1_ns / 1e3 - flashTime;
127 // double distance = std::hypot(flashpos[0]-crtHit->x_pos,
128 // flashpos[1]-crtHit->y_pos, flashpos[2]-crtHit->z_pos);
129 double distance =
130 (flashpos - geo::Point_t{crtHit->x_pos, crtHit->y_pos, crtHit->z_pos})
131 .R();
132 if (abs(tof) >= interval) continue;
133 if (tof < 0) {
134 if (crtHit->plane > 36)
135 sideen++;
136 else
137 topen++;
138 CRTPMT m_match = {tof, distance, crtHit};
139 enteringCRTHits.push_back(m_match);
140 } else if (tof >= 0) {
141 if (crtHit->plane > 36)
142 sideex++;
143 else
144 topex++;
145 CRTPMT m_match = {tof, distance, crtHit};
146 exitingCRTHits.push_back(m_match);
147 }
148 }
149 if (topen == 0 && sideen == 0 && topex == 0 && sideex == 0)
150 MatchType = noMatch;
151 else if (topen == 1 && sideen == 0 && topex == 0 && sideex == 0)
152 MatchType = enTop;
153 else if (topen == 0 && sideen == 1 && topex == 0 && sideex == 0)
154 MatchType = enSide;
155 else if (topen == 1 && sideen == 0 && topex == 0 && sideex == 1)
156 MatchType = enTop_exSide;
157 else if (topen == 0 && sideen == 0 && topex == 1 && sideex == 0)
158 MatchType = exTop;
159 else if (topen == 0 && sideen == 0 && topex == 0 && sideex == 1)
160 MatchType = exSide;
161 else if (topen >= 1 && sideen >= 1 && topex == 0 && sideex == 0)
162 MatchType = enTop_mult;
163 else if (topen >= 1 && sideen >= 1 && topex == 0 && sideex >= 1)
164 MatchType = enTop_exSide_mult;
165 else
166 MatchType = others;
167
168 return {std::move(enteringCRTHits), std::move(exitingCRTHits), MatchType};
169}
