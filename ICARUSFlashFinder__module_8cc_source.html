138void ICARUSFlashFinder::GetFlashLocation(std::vector<double> pePerOpChannel,
139 double& Ycenter,
140 double& Zcenter,
141 double& Ywidth,
142 double& Zwidth)
143{
144
145 // Reset variables
146 Ycenter = Zcenter = 0.;
147 Ywidth = Zwidth = -999.;
148 double totalPE = 0.;
149 double sumy = 0., sumz = 0., sumy2 = 0., sumz2 = 0.;
150
151 for (unsigned int opch = 0; opch < pePerOpChannel.size(); opch++) {
152 /*
153 if (opch > 31 && opch < 200){
154 // std::cout << "Ignoring channel " << opch << " as it's not a real channel" << std::endl;
155 continue;
156 }
157 */
158 // Get physical detector location for this opChannel
159 double PMTxyz[3];
160 ::pmtana::OpDetCenterFromOpChannel(opch, PMTxyz);
161
162 // Add up the position, weighting with PEs
163 sumy += pePerOpChannel[opch]*PMTxyz[1];
164 sumy2 += pePerOpChannel[opch]*PMTxyz[1]*PMTxyz[1];
165 sumz += pePerOpChannel[opch]*PMTxyz[2];
166 sumz2 += pePerOpChannel[opch]*PMTxyz[2]*PMTxyz[2];
167
168 totalPE += pePerOpChannel[opch];
169 }
170
171 Ycenter = sumy/totalPE;
172 Zcenter = sumz/totalPE;
173
174 // This is just sqrt(<x^2> - <x>^2)
175 if ( (sumy2*totalPE - sumy*sumy) > 0. )
176 Ywidth = std::sqrt(sumy2*totalPE - sumy*sumy)/totalPE;
177
178 if ( (sumz2*totalPE - sumz*sumz) > 0. )
179 Zwidth = std::sqrt(sumz2*totalPE - sumz*sumz)/totalPE;
180}
