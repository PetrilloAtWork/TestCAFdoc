30 {
31 std::map< std::pair<raw::ChannelID_t, int>, unsigned int > channelTimeHits; // converting the float PeakTime to int for the second item in pair
32
33 art::Handle<std::vector<recob::Hit>> theHits;
34 evt.getByLabel(label, theHits);
35
36 if (!theHits.isValid()) {

39 }
40 else {
41 mf::LogDebug("LArPandoraHitCollectionToolICARUS") << " Found: " << theHits->size() << " Hits " << std::endl;
42 }
43
44 for (unsigned int i = 0; i < theHits->size(); ++i) {
45 const art::Ptr<recob::Hit> hit(theHits, i);

47 hitVector.push_back(hit);
48 }

50 mf::LogDebug("LArPandoraHitCollectionToolICARUS") << " Passing along " << hitVector.size() << " Hits " << std::endl;
