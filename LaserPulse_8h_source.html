37 //double mu = par[1];
38 double w = par[1];
39 double c = par[2];
40 double a = par[3];
41
42 double t = x[0];
43
44 return a*c/2.0*TMath::Exp(c*c*w*w/2.0)*TMath::Exp(-1.0*c*(t-t0))
