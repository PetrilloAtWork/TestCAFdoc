130void SimChannelROI::produce(art::Event& evt)
131{
132 auto const clockData = art::ServiceHandle<detinfo::DetectorClocksService const>()->DataFor(evt);
133
134 // We need to loop through the list of Wire data we have been given
135 for(size_t labelIdx = 0; labelIdx < fSimChannelLabelVec.size(); labelIdx++)
136 {
137 const art::InputTag& driftEModuleLabel = fSimChannelLabelVec[labelIdx];
138
139 // make a collection of ChannelROI objects
140 std::unique_ptr<std::vector<recob::ChannelROI>> channelROICol(new std::vector<recob::ChannelROI>);
141
142 mf::LogInfo("SimChannelROI") << "SimChannelROI, looking for decon1droi data at " << driftEModuleLabel << std::endl;
143
144 // Read in the collection of full length deconvolved waveforms
145 // Note we assume this list is sorted in increasing channel number!
146 std::vector<const sim::SimChannel*> simChannelHandle;
147 evt.getView(driftEModuleLabel,simChannelHandle);
148
149 mf::LogInfo("SimChannelROI") << "--> Recovered SimChannel data, size: " << simChannelHandle.size() << std::endl;
150
151 if (!simChannelHandle.size())
152 {
153 evt.put(std::move(channelROICol), fOutInstanceLabelVec[labelIdx]);
154
155 continue;
156 }
157
158 // Reserve the room for the output
159 channelROICol->reserve(simChannelHandle.size());
160
161 // we simply loop over simChannels (arranged by channel)
162 size_t numChannels(0);
163
164 for(const auto& simChannel : simChannelHandle)
165 {
166 raw::ChannelID_t channel = simChannel->Channel();
167
168 // Total charge deposition from SimChannel
169 std::vector<short int> chargeVec;
170
171 // vector that will be moved into the Wire object
172 recob::ChannelROI::RegionsOfInterest_t intROIVec;
173
174 size_t startTick(0);
175 size_t gapTicks(0);
176
177 // Here go through the input simchannel and build out the charge array
178 for(size_t tick = 0; tick < fNTimeSamples; tick++)
179 {
180 // Note that we attempt to explicitly remove the time offsets between the planes and leave it to the response
181 // functions to handle this. So we reference the time to the first plane
182 //int tdc = clockData.TPCTick2TDC(tick + detProp.GetXTicksOffset(planeID) - detProp.GetXTicksOffset(geo::PlaneID(cryostat,tpc,0)));
183 int tdc = clockData.TPCTick2TDC(tick);
184
185 // continue if tdc < 0
186 if( tdc < 0 ) continue;
187
188 float charge = simChannel->Charge(tdc) / fGain; // Charge returned in number of electrons
189
190 // Need to insure we don't exceed short int limits
191 if (charge > std::numeric_limits<short>::max()) charge = std::numeric_limits<short>::max();
192
193 if (charge > 0.)
194 {
195 if (chargeVec.empty()) startTick = tick;
196 else if (gapTicks > 0) chargeVec.resize(chargeVec.size()+gapTicks,0);
197
198 chargeVec.push_back(short(std::round(charge)));
199
200 gapTicks = 0;
201 }
202 else
203 {
204 if (gapTicks > 2)
205 {
206 if (!chargeVec.empty())
207 {
208 intROIVec.add_range(startTick,std::move(chargeVec));
209
210 chargeVec.clear();
211 }
212
213 gapTicks = 0;
214 }
215 else gapTicks++;
216 }
217 } // loop over tdcs
218
219 // Look for end case
220 if (!chargeVec.empty()) intROIVec.add_range(startTick,std::move(chargeVec));
221
222 if (!intROIVec.empty()) channelROICol->push_back(recob::ChannelROICreator(std::move(intROIVec),channel).move());
223
224 numChannels++;
225 }
226
227 // Time to stroe everything
228 if(channelROICol->size() == 0) mf::LogWarning("SimChannelROI") << "No wires made for this event.";
229
230 evt.put(std::move(channelROICol), fOutInstanceLabelVec[labelIdx]);
231 }
232
233 fEventCount++;
234
235 return;
236} // produce
