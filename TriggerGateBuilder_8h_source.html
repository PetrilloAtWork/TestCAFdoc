293template <typename Stream>
294void icarus::trigger::TriggerGateBuilder::TriggerGates::dump(Stream& out) const
295{
296
297 auto const nOpenGates = fGates.size() - std::count_if(
298 fGates.begin(), fGates.end(),
299 [](auto const& gate){ return gate.gate().alwaysClosed(); }
300 );
301 out << nOpenGates << "/" << fGates.size() << " trigger gates on threshold "
302 << fThreshold;
303 if (nOpenGates) {
304 out << ":";
305 for (auto const& gate: icarus::trigger::gatesIn(fGates)) {
306 if (gate.alwaysClosed()) continue;
307 out << "\n " << gate;
308 }
309 }
310 out << "\n";
311} // icarus::trigger::TriggerGateBuilder::TriggerGates::dump()
