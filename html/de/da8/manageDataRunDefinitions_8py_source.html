139class DimensionQueryMaker:
140 """Callable object making a SAM query out of a `SampleInfo` object."""
141
142 def __call__(self,
143 info: "SampleInfo object with all the information for the query",
144 minimum: "throws an exception if fewer than these elements are included" = 0,
145 ) -> "a SAM dimensions query string":
146
147 dims = [
148 DimensionQueryMaker.simpleItem('run_number', info.run),
149 DimensionQueryMaker.multiItem(StageDimensions, info.stage, 'stage'),
150 DimensionQueryMaker.simpleItem('data_stream', info.stream),
151 DimensionQueryMaker.simpleItem('icarus_project.version', info.projectVersion),
152 ]
153 if info.minSize > 0:
154 dims.append(
155 DimensionQueryMaker.comparedItem('file_size', info.minSize << 20, ">=")
156 )
157 query = " and ".join(filter(None, dims))
158 if len(dims) < minimum:
159 raise RuntimeError(f"Query resulted in only {len(dims)} constraints: '{query}'")
160 return query
161 # __call__()

163 @staticmethod

166 @staticmethod
167 def simpleItem(key, value, typeName=None):
168 # if typeName is None: typeName = key
169 if SampleInfo.hasOptions(value):
170 return f"{key} in ( { ' '.join(map(str, value)) } )"
171 elif value is not None: return f"{key} {value}"
172 else: return ""
173 # simpleItem()
174
175
176 @staticmethod
177 def comparedItem(key, value, compType=None):
178 if value is not None: return f"({key} {compType} {value})"
179 else: return ""

183 @staticmethod
184 def multiItem(queries, values, typeName):
185 values = SampleInfo.makeOptionList(values)

187 dims = []
188 for value in values:
189 try:
190 dims.append(queries[value])
191 except KeyError:
192 raise RuntimeError(
193 f"{typeName} '{value}' not supported (expected one of: {', '.join(map(str, queries))})."
194 )
195 # try ... except
196 # for
197 if len(dims) > 1: dims = list(map(DimensionQueryMaker.addParentheses, dims))
198 query = " or ".join(dims) if dims else ""
199 if len(dims) > 1: query = DimensionQueryMaker.addParentheses(query)
200 return query
