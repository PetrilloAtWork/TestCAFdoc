286//ROOT::Math::XYZTVector
287TLorentzVector CRTCommonUtils::AvgIDEPoint(sim::AuxDetIDE ide){
288 double x = 0.5*(ide.entryX + ide.exitX);
289 double y = 0.5*(ide.entryY + ide.exitY);
290 double z = 0.5*(ide.entryZ + ide.exitZ);
291 double t = 0.5*(ide.entryT + ide.exitT);
292 //ROOT::Math::XYZTVector
293 TLorentzVector v(x,y,z,t);
294
295 return v;
296}

299//returns the path length in the scintillator strip
300double CRTCommonUtils::LengthIDE(sim::AuxDetIDE ide) {
301 double dx=0., dy=0., dz=0.;
302 dx = ide.entryX - ide.exitX;
303 dy = ide.entryY - ide.exitY;
304 dz = ide.entryZ - ide.exitZ;
305 return sqrt(dx*dx+dy*dy+dz*dz);
306}

521//reads a file generated by CRT geometry generation script and
522// fills a map modID->vector<pair<FEB, FEB channel subset>,+1 dual readoutfor MINOS module>
523// channel subset = 1,2,or 3 (always =1 for c or d modules)
524void CRTCommonUtils::FillFebMap() {
525
526 string fullFileName;// = "/icarus/app/users/chilgenb/ana_icaruscode_v08_52_00/feb_map.txt";
527 cet::search_path searchPath("FW_SEARCH_PATH");
528 searchPath.find_file("feb_map.txt",fullFileName);
529 std::ifstream fin;
530 fin.open(fullFileName,std::ios::in);
531
532 if(!fin.good())
533// std::cout << "opened file 'feb_map.txt' for reading..." << std::endl;
534// else
535 throw cet::exception("CRTDetSim::FillFebMap")
536 << "Unable to find/open file 'feb_map.txt'" << std::endl;
537
538 vector<string> row;
539 string line, word;
540 //each line has pattern
541 // auxDetID, mac5, chan pos, '\n' (CERN, DC, cut MINOS) OR
542 // auxDetID, mac5, chan pos, mac5', chan pos', '\n' (full length MINOS)
543 while(getline(fin,line)) {
544 row.clear();
545 std::stringstream s(line);
546 while (std::getline(s, word, ',')) { //parse line
547 row.push_back(word);
548 }
549 int mod = (size_t)std::stoi(row[0]); //auxDetID
550 uint8_t mac5 = (uint8_t)std::stoi(row[1]); //febID
551 int pos = std::stoi(row[2]); //feb channel block
552 fAuxDetIdToFeb[mod].push_back(std::make_pair(mac5,pos));
553 fAuxDetIdToChanGroup[mod]=pos;
554 fFebToAuxDetId[mac5].push_back(mod);
555// std::cout << "mod: " << mod << ", mac: " << (int)mac5 << ", pos: " << pos;
556 if(row.size()>3) { //if dual ended readout MINOS module
557 mac5 = (uint8_t)std::stoi(row[3]);
558 fAuxDetIdToFeb[mod].push_back(std::make_pair(mac5,pos));
559 fFebToAuxDetId[mac5].push_back(mod);
560// if(pos!=std::stoi(row[4])) //feb channel block same on both febs
561// std::cout << "WARNING in CRTComUtil: 2 unique chan groups for ADId!" << std::endl;
562// std::cout << ", mac: " << (int)mac5;
563 }
564// std::cout << std::endl;
565 }
566// std::cout << "filled febMap with " << fAuxDetIdToFeb.size() << " entries" << std::endl;
567 fin.close();
568}

598string CRTCommonUtils::AuxDetNameToRegion(string name) {
599
600 string base("volAuxDet");
601 const char type = fAuxDetIdToType[fNameToAuxDetId[name]];
602
603 switch( type ){
604 case 'c' : base+= "CERN"; break;
605 case 'd' : base+= "DC"; break;
606 case 'm' : base+= "MINOS"; break;
607 default :
608 throw cet::exception("CRTCommonUtils::Constructor::AuxDetNameToRegion")
609 << "AuxDet type not set!";
610 }
611 base+="module###";
612
613 //keep compatibility with legacy g4
614 name.erase(std::remove(name.begin(), name.end(), '_'), name.end());
615
616 //module name has 2 possible formats
617 // volAuxDet_<subsystem>_module_###_<region>
618 // volAuxDet_<subsystem>_module_###_cut###_<region>
619 string region(name.substr(base.length(),name.length()));
620 if( region.find("cut")==string::npos)
621 return region;
622
623 else
624 return region.substr(region.find("cut")+6,region.length());

650 localpoint.SetY(0);

653 }

669// Minimum distance from infinite track to CRT hit assuming that hit is a 2D square
670double CRTCommonUtils::DistToCrtHit(sbn::crt::CRTHit hit, TVector3 start, TVector3 end){
671
672 // Check if track goes inside hit
673 TVector3 min (hit.x_pos - hit.x_err, hit.y_pos - hit.y_err, hit.z_pos - hit.z_err);
674 TVector3 max (hit.x_pos + hit.x_err, hit.y_pos + hit.y_err, hit.z_pos + hit.z_err);
675 if(CubeIntersection(min, max, start, end).first.X() != -99999) return 0;
676
677 // Calculate the closest distance to each edge of the CRT hit
678 // Assume min error is the fixed position of tagger
679 TVector3 vertex1 (hit.x_pos, hit.y_pos - hit.y_err, hit.z_pos - hit.z_err);
680 TVector3 vertex2 (hit.x_pos, hit.y_pos + hit.y_err, hit.z_pos - hit.z_err);
681 TVector3 vertex3 (hit.x_pos, hit.y_pos - hit.y_err, hit.z_pos + hit.z_err);
682 TVector3 vertex4 (hit.x_pos, hit.y_pos + hit.y_err, hit.z_pos + hit.z_err);
683 if(hit.y_err < hit.x_err && hit.y_err < hit.z_err){
684 vertex1.SetXYZ(hit.x_pos - hit.x_err, hit.y_pos, hit.z_pos - hit.z_err);
685 vertex2.SetXYZ(hit.x_pos + hit.x_err, hit.y_pos, hit.z_pos - hit.z_err);
686 vertex3.SetXYZ(hit.x_pos - hit.x_err, hit.y_pos, hit.z_pos + hit.z_err);
687 vertex4.SetXYZ(hit.x_pos + hit.x_err, hit.y_pos, hit.z_pos + hit.z_err);
688 }
689 if(hit.z_err < hit.x_err && hit.z_err < hit.y_err){
690 vertex1.SetXYZ(hit.x_pos - hit.x_err, hit.y_pos - hit.y_err, hit.z_pos);
691 vertex2.SetXYZ(hit.x_pos + hit.x_err, hit.y_pos - hit.y_err, hit.z_pos);
692 vertex3.SetXYZ(hit.x_pos - hit.x_err, hit.y_pos + hit.y_err, hit.z_pos);
693 vertex4.SetXYZ(hit.x_pos + hit.x_err, hit.y_pos + hit.y_err, hit.z_pos);
694 }
695
696 double dist1 = LineSegmentDistance(vertex1, vertex2, start, end);
697 double dist2 = LineSegmentDistance(vertex1, vertex3, start, end);
698 double dist3 = LineSegmentDistance(vertex4, vertex2, start, end);
699 double dist4 = LineSegmentDistance(vertex4, vertex3, start, end);
700
701 return std::min(std::min(dist1, dist2), std::min(dist3, dist4));
702
703}

706// Distance between infinite line (2) and segment (1)
707// http://geomalgorithms.com/a07-_distance.html
708double CRTCommonUtils::LineSegmentDistance(TVector3 start1, TVector3 end1, TVector3 start2, TVector3 end2){
709
710 double smallNum = 0.00001;
711
712 // 1 is segment
713 TVector3 direction1 = end1 - start1;
714 // 2 is infinite line
715 TVector3 direction2 = end2 - start2;
716
717 TVector3 u = direction1;
718 TVector3 v = direction2;
719 TVector3 w = start1 - start2;
720
721 double a = u.Dot(u);
722 double b = u.Dot(v);
723 double c = v.Dot(v);
724 double d = u.Dot(w);
725 double e = v.Dot(w);
726 double D = a * c - b * b;
727 double sc, sN, sD = D; // sc = sN/sD
728 double tc, tN, tD = D; // sc = sN/sD
729
730 // Compute the line parameters of the two closest points
731 if(D < smallNum){ // Lines are almost parallel
732 sN = 0.0;
733 sD = 1.0;
734 tN = e;
735 tD = c;
736 }
737 else{
738 sN = (b * e - c * d)/D;
739 tN = (a * e - b * d)/D;
740 if(sN < 0.){ // sc < 0, the s = 0 edge is visible
741 sN = 0.;
742 tN = e;
743 tD = c;
744 }
745 else if(sN > sD){ // sc > 1, the s = 1 edge is visible
746 sN = sD;
747 tN = e + b;
748 tD = c;
749 }
750 }
751
752 sc = (std::abs(sN) < smallNum ? 0.0 : sN / sD);
753 tc = (std::abs(tN) < smallNum ? 0.0 : tN / tD);
754 // Get the difference of the two closest points
755 TVector3 dP = w + (sc * u) - (tc * v);
756
757 return dP.Mag();
758
759}

761// Intersection between axis-aligned cube and infinite line
762// (https://www.scratchapixel.com/lessons/3d-basic-rendering/minimal-ray-tracer-rendering-simple-shapes/ray-box-intersection)
763std::pair<TVector3, TVector3> CRTCommonUtils::CubeIntersection(TVector3 min, TVector3 max, TVector3 start, TVector3 end){
764
765 TVector3 dir = (end - start);
766 TVector3 invDir (1./dir.X(), 1./dir.Y(), 1/dir.Z());
767
768 double tmin, tmax, tymin, tymax, tzmin, tzmax;
769
770 TVector3 enter (-99999, -99999, -99999);
771 TVector3 exit (-99999, -99999, -99999);
772
773 // Find the intersections with the X plane
774 if(invDir.X() >= 0){
775 tmin = (min.X() - start.X()) * invDir.X();
776 tmax = (max.X() - start.X()) * invDir.X();
777 }
778 else{
779 tmin = (max.X() - start.X()) * invDir.X();
780 tmax = (min.X() - start.X()) * invDir.X();
781 }
782
783 // Find the intersections with the Y plane
784 if(invDir.Y() >= 0){
785 tymin = (min.Y() - start.Y()) * invDir.Y();
786 tymax = (max.Y() - start.Y()) * invDir.Y();
787 }
788 else{
789 tymin = (max.Y() - start.Y()) * invDir.Y();
790 tymax = (min.Y() - start.Y()) * invDir.Y();
791 }
792
793 // Check that it actually intersects
794 if((tmin > tymax) || (tymin > tmax)) return std::make_pair(enter, exit);
795
796 // Max of the min points is the actual intersection
797 if(tymin > tmin) tmin = tymin;
798
799 // Min of the max points is the actual intersection
800 if(tymax < tmax) tmax = tymax;
801
802 // Find the intersection with the Z plane
803 if(invDir.Z() >= 0){
804 tzmin = (min.Z() - start.Z()) * invDir.Z();
805 tzmax = (max.Z() - start.Z()) * invDir.Z();
806 }
807 else{
808 tzmin = (max.Z() - start.Z()) * invDir.Z();
809 tzmax = (min.Z() - start.Z()) * invDir.Z();
810 }
811
812 // Check for intersection
813 if((tmin > tzmax) || (tzmin > tmax)) return std::make_pair(enter, exit);
814
815 // Find final intersection points
816 if(tzmin > tmin) tmin = tzmin;
817
818 // Find final intersection points
819 if(tzmax < tmax) tmax = tzmax;
820
821 // Calculate the actual crossing points
822 double xmin = start.X() + tmin * dir.X();
823 double xmax = start.X() + tmax * dir.X();
824 double ymin = start.Y() + tmin * dir.Y();
825 double ymax = start.Y() + tmax * dir.Y();
826 double zmin = start.Z() + tmin * dir.Z();
827 double zmax = start.Z() + tmax * dir.Z();
828
829 // Return pair of entry and exit points
830 enter.SetXYZ(xmin, ymin, zmin);
831 exit.SetXYZ(xmax, ymax, zmax);
832 return std::make_pair(enter, exit);
833
834}
