126// Primary working method of service that provides position offsets
127geo::Vector_t spacecharge::SpaceChargeICARUS::GetPosOffsets(geo::Point_t const& point) const
128{
129 std::vector<double> thePosOffsets;
130 double xx=point.X(), yy=point.Y(), zz=point.Z();
131 double cryo_corr=1., tpc_corr=1.;
132
133 if(fRepresentationType == "Voxelized_TH3"){
134 //handle OOAV by projecting edge cases
135 //also only have map for positive cryostat (assume symmetry)
136 //need to invert coordinates for cryo0 (cryo_corr)
137
138 //in larsim, this is how the offsets are used in DriftElectronstoPlane_module
139 // DriftDistance += -1.0 * thePosOffsets[0]
140 // thus need to apply correction to TPCs "left" of cryostat (tpc_corr)
141 // cathode spans x=210.14 and x=210.29 in pos cryostat
142 if(xx>0){
143 cryo_corr=1.0;
144
145 if(f_2D_drift_sim_hack==true)
146 tpc_corr=-1.0;
147
148 if(xx<210.14){
149 tpc_corr=-1.0;
150 }
151
152 }else{
153 cryo_corr=-1.0;
154
155 if(f_2D_drift_sim_hack==true)
156 tpc_corr=-1.0;
157
158 if(xx<-210.29){
159 tpc_corr=-1.0;
160 }
161
162 }
163 fixCoords(&xx, &yy, &zz); //bring into AV and x = abs(x)
164 double offset_x=0., offset_y=0., offset_z=0.;
165 offset_x = tpc_corr*cryo_corr*SCEhistograms.at(0)->Interpolate(xx,yy,zz);
166 offset_y = SCEhistograms.at(1)->Interpolate(xx,yy,zz);
167 offset_z = SCEhistograms.at(2)->Interpolate(xx,yy,zz);
168 thePosOffsets = {offset_x, offset_y, offset_z};
169 }else{
170 thePosOffsets.resize(3, 0.0);
171 }
172
173 return { thePosOffsets[0], thePosOffsets[1], thePosOffsets[2] };
174}

176// Returns the SCE correction at a specific point in the AV
177 geo::Vector_t spacecharge::SpaceChargeICARUS::GetCalPosOffsets(geo::Point_t const& point, int const& TPCid) const
178{
179 std::vector<double> theCalPosOffsets;
180 //make copies of const vars to modify
181 double xx=point.X(), yy=point.Y(), zz=point.Z();
182 int tpcid = TPCid;
183
184 if(fRepresentationType == "Voxelized_TH3"){
185 //handle OOAV by projecting edge cases
186 //also only have map for positive cryostat (assume symmetry)
187 //need to invert coordinates for cryo0
188 double corr=1.;
189 if(xx<0){
190 corr=-1.0;
191 }
192
193 bool x_is_pos = xx > 0;
194
195 fixCoords(&xx, &yy, &zz); //bring into AV and x = abs(x)
196 //handle the depositions that was reconstructed in the wrong TPC
197 //hard code in the cathode faces (got from dump_icarus_geometry.fcl)
198 //
199 //Gray Putnam: update this check to the split-wire Geometry
200 if (x_is_pos && (tpcid == 0 || tpcid == 1) && xx > 210.14 ) { xx = 210.14; }
201 if (x_is_pos && (tpcid == 2 || tpcid == 3) && xx < 210.29 ) { xx = 210.29; }
202
203 if (!x_is_pos && (tpcid == 2 || tpcid == 3) && xx > 210.14 ) { xx = 210.14; }
204 if (!x_is_pos && (tpcid == 0 || tpcid == 1) && xx < 210.29 ) { xx = 210.29; }
205
206 double offset_x=0., offset_y=0., offset_z=0.;
207 offset_x = corr*SCEhistograms.at(3)->Interpolate(xx,yy,zz);
208 offset_y = SCEhistograms.at(4)->Interpolate(xx,yy,zz);
209 offset_z = SCEhistograms.at(5)->Interpolate(xx,yy,zz);
210 theCalPosOffsets = {offset_x, offset_y, offset_z};
211 }else{
212 theCalPosOffsets.resize(3, 0.0);
213 }
214
215 return { theCalPosOffsets[0], theCalPosOffsets[1], theCalPosOffsets[2] };
216}

245void spacecharge::SpaceChargeICARUS::fixCoords(double* xx, double* yy, double* zz) const{
246 //handle the edge cases by projecting SCE corrections onto boundaries
247 *xx = abs(*xx);
248 if(*xx<61.94){*xx=61.94;}
249 if(*xx>358.489){*xx=358.489;}
250 if(*yy<-181.86){*yy=-181.86;}
251 if(*yy>134.96){*yy=134.96;}
252 if(*zz<-894.951){*zz=-894.951;}
253 if(*zz>894.9509){*zz=894.9509;}
254}
