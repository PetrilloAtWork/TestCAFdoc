148std::array<Integer, N> icarusDB::IChannelMapping::splitIntegers
149 (std::string const& s, std::string const& sep)
150{
151 std::array<Integer, N> values;
152
153 auto convert = [](std::string_view s)
154 {
155 Integer value;
156 std::from_chars_result const res
157 = std::from_chars(s.cbegin(), s.cend(), value);
158 if (res.ec != std::errc{}) {
159 std::error_code ec = std::make_error_code(res.ec);
160 throw std::runtime_error{ "Failed to convert '" + std::string{ s }
161 + "' into an integer: " + ec.message() + " (code: "
162 + std::to_string(ec.value()) + ")."
163 };
164 }
165 return value;
166 };
167
168 std::string_view const sv{ s };
169 std::size_t c = 0; // cursor
170 for (std::size_t const i: util::counter(N - 1)) {
171 std::size_t sc = sv.find(sep, c);
172 if (sc == std::string_view::npos) {
173 throw std::runtime_error{ "Separator '" + std::string{ sep }
174 + "' not found after number #" + std::to_string(i) + " of string '"
175 + std::string{ s } + "'"
176 };
177 }
178 values[i] = convert(sv.substr(c, sc));
179 c = sc + sep.length();
180 } // for
181
182 values[N - 1] = convert(sv.substr(c));
183
184 return values;
185} // icarusDB::IChannelMapping::splitIntegers()
