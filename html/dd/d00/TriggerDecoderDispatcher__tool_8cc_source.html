427// -----------------------------------------------------------------------------
428void daq::TriggerDecoderDispatched::createDecoder(std::string const& generator)
429{
430
431 // determine the tool
432 GeneratorParams_t const* newGenParam = nullptr;
433 for (GeneratorParams_t const& genParam: fGenParams) {
434 if (!std::regex_match(generator, genParam.generatorPattern)) continue;
435 newGenParam = &genParam;
436 break;
437 } // for
438
439 if (!newGenParam) {
440 art::Exception e{ art::errors::NotFound };
441 e << "Generator '" << generator << "' not supported (supported: ";
442 auto itGenParam = fGenParams.cbegin();
443 auto gend = fGenParams.cend();
444 e << "'" << itGenParam->generatorName << "'";
445 while (++itGenParam != gend) e << ", '" << itGenParam->generatorName << "'";
446 e << ").\n";
447 throw e;
448 }
449
450 // construct a new tool (only if not the same as the current one)
451 fCurrentGenerator = generator;
452 fCurrentGenParam = newGenParam;
453 fDecoder = art::make_tool<IDecoder>(fCurrentGenParam->toolConfig);
454
455 mf::LogDebug{ fLogCategory }
456 << "Generator '" << fCurrentGenerator << "' matches '"
457 << fCurrentGenParam->generatorName << "' tool configuration.";
458
459} // daq::TriggerDecoderDispatched::createDecoder()

463auto daq::TriggerDecoderDispatched::makeGeneratorParams
464 (std::vector<Config::DecoderConfig> const& genConfigs)
465 -> std::vector<GeneratorParams_t>
466{
467 std::vector<GeneratorParams_t> params;
468 params.reserve(genConfigs.size());
469 for (Config::DecoderConfig const& config: genConfigs) {
470 params.push_back({ // C++20: use named elements
471 config.Generator() // generatorName
472 , std::regex{ config.Generator() } // generatorPattern
473 , config.FragmentsLabel() // fragmentTag
474 , config.ToolConfig.get<fhicl::ParameterSet>() // toolConfig
475 });
476 } // for
477 return params;
