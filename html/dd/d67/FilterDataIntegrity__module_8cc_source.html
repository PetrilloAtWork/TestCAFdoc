47bool filter::FilterDataIntegrity::filter(art::Event &event)
48{
49 bool filterPass = true;
50
51 // get all the artdaq fragment collections in the event.
52 std::vector<art::Handle<std::vector<artdaq::Fragment>>> fragmentHandles;
53 fragmentHandles = event.getMany<std::vector<artdaq::Fragment>>();
54
55 std::set<int> missingFragments(fExpectedFragments);
56 int emptyFragments(0);
57
58 for (const auto& handle : fragmentHandles)
59 {
60 for (const auto& fragment : *handle)
61 {
62 int fragmentID = fragment.fragmentID();
63
64 fExpectedFragments.insert(fragmentID);
65 missingFragments.erase(fragmentID);
66
67 std::string instanceName = handle.provenance()->productInstanceName();
68 std::size_t found = instanceName.find("Empty");
69
70 if (found != std::string::npos) emptyFragments++;
71 }
72 }
73
74 mf::LogDebug("FilterDataIntegrity") << "Expected fragments: " << fExpectedFragments.size() << " expected, has " << missingFragments.size() << " missing. There are " << emptyFragments << " empty fragments";
75
76 if (!missingFragments.empty() || emptyFragments > 0)
77 {
78 mf::LogInfo("FilterDataIntegrity") << "Bad fragments: " << missingFragments.size() << ", " << emptyFragments;
79
80 filterPass = false;
81 }
82
83 return filterPass;
84}
