184void icarus::crt::DumpCRTPMTMatching::analyze(art::Event const& event) {
185
186 //
187 // collect all the input
188 //
189 auto const& matchHandle
190 = event.getValidHandle<std::vector<sbn::crt::CRTPMTMatching>>(fMatchingTag);
191
192 mf::LogInfo(fLogCategory)
193 << "The event contains " << matchHandle->size() << " '"
194 << fMatchingTag.encode() << "' CRT/PMT matches.";
195
196 auto const flashAssns = fPrintFlashAssociations
197 ? std::make_optional
198 (art::FindOneP<recob::OpFlash>(matchHandle, event, fMatchingTag))
199 : std::nullopt
200 ;
201 if (flashAssns && !flashAssns->isValid()) {
202 throw art::Exception(art::errors::ProductNotFound)
203 << "No flash associated with '" << fMatchingTag.encode() << "' matches.\n";
204 }
205
206 auto const hitAssns = fPrintCRTHitAssociations
207 ? std::make_optional
208 (art::FindManyP<sbn::crt::CRTHit>(matchHandle, event, fMatchingTag))
209 : std::nullopt
210 ;
211 if (hitAssns && !hitAssns->isValid()) {
212 throw art::Exception(art::errors::ProductNotFound)
213 << "No CRT hits associated with '" << fMatchingTag.encode() << "' matches.\n";
214 }
215
216 ProductTagIndex tagIndex{ event };
217
218 for (auto const& [ iMatch, match ]: util::enumerate(*matchHandle)) {
219
220 if (flashAssns) tagIndex.add(flashAssns->at(iMatch));
221 if (hitAssns) {
222 for (art::Ptr<sbn::crt::CRTHit> const& hitPtr: hitAssns->at(iMatch))
223 tagIndex.add(hitPtr);
224 }
225 DumpMatching(iMatch, match
226 , flashAssns? flashAssns->at(iMatch): art::Ptr<recob::OpFlash>{}
227 , hitAssns? hitAssns->at(iMatch): std::vector<art::Ptr<sbn::crt::CRTHit>>{}
228 , tagIndex
229 );
230
231 } // for matches
232
233} // icarus::crt::DumpCRTPMTMatching::analyze()

236//---------------------------------------------------------------------------
237void icarus::crt::DumpCRTPMTMatching::DumpMatching(
238 unsigned int iMatching,
239 sbn::crt::CRTPMTMatching const& matchInfo,
240 art::Ptr<recob::OpFlash> const& flash,
241 std::vector<art::Ptr<sbn::crt::CRTHit>> const& CRThits,
242 ProductTagIndex const& tagIndex,
243 std::string const& indent /* = "" */
244) const {
245
246 mf::LogVerbatim log(fLogCategory);
247
248 log << "[#" << iMatching << "] match:";
249 if (flash) log << " 1 flash";
250 if (!CRThits.empty()) {
251 if (flash) log << " and";
252 log << " " << CRThits.size() << " CRT hits";
253 }
254
255 log << " (" << (
256 icarus::crt::MatchTypeSelector.hasOption(matchInfo.flashClassification)
257 ? icarus::crt::MatchTypeSelector.get(matchInfo.flashClassification).name()
258 : "<unknown [#" + std::to_string(static_cast<int>(matchInfo.flashClassification)) + "]>"
259 ) << ")";
260
261 auto const hasCount
262 = [](unsigned int count){ return count != CRTPMTMatching::NoCount; };
263
264 if (hasCount(matchInfo.nTopCRTHitsBefore) || hasCount(matchInfo.nTopCRTHitsAfter))
265 {
266 log << " with top CRT hits";
267 if (hasCount(matchInfo.nTopCRTHitsBefore))
268 log << " " << matchInfo.nTopCRTHitsBefore << " before";
269 if (hasCount(matchInfo.nTopCRTHitsAfter)) {
270 if (hasCount(matchInfo.nTopCRTHitsBefore)) log << " and";
271 log << " " << matchInfo.nTopCRTHitsAfter << " after";
272 }
273 log << " the flash";
274 }
275 if (hasCount(matchInfo.nSideCRTHitsBefore) || hasCount(matchInfo.nSideCRTHitsAfter))
276 {
277 if (hasCount(matchInfo.nTopCRTHitsBefore) || hasCount(matchInfo.nTopCRTHitsAfter))
278 log << ",";
279 log << " with side CRT hits";
280 if (hasCount(matchInfo.nSideCRTHitsBefore))
281 log << " " << matchInfo.nSideCRTHitsBefore << " before";
282 if (hasCount(matchInfo.nSideCRTHitsAfter)) {
283 if (hasCount(matchInfo.nSideCRTHitsBefore)) log << " and";
284 log << " " << matchInfo.nSideCRTHitsAfter << " after";
285 }
286 log << " the flash";
287 }
288
289 // flash ID
290 log << "\n" << indent << " flash";
291 if (matchInfo.flashID == CRTPMTMatching::NoID) log << " <unknown ID>";
292 else log << " ID=" << matchInfo.flashID;
293
294 // flash times
295 if (matchInfo.flashTime != CRTPMTMatching::NoTime)
296 log << " at " << matchInfo.flashTime << " us";
297 if (matchInfo.flashGateTime != CRTPMTMatching::NoTime)
298 log << ", " << (matchInfo.flashGateTime*1000) << " ns after beam gate";
299
300 // optical hit times
301 if (matchInfo.firstOpHitStartTime != CRTPMTMatching::NoTime
302 || matchInfo.firstOpHitPeakTime != CRTPMTMatching::NoTime)
303 {
304 log << " (";
305 if (matchInfo.firstOpHitStartTime != CRTPMTMatching::NoTime)
306 log << "first hit start at " << matchInfo.firstOpHitStartTime;
307 if (matchInfo.firstOpHitPeakTime != CRTPMTMatching::NoTime) {
308 if (matchInfo.firstOpHitStartTime != CRTPMTMatching::NoTime)
309 log << ", ";
310 log << "peak at " << matchInfo.firstOpHitPeakTime << " us";
311 }
312 log << ")";
313 }
314
315 // gates and p.e.
316 log << "," << (matchInfo.flashInGate? "": " not") << " in beam gate,"
317 << (matchInfo.flashInBeam? "": " not") << " in spill";
318 if (matchInfo.flashPE >= 0.0)
319 log << ", with " << matchInfo.flashPE << " p.e.";
320
321 // position and width
322 log << ", at " << matchInfo.flashPosition << " cm";
323 if (matchInfo.flashYWidth != CRTPMTMatching::NoWidth
324 || matchInfo.flashZWidth != CRTPMTMatching::NoWidth)
325 {
326 log << " (width ";
327 if (matchInfo.flashZWidth != CRTPMTMatching::NoWidth)
328 log << "z: " << matchInfo.flashZWidth << " cm";
329 if (matchInfo.flashYWidth != CRTPMTMatching::NoWidth) {
330 if (matchInfo.flashZWidth != CRTPMTMatching::NoWidth)
331 log << ", ";
332 log << "y: " << matchInfo.flashYWidth << " cm";
333 }
334 log << ")";
335 }
336
337 // CRT hits
338 for (auto const& [ iCRT, matchedHit ]: util::enumerate(matchInfo.matchedCRTHits)) {
339 log << "\n" << indent << " CRT hit #" << iCRT;
340 if (matchedHit.time != MatchedCRT::NoTime)
341 log << " at " << matchedHit.time << " us";
342 if (matchedHit.PMTTimeDiff != MatchedCRT::NoTime)
343 log << " (" << (matchedHit.PMTTimeDiff * 1000) << " ns from flash)";
344 log << " at " << matchedHit.position << " cm";
345 if (matchedHit.sys != MatchedCRT::NoLocation
346 || matchedHit.region != MatchedCRT::NoLocation)
347 {
348 log << "; sys/region: " << matchedHit.sys << "/" << matchedHit.region;
349 }
350 } // for
351
352 if (flash) {
353 log << "\n" << indent << " associated flash: "
354 << tagIndex(flash).encode() << "#"
355 << flash.key() << " at " << flash->Time() << " us and (";
356 if (flash->hasXCenter()) log << flash->XCenter();
357 else log << "n/a";
358 log << ", " << flash->YCenter() << ", " << flash->ZCenter() << ") cm";
359 }
360 if (!CRThits.empty()) {
361 log << "\n" << indent << " associated CRT hits:";
362 for (art::Ptr<sbn::crt::CRTHit> const& hitPtr: CRThits) {
363 log
364 << " " << tagIndex(hitPtr).encode() << "#" << hitPtr.key()
365 << " at T0=" << hitPtr->ts0() << " ns, T1="
366 << hitPtr->ts1()/1000.0 << " us, (" << hitPtr->x_pos << ", "
367 << hitPtr->y_pos << ", " << hitPtr->z_pos << ") cm, " << hitPtr->peshit
368 << " p.e., on " << hitPtr->tagger << ";";
369 }
370 }

372} // icarus::crt::DumpCRTPMTMatching::DumpMatching()
