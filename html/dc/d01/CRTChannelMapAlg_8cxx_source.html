290 //----------------------------------------------------------------------------
291 geo::Point_t CRTChannelMapAlg::AuxDetChannelToPosition(
292 uint32_t const channel,
293 std::string const& auxDetName,
294 std::vector<geo::AuxDetGeo> const& auxDets) const {
295
296 // Figure out which detector we are in
297 size_t ad = UINT_MAX;
298 if (fNameToADGeo.count(auxDetName) > 0) {
299 ad = fNameToADGeo.find(auxDetName)->second;
300 }
301 else {
302 throw cet::exception("CRTChannelMapAlg")
303 << "No AuxDetGeo with name " << auxDetName;
304 }
305
306 // Get the vector of channel and sensitive volume pairs
307 auto csvItr = fADGeoToChannelAndSV.find(ad);
308
309 if (csvItr == fADGeoToChannelAndSV.end()) {
310 throw cet::exception("CRTChannelMapAlg")
311 << "No entry in channel and sensitive volume"
312 << " map for AuxDet index " << ad << " bail";
313 }
314
315 // Loop over the vector of channel and sensitive volumes to determine the
316 // sensitive volume for this channel. Then get the origin of the sensitive
317 // volume in the world coordinate system.
318 for (auto csv : csvItr->second) {
319 if (csv.first == channel) {
320 // Get the center of the sensitive volume for this channel
321 return auxDets[ad].SensitiveVolume(csv.second).GetCenter();
322 }
323 }
324
325 return {};
326 }
