846// -----------------------------------------------------------------------------
847template <typename Event>
848template <typename Handle>
849auto util::ArtHandleTrackerManager<Event>::registerHandle
850 (Handle&& handle) -> decltype(auto)
851{
852 using util::details::ProvenanceGetter;
853
854 using Handle_t = std::decay_t<Handle>;
855
856 if (!handle.isValid()) {
857 mf::LogDebug{ fConfig.logCategory }
858 << "Attempt to register an invalid handle.";
859 return std::forward<Handle>(handle);
860 }
861
862 canOperate("registerHandle()");
863
864 using Tracker_t = util::details::ArtHandleTracker<Event_t, Handle_t>;
865
866 // if it's already registered, we don't want to have it again
867 if (auto ptr = findHandle(handle)) {
868 auto const& registeredHandle = *ptr;
869 mf::LogDebug msg { fConfig.logCategory };
870 msg
871 << "Handle<" << details::productClassOf(registeredHandle)
872 << ">(" << details::inputTagOf(registeredHandle).encode()
873 << ") was already registered";
874 if (typeid(registeredHandle) != typeid(Handle_t)) {
875 msg << " as a different handle type (" << typeid(registeredHandle).name()
876 << ", now " << typeid(Handle_t).name() << ")";
877 }
878 msg << ".";
879 }
880 else {
881 mf::LogDebug{ fConfig.logCategory }
882 << "Registering handle<" << details::productClassOf(handle)
883 << ">(" << details::inputTagOf(handle).encode()
884 << ") (handle type: " << typeid(Handle_t).name() << ")."
885 ;
886
887 fTrackers.push_back(std::make_unique<Tracker_t>(*fEvent, handle));
888 }
889
890 return std::forward<Handle>(handle);
891
892} // util::ArtHandleTrackerManager<>::registerHandle()

948// -----------------------------------------------------------------------------
949template <typename Event>
950template <typename Handle>
951Handle const* util::ArtHandleTrackerManager<Event>::findHandle
952 (Handle const& like) const
953{
954 if (!like.isValid()) return nullptr;
955
956 // look for it, one by one
957 for (TrackerPtr const& tracker: fTrackers) {
958
959 std::any anyHandlePtr = tracker->handlePtr();
960 Handle const** handlePtr = std::any_cast<Handle const*>(&anyHandlePtr);
961 if (!handlePtr) continue; // not the right type
962
963 if ((*handlePtr)->provenance()->inputTag() != like.provenance()->inputTag())
964 continue; // different tag
965
966 return *handlePtr;
967 } // for
968
969 return nullptr; // failed
970
971} // util::ArtHandleTrackerManager<Event>::findTracker()
