50icarus::crt::CRTMatches icarus::crt::CRTHitmatched(
51 double flashTime, geo::Point_t const& flashpos,
52 std::vector<art::Ptr<sbn::crt::CRTHit>>& crtHits, double interval, bool isRealData, double globalT0Offset, bool MatchBottomCRT) {
53
54 std::vector<icarus::crt::CRTPMT> enteringCRTHits;
55 std::vector<icarus::crt::CRTPMT> exitingCRTHits;
56 MatchType flashType;
57 uint topen = 0, topex = 0, sideen = 0, sideex = 0, bottomen = 0, bottomex = 0;
58 for (auto const& crtHit : crtHits) {
59 if (!MatchBottomCRT && crtHit->plane > 49) continue; // For now, we are skipping bottom CRT Hits as they are not present in data.
60 // care with conversions: if either side of a subtraction is a `double`,
61 // the other side is also converted to `double` just before the subtraction,
62 // and in this conversion it may lose precision; subtraction itself must
63 // operate between 64-bit integers, then the conversion of the result may happen
64 double const CRTHitTime_ns = CRTHitTime(*crtHit, globalT0Offset, isRealData);
65 double tof = CRTHitTime_ns - flashTime * 1e3;
66 double distance =
67 (flashpos - geo::Point_t{crtHit->x_pos, crtHit->y_pos, crtHit->z_pos})
68 .R();
69 if (abs(tof) >= interval) continue;
70 if (tof < 0) {
71 if (MatchBottomCRT && crtHit->plane > 49)
72 bottomen++;
73 else if (crtHit->plane > 36)
74 sideen++;
75 else
76 topen++;
77 CRTPMT m_match = {tof, distance, crtHit};
78 enteringCRTHits.push_back(std::move(m_match));
79 } else if (tof >= 0) {
80 if (MatchBottomCRT && crtHit->plane > 49)
81 bottomex++;
82 else if (crtHit->plane > 36)
83 sideex++;
84 else
85 topex++;
86 CRTPMT m_match = {tof, distance, crtHit};
87 exitingCRTHits.push_back(std::move(m_match));
88 }
89 }
90 if (topen == 0 && sideen == 0 && topex == 0 && sideex == 0){
91 if(bottomex==0 && bottomen==0) flashType = MatchType::noMatch;
92 else if (bottomex>=1 && bottomen==0) flashType = MatchType::exBottom;
93 else if (bottomex==0 && bottomen>=1) flashType = MatchType::enBottom;
94 else flashType = MatchType::others;
95 }
96 else if (topen == 1 && sideen == 0 && topex == 0 && sideex == 0){
97 if(bottomex==1 && bottomen==0) flashType = MatchType::enTop_exBottom;
98 else flashType = MatchType::enTop;
99 }
100 else if (topen == 0 && sideen == 1 && topex == 0 && sideex == 0)
101 if(bottomex==1 && bottomen==0) flashType = MatchType::enSide_exBottom;
102 else flashType = MatchType::enSide;
103 else if (topen == 1 && sideen == 0 && topex == 0 && sideex == 1)
104 flashType = MatchType::enTop_exSide;
105 else if (topen == 0 && sideen == 0 && topex == 1 && sideex == 0){
106 if(bottomex==0 && bottomen==1) flashType = MatchType::exTop_enBottom;
107 else flashType = MatchType::exTop;
108 }
109 else if (topen == 0 && sideen == 0 && topex == 0 && sideex == 1)
110 if(bottomex==0 && bottomen==1) flashType = MatchType::exSide_enBottom;
111 else flashType = MatchType::exSide;
112 else if (topen >= 1 && sideen >= 1 && topex == 0 && sideex == 0) // could also add `if (MatchBottomCRT)` here
113 flashType = MatchType::enTop_mult;
114 else if (topen >= 1 && sideen >= 1 && topex == 0 && sideex >= 1) // and here
115 flashType = MatchType::enTop_exSide_mult;
116 else
117 flashType = MatchType::others;
118 return {std::move(enteringCRTHits), std::move(exitingCRTHits), flashType};
119}
