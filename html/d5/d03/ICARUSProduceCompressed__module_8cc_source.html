85 uint64_t getFragmentWord(artdaq::Fragment const& f, size_t nWord)
86 {
87 if (nWord > f.dataSize())
88 throw cet::exception("PhysCrateFragment:getFragmentWord")
89 << "Asked for 64-bit word " << nWord << " of the fragment, but there are only " << f.dataSize() << " in the payload" << '\n';
90
91 return *(f.dataBegin() + nWord);
92 }
93 uint16_t getA2795Word(artdaq::Fragment const& f, size_t nWord)
94 {
95 size_t nFragmentWord = nWord / 4;
96 size_t shift = 16*(nWord % 4);
97 uint64_t longWord = getFragmentWord(f, nFragmentWord);
98 uint16_t shortWord = (longWord >> shift) & 0xFFFF;
99 return shortWord;
100 }
101 void setFragmentWord(artdaq::Fragment& f, size_t nWord, uint64_t value)
102 {
103 *(f.dataBegin() + nWord) = value;
104 }
105 void setA2795Word(artdaq::Fragment& f, size_t nWord, uint16_t value)
106 {
107 uint64_t value64 = value;
108 size_t nFragmentWord = nWord / 4;
109 size_t shift = 16*(nWord % 4);
110 uint64_t oldWord = getFragmentWord(f, nFragmentWord);
111 uint64_t filterBlock = 0xFFFF;
112 uint64_t filter = ~(filterBlock << shift);
113 uint64_t newWord = (oldWord & filter) + ((value64 << shift) & ~filter);
114 setFragmentWord(f, nWord / 4, newWord);
115 }
116 size_t SampleBytesFromKey(std::array<bool, 16> const& key)
117 {
118 size_t nCompressed = 0;
119 for (auto const& bit : key)
120 nCompressed += bit;
121
122 return 128 - 6*nCompressed + 2*(nCompressed % 2);
123 }
124 uint16_t adc_val(artdaq::Fragment const& f, size_t b, size_t c, size_t s)
125 {
126 size_t nChannels = f.metadata<MetaData>()->channels_per_board();
127 size_t nSamples = f.metadata<MetaData>()->samples_per_channel();
128 uint16_t const* boardData = reinterpret_cast<uint16_t const*>(f.dataBeginBytes() + (b+1)*28 + (b+1)*8 + b*(nChannels*nSamples + 4)*sizeof(uint16_t));
129 return *(boardData + s*nChannels + c) & (~(1<<(f.metadata<MetaData>()->num_adc_bits()+1)));
130 }

164 artdaq::Fragment ICARUSProduceCompressed::compressArtdaqFragment(artdaq::Fragment const & f)
165 {
166 // get boards, channels, samples in Fragment
167 size_t nBoards = f.metadata<MetaData>()->num_boards();
168 size_t nChannels = f.metadata<MetaData>()->channels_per_board();
169 size_t nSamples = f.metadata<MetaData>()->samples_per_channel();
170
171 // initialize the output fragment
172 artdaq::Fragment compressed_fragment(f);
173
174 // go through board/channel/sample and load up the data
175 // data is stored in channel->sample->board
176 // keep track of where we are reading from/writing to
177 size_t compressedDataOffset = 0;
178 size_t uncompressedDataOffset = 0;
179 std::vector<std::array<bool, 16>> compressionKeys;
180 compressionKeys.resize(nBoards * nSamples);
181 size_t totalDataTileSize = 0;
182 for (size_t board = 0; board < nBoards; ++board)
183 {
184 // each board has a header...
185 uint32_t boardDataTileSize = 36;
186 for (size_t boardHeaderWord = 0; boardHeaderWord < 18; ++boardHeaderWord)
187 {
188 setA2795Word(compressed_fragment, compressedDataOffset, getA2795Word(f, uncompressedDataOffset));
189 ++compressedDataOffset;
190 ++uncompressedDataOffset;
191 } // done with board header
192
193 // since the 0th sample is used as a reference, store it as is
194 compressionKeys[board*nSamples] = {};
195 for (size_t channel = 0; channel < nChannels; ++channel)
196 {
197 uint16_t sampleADC = (adc_val(f, board, channel, 0) & 0x0FFF);
198 sampleADC += 0x8000;
199 setA2795Word(compressed_fragment, compressedDataOffset, sampleADC);
200 ++compressedDataOffset;
201 ++uncompressedDataOffset;
202 } // done with channel in sample 0
203 boardDataTileSize += SampleBytesFromKey({});
204
205 // from here on we store the difference between samples for each channel
206 for (size_t sample = 1; sample < nSamples; ++sample)
207 {
208 bool oddCompressions = false;
209 // work in blocks of 4 channels
210 for (size_t channelBlock = 0; channelBlock < nChannels / 4; ++channelBlock)
211 {
212 int16_t adcDiff_0 = adc_val(f, board, 4*channelBlock + 0, sample) - adc_val(f, board, 4*channelBlock + 0, sample - 1);
213 int16_t adcDiff_1 = adc_val(f, board, 4*channelBlock + 1, sample) - adc_val(f, board, 4*channelBlock + 1, sample - 1);
214 int16_t adcDiff_2 = adc_val(f, board, 4*channelBlock + 2, sample) - adc_val(f, board, 4*channelBlock + 2, sample - 1);
215 int16_t adcDiff_3 = adc_val(f, board, 4*channelBlock + 3, sample) - adc_val(f, board, 4*channelBlock + 3, sample - 1);
216
217 bool isComp = (std::abs(adcDiff_0) < 8) &&
218 (std::abs(adcDiff_1) < 8) &&
219 (std::abs(adcDiff_2) < 8) &&
220 (std::abs(adcDiff_3) < 8);
221
222 uncompressedDataOffset += 4;
223
224 if (isComp)
225 {
226 compressionKeys[board*nSamples + sample][channelBlock] = true;
227 oddCompressions = (not oddCompressions);
228 uint16_t new_word = (adcDiff_3 & 0x000F);
229 new_word <<= 4;
230 new_word += (adcDiff_2 & 0x000F);
231 new_word <<= 4;
232 new_word += (adcDiff_1 & 0x000F);
233 new_word <<= 4;
234 new_word += (adcDiff_0 & 0x000F);
235 setA2795Word(compressed_fragment, compressedDataOffset, new_word);
236 ++compressedDataOffset;
237 } else {
238 compressionKeys[board*nSamples + sample][channelBlock] = false;
239 setA2795Word(compressed_fragment, compressedDataOffset + 0, (adcDiff_0 & 0x0FFF) + 0x8000);
240 setA2795Word(compressed_fragment, compressedDataOffset + 1, (adcDiff_1 & 0x0FFF) + 0x8000);
241 setA2795Word(compressed_fragment, compressedDataOffset + 2, (adcDiff_2 & 0x0FFF) + 0x8000);
242 setA2795Word(compressed_fragment, compressedDataOffset + 3, (adcDiff_3 & 0x0FFF) + 0x8000);
243 compressedDataOffset += 4;
244 }
245 } // done with channels
246
247 // if there are an odd number of words in the difference then there needs to be a spacer added
248 if (oddCompressions)
249 {
250 setA2795Word(compressed_fragment, compressedDataOffset, 0);
251 ++compressedDataOffset;
252 }
253
254 // add up the size of the differences
255 boardDataTileSize += SampleBytesFromKey(compressionKeys[board*nSamples + sample]);
256 } // done with samples
257
258 // ...and each board has a trailer
259 boardDataTileSize += 4*sizeof(uint16_t);
260 for (size_t boardTrailerWord = 0; boardTrailerWord < 4; ++boardTrailerWord)
261 {
262 setA2795Word(compressed_fragment, compressedDataOffset, getA2795Word(f, uncompressedDataOffset));
263 ++compressedDataOffset;
264 ++uncompressedDataOffset;
265 }
266
267 // now that we know the size for the data tile, store that in the tile header
268 // endianness is weird for this, hence the htonl...
269 TileHeader* boardHeader = reinterpret_cast<TileHeader*>(compressed_fragment.dataBeginBytes() + totalDataTileSize);
270 boardHeader->packSize = htonl(boardDataTileSize);
271 totalDataTileSize += boardDataTileSize;
272 } // done with boards
273
274 // resize the fragment down to the compressed size
275 compressed_fragment.resizeBytes(totalDataTileSize);
276
277 // updated the metadata to reflect the compression
278 compressed_fragment.metadata<MetaData>()->SetCompressionScheme(1);
279
280 return compressed_fragment;
281 }

283 void ICARUSProduceCompressed::produceForLabel(art::Event& evt, art::InputTag fFragmentLabel)
284 {
285 // get the old fragments
286 art::Handle<std::vector<artdaq::Fragment>> fragHandle;
287 evt.getByLabel(fFragmentLabel, fragHandle);
288 auto const& old_fragments(*fragHandle);
289
290 // make a vector to put the new fragments in
291 std::unique_ptr<std::vector<artdaq::Fragment>> new_fragments(new std::vector<artdaq::Fragment>());
292
293 // fill the new fragments vector
294 for (auto const& old_fragment : old_fragments)
295 {
296 size_t nBoards = old_fragment.metadata<MetaData>()->num_boards();
297 size_t nChannels = old_fragment.metadata<MetaData>()->channels_per_board();
298 size_t nSamples = old_fragment.metadata<MetaData>()->samples_per_channel();
299
300 artdaq::Fragment new_fragment = compressArtdaqFragment(old_fragment);
301 new_fragments->emplace_back(new_fragment);
302
303 if (fDebug)
304 {
305 // store ADC vals
306 std::vector<uint16_t> adcValues(nBoards*nChannels*nSamples, std::numeric_limits<uint16_t>::min());
307
308 // loop to fill
309 uint64_t fragWord = 0;
310 for (size_t b = 0; b < nBoards; b++)
311 {
312 // skip board headers...
313 fragWord += ((28 + 8) / sizeof(uint16_t));
314 for (size_t s = 0; s < nSamples; s++)
315 {
316 size_t keyCount = 0;
317 for (size_t bit = 0; bit < nChannels/4; bit++)
318 {
319 uint16_t word = getA2795Word(new_fragment, fragWord);
320 bool isCompressed = (((word & 0xF000) != 0x8000));
321 for (size_t cInSet = 0; cInSet < 4; ++ cInSet)
322 {
323 size_t c = 4*bit + cInSet;
324 if (not isCompressed)
325 {
326 uint16_t tempWord = getA2795Word(new_fragment, fragWord + cInSet);
327 int16_t twelveBitDiff = (tempWord & 0x0FFF);
328 bool isNeg = (twelveBitDiff >> 11) && (s != 0);
329 uint16_t prevSample = (s != 0) ? adcValues[b*nSamples*nChannels + c*nSamples + s - 1] : 0;
330 uint16_t currSample = (isNeg*0xF000 + twelveBitDiff + prevSample);
331 adcValues.at(b*nSamples*nChannels + c*nSamples + s) = currSample;
332 } else {
333 int16_t fourBitDiff = (word >> (4*cInSet)) & 0x000F;
334 bool isNeg = (fourBitDiff >> 3);
335 uint16_t prevSample = (s != 0) ? adcValues[b*nSamples*nChannels + c*nSamples + s - 1] : 0;
336 uint16_t currSample = (isNeg*0xFFF0 + fourBitDiff + prevSample);
337 adcValues.at(b*nSamples*nChannels + c*nSamples + s) = currSample;
338 }
339 }
340 fragWord += (isCompressed) ? 1 : 4;
341 keyCount += isCompressed;
342 }
343 if ((keyCount % 2) == 1)
344 fragWord += 1;
345 }
346 // ...and skip board trailers
347 fragWord += 4;
348 }
349 for (size_t board = 0; board < nBoards; ++board)
350 for (size_t sample = 0; sample < nSamples; ++sample)
351 for (size_t channel = 0; channel < nChannels; ++channel)
352 {
353 uint16_t oldADC = adc_val(old_fragment, board, channel, sample);
354 uint16_t newADC = adcValues.at(board*nSamples*nChannels + channel*nSamples + sample);
355 if (oldADC != newADC)
356 MF_LOG_VERBATIM("ICARUSProduceCompressed")
357 << "ERROR - ADC Mismatch in board " << board << ", sample " << sample << ", channel " << channel << '\n'
358 << " old: " << oldADC << '\n'
359 << " new: " << newADC;
360 }
361 }
362 }
363
364 // put the new fragments into the event
365 evt.put(std::move(new_fragments), fFragmentLabel.instance());
366 }
