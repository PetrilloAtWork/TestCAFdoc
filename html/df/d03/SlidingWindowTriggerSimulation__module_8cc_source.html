583icarus::trigger::SlidingWindowTriggerSimulation::SlidingWindowTriggerSimulation
584 (Parameters const& config)
585 : art::EDProducer (config)
586 // configuration
587 , fPattern (config().Pattern())
588 , fBeamGateDuration (config().BeamGateDuration())
589 , fBeamGateStart (config().BeamGateStart())
590 , fBeamBits (config().BeamBits())
591 , fTriggerTimeResolution(config().TriggerTimeResolution())
592 , fEventTimeBinning (config().EventTimeBinning())
593 , fLogCategory (config().LogCategory())
594 // services
595 , fOutputDir (*art::ServiceHandle<art::TFileService>())
596 // internal and cached
597 , fWindowMapMan
598 { *lar::providerFrom<geo::Geometry>(), fLogCategory + "_WindowMapManager" }
599 , fPlots(
600 fOutputDir, "", "requirement: " + fPattern.description()
601 )
602 , fEventPlotInfo{
603 BinnedContent_t{ fEventTimeBinning } // eventTimes
604 , BinnedContent_t{ // HWtrigTimeVsBeam
605 fTriggerTimeResolution.convertInto
606 <detinfo::timescales::trigger_time::interval_t>().value()
607 }
608 , BinnedContent_t{ // triggerTimesVsHWtrig
609 fTriggerTimeResolution.convertInto
610 <detinfo::timescales::trigger_time::interval_t>().value()
611 }
612 , BinnedContent_t{ // triggerTimesVsBeam
613 fTriggerTimeResolution.convertInto
614 <detinfo::timescales::trigger_time::interval_t>().value()
615 }
616 }
617{
618
619 //
620 // more complex parameter parsing
621 //
622 std::string const& discrModuleLabel = config().TriggerGatesTag();
623 for (std::string const& threshold: config().Thresholds())
624 fADCthresholds[threshold] = art::InputTag{ discrModuleLabel, threshold };
625
626 // initialization of a vector of atomic is not as trivial as it sounds...
627 fTriggerCount = std::vector<std::atomic<unsigned int>>(fADCthresholds.size());
628 std::fill(fTriggerCount.begin(), fTriggerCount.end(), 0U);
629
630 //
631 // input data declaration
632 //
633 using icarus::trigger::OpticalTriggerGateData_t; // for convenience
634
635 // trigger primitives
636 for (art::InputTag const& inputDataTag: util::const_values(fADCthresholds)) {
637 consumes<std::vector<OpticalTriggerGateData_t>>(inputDataTag);
638 consumes<art::Assns<OpticalTriggerGateData_t, sbn::OpDetWaveformMeta>>
639 (inputDataTag);
640 } // for
641
642 //
643 // output data declaration
644 //
645 // keepThresholdName is true if we write instance name in output data products
646 bool const keepThresholdName
647 = config().KeepThresholdName().value_or(config().Thresholds().size() > 1);
648 if (!keepThresholdName && (config().Thresholds().size() > 1)) {
649 throw art::Exception(art::errors::Configuration)
650 << config().KeepThresholdName.name()
651 << " can be set to `true` only when a single threshold is specified ("
652 << config().Thresholds.name() << " has " << config().Thresholds().size()
653 << ")";
654 }
655
656 for (auto const& inputDataTag: util::const_values(fADCthresholds)) {
657 std::string const outputInstance
658 = keepThresholdName? inputDataTag.instance(): "";
659 produces<std::vector<raw::Trigger>>(outputInstance);
660 fOutputInstances.push_back(outputInstance);
661 }
662
663 {
664 mf::LogInfo log(fLogCategory);
665 log << "\nConfigured " << fADCthresholds.size() << " thresholds (ADC):";
666 for (auto const& [ thresholdTag, dataTag ]: fADCthresholds)
667 log << "\n * " << thresholdTag << " (from '" << dataTag.encode() << "')";
668
669 } // local block
670
671
672} // icarus::trigger::SlidingWindowTriggerSimulation::SlidingWindowTriggerSimulation()

683//------------------------------------------------------------------------------
684void icarus::trigger::SlidingWindowTriggerSimulation::produce(art::Event& event)
685{
686
687 detinfo::DetectorClocksData const clockData
688 = art::ServiceHandle<detinfo::DetectorClocksService const>()->DataFor(event);
689 detinfo::DetectorTimings const detTimings{clockData};
690 auto const beamGate = makeMyBeamGate(clockData);
691
692 if (auto oldGate = fGateChangeCheck(beamGate); oldGate) {
693 MF_LOG_DEBUG(fLogCategory)
694 << "Beam gate has changed from " << *oldGate << " to " << beamGate;
695 }
696
697
698 mf::LogDebug log(fLogCategory); // this will print at the end of produce()
699 log << "Event " << event.id() << ":";
700
701 for (auto const& [ iThr, thrTag ]
702 : util::enumerate(util::get_elements<0U>(fADCthresholds))
703 ) {
704
705 WindowTriggerInfo_t const triggerInfo
706 = produceForThreshold(event, detTimings, beamGate, iThr, thrTag);
707
708 log << "\n * threshold " << thrTag << ": ";
709 if (triggerInfo) log << "trigger at " << triggerInfo.info.atTick();
710 else log << "not triggered";
711
712 } // for
713
714 plotEvent(event, detTimings);
715
716 ++fTotalEvents;
717
718} // icarus::trigger::SlidingWindowTriggerSimulation::produce()

731//------------------------------------------------------------------------------
732void icarus::trigger::SlidingWindowTriggerSimulation::initializePlots() {
733
734 //
735 // overview plots with different settings
736 //
737
738 std::vector<std::string> thresholdLabels;
739 thresholdLabels.reserve(size(fADCthresholds));
740 for (std::string thr: util::get_elements<0U>(fADCthresholds))
741 thresholdLabels.push_back(std::move(thr));
742
743 auto const beamGate = makeMyBeamGate();
744 fGateChangeCheck(beamGate);
745 mf::LogInfo(fLogCategory)
746 << "Beam gate for plots: " << beamGate.asSimulationTime()
747 << " (simulation time), " << beamGate.tickRange()
748 << " (optical ticks)"
749 ;
750
751 //
752 // Triggering efficiency vs. ADC threshold.
753 //
754 auto* NTriggers = fPlots.make<TH1F>(
755 "NTriggers",
756 "Number of triggering events"
757 ";PMT discrimination threshold [ ADC counts ]"
758 ";events",
759 thresholdLabels.size(), 0.0, double(thresholdLabels.size())
760 );
761 util::ROOT::applyAxisLabels(NTriggers->GetXaxis(), thresholdLabels);
762
763 auto* Eff = fPlots.make<TEfficiency>(
764 "Eff",
765 "Triggering pass fraction"
766 ";PMT discrimination threshold [ ADC counts ]"
767 ";trigger pass fraction",
768 thresholdLabels.size(), 0.0, double(thresholdLabels.size())
769 );
770 // people are said to have earned hell for things like this;
771 // but TEfficiency really does not expose the interface to assign labels to
772 // its axes, which supposedly could be done had we chosen to create it by
773 // histograms instead of directly as recommended.
774 util::ROOT::applyAxisLabels
775 (const_cast<TH1*>(Eff->GetTotalHistogram())->GetXaxis(), thresholdLabels);
776
777 detinfo::timescales::optical_time_ticks const triggerResolutionTicks{
778 icarus::ns::util::makeDetTimings().toOpticalTicks(fTriggerTimeResolution)
779 };
780
781 auto const& beamGateTicks = beamGate.tickRange();
782 auto* TrigTime = fPlots.make<TH2F>(
783 "TriggerTick",
784 "Trigger time tick"
785 ";optical time tick [ /" + util::to_string(triggerResolutionTicks) + " ]"
786 ";PMT discrimination threshold [ ADC counts ]"
787 ";events",
788 static_cast<int>(std::ceil(beamGate.lengthTicks()/triggerResolutionTicks)),
789 beamGateTicks.start().value(),
790 icarus::ns::util::roundup
791 (beamGateTicks.start() + beamGate.lengthTicks(), triggerResolutionTicks)
792 .value(),
793 thresholdLabels.size(), 0.0, double(thresholdLabels.size())
794 );
795 util::ROOT::applyAxisLabels(TrigTime->GetYaxis(), thresholdLabels);
796
797
798 // we allow some fixed margin in the plot, just in case:
799 constexpr microseconds beamPlotPadding { 4_us };
800
801 using detinfo::timescales::trigger_time;
802
803 // hardware trigger may happen at any place within the beam gate,
804 // and in this plot range I want to include the full beam gate;
805 // so I take a beam gate worth of time before the trigger time,
806 // and as much after it; since this plot is relative to the hardware trigger,
807 // the hardware trigger time itself is 0
808 icarus::ns::util::BinningSpecs const HWtrigBinning = alignBinningTo(
809 icarus::ns::util::BinningSpecs{
810 (- beamGate.length() - beamPlotPadding).value(),
811 (beamGate.length() + beamPlotPadding).value(),
812 fTriggerTimeResolution.convertInto<trigger_time::interval_t>().value()
813 },
814 0.0
815 );
816 fPlots.make<TH2F>(
817 "TriggerTimeVsHWTrig",
818 "Time of the trigger"
819 ";trigger time (relative to hardware trigger) [ #mus ]"
820 ";PMT discrimination threshold [ ADC counts ]"
821 ";triggers",
822 HWtrigBinning.nBins(), HWtrigBinning.lower(), HWtrigBinning.upper(),
823 thresholdLabels.size(), 0.0, double(thresholdLabels.size())
824 );
825
826 icarus::ns::util::BinningSpecs const beamGateBinning = alignBinningTo(
827 icarus::ns::util::BinningSpecs{
828 (-beamPlotPadding).value(),
829 (beamGate.length() + beamPlotPadding).value(),
830 fTriggerTimeResolution.convertInto<trigger_time::interval_t>().value()
831 },
832 0.0
833 );
834 fPlots.make<TH2F>(
835 "TriggerTimeVsBeamGate",
836 "Time of the trigger"
837 ";trigger time (relative to beam gate opening) [ #mus ]"
838 ";PMT discrimination threshold [ ADC counts ]"
839 ";triggers",
840 beamGateBinning.nBins(), beamGateBinning.lower(), beamGateBinning.upper(),
841 thresholdLabels.size(), 0.0, double(thresholdLabels.size())
842 );
843
844 //
845 // per-threshold plots; should this initialization be set into its own method?
846 //
847 fThresholdPlots.resize(
848 size(fADCthresholds),
849 {
850 BinnedContent_t{ fEventTimeBinning }, // eventTimes
851 BinnedContent_t{ HWtrigBinning.binWidth() }, // HWtrigTimeVsBeam
852 BinnedContent_t{ HWtrigBinning.binWidth() }, // triggerTimesVsHWtrig
853 BinnedContent_t{ beamGateBinning.binWidth() } // triggerTimesVsBeam
854 }
855 );
856
857 for (std::string const& thr: util::get_elements<0U>(fADCthresholds)) {
858 PlotSandbox_t& plots
859 = fPlots.addSubSandbox("Thr" + thr, "Threshold: " + thr);
860
861 plots.make<TGraph>(
862 "TriggerTimeVsHWTrigVsBeam",
863 "Time of the trigger: emulated vs. hardware"
864 ";hardware trigger time (relative to beam gate opening) [ #mus ]"
865 ";emulated trigger time (relative to beam gate opening) [ #mus ]",
866 PlotSandbox_t::NoNameTitle
867 );
868
869 } // for thresholds
870
871} // icarus::trigger::SlidingWindowTriggerSimulation::initializePlots()
