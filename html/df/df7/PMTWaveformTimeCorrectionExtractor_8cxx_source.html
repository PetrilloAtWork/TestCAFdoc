137template<typename T>
138 size_t icarus::timing::PMTWaveformTimeCorrectionExtractor::getStartSample( std::vector<T> const& vv ){
139
140 // NOTE: when changing this algorithm, also update the documentation
141 // in the section "Signal timing extraction" of the class documentation
142
143 // We are thinking in inverted polarity
144 size_t minbin = getMinBin( vv, 0, vv.size() );
145
146 //Search only a cropped region of the waveform backward from the min
147 size_t maxbin = minbin-20; //getMaxBin( wave, minbin-20, minbin );
148
149 // Now we crawl betweem maxbin and minbin and we stop when:
150 //( maxbin value - minbin value )*0.05 > (maxbin value - bin value )
151 size_t startbin = maxbin;
152 auto delta = vv[maxbin]-vv[minbin];
153 for( size_t bin=maxbin; bin<minbin; bin++ ){
154 auto val = vv[maxbin]-vv[bin];
155 if( val >= 0.2*delta ){
156 startbin = bin - 1;
157 break;
158 }
159 }
160
161 if( startbin < maxbin ){
162 startbin=maxbin;
163 }
164
165 return startbin;
166}

172void icarus::timing::PMTWaveformTimeCorrectionExtractor::findWaveformTimeCorrections
173( raw::OpDetWaveform const & wave,
174 bool correctCableDelay,
175 std::vector<PMTWaveformTimeCorrection> & corrections ) const
176{
177 if (!fPMTTimingCorrectionsService && correctCableDelay) {
178 throw Error{ "Requested cable delay correction without providing a correction database!\n" };
179 }
180
181 unsigned int const waveChannelID = wave.ChannelNumber();
182 if ((waveChannelID & 0xF000) == 0)
183 throw NotASpecialChannel{ waveChannelID };
184
185 unsigned int crateSignalID = waveChannelID & 0x00F0;
186
187 auto const itCrateFragment = fCrateFragmentMap.find(crateSignalID);
188 if( itCrateFragment == fCrateFragmentMap.end() )
189 throw UnknownCrate{ waveChannelID };
190
191 // This will be the first sample of the falling edge of the special channel signal
192 // Which corresponds to the global trigger time.
193 int startSampleSignal = static_cast<int>( getStartSample( wave ) );
194
195 // allocates room for correction for `channel`; intermediate ones are defaulted
196 auto correctionFor
197 = [&corrections](unsigned int channel) -> PMTWaveformTimeCorrection&
198 {
199 if (channel >= corrections.size()) corrections.resize(channel + 1);
200 return corrections[channel];
201 };
202
203 // we now access the channels that we need
204 for( auto const & crateFragID : itCrateFragment->second ){
205
206 for( auto const & mapRow : fChannelMap.getPMTchannelInfo(crateFragID) ){
207
208 unsigned int channelID = mapRow.channelID;
209
210 double cableDelay = 0;
211
212 if( correctCableDelay ){
213 cableDelay = fPMTTimingCorrectionsService->getTriggerCableDelay(channelID);
214 }
215
216 // time in electronics scale when trigger signal arrived to readout;
217 // ideally, it would be fClocksData.TriggerTime()
218 double newStartTime = wave.TimeStamp()
219 + startSampleSignal * fClocksData.OpticalClock().TickPeriod()
220 + cableDelay; // << The correction is saved already with a minus sign
221
222 PMTWaveformTimeCorrection& correction = correctionFor(channelID);
223 if (correction.isValid()) {
224 throw MultipleCorrectionsForChannel
225 { channelID, correction.channelID, waveChannelID };
226 }
227
228 correction.channelID = waveChannelID;
229 correction.sample = startSampleSignal * fClocksData.OpticalClock().TickPeriod();
230 correction.startTime = fClocksData.TriggerTime() - newStartTime;
231
232 if(fVerbose){
233 std::cout << channelID << ", "
234 << wave.TimeStamp() << ", "
235 << startSampleSignal << ", "
236 << fClocksData.OpticalClock().TickPeriod() << ", "
237 << startSampleSignal*fClocksData.OpticalClock().TickPeriod() << ", "
238 << newStartTime << ", "
239 << cableDelay << ", "
240 << fClocksData.TriggerTime() - newStartTime << std::endl;
241 }
242 }
243 }
