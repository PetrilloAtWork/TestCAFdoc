313 void CRTDataAnalysis::FillFebMap() {
314 if(!this->fFebMap.empty())
315 return;
316 std::string fullFileName;
317 cet::search_path searchPath("FW_SEARCH_PATH");
318 searchPath.find_file("feb_map.txt",fullFileName);
319 std::ifstream fin;
320 fin.open(fullFileName,std::ios::in);
321 if(fin.good()) mf::LogError("CRTDataAnalysis") << "opened file 'feb_map.txt' for reading..." << std::endl;
322 else // mf::LogError("CRTDataAnalysis") << "could not open file 'feb_map.txt' for reading!" << std::endl;
323 throw cet::exception("CRTDataAnalysis::FillFebMap") << "Unable to find/open file 'feb_map.txt'" << std::endl;
324 std::vector<std::string> row;
325 std::string line, word;
326 while(getline(fin,line)) {
327 row.clear();
328 std::stringstream s(line);
329 int mod;
330 while (std::getline(s, word, ',')) {
331 row.push_back(word);
332 }
333 mod = std::stoi(row[0]);
334 (this->fFebMap)[mod].push_back(std::make_pair(std::stoi(row[1]),std::stoi(row[2])));
335 if(row.size()>3)
336 (this->fFebMap)[mod].push_back(std::make_pair(std::stoi(row[3]),std::stoi(row[4])));
337 }
338 // mf::LogError("CRTDataAnalysis") << "filled febMap with " << (this->fFebMap).size() << " entries" << std::endl;
339 fin.close();
340 }

445 void CRTDataAnalysis::analyze(const art::Event& event)
446 {
447 MF_LOG_DEBUG("CRTDataAnalysis") << "beginning analyis" << '\n';
448
449 // Start by fetching some basic event information for our n-tuple.
450 fEvent = event.id().event();
451 fRun = event.run();
452 fSubRun = event.subRun();
453 FillFebMap();//febMap);
454
455 //add trigger info
456 m_gate_type = value(sbn::triggerSource::Unknown);
457 if( !fTriggerLabel.empty() ) {
458
459 art::Handle<sbn::ExtraTriggerInfo> trigger_handle;
460 event.getByLabel( fTriggerLabel, trigger_handle );
461 if( trigger_handle.isValid() ) {
462 sbn::triggerSource bit = trigger_handle->sourceType;
463 m_gate_type = value(bit);
464 m_gate_name = bitName(bit);
465 m_trigger_timestamp = trigger_handle->triggerTimestamp;
466 m_gate_start_timestamp = trigger_handle->beamGateTimestamp;
467 m_trigger_gate_diff = trigger_handle->triggerTimestamp - trigger_handle->beamGateTimestamp;
468
469 }
470 else{
471 mf::LogError("CRTDataAnalysis") << "No raw::Trigger associated to label: " << fTriggerLabel.label() << "\n" ;
472 }
473 }
474 else {
475 mf::LogError("CRTDataAnalysis") << "Trigger Data product " << fTriggerLabel.label() << " not found!\n" ;
476 }
477
478 art::Handle<vector<icarus::crt::CRTData>> crtDAQHandle;
479 vector<art::Ptr<icarus::crt::CRTData> > crtList;
480 if ( event.getByLabel(fCRTDAQProducerLabel, crtDAQHandle))
481 art::fill_ptr_vector(crtList, crtDAQHandle);
482
483 vector<art::Ptr<icarus::crt::CRTData>> crtData;
484 bool presel = false;
485
486 for (size_t febdat_i=0; febdat_i<crtList.size(); febdat_i++) {
487
488 uint8_t mac = crtList[febdat_i]->fMac5;
489 int adid = fCrtutils->MacToAuxDetID(mac,0);
490 char type = fCrtutils->GetAuxDetType(adid);
491 /*
492 for(int chan=0; chan<32; chan++) {
493 mf::LogError("CRTDataAnalysis") << "\nfebP (mac5, channel, adc, type, adid) = (" << (int)crtList[febdat_i]->fMac5 << " , " << chan << " , "
494 << crtList[febdat_i]->fAdc[chan] << " , " << type << " , " << adid << ")\n";
495 }
496 */
499 // if (std::fabs(int64_t(crtList[febdat_i]->fTs0 - m_trigger_timestamp) + 1e9) > fCrtWindow) continue;
500 if ( type == 'm'){
501 for(int chan=0; chan<32; chan++) {
502 std::pair<double,double> const chg_cal = fChannelMap->getSideCRTCalibrationMap((int)crtList[febdat_i]->fMac5,chan);
503 float pe = (crtList[febdat_i]->fAdc[chan]-chg_cal.second)/chg_cal.first;
504 // In order to have Reset TS1 hits in CRTData from Side CRT, we have to explicitly include them
505 // The current threshold cut (6.5 PE) was applied to filter out noise, but this also filters out
506 // Reset events which are random trigger around the pedestal. These Reset hits are removed in
507 // CRT Hit reconstruction. Top CRT has in internal triggering logic and threshold that screens
508 // from the noise (hence presel = true for all the hits).
509 // Please revise this in the future if also T0 Reset hits need to be kept in CRTData.
510 // To do so, include !0crtList[febdat_i]->IsReference_TS0()
511 if(pe<=fPEThresh && !crtList[febdat_i]->IsReference_TS1()) continue;
512 presel = true;
513 }
514 }else if ( type == 'c' ) {
515
516 presel = true;
517
518 }else if ( type == 'd'){
519 for(int chan=0; chan<64; chan++) {
520 float pe = (crtList[febdat_i]->fAdc[chan]-fQPed)/fQSlope;
521 if(pe<=fPEThresh) continue;
522 presel = true;
523 }
524 }
525 if (presel) crtData.push_back(crtList[febdat_i]);
526 presel = false;
527 } // end of crtList
528
529
530 mf::LogError("CRTDataAnalysis") << "about to loop over " << crtData.size() <<" crtData entries \n";
531 for (size_t febdat_i=0; febdat_i<crtData.size(); febdat_i++) {
532
533
534 fDetEvent = fEvent;
535 fDetRun = fRun;
536 fDetSubRun = fSubRun;
537 fMac5 = crtData[febdat_i]->fMac5;
538 fEntry = crtData[febdat_i]->fEntry;
539 fFEBReg = fCrtutils->AuxDetRegionNameToNum(fCrtutils->MacToRegion(fMac5));
540 fNChan = 0;
541 fDetSubSys = fCrtutils->MacToTypeCode(fMac5);
542 fT0 = crtData[febdat_i]->fTs0;
543 fT1 = crtData[febdat_i]->fTs1;
544 fFlags = crtData[febdat_i]->fFlags;
545 int maxchan =0;
546 if(fDetSubSys!=2) maxchan=32;
547 else maxchan = 64;
548 fNMaxCh = maxchan;
549 for(int ch=0; ch<maxchan; ch++) {
550 fADC[ch] = crtData[febdat_i]->fAdc[ch];
551 std::pair<double,double> const chg_cal = fChannelMap->getSideCRTCalibrationMap((int)fMac5,ch);
552 if (fDetSubSys == 0 || fDetSubSys == 1){
553 float pe = (fADC[ch]-chg_cal.second)/chg_cal.first;
554 if (pe < 0) continue;
555 fPE[ch] = pe;
556 }else{
557 float pe = (fADC[ch]-fQPed)/fQSlope;
558 if (pe < 0) continue;
559 fPE[ch] = pe;
560 }
561
562 }
563
564 fDAQNtuple->Fill();
565
566 } //for CRT FEB events
567
568
569 // Fill CRT Hit Tree
570 art::Handle<std::vector<sbn::crt::CRTHit>> crtHitHandle;
571
572 bool isCRTHit = event.getByLabel(fCRTHitProducerLabel, crtHitHandle);
573 std::vector<int> ids;
574 fNHit = 0;
575 if (isCRTHit) {
