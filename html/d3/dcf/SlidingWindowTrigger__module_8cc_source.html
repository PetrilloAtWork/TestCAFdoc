414icarus::trigger::SlidingWindowTrigger::SlidingWindowTrigger
415 (Parameters const& config)
416 : art::EDProducer(config)
417 // configuration
418 , fWindowSize(config().WindowSize())
419 , fWindowStride(config().Stride().value_or(fWindowSize))
420 , fWindowChannels(defineWindows())
421 , fEnabledWindows(makeEnabledWindowIndices(
422 fWindowChannels.size(),
423 config().EnableOnlyWindows, config().DisableWindows
424 ))
425 , fProduceWaveformAssns(config().ProduceWaveformAssns())
426 , fIgnoreErrors(config().IgnoreErrors())
427 , fLogCategory(config().LogCategory())
428 // demand full PMT coverage only if no window was disabled:
429 , fCombiner(
430 filter(fWindowChannels, fEnabledWindows),
431 config().MissingChannels(),
432 nDisabledWindows() == 0U
433 )
434{
435 //
436 // more complex parameter parsing
437 //
438 std::string const discrModuleLabel = config().TriggerGatesTag();
439 for (std::string const& threshold: config().Thresholds())
440 fADCthresholds[threshold] = art::InputTag{ discrModuleLabel, threshold };
441
442 //
443 // configuration report (short)
444 //
445 {
446 mf::LogInfo log(fLogCategory);
447 log << "Trigger configuration: " << fWindowChannels.size() << " windows";
448 if (fEnabledWindows.size() < fWindowChannels.size())
449 log << " (only " << fEnabledWindows.size() << " enabled)";
450 log << ":";
451 for (auto const& [ iWindow, window ]: util::enumerate(fWindowChannels)) {
452 log << "\n #" << iWindow << ": "
453 << icarus::trigger::dumpTriggerWindowChannels(window);
454 if (std::find(fEnabledWindows.begin(), fEnabledWindows.end(), iWindow)
455 == fEnabledWindows.end())
456 {
457 log << " (disabled)";
458 }
459 } // for
460
461 log << "\nConfigured " << fADCthresholds.size() << " thresholds (ADC):";
462 for (auto const& [ threshold, dataTag ]: fADCthresholds)
463 log << "\n * " << threshold << " (from '" << dataTag.encode() << "')";
464 } // local block
465
466
467 using icarus::trigger::OpticalTriggerGateData_t; // for convenience
468
469 //
470 // input data declaration
471 //
472 for (art::InputTag const& inputDataTag: util::const_values(fADCthresholds)) {
473 consumes<std::vector<OpticalTriggerGateData_t>>(inputDataTag);
474 consumes<art::Assns<OpticalTriggerGateData_t, sbn::OpDetWaveformMeta>>
475 (inputDataTag);
476 } // for
477
478
479 //
480 // output data declaration
481 //
482 for (art::InputTag const& inputDataTag: util::const_values(fADCthresholds)) {
483 produces<std::vector<OpticalTriggerGateData_t>>(inputDataTag.instance());
484 produces<art::Assns<OpticalTriggerGateData_t, sbn::OpDetWaveformMeta>>
485 (inputDataTag.instance());
486 if (fProduceWaveformAssns) {
487 produces<art::Assns<OpticalTriggerGateData_t, raw::OpDetWaveform>>
488 (inputDataTag.instance());
489 }
490 } // for
491
492
493} // icarus::trigger::SlidingWindowTrigger::SlidingWindowTrigger()

539//------------------------------------------------------------------------------
540void icarus::trigger::SlidingWindowTrigger::produceThreshold(
541 art::Event& event,
542 icarus::trigger::OpDetWaveformMetaDataProductMap_t& waveformMap,
543 std::string const& threshold,
544 art::InputTag const& dataTag
545) const {
546
547 mf::LogDebug(fLogCategory)
548 << "Processing threshold " << threshold
549 << " from '" << dataTag.encode() << "'"
550 ;
551
552 using icarus::trigger::OpticalTriggerGateData_t; // for convenience
553 using TrackedTriggerGate_t
554 = icarus::trigger::TrackedOpticalTriggerGate<sbn::OpDetWaveformMeta>;
555
556 std::vector<TrackedTriggerGate_t> const& gates
557 = ReadTriggerGates(event, dataTag, waveformMap);
558
559 std::vector<TrackedTriggerGate_t> combinedGates = fCombiner.combine(gates);
560
561 // transform the data; after this line, `gates` is not usable any more
562 art::PtrMaker<OpticalTriggerGateData_t> const makeGatePtr
563 (event, dataTag.instance());
564 auto [ outputGates, outputAssns ]
565 = icarus::trigger::transformIntoOpticalTriggerGate
566 (std::move(combinedGates), makeGatePtr, waveformMap);
567
568 {
569 mf::LogTrace log { fLogCategory };
570 log
571 << "Threshold " << threshold << " ('" << dataTag.encode() << "'): "
572 << gates.size() << " input channels, "
573 << outputGates.size() << " output channels (+"
574 << outputAssns.size() << " associations to waveforms) into '"
575 << moduleDescription().moduleLabel() << ":" << dataTag.instance() << "'"
576 ;
577 for (auto const& [ iGate, gate ]: util::enumerate(outputGates)) {
578 log << "\n [#" << iGate << "] " << compactdump(gate);
579 } // for
580 } // local scope
581
582 if (fProduceWaveformAssns) {
583
584 // produce one gate-waveform association for each gate-metadata one;
585 // do it now while the gate/metadata association is still locally available
586 icarus::trigger::OpDetWaveformMetaMatcher waveformMetaMatcher{ event };
587 art::Assns<OpticalTriggerGateData_t, raw::OpDetWaveform> outputWaveAssns;
588 for (auto const &[ gatePtr, metaPtr ]: outputAssns)
589 outputWaveAssns.addSingle(gatePtr, waveformMetaMatcher(metaPtr));
590
591 event.put(
592 std::make_unique<art::Assns<OpticalTriggerGateData_t, raw::OpDetWaveform>>
593 (std::move(outputWaveAssns)),
594 dataTag.instance()
595 );
596 } // if fProduceWaveformAssns
597
598 event.put(
599 std::make_unique<std::vector<OpticalTriggerGateData_t>>
600 (std::move(outputGates)),
601 dataTag.instance()
602 );
603 event.put(
604 std::make_unique<art::Assns<OpticalTriggerGateData_t, sbn::OpDetWaveformMeta>>
605 (std::move(outputAssns)),
606 dataTag.instance()
607 );
608
609} // icarus::trigger::SlidingWindowTrigger::produceThreshold()

612//------------------------------------------------------------------------------
613auto icarus::trigger::SlidingWindowTrigger::makeEnabledWindowIndices(
614 std::size_t nWindows
615 , fhicl::OptionalSequence<std::size_t> const& enabled
616 , fhicl::OptionalSequence<std::size_t> const& disabled
617) -> std::vector<std::size_t> {
618
619 std::vector<std::size_t> selection; // indices of the enabled windows
620
621 // add all enabled windows (or all if no enabled list is specified)
622 if (!enabled(selection)) {
623 selection.resize(nWindows);
624 std::iota(selection.begin(), selection.end(), 0U);
625 }
626
627 // remove all disabled windows (if any)
628 std::vector<std::size_t> removeWindows;
629 disabled(removeWindows);
630 auto const find = [&selection](std::size_t index)
631 { return std::find(selection.begin(), selection.end(), index); };
632 for (std::size_t const index: removeWindows) {
633 if (auto const it = find(index); it != selection.end()) selection.erase(it);
634 } // for
635
636 return selection;
637} // icarus::trigger::SlidingWindowTrigger::makeEnabledWindowIndices()
