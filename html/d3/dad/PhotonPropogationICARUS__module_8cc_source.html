142void PhotonPropogationICARUS::produce(art::Event & event)
143{
144 ++fNumEvent;
145
146 /*
147 * In LArSoft, detected photons are organised optical channel by channel,
148 * each channel with its own sim::SimPhotons.
149 * Each channel contains individual information for each of the photons
150 * it has detected.
151 * Here we go through all detected photons from all the channels,
152 * and for each we assign a delay to the start time, parametrized by the
153 * distance between its generation point and the PMT.
154 *
155 */
156 // Agreed convention is to ALWAYS output to the event store so get a pointer to our collection
157 auto simPhotonVec = std::make_unique<std::vector<sim::SimPhotons>>();
158
159 // Read in the digit List object(s).
160 auto const& srcSimPhotons
161 = *(event.getValidHandle< std::vector<sim::SimPhotons> >(fSimPhotonModuleLabel));
162
163 if (srcSimPhotons.empty()) {
164 mf::LogWarning("PhotonPropagationICARUS") << "No photons! Nice event you have here.";
165 event.put(std::move(simPhotonVec));
166 return;
167 }
168
169 // get hold of all needed services
170// auto const& pvs = *(art::ServiceHandle<phot::PhotonVisibilityService>());
171 CLHEP::RandLandau landauGen(fPhotonEngine);
172
173 // Loop through the input photons (this might need to be more complicated...)
174 for(const auto& simPhoton : srcSimPhotons)
175 {
176 auto const channel = simPhoton.OpChannel();
177 if (simPhoton.empty()) continue;
178
179 auto const& PMTcenter = fGeometry->OpDetGeoFromOpChannel(channel).GetCenter();
180
181 // TODO restore the "MF_LOG_TRACE" line for normal operations
182 // (MF_LOG_TRACE will print only in debug mode, with debug qualifiers and proper messagefacility settings)
183 // MF_LOG_TRACE("LightPropagationICARUS")
184// mf::LogVerbatim("LightPropagationICARUS")
185// << "Processing photon channel #" << channel << ", detector center: " << PMTcenter;
186 //
187 // fix the arrival time
188 //
189 sim::SimPhotons localPhoton = simPhoton; // modify a copy of the original photon
190 unsigned int photonNo = 0;
191 for (auto& onePhoton: localPhoton)
192 {
193 //
194 // sanity check: if there are too many photons we are in trouble
195 // (like in: not enough computing resources)
196 //
197 if (photonNo++ >= MaxPhotons) {
198 mf::LogError("LightPropagationICARUS")
199 << "Too many photons to handle! only " << (photonNo - 1) << " saved.";
200 break;
201 }
202
203 geo::Point_t const position = geo::vect::toPoint(onePhoton.InitialPosition);
204 double const dis = (position - PMTcenter).R();
205
206 double mean = 0.18*dis; // TODO
207 double sigma = 0.75*dis; // TODO
208
209 //double time_plus = -1;
210
211 //while (time_plus<dis/21.74){time_plus=mean + landauGen.fire() * sigma; }// TODO
212
213 const double minPropTime = dis / 21.74; // d / (c/n) in [ns]
214 //const double maxPropTime = 2000 / 21.74; // dimension / (c/n) in [ns]
215 double time_plus;
216 do {
217 time_plus = mean + landauGen.fire() * sigma; // TODO
218 //time_plus = landauGen.fire(mean,sigma); // TODO
219 } while (time_plus < minPropTime);
220
221 // TODO restore the "MF_LOG_TRACE" line for normal operations
222 MF_LOG_TRACE("LightPropagationICARUS") << "";
223// mf::LogVerbatim("LightPropagationICARUS")
224// << "Photon #" << photonNo
225// << " (at " << position << ", " << dis << " cm far from PMT) given offset "
226// << time_plus;
227 //double time_plus = dis/21.74;
228 //TRandom r;
229
230 onePhoton.Time += time_plus;
231 //onePhoton.Time = time_plus;
232
233 } // for photons in simPhoton
234 // move the photon with the new arrival time into the new output collection
235 simPhotonVec->push_back(std::move(localPhoton));
236 } // for all simPhoton channels
237
238 // Add tracks and associations to event.
239 event.put(std::move(simPhotonVec));
240
241} // LightPropagationICARUS::produce()
