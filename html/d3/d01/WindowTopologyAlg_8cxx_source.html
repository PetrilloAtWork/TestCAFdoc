88//------------------------------------------------------------------------------
89auto icarus::trigger::WindowTopologyAlg::createFromGates
90 (TriggerGates_t const& gates) const -> WindowChannelMap
91{
92
93 // split gates by cryostat
94 TriggerGatesPerCryostat_t gatesByCryostat { fGeom->Ncryostats() };
95 for (InputTriggerGate_t const& gate: gates) {
96
97 geo::CryostatID cid; // invalid
98 assert(!cid.isValid);
99
100 for (raw::Channel_t const channel: gate.channels()) {
101 geo::OpDetGeo const& opDet = fGeom->OpDetGeoFromOpChannel(channel);
102 geo::OpDetID const oid = opDet.ID();
103 if (!cid) cid = oid;
104 else if (cid != oid) { // just in case
105 throw cet::exception("WindowTopologyAlg")
106 << "Input gate includes gates from different cryostats!!\n";
107 }
108 } // for channels in gate
109 if (!cid) continue; // gate with no channels does not contribute
110
111 gatesByCryostat.at(cid.Cryostat).push_back(gate); // (copy)
112 } // for gates
113
114 return createFromGates(gatesByCryostat);
115
116} // icarus::trigger::WindowTopologyAlg::createFromGates()

150//------------------------------------------------------------------------------
151auto icarus::trigger::WindowTopologyAlg::createWindowsFromCryostat(
152 WindowChannelColl_t const& windowChannels,
153 geo::CryostatGeo const& cryo,
154 geo::GeometryCore const& geom,
155 std::size_t firstWindowIndex /* = 0U */
156 ) -> std::vector<WindowChannelMap::WindowInfo_t>
157{
158 /*
159 * 1. fill the window information with local information
160 * 2. sort the windows in drift plane (first cryostat TPC as reference)
161 * 3. split the windows per plane
162 * 4. for each window plane:
163 * 4.1. sort windows on beam direction (TPC width direction)
164 * 4.2. fill the neighbour information on each window
165 */
166 std::vector<WindowChannelMap::WindowInfo_t> windows;
167 windows.reserve(windowChannels.size());
168
169 // use the first TPC of the cryostat as reference for directions
170 assert(cryo.NTPC() > 0U);
171 geo::TPCGeo const& refTPC = cryo.TPC(0U);
172
173 //
174 // 1. fill the window information with local information
175 //
176 using WindowInfoPtrs_t = std::vector<WindowChannelMap::WindowInfo_t*>;
177
178 WindowInfoPtrs_t cryoWindowInfo;
179 cryoWindowInfo.reserve(windowChannels.size());
180
181 std::size_t iWindow = firstWindowIndex;
182 for (auto const& channels: windowChannels) {
183
184 WindowChannelMap::WindowInfo_t wInfo;
185 WindowChannelMap::WindowComposition_t& wComp = wInfo.composition;
186
187 wInfo.topology.index = iWindow++;
188 wComp.channels = channels;
189 std::sort(wComp.channels.begin(), wComp.channels.end());
190 wComp.cryoid = channels.empty()
191 ? geo::CryostatID{}
192 : geom.OpDetGeoFromOpChannel(channels.front()).ID().asCryostatID()
193 ;
194
195 geo::vect::MiddlePointAccumulator middlePoint;
196 for (raw::Channel_t const channel: channels) {
197 // documentation of OpDetGeoFromOpChannel() does not say what on error...
198 geo::OpDetGeo const& opDet = geom.OpDetGeoFromOpChannel(channel);
199 middlePoint.add(opDet.GetCenter());
200 if (opDet.ID() != wComp.cryoid) wComp.cryoid = geo::CryostatID{};
201 } // for channel
202 wComp.center = middlePoint.middlePoint();
203
204 windows.push_back(std::move(wInfo));
205 cryoWindowInfo.push_back(&windows.back());
206
207 } // for windows
208
209 //
210 // 2. sort the windows in drift plane (first cryostat TPC as reference)
211 //
212 auto const normalProjection = [&refTPC](auto const* info)
213 { return refTPC.DistanceFromReferencePlane(info->composition.center); };
214 WindowInfoPtrs_t const windowsByNormal
215 = util::sortCollBy(cryoWindowInfo, normalProjection);
216
217 //
218 // 3. split the windows per plane
219 //
220 // split the list in two; there is a good deal of faith here
221 auto const beamCoordinate
222 = [&refPlane=refTPC.ReferencePlane()](auto const* info)
223 {
224 return refPlane.PointWidthDepthProjection(info->composition.center).X();
225 }
226 ;
227
228 //
229 // 4. for each plane:
230 //
231 // 4.1. sort windows on beam direction (TPC width direction)
232 //
233 auto const iMiddleWindow
234 = std::next(windowsByNormal.cbegin(), windowsByNormal.size() / 2U);
235 std::array<WindowInfoPtrs_t, 2U> const windowsByPlane = {
236 util::sortBy(windowsByNormal.cbegin(), iMiddleWindow, beamCoordinate),
237 util::sortBy(iMiddleWindow, windowsByNormal.cend(), beamCoordinate)
238 };
239
240 for (auto const& [ iPlane, planeWindows ]: util::enumerate(windowsByPlane)) {
241 //
242 // 4.2. fill the neighbour information on each window
243 //
244 auto const& otherPlaneWindows
245 = windowsByPlane.at(windowsByPlane.size() - 1U - iPlane);
246 std::size_t const iLastPlaneWindow = planeWindows.size() - 1U;
247 for (auto [ iPlaneWindow, windowInfo ]: util::enumerate(planeWindows)) {
248
249 // assumes all topology information is InvalidWindowIndex by default
250
251 if (WindowChannelMap::WindowInfo_t const* closestOppositeWindow
252 = findClosestWindow(otherPlaneWindows, windowInfo)
253 ) {
254 windowInfo->topology.opposite = closestOppositeWindow->topology.index;
255 }
256
257 if (iPlaneWindow > 0U) {
258 windowInfo->topology.upstream
259 = planeWindows[iPlaneWindow - 1U]->topology.index;
260 }
261
262 if (iPlaneWindow < iLastPlaneWindow) {
263 windowInfo->topology.downstream
264 = planeWindows[iPlaneWindow + 1U]->topology.index;
265 }
266
267 } // for window in plane
268 } // for planes
269
270 return windows;
271
272} // icarus::trigger::WindowTopologyAlg::createWindowsFromCryostat()

293//------------------------------------------------------------------------------
294auto icarus::trigger::WindowTopologyAlg::findClosestWindow(
295 std::vector<WindowChannelMap::WindowInfo_t*> const& windowList,
296 WindowChannelMap::WindowInfo_t const* targetWindow
297 ) -> WindowChannelMap::WindowInfo_t const*
298{
299
300 if (!targetWindow || windowList.empty()) return nullptr;
301
302 WindowChannelMap::WindowInfo_t const* closest = nullptr;
303 double minDistance2 = std::numeric_limits<double>::max();
304 for (auto const* window: windowList) {
305 if (!window) continue;
306
307 double const d2
308 = (window->composition.center - targetWindow->composition.center).mag2();
309 if (minDistance2 <= d2) continue;
310 minDistance2 = d2;
311 closest = window;
312 } // for
313
314 return closest;
315} // icarus::trigger::WindowTopologyAlg::findClosestWindow()

382//------------------------------------------------------------------------------
383std::string icarus::trigger::WindowTopologyVerification::verifyGate
384 (std::size_t iWindow, InputTriggerGate_t const& gate) const
385{
386 std::string errors; // if this stays `empty()` there is no error
387
388 WindowChannelMap::WindowInfo_t const& windowInfo = fWindowMap->info(iWindow);
389
390 auto const channelInWindow = [
391 begin=windowInfo.composition.channels.cbegin(),
392 end=windowInfo.composition.channels.cend()
393 ](raw::Channel_t channel)
394 { return std::binary_search(begin, end, channel); }
395 ;
396
397 for (raw::Channel_t const channel: gate.channels()) {
398 if (channelInWindow(channel)) continue;
399 if (errors.empty()) {
400 errors =
401 "channels not in window #" + std::to_string(windowInfo.topology.index)
402 + ":";
403 } // if first error
404 errors += " " + std::to_string(channel);
405 } // for all channels in gate
406
407 return errors;
408} // icarus::trigger::WindowTopologyVerification::verifyGate()
