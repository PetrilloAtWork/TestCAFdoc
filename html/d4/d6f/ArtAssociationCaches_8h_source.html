184template <typename Left, typename Right>
185void util::OneToOneAssociationCache<Left, Right>::mergeCache
186 (Cache_t const& other)
187{
188 // associations
189 fCache.reserve(fCache.size() + other.fCache.size());
190 auto const itNew = fCache.end();
191 fCache.insert(itNew, other.fCache.begin(), other.fCache.end());
192 std::inplace_merge(fCache.begin(), itNew, fCache.end());
193 // product IDs
194 std::vector<art::ProductID> newProducts;
195 std::set_union(
196 fProducts.cbegin(), fProducts.cend(),
197 other.fProducts.begin(), other.fProducts.end(),
198 back_inserter(newProducts)
199 );
200 fProducts = std::move(newProducts);
201} // util::OneToOneAssociationCache<>::mergeCache()

255template <typename Left, typename Right>
256void util::OneToOneAssociationCache<Left, Right>::updateProducts() {
257 fProducts.clear();
258 if (fCache.empty()) return;
259 if (fCache.front().first == fCache.back().first) // worth a special case
260 fProducts.push_back(fCache.front().first.id());
261 else {
262 auto it = fCache.cbegin();
263 auto const cend = fCache.cend();
264 fProducts.push_back(it->first.id());
265 while (++it != cend) {
266 if (it->first.id() != fProducts.back())
267 fProducts.push_back(it->first.id());
268 }
269 }
