561//------------------------------------------------------------------------------
562void icarus::PMTWaveformBaselinesFromChannelData::produce
563 (art::Event& event, art::ProcessingFrame const& frame)
564{
565
566 detinfo::DetectorTimings const detTimings{
567 frame.serviceHandle<detinfo::DetectorClocksService const>()->DataFor(event)
568 };
569
570 detinfo::timescales::electronics_time const triggerTime
571 = detTimings.TriggerTime();
572
573 mf::LogDebug{ fLogCategory }
574 << event.id() << " trigger time: " << triggerTime;
575
576 //
577 // fetch input
578 //
579 auto const& waveformHandle
580 = event.getValidHandle<std::vector<raw::OpDetWaveform>>(fOpDetWaveformTag);
581 auto const& waveforms = *waveformHandle;
582
583 //
584 // compute all the baselines (and plot them)
585 //
586 opdet::SharedWaveformBaseline const sharedWaveformBaselineAlgo
587 { fAlgoParams, fLogCategory };
588
589 double const eventTime = static_cast<double>(event.time().timeHigh())
590 + static_cast<double>(event.time().timeHigh()) * 1e-9;
591
592 std::vector<BaselineInfo_t> channelBaselines;
593 VectorExpander baselineForChannel { channelBaselines };
594
595 icarus::ns::util::GroupByIndex<raw::OpDetWaveform> waveformsByChannel
596 { waveforms, std::mem_fn(&raw::OpDetWaveform::ChannelNumber) };
597
598 for
599 (auto const& [ channel, allWaveforms ]: util::enumerate(waveformsByChannel))
600 {
601 if (allWaveforms.empty()) continue;
602
603 mf::LogTrace{ fLogCategory }
604 << "Processing " << allWaveforms.size() << " waveforms for channel "
605 << channel;
606
607 //
608 // remove global trigger waveform
609 //
610 std::vector<raw::OpDetWaveform const*> waveforms{ allWaveforms };
611 if (waveforms.size() >= fExcludeSpillTimeIfMoreThan) {
612
613 unsigned int const nExcluded
614 = removeWaveformsAround(waveforms, triggerTime.value());
615 if (nExcluded > 0U) {
616 mf::LogTrace{ fLogCategory }
617 << "Removed " << nExcluded << "/" << (waveforms.size() + nExcluded)
618 << " waveforms at trigger time " << triggerTime;
619 }
620
621 } // if many waveforms
622
623 //
624 // extract baseline
625 //
626 opdet::SharedWaveformBaseline::BaselineInfo_t const baseline
627 = sharedWaveformBaselineAlgo(waveforms);
628
629 mf::LogTrace{ fLogCategory }
630 << "Channel " << channel << ": baseline " << baseline.baseline
631 << " ADC# from " << baseline.nSamples << " samples in "
632 << baseline.nWaveforms << "/" << waveforms.size()
633 << " waveforms; found RMS=" << baseline.RMS << " ADC#";
634
635 auto const chIndex = static_cast<std::size_t>(channel);
636 baselineForChannel(chIndex) = { baseline.baseline, baseline.RMS };
637
638 if (fHBaselines) fHBaselines->Fill(double(channel), baseline.baseline);
639 if (chIndex < fBaselinesVsTime.size())
640 fBaselinesVsTime[chIndex].emplace_back(eventTime, baseline.baseline);
641
642 } // for grouped waveforms
643
644 //
645 // assign baselines to waveforms
646 //
647
648 std::vector<icarus::WaveformBaseline> baselines;
649 baselines.reserve(waveforms.size());
650 art::Assns<icarus::WaveformBaseline, raw::OpDetWaveform> baselineToWaveforms;
651 art::PtrMaker<icarus::WaveformBaseline> const makeBaselinePtr(event);
652
653 std::vector<icarus::WaveformRMS> RMSs;
654 RMSs.reserve(waveforms.size());
655 art::Assns<icarus::WaveformRMS, raw::OpDetWaveform> RMStoWaveforms;
656 art::PtrMaker<icarus::WaveformRMS> const makeRMSPtr(event);
657
658 for (auto const& [ iWaveform, waveform ]: util::enumerate(waveforms)) {
659
660 BaselineInfo_t baselineInfo
661 = channelBaselines[static_cast<std::size_t>(waveform.ChannelNumber())];
662
663 art::Ptr<raw::OpDetWaveform> const waveformPtr{ waveformHandle, iWaveform };
664
665 baselines.push_back(baselineInfo.baseline);
666 RMSs.push_back(baselineInfo.RMS);
667 baselineToWaveforms.addSingle(makeBaselinePtr(iWaveform), waveformPtr);
668 RMStoWaveforms.addSingle(makeRMSPtr(iWaveform), waveformPtr);
669
670 } // for
671
672 //
673 // output
674 //
675 event.put(
676 std::make_unique<std::vector<icarus::WaveformBaseline>>
677 (std::move(baselines))
678 );
679 event.put
680 (std::make_unique<std::vector<icarus::WaveformRMS>>(std::move(RMSs)));
681 event.put(
682 std::make_unique<art::Assns<icarus::WaveformBaseline, raw::OpDetWaveform>>
683 (std::move(baselineToWaveforms))
684 );
685 event.put(
686 std::make_unique<art::Assns<icarus::WaveformRMS, raw::OpDetWaveform>>
687 (std::move(RMStoWaveforms))
688 );
689
690
691} // icarus::PMTWaveformBaselinesFromChannelData::produce()

731unsigned int icarus::PMTWaveformBaselinesFromChannelData::getPretriggerBuffer
732 (sbn::PMTconfiguration const& PMTconfig) const
733{
734 lar::util::MinMaxCollector<unsigned int> prebuffer;
735 for (sbn::V1730Configuration const& board: PMTconfig.boards)
736 prebuffer.add(board.preTriggerTicks());
737 if (!prebuffer.has_data()) {
738 throw art::Exception{ art::errors::Unknown }
739 << "No boards found in the PMT configuration!\n";
740 }
741 else if (prebuffer.max() > prebuffer.min()) {
742 throw art::Exception{ art::errors::Unknown }
743 << "Found different pre-trigger readout buffer sizes between "
744 << prebuffer.min() << " and " << prebuffer.max() << " ticks.\n";
745 }
746 return prebuffer.min();

750//------------------------------------------------------------------------------
751unsigned int icarus::PMTWaveformBaselinesFromChannelData::removeWaveformsAround
752 (std::vector<raw::OpDetWaveform const*>& waveforms, double time) const
753{
754
755 // tick duration in the same unit as the electronics time scale times
756 // (that is microseconds, actually)
757 double const tickDuration
758 = detinfo::timescales::electronics_time::interval_t{ fOpticalTick }.value();
759
760 auto const hasTime = [tickDuration,time](raw::OpDetWaveform const* waveform)
761 {
762 double const relTime = time - waveform->TimeStamp();
763 return (relTime >= 0.0) && (relTime < (waveform->size() * tickDuration));
764 };
765
766 auto const iLast
767 = std::remove_if(waveforms.begin(), waveforms.end(), hasTime);
768
769 unsigned int const nRemoved = std::distance(iLast, waveforms.end());
770 waveforms.erase(iLast, waveforms.end());
771
772 return nRemoved;
773
774} // icarus::PMTWaveformBaselinesFromChannelData::removeWaveformsAround()

777//------------------------------------------------------------------------------
778void icarus::PMTWaveformBaselinesFromChannelData::buildBaselineGraphs
779 (art::ProcessingFrame const& frame)
780{
781
782 auto& tfs = *(frame.serviceHandle<art::TFileService>());
783
784 art::TFileDirectory graphDir = tfs.mkdir("graphs", "Baseline vs. time");
785 art::TFileDirectory profileDir
786 = tfs.mkdir("profiles", "Baseline profiles vs. time");
787
788 for (auto const channel: util::counter(fBaselinesVsTime.size())) {
789
790 auto& timeAndBaselines = fBaselinesVsTime[channel];
791 if (timeAndBaselines.empty()) continue;
792
793 // sort by time (entries with the same time would be sorted by baseline,
794 // but that does not really happen nor it mattered if it happened)
795 std::sort(timeAndBaselines.begin(), timeAndBaselines.end());
796
797 // graph, one point per event
798 auto* const graph = graphDir.makeAndRegister<TGraph>(
799 "BaselineCh" + std::to_string(channel),
800 "PMT channel #" + std::to_string(channel) + ": baseline vs. run time",
801 timeAndBaselines.size()
802 );
803 assert(graph->GetXaxis());
804 assert(graph->GetYaxis());
805 graph->GetXaxis()->SetTitle("event time");
806 graph->GetYaxis()->SetTitle("baseline [ ADC ]");
807
808
809 // profile, one point every 10 minutes (or best offer)
810 double const startTime = timeAndBaselines.front().first;
811 double const totalTime = timeAndBaselines.back().first - startTime;
812 auto const nBins = std::max(1U,
813 static_cast<unsigned int>(std::ceil(totalTime / fBaselineTimeAverage))
814 );
815 double const endTime = startTime + nBins * fBaselineTimeAverage;
816
817 auto* const profile = profileDir.make<TProfile>(
818 ("BaselineCh" + std::to_string(channel) + "profile").c_str(),
819 ("PMT channel #" + std::to_string(channel) + ": baseline vs. run time")
820 .c_str(),
821 nBins, startTime, endTime
822 );
823 assert(profile->GetXaxis());
824 assert(profile->GetYaxis());
825 profile->GetXaxis()->SetTitle(
826 ("event time [ / " + std::to_string(fBaselineTimeAverage) + " s ]")
827 .c_str()
828 );
829 profile->GetYaxis()->SetTitle("average baseline [ ADC ]");
830
831 for (auto const& [ i, data ]: util::enumerate(timeAndBaselines)) {
832 auto const [ time, baseline ] = data;
833 graph->SetPoint(i, time, baseline);
834 profile->Fill(time, baseline);
835 } // for
836
837 } // for
838
839} // icarus::PMTWaveformBaselinesFromChannelData::buildBaselineGraphs()
