154template <typename TPCCont>
155geo::Vector_t icarus::trigger::PMTverticalSlicingAlg::determineDriftDir
156 (TPCCont const& TPCcont)
157{
158
159 auto iTPC = util::begin(TPCcont);
160 auto const tend = util::end(TPCcont);
161
162 if (iTPC == tend) {
163 throw cet::exception("PMTverticalSlicingAlg")
164 << "determineDriftDir(): no TPC in the specified object!";
165 }
166
167 auto const dir = iTPC->DriftDir();
168 while(++iTPC != tend) {
169 if (areParallel(iTPC->DriftDir(), dir))
170 continue;
171 throw cet::exception("PMTverticalSlicingAlg")
172 << "determineDriftDir(): TPC " << iTPC->ID()
173 << " has a drift direction incompatible with " << dir << ".\n";
174 } // while
175
176 return dir;
177
178} // icarus::trigger::PMTverticalSlicingAlg::determineDriftDir()
