223void SaveConfigurationIntoTFile::storeString
224 (std::string const& key, std::string const& value)
225{
226 /*
227 * The following `TFileService::makeAndRegister()` call does two things:
228 * 1) constructs a `TNamed` object with the _default constructor_
229 * 2) assigns it the specified name and title
230 * 3) adds the new object to the ROOT directory we are given.
231 *
232 * The first two operations might be combined by calling the proper
233 * constructor, but the generic `makeAndRegister()` will always explicitly
234 * perform action (2) anyway, while `make()` will not perform action (3)
235 * at all, which appears to be required for successfully writing `TNamed`.
236 *
237 */
238 auto const& fs = *(art::ServiceHandle<art::TFileService>());
239 fs.makeAndRegister<TNamed>(key.c_str(), value.c_str());
240
241} // SaveConfigurationIntoTFile::storeString()

244//------------------------------------------------------------------------------
245fhicl::ParameterSet SaveConfigurationIntoTFile::currentProcessConfiguration
246 (std::string const& procName)
247{
248
249 /*
250 * The strategy is to parse the global FHiCL registry (yes, there is such a
251 * thing), looking for the table which contains 'process_name'.
252 * This heuristic is not fool proof, since there might be other tables with
253 * the same atom. So:
254 * TODO find a way to ask art which is the root parameter set
255 * The registry contains all root-level tables, including the main
256 * configuration and other tables that are used to resolve references
257 * (internally, references are not resolved to save space).
258 * It turns out that when a fhicl::ParameterSet is created, all references
259 * are resolved, so by the time we get the parameter sets we don't have to
260 * worry to manually resolve the references. Which would not be fun.
261 *
262 */
263
264 static std::string const rootKey { "process_name" };
265
266 fhicl::ParameterSet const* rootConfig = nullptr;
267 for (auto const& idAndSet: fhicl::ParameterSetRegistry::get()) {
268 auto const& pset = idAndSet.second;
269 if (isProcessConfiguration(pset, procName)) {
270 if (rootConfig) {
271 throw art::Exception(art::errors::LogicError)
272 << "Found two candidate process parameter sets: with " << rootKey
273 << " '" << rootConfig->get<std::string>(rootKey) << "' and '"
274 << pset.get<std::string>(rootKey) << "'!\n";
275 }
276 rootConfig = &pset;
277 } // if found
278 }
279 if (!rootConfig) {
280 throw art::Exception(art::errors::LogicError)
281 << "No parameter set with 'process_name' atom found!\n";
282 }
283
284 return *rootConfig;
285} // SaveConfigurationIntoTFile::currentProcessConfiguration()

288//------------------------------------------------------------------------------
289bool SaveConfigurationIntoTFile::isProcessConfiguration
290 (fhicl::ParameterSet const& pset)
291{
292 /*
293 * We are forced to parse the parameter set heuristically.
294 * If a configuration is something like:
295 *
296 * full_configuration: { ... }
297 *
298 * @table::full_configuration
299 *
300 * the heuristic will match both the actual root configuration *and*
301 * `full_configuration` parameter set (hint: put `full_configuration` within a
302 * prolog). This may cause trouble downstream.
303 * Barred that case, the more detailed the logic here is, the less likely
304 * a false positive is to happen.
305 *
306 */
307 if (!::has_table (pset, "services" )) return false;
308 if (!::has_table (pset, "physics" )) return false;
309 if (!::has_table (pset, "source" )) return false;
310 if (!::has_atom (pset, "process_name")) return false;
311 if (!::has_sequence(pset, "physics.trigger_paths")) return false;
312 if (!::has_sequence(pset, "physics.end_paths")) return false;
313
314 return true;
315} // SaveConfigurationIntoTFile::isProcessConfiguration()
