152void FakeFlash::GenPosition(double& x, double& y, double& z) {
153
154 size_t tpc_id = (size_t)(fFlatRandom->fire(0,_tpc_v.size()));
155 bool found = false;
156 // Implementation of required member function here.
157 auto geop = lar::providerFrom<geo::Geometry>();
158 for(auto const& cryostat : geop->Iterate<geo::CryostatGeo>()) {
159 if(!cryostat.HasTPC(tpc_id)) continue;
160 auto const& tpc = cryostat.TPC(tpc_id);
161 auto const& tpcabox = tpc.ActiveBoundingBox();
162 double xmin = tpcabox.MinX() + (tpcabox.MaxX() - tpcabox.MinX()) * _xmin;
163 double xmax = tpcabox.MinX() + (tpcabox.MaxX() - tpcabox.MinX()) * _xmax;
164 double ymin = tpcabox.MinY() + (tpcabox.MaxY() - tpcabox.MinY()) * _ymin;
165 double ymax = tpcabox.MinY() + (tpcabox.MaxY() - tpcabox.MinY()) * _ymax;
166 double zmin = tpcabox.MinZ() + (tpcabox.MaxZ() - tpcabox.MinZ()) * _zmin;
167 double zmax = tpcabox.MinZ() + (tpcabox.MaxZ() - tpcabox.MinZ()) * _zmax;
168 x = fFlatRandom->fire(xmin,xmax);

177void FakeFlash::FillSimPhotons(std::vector<sim::SimPhotons>& simph_v,
178 int mother_trackid,
179 size_t nphotons,
180 const TLorentzVector& pos)
181{
182 art::ServiceHandle<phot::PhotonVisibilityService const> pvs;
183 double xyz[3];
184 xyz[0] = pos.X();
185 xyz[1] = pos.Y();
186 xyz[2] = pos.Z();
187 auto const& Visibilities = pvs->GetAllVisibilities(xyz);
188 // Now loop over optical channels and fill SimPhotons array
189 if(simph_v.empty()) {
190 for(size_t opch=_ch_min; opch<=_ch_max; ++opch)
191 simph_v.push_back(sim::SimPhotons(opch));
192 }
193 for(size_t opch=_ch_min; opch <= _ch_max; ++opch) {
194 // get visibility
195 size_t detected = fPoisRandom->fire((double)(nphotons) * Visibilities[opch]);
196 //std::cout<<opch<<","<<detected<<std::endl;
197 auto time_array = this->GenerateTime(detected);
198 assert(time_array.size() == detected);
199 // record
200 auto& simph = simph_v[opch];
201 for(size_t idx=0; idx<detected; ++idx) {
202 sim::OnePhoton phot;
203 phot.Time = pos.T() + time_array[idx];
204 phot.InitialPosition.SetX(pos.X());
205 phot.InitialPosition.SetY(pos.Y());
206 phot.InitialPosition.SetZ(pos.Z());
207 phot.MotherTrackID = mother_trackid;
208 simph.emplace_back(phot);
209 }
210 }

213std::vector<double> FakeFlash::GenerateTime(size_t numphotons) {
214
215 std::vector<double> res(numphotons);
216 for(int i=0; i<((int)numphotons); ++i) {
217 if(fFlatRandom->fire(0.,1.) < _fast_frac)
218 res[i] = fExpoRandom->fire(_fast_tau);
219 else res[i] = fExpoRandom->fire(_slow_tau);
220 }

224void FakeFlash::produce(art::Event& e)
225{
226
227 auto simph_v = std::unique_ptr<std::vector<sim::SimPhotons> >(new std::vector<sim::SimPhotons>());
228 auto mct_v = std::unique_ptr<std::vector<simb::MCTruth> >(new std::vector<simb::MCTruth>());
229 std::vector<simb::MCParticle> mcp_v;
230 double clock = 0.;
231 while(clock <= _duration) {
232 // Determine photon count
233 int nphotons = _min_photons + (_max_photons - _min_photons) * (fFlatRandom->fire(0,1));
234
235 // Generate position 4 vector
236 double time = _tstart + clock * 1.e3;
237 double x,y,z;
238 this->GenPosition(x,y,z);
239 TLorentzVector pos(x,y,z,time);
240
241 // Fill SimPhotons
242 this->FillSimPhotons(*simph_v, (int)(mcp_v.size()), nphotons, pos);
243
244 // Record a fake photon particle
245 simb::MCParticle part(mcp_v.size(), 22, "primary", 0, 0., 1);
246 TLorentzVector mom(0.,0.,0.,(double)nphotons);
247 part.AddTrajectoryPoint(pos,mom);
248 mcp_v.emplace_back(std::move(part));
249
250 // count time
251 clock += (1./_frequency);
252 }
253
254 simb::MCTruth mct;
255 for(auto& part : mcp_v)
256 mct.Add(part);
257 mct_v->emplace_back(std::move(mct));
258
259 e.put(std::move(simph_v));
260 e.put(std::move(mct_v));
261}
