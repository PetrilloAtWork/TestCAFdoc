140void BasicTrackAnalysis::initializeHists(art::ServiceHandle<art::TFileService>& tfs, const std::string& dirName)
141{
142 // Make a directory for these histograms
143 art::TFileDirectory dir = tfs->mkdir(dirName.c_str());
144
145 auto const n_wires_0 = fGeometry->Nwires(geo::PlaneID{0, 0, 0});
146 auto const n_wires_1 = fGeometry->Nwires(geo::PlaneID{0, 0, 1});
147 auto const n_wires_2 = fGeometry->Nwires(geo::PlaneID{0, 0, 2});
148 fHitsByWire[0] = dir.make<TH1D>("HitsByWire0", ";Wire #", n_wires_0, 0., n_wires_0);
149 fHitsByWire[1] = dir.make<TH1D>("HitsByWire1", ";Wire #", n_wires_1, 0., n_wires_1);
150 fHitsByWire[2] = dir.make<TH1D>("HitsByWire2", ";Wire #", n_wires_2, 0., n_wires_2);
151
152 fPulseHVsWidth[0] = dir.make<TH2D>("PHVsWidth0", ";PH;Width", 100, 0., 100., 100, 0., 20.);
153 fPulseHVsWidth[1] = dir.make<TH2D>("PHVsWidth1", ";PH;Width", 100, 0., 100., 100, 0., 20.);
154 fPulseHVsWidth[2] = dir.make<TH2D>("PHVsWidth2", ";PH;Width", 100, 0., 100., 100, 0., 20.);
155
156 fPulseHVsHitNo[0] = dir.make<TProfile>("PHVsNo0", ";Hit #;PH", 1000, 0., 1000., 0., 100.);
157 fPulseHVsHitNo[1] = dir.make<TProfile>("PHVsNo1", ";Hit #;PH", 1000, 0., 1000., 0., 100.);
158 fPulseHVsHitNo[2] = dir.make<TProfile>("PHVsNo2", ";Hit #;PH", 1000, 0., 1000., 0., 100.);
159
160 return;
161}

163void BasicTrackAnalysis::fillHistograms(const art::Event& event) const
164{
165 // The game plan for this module is to look at recob::Tracks and objects associated to tracks
166 // To do this we need a valid track collection for those we are hoping to look at
167 for(const auto& trackLabel : fTrackProducerLabelVec)
168 {
169 art::Handle<std::vector<recob::Track> > trackHandle;
170 event.getByLabel(trackLabel, trackHandle);
171
172 if (trackHandle.isValid())
173 {
174 // Recover the collection of associations between tracks and hits
175 art::FindManyP<recob::Hit> trackHitAssns(trackHandle, event, trackLabel);
176
177 for(size_t trackIdx = 0; trackIdx < trackHandle->size(); trackIdx++)
178 {
179 art::Ptr<recob::Track> track(trackHandle,trackIdx);
180
181 }
182 }
183 }
184
185 return;
186}
