488//------------------------------------------------------------------------------
489void icarus::trigger::LVDSgates::produceThreshold(
490 art::Event& event,
491 icarus::trigger::OpDetWaveformMetaDataProductMap_t& waveformMap,
492 std::string const& thresholdStr,
493 SourceInfo_t const& srcInfo
494) const {
495
496 auto const& [ dataTag, outputInstanceName ] = srcInfo;
497
498 mf::LogDebug(fLogCategory)
499 << "Processing threshold " << thresholdStr
500 << " from '" << dataTag.encode() << "'";
501
502 using icarus::trigger::OpticalTriggerGateData_t; // for convenience
503
504 std::vector<TrackedTriggerGate_t> const& gates
505 = ReadTriggerGates(event, dataTag, waveformMap);
506
507 checkInput(gates);
508
509 std::vector<TrackedTriggerGate_t> combinedGates;
510
511 std::vector<std::vector<raw::Channel_t>> const cleanedChannels
512 = removeChannels(fChannelPairing, fIgnoreChannels);
513
514 for (std::vector<raw::Channel_t> const& pairing: cleanedChannels) {
515 if (pairing.empty()) continue; // ???
516
517 combinedGates.push_back(combineChannels(gates, pairing, fComboMode));
518
519 } // for
520
521 // transform the data; after this line, `gates` is not usable any more
522 art::PtrMaker<OpticalTriggerGateData_t> const makeGatePtr
523 (event, outputInstanceName);
524 auto [ outputGates, outputAssns ]
525 = icarus::trigger::transformIntoOpticalTriggerGate
526 (std::move(combinedGates), makeGatePtr, waveformMap);
527
528 mf::LogTrace(fLogCategory)
529 << "Threshold " << thresholdStr << " ('" << dataTag.encode() << "'): "
530 << gates.size() << " input channels, "
531 << outputGates.size() << " output channels (+"
532 << outputAssns.size() << " associations to waveforms) into '"
533 << moduleDescription().moduleLabel() << ":" << outputInstanceName << "'"
534 ;
535
536
537 if (fProduceWaveformAssns) {
538
539 // produce one gate-waveform association for each gate-metadata one;
540 // do it now while the gate/metadata association is still locally available
541 icarus::trigger::OpDetWaveformMetaMatcher waveformMetaMatcher{ event };
542 art::Assns<OpticalTriggerGateData_t, raw::OpDetWaveform> outputWaveAssns;
543 for (auto const & [ gatePtr, metaPtr ]: outputAssns)
544 outputWaveAssns.addSingle(gatePtr, waveformMetaMatcher(metaPtr));
545
546 event.put(
547 std::make_unique<art::Assns<OpticalTriggerGateData_t, raw::OpDetWaveform>>
548 (std::move(outputWaveAssns)),
549 outputInstanceName
550 );
551
552 } // if fProduceWaveformAssns
553
554 event.put(
555 std::make_unique<std::vector<OpticalTriggerGateData_t>>
556 (std::move(outputGates)),
557 outputInstanceName
558 );
559 event.put(
560 std::make_unique<art::Assns<OpticalTriggerGateData_t, sbn::OpDetWaveformMeta>>
561 (std::move(outputAssns)),
562 outputInstanceName
563 );
564
565} // icarus::trigger::LVDSgates::produceThreshold()

568//------------------------------------------------------------------------------
569auto icarus::trigger::LVDSgates::combineChannels(
570 std::vector<TrackedTriggerGate_t> const& gates,
571 std::vector<raw::Channel_t> const& pairing,
572 ComboMode comboMode
573) const -> TrackedTriggerGate_t {
574
575 namespace GateOps = icarus::trigger::GateOps;
576
577 switch (comboMode) {
578 case ComboMode::disable:
579 return discardChannels(gates, pairing);
580 case ComboMode::AND:
581 return binaryCombineChannel(gates, pairing, GateOps::Min);
582 case ComboMode::OR:
583 return binaryCombineChannel(gates, pairing, GateOps::Max);
584 case ComboMode::Input1:
585 return selectChannel(gates, pairing, 0U);
586 case ComboMode::Input2:
587 return selectChannel(gates, pairing, 1U);
588 default:
589 throw art::Exception(art::errors::LogicError)
590 << "Unexpected combination mode (#" << static_cast<int>(comboMode)
591 << ").\n";
592 } // switch(comboMode)
593} // icarus::trigger::LVDSgates::combineChannels()

694//------------------------------------------------------------------------------
695unsigned int icarus::trigger::LVDSgates::checkPairings() const {
696
697 //
698 // collect all the errors first
699 //
700 auto const& geom = *(lar::providerFrom<geo::Geometry>());
701
702
703 // collect configured and duplicate channel numbers
704 unsigned int nEmptyGroups = 0U;
705 std::set<raw::Channel_t> configuredChannels, duplicateChannels;
706 for (auto const& pairing: fChannelPairing) {
707 if (pairing.empty()) {
708 ++nEmptyGroups;
709 continue;
710 }
711 for (raw::Channel_t channel: pairing) {
712 if (!configuredChannels.insert(channel).second)
713 duplicateChannels.insert(channel);
714 } // for channel
715 } // for pairing
716
717 // collect invalid channel numbers
718 std::vector<raw::Channel_t> invalidChannels;
719 for (raw::Channel_t channel: configuredChannels) {
720 if (!geom.IsValidOpChannel(channel)) invalidChannels.push_back(channel);
721 } // for configured channels
722
723 // collect missing channels
724 auto const endChannel = geom.MaxOpChannel() + 1U;
725 std::vector<raw::Channel_t> missingChannels;
726 for (auto channel: util::counter<raw::Channel_t>(endChannel)) {
727 if (!geom.IsValidOpChannel(channel)) continue;
728 if (configuredChannels.count(channel) > 0U) continue;
729 missingChannels.push_back(channel);
730 } // for channel
731
732 // no error? good we are
733 if (invalidChannels.empty()
734 && missingChannels.empty()
735 && duplicateChannels.empty()
736 && (nEmptyGroups == 0U)
737 ) {
738 return configuredChannels.size();
739 }
740
741 // error? good we are not
742 art::Exception e(art::errors::Configuration);
743 e << "The pairing configuration of " << configuredChannels.size()
744 << " optical channels in `ChannelPairing` presents the following errors.\n";
745 if (!invalidChannels.empty()) {
746 auto iChannel = invalidChannels.cbegin();
747 auto const cend = invalidChannels.cend();
748 e
749 << invalidChannels.size()
750 << " channel numbers do not represent valid channels: " << *iChannel;
751 while (++iChannel != cend) e << ", " << *iChannel;
752 e << ".\n";
753 } // if invalid channels
754
755 if (!missingChannels.empty()) {
756 auto iChannel = missingChannels.cbegin();
757 auto const cend = missingChannels.cend();
758 e
759 << missingChannels.size()
760 << " channels do not appear in the configuration: " << *iChannel;
761 while (++iChannel != cend) e << ", " << *iChannel;
762 e << ".\n";
763 } // if missing channels
764
765 if (!duplicateChannels.empty()) {
766 auto iChannel = duplicateChannels.cbegin();
767 auto const cend = duplicateChannels.cend();
768 e
769 << duplicateChannels.size()
770 << " channels are specified more than once: " << *iChannel;
771 while (++iChannel != cend) e << ", " << *iChannel;
772 e << ".\n";
773 } // if duplicate channels
774
775 if (nEmptyGroups > 0U) {
776 e << nEmptyGroups << " channel groups are empty.";
777 }
778
779 throw e;
780} // icarus::trigger::LVDSgates::checkPairings()

783//------------------------------------------------------------------------------
784void icarus::trigger::LVDSgates::checkInput
785 (std::vector<TrackedTriggerGate_t> const& gates) const
786{
787 /*
788 * check that
789 * * each gate has a single channel
790 * * each gate is in the position matching its channel number
791 */
792
793 using icarus::trigger::gatesIn;
794
795 for (auto const& [ iGate, gate ]: util::enumerate(gatesIn(gates))) {
796 // gate is a OpticalTriggerGateData_t
797
798 auto const expectedChannel = static_cast<raw::Channel_t>(iGate);
799
800 if (gate.nChannels() != 1U) {
801 cet::exception e("LVDSgates");
802 e << "Requirement failure: gate #" << iGate << " includes ";
803 if (!gate.hasChannels())
804 e << "no channels";
805 else {
806 auto const& channels = gate.channels();
807 auto iChannel = channels.begin();
808 auto const cend = channels.end();
809 e << gate.nChannels() << " channels (" << *iChannel;
810 while (++iChannel != cend) e << ", " << *iChannel;
811 e << ")";
812 }
813 e << " should have exactly one channel (" << expectedChannel << ").\n";
814 throw e;
815 }
816
817 if (gate.channel() != expectedChannel) {
818 throw cet::exception("LVDSgates")
819 << "Requirement failure: gate #" << iGate << " gate covers channel "
820 << gate.channel() << ", should cover " << expectedChannel
821 << " instead.\n";
822 }
823
824 } // for
825
826 MF_LOG_TRACE(fLogCategory)
827 << "LVDSgates[" << moduleDescription().moduleLabel() << "]: input checked.";
828
829} // icarus::trigger::LVDSgates::checkInput()
