513template <typename OpDetInfo /* = sbn::OpDetWaveformMeta */>
514std::tuple<
515 std::vector<icarus::trigger::OpticalTriggerGateData_t>,
516 art::Assns<icarus::trigger::OpticalTriggerGateData_t, OpDetInfo>
517 >
518icarus::trigger::transformIntoOpticalTriggerGate(
519 std::vector<icarus::trigger::TrackedOpticalTriggerGate<OpDetInfo>>&& gates,
520 art::PtrMaker<icarus::trigger::OpticalTriggerGateData_t> const& makeGatePtr,
521 util::DataProductPointerMap_t<OpDetInfo> const& opDetInfoPtrs
522) {
523 using TriggerGateData_t = icarus::trigger::OpticalTriggerGateData_t;
524
525 // note that we rely on the fact that `gates` still contains trigger gates,
526 // and those trigger gates still know about the associated waveforms;
527 // attempting to access their levels, though, would be an error
528 // (if really needed, we can still find them in `gateData`)
529 art::Assns<TriggerGateData_t, OpDetInfo> gateToWaveforms;
530
531 for (auto const& [ iGate, gate ]: util::enumerate(gates)) {
532
533 if (!gate.tracking().hasTracked()) continue;
534
535 //
536 // produce the associations
537 //
538
539 // pointer to the gate data we have just added:
540 art::Ptr<TriggerGateData_t> const& gatePtr = makeGatePtr(iGate);
541 for (OpDetInfo const* waveform: gate.tracking().getTracked()) {
542
543 gateToWaveforms.addSingle(gatePtr, opDetInfoPtrs.at(waveform));
544
545 } // for all waveform info
546
547 } // for all gates
548
549 //
550 // create objects for the data products
551 //
552 std::vector<TriggerGateData_t> gateData
553 = transformIntoOpticalTriggerGate(icarus::trigger::gatesIn(gates));
554
555 return { std::move(gateData), std::move(gateToWaveforms) };
556
557} // icarus::trigger::transformIntoOpticalTriggerGate()

561template <typename OpDetInfo>
562std::vector<icarus::trigger::TrackedOpticalTriggerGate<OpDetInfo>>
563icarus::trigger::FillTriggerGates(
564 std::vector<icarus::trigger::OpticalTriggerGateData_t> const& gates,
565 art::Assns<icarus::trigger::OpticalTriggerGateData_t, OpDetInfo> const& gateToWaveformInfo
566) {
567
568 std::vector<icarus::trigger::TrackedOpticalTriggerGate<OpDetInfo>>
569 allGates; // the output
570
571 auto iGateToWaveform = gateToWaveformInfo.begin();
572 auto const gwend = gateToWaveformInfo.end();
573
574 for (auto const& [ iGate, gate ]: util::enumerate(gates)) {
575
576 icarus::trigger::TrackedOpticalTriggerGate<OpDetInfo> trackedGate { gate };
577
578 //
579 // find the relevant waveforms for this gate;
580 // match by index of the element in the data product collection
581 //
582 while (iGateToWaveform != gwend) {
583 if (iGateToWaveform->first.key() == iGate) break;
584 ++iGateToWaveform;
585 } // while
586
587 while (iGateToWaveform != gwend) {
588 if (iGateToWaveform->first.key() != iGate) break;
589
590 OpDetInfo const& wfInfo = *(iGateToWaveform->second);
591
592 // add the associated information to the gate:
593 trackedGate.gate().addChannel(wfInfo.ChannelNumber());
594 trackedGate.tracking().add(&wfInfo);
595
596 ++iGateToWaveform;
597 } // while
598
599 allGates.push_back(std::move(trackedGate));
600
601 } // for gates
602
603 return allGates;
604} // icarus::trigger::FillTriggerGates()
