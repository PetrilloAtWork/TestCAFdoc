423 void CRTPMTMatchingProducer::produce(art::Event & e, art::ProcessingFrame const&)
424 {
425 mf::LogDebug("CRTPMTMatchingProducer") << "beginning CRTPMTProducer";
426 // Trigger data product variables
427 // add trigger info
428
429 sbn::ExtraTriggerInfo const* trigInfo = nullptr;
430 if (!fTriggerLabel.empty()) {
431 if (auto const trigger_handle = e.getHandle<sbn::ExtraTriggerInfo>(fTriggerLabel))
432 {
433 trigInfo = trigger_handle.product();
434 } else {
435 mf::LogDebug("CRTPMTMatchingProducer") << "No sbn::ExtraTriggerInfo associated to label: " << fTriggerLabel.encode() << "\n";
436 }
437 }
438 else {
439 mf::LogDebug("CRTPMTMatchingProducer") << "No TriggerLabel in : " << fTriggerLabel.encode() << "\n";
440 }
441 sbn::triggerSource const gateType
442 = trigInfo? trigInfo->sourceType: sbn::triggerSource::Unknown;
443 int64_t const triggerGateDiff
444 = trigInfo? trigInfo->triggerFromBeamGate(): 0;
445
446 double BeamGateMin = 0.0, BeamGateMax = 0.0;
447 double inBeamMin = 0.0, inBeamMax = 0.0;
448 switch (gateType) {
449 case sbn::triggerSource::BNB:
450 case sbn::triggerSource::OffbeamBNB:
451 BeamGateMin = fBNBBeamGateMin;
452 BeamGateMax = fBNBBeamGateMax;
453 inBeamMin = fBNBinBeamMin;
454 inBeamMax = fBNBinBeamMax;
455 break;
456 case sbn::triggerSource::NuMI:
457 case sbn::triggerSource::OffbeamNuMI:
458 BeamGateMin = fNuMIBeamGateMin;
459 BeamGateMax = fNuMIBeamGateMax;
460 inBeamMin = fNuMIinBeamMin;
461 inBeamMax = fNuMIinBeamMax;
462 break;
463 case sbn::triggerSource::Unknown:
464 mf::LogWarning("CRTPMTMatchingProducer")
465 << "Unknown beam gate type: in-gate flags won't be available.";
466 break;
467 default:
468 mf::LogWarning("CRTPMTMatchingProducer")
469 << "Unsupported beam gate type: " << name(gateType)
470 << "; in-gate flags won't be available.";
471 break;
472 } // switch
473
474 auto CRTPMTMatchesColl = std::make_unique<std::vector<CRTPMTMatching>>();
475
476 art::PtrMaker<CRTPMTMatching> const makeInfoPtr(e);
477 auto FlashAssociation = std::make_unique<art::Assns<CRTPMTMatching, recob::OpFlash>>();
478 auto CRTAssociation = std::make_unique<art::Assns<CRTPMTMatching, sbn::crt::CRTHit>>();
479 auto FlashCRTAssociation = std::make_unique<art::Assns<recob::OpFlash, sbn::crt::CRTHit, CRTPMTMatchingInfo>>();
480
481 // add CRTHits
482 std::vector<art::Ptr<CRTHit>> crtHitList;
483 if (auto crtHitListHandle = e.getHandle<std::vector<CRTHit>>(fCrtHitModuleLabel))
484 art::fill_ptr_vector(crtHitList, crtHitListHandle);
485 {
486 mf::LogTrace log("CRTPMTMatchingProducer");
487 log << fCrtHitModuleLabel.encode() << " has " << crtHitList.size() << " CRT hits";
488 for (art::Ptr<sbn::crt::CRTHit> const& hitPtr: crtHitList) {
489 log << "\n [#" << hitPtr.key() << "] at T0=" << hitPtr->ts0() << " ns, T1="
490 << hitPtr->ts1()/1000.0 << " us, (" << hitPtr->x_pos << ", "
491 << hitPtr->y_pos << ", " << hitPtr->z_pos << ") cm, " << hitPtr->peshit
492 << " p.e., on " << hitPtr->tagger;
493 } // for
494 }
495 bool const isRealData = e.isRealData();
496 int n_entering_matches = 0;
497 int n_exiting_matches = 0;
498 mf::LogTrace("CRTPMTMatchingProducer") << "is this real data? " << std::boolalpha << isRealData;
499
500 //
501 // prepare the flashes (sorted by time)
502 //
503 std::vector
504 <std::tuple<art::Ptr<recob::OpFlash>, std::vector<recob::OpHit const*>>>
505 flashesAndHits;
506 for (art::InputTag const& flashLabel : fFlashLabels) {
507 auto const flashHandle =
508 e.getHandle<std::vector<recob::OpFlash>>(flashLabel);
509
510 mf::LogTrace("CRTPMTMatchingProducer")
511 << "Matching " << flashHandle->size() << " flashes from " << flashLabel.encode();
512
513 art::FindMany<recob::OpHit> const findManyHits(flashHandle, e, flashLabel);
514 for (std::size_t const iFlash: util::counter(flashHandle->size())) {
515 flashesAndHits.emplace_back(
516 art::Ptr<recob::OpFlash>{ flashHandle, iFlash },
517 findManyHits.at(iFlash)
518 );
519 } // for flashes
520 } // for flash labels
521
522 // sort the flashes by Time(); their associated hits follow
523 std::sort(flashesAndHits.begin(), flashesAndHits.end(),
524 [](auto const& flashAndHitsA, auto const& flashAndHitsB)
525 {
526 art::Ptr<recob::OpFlash> const& flashPtrA = std::get<0U>(flashAndHitsA);
527 art::Ptr<recob::OpFlash> const& flashPtrB = std::get<0U>(flashAndHitsB);
528 if (flashPtrA == flashPtrB) return false; // or we would emit warnings
529 if (!flashPtrA || !flashPtrB) return flashPtrA < flashPtrB;
530 if (flashPtrA->Time() != flashPtrB->Time())
531 return flashPtrA->Time() < flashPtrB->Time();
532 // this should be very, very rare
533 mf::LogPrint("CRTPMTMatchingProducer")
534 << "Flashes " << flashPtrA << " and " << flashPtrB
535 << " both are at time " << flashPtrA->Time() << " us";
536 if (flashPtrA->TotalPE() != flashPtrB->TotalPE())
537 return flashPtrA->TotalPE() > flashPtrB->TotalPE();
538 mf::LogPrint("CRTPMTMatchingProducer")
539 << " ... and they both have " << flashPtrA->TotalPE() << "?!?";
540 return flashPtrA < flashPtrB;
541 }
542 );
543
544 //
545 // process all the flashes
546 //
547 for (auto const& [ flashPtr, hits ]: flashesAndHits) {
548
549 double const tflash = flashPtr->Time();
550 double const totPe = flashPtr->TotalPE();
551 double const flashYWidth = flashPtr->YWidth();
552 double const flashZWidth = flashPtr->ZWidth();
553
554 int nPMTsTriggering = 0;
555 double firstOpHitPeakTime = 999999;
556 double firstOpHitStartTime = 999999;
557
558 geo::vect::MiddlePointAccumulator flashCentroid;
559 for (auto const& hit : hits) {
560 if (hit->Amplitude() > fPMTADCThresh) nPMTsTriggering++;
561 if (firstOpHitPeakTime > hit->PeakTime()) firstOpHitPeakTime = hit->PeakTime();
562 if(hit->HasStartTime()){
563 if (firstOpHitStartTime > hit->StartTime()) firstOpHitStartTime = hit->StartTime();
564 //mf::LogTrace("CRTPMTMatchingProducer") << "OpHit has a starttime " << hit->StartTime() << ", saving " << firstOpHitStartTime << " to firstOpHitStartTime.\n";
565 }
566
567 geo::Point_t const pos =
568 fGeometryService->OpDetGeoFromOpChannel(hit->OpChannel())
569 .GetCenter();
570 double const amp = hit->Amplitude();
571 flashCentroid.add(pos, amp);
572 }
573 geo::Point_t const flash_pos = flashCentroid.middlePoint();
574 {
575 mf::LogTrace log("CRTPMTMatchingProducer");
576 log << "Now matching flash #" << flashPtr.key()
577 << " at " << flashPtr->Time() << " us and (";
578 if (flashPtr->hasXCenter()) log << flashPtr->XCenter();
579 else log << "n/a";
580 log << ", " << flashPtr->YCenter() << ", " << flashPtr->ZCenter()
581 << ") cm [" << hits.size() << " op.hits] -> first time: "
582 << firstOpHitPeakTime << " us, centroid: " << flash_pos << " cm";
583 }
584
585 if (nPMTsTriggering < fnOpHitToTrigger) {
586 mf::LogTrace("CRTPMTMatchingProducer")
587 << " => skipped (only " << nPMTsTriggering << " < " << fnOpHitToTrigger
588 << " hits above threshold)";
589 continue;
590 }
591
592 double const thisRelGateTime = triggerGateDiff + tflash * 1e3; // ns
593 bool const thisInTime_gate
594 = thisRelGateTime > BeamGateMin && thisRelGateTime < BeamGateMax;
595 bool const thisInTime_beam
596 = thisRelGateTime > inBeamMin && thisRelGateTime < inBeamMax;
597
598 icarus::crt::CRTMatches const crtMatches = icarus::crt::CRTHitmatched(
599 firstOpHitPeakTime, flash_pos, crtHitList, fTimeOfFlightInterval, isRealData, fGlobalT0Offset, fMatchBottomCRT);
600
601 std::vector<MatchedCRT> thisFlashCRTmatches;
602 std::vector<art::Ptr<sbn::crt::CRTHit>> CRTPtrs; // same order as thisFlashCRTmatches
603 MatchType const eventType = crtMatches.flashType;
604 if (!crtMatches.entering.empty()) {
605 mf::LogTrace("CRTPMTMatchingProducer")
606 << "Entering matches (" << crtMatches.entering.size() << "):";
607 for (auto const& entering : crtMatches.entering) {
608 n_entering_matches++;
609 CRTPtrs.push_back(entering.CRTHit);
610 thisFlashCRTmatches.push_back(makeMatchedCRT(*entering.CRTHit, tflash, isRealData));
611 }
612 }
613 if (!crtMatches.exiting.empty()) {
614 mf::LogTrace("CRTPMTMatchingProducer")
615 << "Exiting matches (" << crtMatches.exiting.size() << "):";
616 for (auto const& exiting : crtMatches.exiting) {
617 n_exiting_matches++;
618 CRTPtrs.push_back(exiting.CRTHit);
619 thisFlashCRTmatches.push_back(makeMatchedCRT(*exiting.CRTHit, tflash, isRealData));
620 }
621 }
622 if (!thisFlashCRTmatches.empty() ) {
623 mf::LogTrace("CRTPMTMatchingProducer") << "pushing back flash with "
624 << thisFlashCRTmatches.size() << " CRT Matches. --> Match classification = " << std::to_string(static_cast<int>(eventType))<< "\n" ;
625 mf::LogTrace("CRTPMTMatchingProducer") << "\tfirstOpHitPeakTime " << firstOpHitPeakTime << ", firstOpHitStartTime = " << firstOpHitStartTime << " (us)\n\ttotPe = " << totPe << "\n\tflashYWidth = " << flashYWidth << ", flashZWidth = " << flashZWidth << " (cm)\n";
626
627 }
628
629 int const flashID = icarus::crt::makeFlashID(flashPtr.get());
630 mf::LogTrace("CRTPMTMatchingProducer") << "saving Flash with FlashID = " << flashID << "\n";
631 FlashType thisFlashType = { /* .flashID = */ flashID, // C++20: restore initializers
632 /* .flashPos = */ flash_pos,
633 /* .flashTime = */ tflash,
634 /* .flashGateTime = */ thisRelGateTime / 1000.0, // -> us
635 /* .firstOpHitPeakTime = */ firstOpHitPeakTime,
636 /* .firstOpHitStartTime = */ firstOpHitStartTime,
637 /* .inBeam = */ thisInTime_beam,
638 /* .inGate = */ thisInTime_gate,
639 /* .flashPE = */ totPe,
640 /* /.flashYWidth = */ flashYWidth,
641 /* /.flashZWidth = */ flashZWidth,
642 /* .classification = */ eventType,
643 /* .CRTmatches = */ std::move(thisFlashCRTmatches)};
644
645 // add to the data products
646
647 // pointer to the matching info object we are going to add to the collection:
648 art::Ptr<CRTPMTMatching> const infoPtr
649 = makeInfoPtr(CRTPMTMatchesColl->size());
650
651 FlashAssociation->addSingle(infoPtr, flashPtr);
652
653 for (icarus::crt::CRTPMT const& CRTmatch: crtMatches.entering) {
654 CRTAssociation->addSingle(infoPtr, CRTmatch.CRTHit);
655 FlashCRTAssociation->addSingle(
656 flashPtr, CRTmatch.CRTHit,
657 CRTPMTMatchingInfo{
658 /* .direction = */ CRTPMTMatchingInfo::Dir::entering, // C++20: restore initializers
659 /* .timeOfFlight = */ CRTmatch.tof,
660 /* .distance = */ CRTmatch.distance
661 }
662 );
663 }
664 for (icarus::crt::CRTPMT const& CRTmatch: crtMatches.exiting) {
665 CRTAssociation->addSingle(infoPtr, CRTmatch.CRTHit);
666 FlashCRTAssociation->addSingle(
667 flashPtr, CRTmatch.CRTHit,
668 CRTPMTMatchingInfo{
669 /* .direction = */ CRTPMTMatchingInfo::Dir::exiting, // C++20: restore initializers
670 /* .timeOfFlight = */ CRTmatch.tof,
671 /* .distance = */ CRTmatch.distance
672 }
673 );
674 }
675
676 CRTPMTMatching matchInfo = sbn::crt::FillCRTPMT(thisFlashType);
677 CRTPMTMatchesColl->push_back(std::move(matchInfo));
678
679 } // for flash data products
680 mf::LogTrace("CRTPMTMatchingProducer") << "in total " << n_entering_matches << " entering CRT Hit matches, " << n_exiting_matches << " exiting CRT Hit matches\n";
681 mf::LogTrace("CRTPMTMatchingProducer") <<"This Event has "<<CRTPMTMatchesColl->size()<<" Flashes candidate for CRT matching."<<std::endl;
682 e.put(std::move(CRTPMTMatchesColl));
683 e.put(std::move(FlashAssociation));
684 e.put(std::move(CRTAssociation));
685 e.put(std::move(FlashCRTAssociation));
686
687 } // CRTPMTMatchingProducer::produce()

691 sbn::crt::MatchedCRT CRTPMTMatchingProducer::makeMatchedCRT
692 (sbn::crt::CRTHit const& hit, double tflash, bool isRealData) const
693 {
694 geo::Point_t const thisCRTpos {hit.x_pos, hit.y_pos, hit.z_pos};
695 // fGlobalT0Offset = 1.6e6 ns = 1600000 ns.
696 double const CRTtime // ns -> us
697 = icarus::crt::CRTHitTime(hit, fGlobalT0Offset, isRealData) / 1000.0;
698 double const CRTTof_opflash = CRTtime - tflash; // us
699 int const CRTRegion = hit.plane;
700 // Very lazy way to determine if the Hit is a Top or a Side, bottom CRT is only set if fMatchBottomCRT true;
701 int const CRTSys = (fMatchBottomCRT && CRTRegion >= 49) ? 2 : ((CRTRegion >= 36)? 1: 0);
702
703 mf::LogTrace("CRTPMTMatchingProducer")
704 << " Match: tof = crtTime - tflash = " << CRTtime
705 << " - "<< tflash << " = " << CRTTof_opflash << " (us) in region " << CRTRegion << "";
706 return { /* .position =*/ thisCRTpos, // C++20: restore initializers
707 /* .PMTTimeDiff = */ CRTTof_opflash,
708 /* .time = */ CRTtime,
709 /* .sys = */ CRTSys,
710 /* .region = */ CRTRegion};
711 }
