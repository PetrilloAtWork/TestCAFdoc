77template <typename TK, typename TI>
78void icarus::trigger::TriggerGateData<TK, TI>::compact() {
79
80 /*
81 * Removes:
82 * * multiple sets at the same tick
83 * * shifts that don't change opening level
84 *
85 * Keeps:
86 * * unknown stati (because they are unknown and they might have reason to be)
87 *
88 */
89
90 auto const send = fGateLevel.end();
91 auto iLast = fGateLevel.begin(); // last good status
92 auto iDest = std::next(iLast); // the status next to be assigned
93 auto iTest = iDest; // the next status candidate
94
95 // declare the current candidate status good:
96 auto accept = [&iDest, &iTest, &iLast]()
97 { if (iDest != iTest) *iDest = *iTest; iLast = iDest; ++iDest; };
98
99 while (iTest != send) {
100
101 switch (iTest->event) {
102 case EventType::Shift:
103 if (iLast->opening != iTest->opening) accept();
104 break;
105 case EventType::Set:
106 if (iTest->tick > iLast->tick) accept();
107 break;
108 case EventType::Unknown:
109 accept();
110 break;
111 } // switch
112
113 ++iTest;
114
115 } // while
116
117 fGateLevel.erase(iDest, send);
118
119} // icarus::trigger::TriggerGateData<>::compact()

123template <typename TK, typename TI>
124auto icarus::trigger::TriggerGateData<TK, TI>::openingRange
125 (ClockTick_t start, ClockTick_t end) const
126 -> std::pair<OpeningCount_t, OpeningCount_t>

128 /*
129 * parse from the first to the last relevant status of the gate and collect
130 * the extrema of their opening counts
131 */
132 assert(!fGateLevel.empty());

134 if (start >= end) return { {}, {} }; // this should rank pretty high in the
135 // competition of weird C++ constructs
136
137 // look where to start from; if the start tick is before the start of the
138 // gate, fast forward to the actual start of the gate
139 auto const send = fGateLevel.end();
140 auto const maybeStatusIter = findLastStatusFor(start);
141 auto iStatus = maybeStatusIter? maybeStatusIter.value(): fGateLevel.begin();
142 assert(iStatus != fGateLevel.end());
143
144 // so the requested range is fully before the start of the gate: bail out!
145 if (iStatus->tick >= end) return { {}, {} };

147 OpeningCount_t const startOpening = iStatus->opening;
148 std::pair<OpeningCount_t, OpeningCount_t> limits
149 { startOpening, startOpening + 1 };
150 while (++iStatus != send) {
151 if (iStatus->tick >= end) break; // we reached the end of the tick interval
152
153 auto const opening = iStatus->opening;
154 if (opening < limits.first) limits.first = opening;
155 if (opening >= limits.second) limits.second = opening + 1;
156
157 } // while

159 return limits;

173template <typename TK, typename TI>
174void icarus::trigger::TriggerGateData<TK, TI>::setOpeningAt
175 (ClockTick_t tick, OpeningCount_t openingCount)
176{
177 //
178 // first find where to start acting
179 //
180 auto iStatus = findLastStatusForTickOrThrow(tick); // may be before the tick
181
182 //
183 // set the current status (possibly a new one)
184 //
185 if (iStatus->tick == tick) { // overwrite the previous action
186 iStatus->event = EventType::Set;
187 iStatus->opening = openingCount;
188 }
189 else { // no status exactly at this tick, iStatus is just before
190 // insert the new status after iStatus
191 iStatus = fGateLevel.insert
192 (++iStatus, { EventType::Set, tick, openingCount });
193 }
194
195 //
196 // update all the following stati;
197 // the "Set" action stops when another set happens, but overrides shifts
198 //
199 ++iStatus; // start from the status next to the one we just changed/added
200 while (iStatus != fGateLevel.end()) {
201 switch (iStatus->event) {
202 case EventType::Shift: // just remove the shift event
203 iStatus = fGateLevel.erase(iStatus);
204 break;
205 case EventType::Set: // the later Set event takes over, we are done
206 return;
207 case EventType::Unknown: // not sure about this... let's keep going
208 ++iStatus;
209 break;
210 } // switch event type
211 }

217template <typename TK, typename TI>
218void icarus::trigger::TriggerGateData<TK, TI>::openBetween
219 (ClockTick_t start, ClockTick_t end, OpeningDiff_t count /* = 1 */)
220{
221 /*
222 * The plan is:
223 * 1) find the even just before (or at) the tick where we want the gate opened
224 * 2) open the gate at that location
225 * 3) find the last existing event in the gate that we need to change, by tick
226 * 4) update all the events to reflect the level change
227 * 1) if the event is a (re)set event, it takes priority and we are done
228 * 5) insert the gate close event at the right tick
229 */
230 if (start >= end) return; // weird, yet valid
231
232 //
233 // (1) first find where to start acting
234 //
235 auto iStatus = findLastStatusForTickOrThrow(start); // may be before the start
236
237 //
238 // (2) set the current status (possibly a new one)
239 //
240 if (iStatus->tick == start) { // there is already something here
241 switch (iStatus->event) {
242 case EventType::Unknown: // the shift overrides... we don't even know what
243 iStatus->event = EventType::Shift;
244 [[fallthrough]];
245 case EventType::Shift:
246 case EventType::Set: // in case of Set, we change the set value
247 iStatus->opening += count;
248 break;
249 } // switch
250 }
251 else { // no status exactly at start tick, iStatus is just before
252 // insert the new status after iStatus
253 auto const opening = iStatus->opening + count;
254 iStatus = fGateLevel.insert
255 (++iStatus, { EventType::Shift, start, opening });
256 }
257
258 //
259 // (3) also find where to stop acting
260 //
261 auto send = findLastStatusForTickOrThrow(end); // may be before `end`
262 // if `send` is stricly earlier than `end`, we want it affected too
263 if (send->tick < end) ++send;
264
265 //
266 // (4) update all the following stati;
267 // the "Shift" action stops when a set happens, and stacks with other shifts
268 //
269 using std::to_string;
270 while (++iStatus != send) {
271 switch (iStatus->event) {
272 case EventType::Shift: // change the resulting opening
273 if ((count < 0) && (iStatus->opening < OpeningCount_t(-count))) {
274 throw std::runtime_error(
275 "icarus::trigger::TriggerGateData::openBetween(): "
276 "asked to close " + to_string(-count)
277 + " gate counts starting at "
278 + to_string(start) + " but at time "
279 + to_string(iStatus->tick) + " only "
280 + to_string(iStatus->opening) + " are still open"
281 );
282 }
283 iStatus->opening += count;
284 break;
285 case EventType::Set: // (4.1) Set event takes over, we are done
286 return;
287 case EventType::Unknown: // not sure about this... let's keep going
288 break;
289 } // switch event type
290 } // while
291
292 //
293 // (5) finally close the gate: mark the end, and leave the following alone
294 // Note that gate is unlimited in time.
295 //
296
297 if ((iStatus != fGateLevel.end()) && (iStatus->tick == end)) {
298 // there is already something here, and it must be at the right level
299 switch (iStatus->event) {
300 case EventType::Unknown: // the shift overrides... we don't even know what
301 iStatus->event = EventType::Shift;
302 [[fallthrough]];
303 case EventType::Shift:
304 case EventType::Set:
305 break;
306 } // switch
307 }
308 else { // no status exactly at this tick, just need to insert one
309 // insert the new status before iStatus;
310 // the correct opening now is the last one, minus what we added;
311 // it is also the status in iStatus, if it exists
312 assert(iStatus != fGateLevel.begin()); // we must have added one status!
313
314 auto const opening = (iStatus == fGateLevel.end())
315 ? (std::prev(iStatus)->opening - count): (iStatus->opening);
316
317 // checking that if the two branches above are both available, they match;
318 // the branch above just takes the easier way when available
319 assert((iStatus == fGateLevel.end())
320 || (opening == (std::prev(iStatus)->opening - count))
321 );
322
323 iStatus = fGateLevel.insert(iStatus, { EventType::Shift, end, opening });
324 }
325 // if we get here, now iStatus contains the gate closing (for what we care)
326
327} // icarus::trigger::TriggerGateData<>::openFor()

573 }

575 } // while
