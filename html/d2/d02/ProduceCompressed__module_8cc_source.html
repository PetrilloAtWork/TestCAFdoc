63 void ProduceCompressed::produceForLabel(art::Event& evt, art::InputTag fFragmentLabel)
64 {
65 // get the old fragments
66 art::Handle<std::vector<artdaq::Fragment>> fragHandle;
67 evt.getByLabel(fFragmentLabel, fragHandle);
68 auto const& old_fragments(*fragHandle);
69
70 // make a vector to put the new fragments in
71 std::unique_ptr<std::vector<artdaq::Fragment>> new_fragments(new std::vector<artdaq::Fragment>());
72
73 // fill the new fragments vector
74 for (auto const& old_fragment : old_fragments)
75 {
76
77 artdaq::Fragment new_fragment = icarus::PhysCrateFragment::fragmentSwitch(old_fragment, true);
78 if (fDebug)
79 MF_LOG_VERBATIM("ProduceCompressed")
80 << "-------------------------" << '\n'
81 << " size of old_fragment (in bytes) = " << old_fragment.sizeBytes() << '\n'
82 << " size of payload: " << old_fragment.dataSizeBytes() << '\n'
83 << " 64-bit words in payload: " << old_fragment.dataSize() << '\n'
84 << " compression scheme: " << old_fragment.metadata<icarus::PhysCrateFragmentMetadata>()->compression_scheme() << '\n'
85 << " size of new_fragment (in byets) = " << new_fragment.sizeBytes() << '\n'
86 << " size of payload: " << new_fragment.dataSizeBytes() << '\n'
87 << " 64-bit words in payload: " << new_fragment.dataSize() << '\n'
88 << " compression scheme: " << new_fragment.metadata<icarus::PhysCrateFragmentMetadata>()->compression_scheme() << '\n'
89 << "-------------------------";
90 new_fragments->emplace_back(new_fragment);
91
92 if (fDebug)
93 {
94 // check adc vals
95 icarus::PhysCrateFragment old_overlay(old_fragment);
96 icarus::PhysCrateFragment new_overlay(new_fragment);
97 artdaq::detail::RawFragmentHeader::fragment_id_t fragmentID = new_fragment.fragmentID();
98 const std::string& crateName = fChannelMap->getCrateName(fragmentID);
99 MF_LOG_VERBATIM("ProduceCompressed")
100 << "########################################################" << '\n'
101 << "Checking compression for FragmentID " << std::hex << fragmentID << std::dec << ", Crate " << crateName;
102 bool adcsGood = true;
103 for (size_t b = 0; b < old_overlay.nBoards(); ++b)
104 {
105 for (size_t s = 0; s < old_overlay.nSamplesPerChannel(); ++s)
106 {
107 for (size_t c = 0; c < old_overlay.nChannelsPerBoard(); ++c)
108 {
109 uint16_t old_adc_val = old_overlay.adc_val(b, c, s);
110 uint16_t new_adc_val = new_overlay.adc_val(b, c, s);
111 int16_t adc_difference = (old_adc_val - new_adc_val);
112 if (old_adc_val != new_adc_val)
113 {
114 adcsGood = false;
115 MF_LOG_VERBATIM("ProduceCompressed")
116 << "ADC mismatch in board " << b << ", channel "<< c << ", sample " << s << '\n'
117 << " Old value: " << old_adc_val << '\n'
118 << " New value: " << new_adc_val << '\n'
119 << " Difference: " << adc_difference;
120 }
121 }
122 }
123 }
124 if (adcsGood)
125 {
126 MF_LOG_VERBATIM("ProduceCompressed")
127 << "Fragments for FragmentID " << std::hex << fragmentID << std::dec << ", Crate " << crateName << " agree!" << '\n'
128 << "########################################################";
129 } else {
130 MF_LOG_VERBATIM("ProduceCompressed")
131 << "Fragments for FragmentID " << std::hex << fragmentID << std::dec << ", Crate " << crateName << " DO NOT MATCH" << '\n'
132 << " Checking fragment word-by-word:";
133 uint64_t dataOffset = 0;
134 for (size_t b = 0; b < old_overlay.nBoards(); ++b)
135 {
136 uint32_t boardDataTileSize = sizeof(icarus::PhysCrateDataTileHeader) + sizeof(icarus::A2795DataBlock::Header);
137 for (size_t boardHeaderWord = 0; boardHeaderWord < (boardDataTileSize / sizeof(icarus::A2795DataBlock::data_t)); ++boardHeaderWord)
138 {
139 MF_LOG_VERBATIM("ProduceCompressed")
140 << " " << std::bitset<16>(icarus::PhysCrateFragment::getA2795Word(new_fragment, dataOffset)) << " | board " << b << " header word (" << dataOffset << ")";
141 ++dataOffset;
142 }
143 std::vector<uint16_t> prevADCVals(64, 0);
144 for (size_t s = 0; s < old_overlay.nSamplesPerChannel(); ++s)
145 {
146 bool oddCompressions = false;
147 size_t c = 0;
148 while (c < 64)
149 {
150 uint16_t word = icarus::PhysCrateFragment::getA2795Word(new_fragment, dataOffset);
151 if ((word & 0xF000) == 0x8000)
152 {
153 int16_t delta = (word & 0x0FFF) + (0xF000 * ((word & 0x0800) == 0x0800));
154 MF_LOG_VERBATIM("ProduceCompressed")
155 << " " << std::bitset<16>(word) << " | uncompressed word (" << dataOffset << ")" << '\n'
156 << " board " << b << ", channel " << c << ", sample " << s << " has ADC value " << prevADCVals[c] << " + " << delta << " = " << (prevADCVals[c] + delta) << '\n'
157 << " ADC value from the old fragment is " << old_overlay.adc_val(b, c, s);
158 prevADCVals[c] += delta;
159 ++c;
160 } else {
161 oddCompressions = (not oddCompressions);
162 int16_t delta0 = ((word >> (4*0)) & 0x000F) + (0xFFF0 * ((word & 0x0008) == 0x0008));
163 int16_t delta1 = ((word >> (4*1)) & 0x000F) + (0xFFF0 * ((word & 0x0080) == 0x0080));
164 int16_t delta2 = ((word >> (4*2)) & 0x000F) + (0xFFF0 * ((word & 0x0800) == 0x0800));
165 int16_t delta3 = ((word >> (4*3)) & 0x000F) + (0xFFF0 * ((word & 0x8000) == 0x8000));
166 MF_LOG_VERBATIM("ProduceCompressed")
167 << " " << std::bitset<16>(word) << " | compressed word (" << dataOffset << ")" << '\n'
168 << " board " << b << ", channel " << c + 0 << ", sample " << s << " has ADC value " << prevADCVals[c + 0] << " + " << delta0 << " = " << (prevADCVals[c + 0] + delta0) << '\n'
169 << " ADC value from the old fragment is " << old_overlay.adc_val(b, c + 0, s) << '\n'
170 << " board " << b << ", channel " << c + 1 << ", sample " << s << " has ADC value " << prevADCVals[c + 1] << " + " << delta1 << " = " << (prevADCVals[c + 1] + delta1) << '\n'
171 << " ADC value from the old fragment is " << old_overlay.adc_val(b, c + 1, s) << '\n'
172 << " board " << b << ", channel " << c + 2 << ", sample " << s << " has ADC value " << prevADCVals[c + 2] << " + " << delta2 << " = " << (prevADCVals[c + 2] + delta2) << '\n'
173 << " ADC value from the old fragment is " << old_overlay.adc_val(b, c + 2, s) << '\n'
174 << " board " << b << ", channel " << c + 3 << ", sample " << s << " has ADC value " << prevADCVals[c + 3] << " + " << delta3 << " = " << (prevADCVals[c + 3] + delta3) << '\n'
175 << " ADC value from the old fragment is " << old_overlay.adc_val(b, c + 3, s);
176 prevADCVals[c + 0] += delta0;
177 prevADCVals[c + 1] += delta1;
178 prevADCVals[c + 2] += delta2;
179 prevADCVals[c + 3] += delta3;
180 c += 4;
181 }
182 ++dataOffset;
183 }
184 if (oddCompressions)
185 {
186 MF_LOG_VERBATIM("ProduceCompressed")
187 << " " << std::bitset<16>(icarus::PhysCrateFragment::getA2795Word(new_fragment, dataOffset)) << " | board " << b << " offset word for odd compressions in sample " << s << "(" << dataOffset << ")";
188 ++dataOffset;
189 }
190 }
191 for (size_t trailerWord = 0; trailerWord < 4; ++trailerWord)
192 {
193 MF_LOG_VERBATIM("ProduceCompressed")
194 << " " << std::bitset<16>(icarus::PhysCrateFragment::getA2795Word(new_fragment, dataOffset)) << " | board " << b << " trailer word (" << dataOffset << ")";
195 ++dataOffset;
196 }
197 }
198 MF_LOG_VERBATIM("ProduceCompressed")
199 << "########################################################";
200 }
201 }
202 }
203
204 // put the new fragments into the event
205 evt.put(std::move(new_fragments), fFragmentLabel.instance());
206 }
