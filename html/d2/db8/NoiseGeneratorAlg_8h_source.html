534std::size_t icarus::opdet::NoiseGeneratorAlg<ADCT>::_addByFilling(
535 raw::Channel_t channel, Timestamp_t time,
536 ADCcount_t* begin, std::size_t n
537) {
538 std::vector<ADCcount_t> noise(n);
539 std::size_t const nSamples = fill(channel, time, noise);
540 assert(nSamples <= n);
541
542 auto it = noise.cbegin();
543 auto const nend = it + nSamples;
544 while (it != nend) *(begin++) += *it++;
545
546 return nSamples;
