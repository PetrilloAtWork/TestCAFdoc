546 {
547 MF_LOG_DEBUG("CRT") << "beginning analyis" << '\n';
548
549 // Check that lists for momenta limits is same size as last of PDGs from FHiCL
550 if (fPDGs.size() != fMinMomenta.size() || fPDGs.size() != fMaxMomenta.size())
551 throw cet::exception("CRTSimAnalysis")
552 << " PDG/Momtenta values not set correctly in fhicl - lists have different sizes"
553 << " Line " << __LINE__ << " in file " << __FILE__ << std::endl;
554
555
556 // Start by fetching some basic event information for our n-tuple.
557 fEvent = event.id().event();
558 fRun = event.run();
559 fSubRun = event.subRun();
560
561 //CRTBackTracker matches CRTProducts to the true trackIDs
562 //bt.Initialize(event);
563
564 // Define "handle" to Generator level MCTruth objects
565 art::Handle< vector<simb::MCTruth>> genHandle;
566
567 // Define a "handle" to point to a vector of MCParticle objects.
568 art::Handle< vector<simb::MCParticle> > particleHandle;
569 map< int, const simb::MCParticle*> particleMap;
