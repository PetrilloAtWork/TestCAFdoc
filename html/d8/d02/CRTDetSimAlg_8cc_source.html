786 // for CERN modules only. Given position in local strip coordinates,
787 // calculate the transverse attentuation factor
788 std::pair<double,double> CRTDetSimAlg::GetTransAtten(const double pos) {
789 //attenuation coefficients for each fiber
790 double abs0=0.0, abs1=0.0;
791
792 //coefficiencts for transverse attenuation in CERN modules
793 //from early beta-source tranverse scan data
794 double at0_c = 0.682976;
795 double at1_c = -0.0204477;
796 double at2_c = -0.000707564;
797 double at3_c = 0.000636617;
798 double at4_c = 0.000147957;
799 double at5_c = -3.89078e-05;
800
801 double at0_r = 0.139941;
802 double at1_r = 0.168238;
803 double at2_r = -0.0198199;
804 double at3_r = 0.000781752;
805
806 double at0_l = 8.78875;
807 double at1_l = 3.54602;
808 double at2_l = 0.595592;
809 double at3_l = 0.0449169;
810 double at4_l = 0.00127892;
811
812 //hit between both fibers
813 if ( abs(pos) <= 5.5 ) {
814 abs0 = at5_c*pow(pos,5) + at4_c*pow(pos,4) + at3_c*pow(pos,3)
815 + at2_c*pow(pos,2) + at1_c*pos + at0_c;
816 abs1 = -1*at5_c*pow(pos,5) + at4_c*pow(pos,4) - at3_c*pow(pos,3)
817 + at2_c*pow(pos,2) - at1_c*pos + at0_c;
818 }
819
820 //hit to right of both fibers
821 if ( pos > 5.5 ) {
822 abs0 = at3_r*pow(pos,3) + at2_r*pow(pos,2) + at1_r*pos + at0_r;
823 abs1 = at4_l*pow(pos,4) - at3_l*pow(pos,3)
824 + at2_l*pow(pos,2) - at1_l*pos + at0_l;
825 }
826
827 //hit to left of both fibers
828 if ( pos < -5.5 ) {
829 abs0 = at4_l*pow(pos,4) + at3_l*pow(pos,3) \
830 + at2_l*pow(pos,2) + at1_l*pos + at0_l;
831 abs1 = -1*at3_r*pow(pos,3) + at2_r*pow(pos,2) - at1_r*pos + at0_r;
832 }
833
834 return std::make_pair(abs0,abs1);
835
836 }

838 //---------------------------------------------------------------
839 double CRTDetSimAlg::GetLongAtten(const double dist) {
840
841 double pes=0.0;
842
843 //coefficients for quadratic fit to MINOS test data w/S14
844 //obtained for normally incident cosmic muons
845 //p2_m * x^2 + p1_m * x + p0_m, x is distance from readout [m]
846 double p0_m = 36.5425; //initial light yield (pe) before any attenuation in m scintillator
847 double p1_m = -6.3895;
848 double p2_m = 0.3742;
849
850 pes = p2_m * pow(dist,2) + p1_m * dist + p0_m;
851
852 return pes;
853
854 }

856 //function for simulating time response
857 // takes true hit time, LY(PE) observed, and intitudinal distance from readout
858 // uses 12 FHiCL configurable parameters
859 // returns simulated time in units of clock ticks
860 uint64_t CRTDetSimAlg::GetChannelTriggerTicks(detinfo::ElecClock& clock,
861 double t0, float npeMean, float r)
862 {
863 // Hit timing, with smearing and NPE dependence
864 double tDelayMean =
865 fTDelayNorm *
866 exp(-0.5 * pow((npeMean - fTDelayShift) / fTDelaySigma, 2)) +
867 fTDelayOffset;
868
869 double tDelayRMS =
870 fTDelayRMSGausNorm *
871 exp(-pow(npeMean - fTDelayRMSGausShift, 2) / fTDelayRMSGausSigma) +
872 fTDelayRMSExpNorm *
873 exp(-(npeMean - fTDelayRMSExpShift) / fTDelayRMSExpScale);
874
875 double tDelay = CLHEP::RandGauss::shoot(&fRandEngine, tDelayMean, tDelayRMS);
876
877 // Time resolution of the interpolator
878 tDelay += CLHEP::RandGauss::shoot(&fRandEngine, 0, fTResInterpolator);
879
880 // Propagation time
881 double tProp = CLHEP::RandGauss::shoot(fPropDelay, fPropDelayError) * r;
882
883 double t = t0 + tProp + tDelay;
884
885 // Get clock ticks
886 clock = clock.WithTime(t / 1e3); // WithTime takes microseconds
887
888 if (fUltraVerbose) mf::LogInfo("CRT")
889 << "CRT TIMING: t0=" << t0
890 << ", tDelayMean=" << tDelayMean << ", tDelayRMS=" << tDelayRMS
891 << ", tDelay=" << tDelay << ", tDelay(interp)="
892 << tDelay << ", tProp=" << tProp << ", t=" << t
893 << ", ticks=" << clock.Ticks() << "\n";
894
895 return (uint64_t)t;//clock.Ticks();
896 }//CRTDetSim::GetChannelTriggerTicks()
