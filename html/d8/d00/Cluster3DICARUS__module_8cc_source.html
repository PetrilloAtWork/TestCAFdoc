622 void
623 Cluster3DICARUS::produce(art::Event& evt)
624 {
625 mf::LogInfo("Cluster3DICARUS") << " *** Cluster3D::produce(...) [Run=" << evt.run()
626 << ", Event=" << evt.id().event() << "] Starting Now! *** "
627 << std::endl;
628
629 // Set up for monitoring the timing... at some point this should be removed in favor of
630 // external profilers
631 cet::cpu_timer theClockTotal;
632 cet::cpu_timer theClockFinish;
633
634 if (m_enableMonitoring) theClockTotal.start();
635
636 // This really only does anything if we are monitoring since it clears our tree variables
637 this->PrepareEvent(evt);
638
639 // Get instances of the primary data structures needed
640 reco::ClusterParametersList clusterParametersList;
641 IHit3DBuilder::RecobHitToPtrMap clusterHitToArtPtrMap;
642 std::unique_ptr<reco::HitPairList> hitPairList(new reco::HitPairList); // Potentially lots of hits, use heap instead of stack
643
644 // Call the algorithm that builds 3D hits and stores the hit collection
645 m_hit3DBuilderAlg->Hit3DBuilder(evt, *hitPairList, clusterHitToArtPtrMap);
646
647 // Only do the rest if we are not in the mode of only building space points (requested by ML folks)
648 if (!m_onlyMakSpacePoints)
649 {
650 // Call the main workhorse algorithm for building the local version of candidate 3D clusters
651 m_clusterAlg->Cluster3DHits(*hitPairList, clusterParametersList);
652
653 // Try merging clusters
654 m_clusterMergeAlg->ModifyClusters(clusterParametersList);
655
656 // Run the path finding
657 m_clusterPathAlg->ModifyClusters(clusterParametersList);
658 }
659
660 if (m_enableMonitoring) theClockFinish.start();
661
662 // Get the art ouput object
663 ArtOutputHandler output(evt, m_pathInstance, m_vertexInstance, m_extremeInstance);
664
665 // Call the module that does the end processing (of which there is quite a bit of work!)
666 // This goes here to insure that something is always written to the data store
667 auto const clockData = art::ServiceHandle<detinfo::DetectorClocksService const>()->DataFor(evt);
668 auto const detProp = art::ServiceHandle<detinfo::DetectorPropertiesService const>()->DataFor(evt, clockData);
669 util::GeometryUtilities const gser{*lar::providerFrom<geo::Geometry>(), clockData, detProp};
670
671 ProduceArtClusters(gser, output, *hitPairList, clusterParametersList, clusterHitToArtPtrMap);
672
673 // Output to art
674 output.outputObjects(m_onlyMakSpacePoints);

758 void
759 Cluster3DICARUS::findTrackSeeds(art::Event& evt,
760 reco::ClusterParameters& cluster,
761 IHit3DBuilder::RecobHitToPtrMap& hitToPtrMap,
762 std::vector<recob::Seed>& seedVec,
763 art::Assns<recob::Seed, recob::Hit>& seedHitAssns) const
764 {
770 // Make sure we are using the right pca
771 reco::PrincipalComponents& fullPCA = cluster.getFullPCA();
772 reco::PrincipalComponents& skeletonPCA = cluster.getSkeletonPCA();
773 reco::HitPairListPtr& hitPairListPtr = cluster.getHitPairListPtr();
774 reco::HitPairListPtr skeletonListPtr;
775
776 // We want to work with the "skeleton" hits so first step is to call the algorithm to
777 // recover only these hits from the entire input collection
778 m_skeletonAlg.GetSkeletonHits(hitPairListPtr, skeletonListPtr);
779
780 // Skeleton hits are nice but we can do better if we then make a pass through to "average"
781 // the skeleton hits position in the Y-Z plane
782 m_skeletonAlg.AverageSkeletonPositions(skeletonListPtr);
783
784 SeedHitPairListPairVec seedHitPairVec;
785
786 // Some combination of the elements below will be used to determine which seed finding algorithm
787 // to pursue below
788 float eigenVal0 = 3. * sqrt(skeletonPCA.getEigenValues()[0]);
789 float eigenVal1 = 3. * sqrt(skeletonPCA.getEigenValues()[1]);
790 float eigenVal2 = 3. * sqrt(skeletonPCA.getEigenValues()[2]);
791 float transRMS = std::hypot(eigenVal0, eigenVal1);
792
793 bool foundGoodSeed(false);
794
795 // Choose a method for finding the seeds based on the PCA that was run...
796 // Currently we have an ad hoc if-else block which I hope will be improved soon!
797 if (aParallelHitsCluster(fullPCA)) {
798 // In this case we have a track moving relatively parallel to the wire plane with lots of
799 // ambiguous 3D hits. Your best bet here is to use the "parallel hits" algorithm to get the
800 // best axis and seeds
801 // This algorithm does not fail (foundGoodSeed will always return true)
802 foundGoodSeed = m_parallelHitsAlg.findTrackSeeds(hitPairListPtr, skeletonPCA, seedHitPairVec);
803 }
804 else if (eigenVal2 > 40. && transRMS < 5.) {
805 // If the input cluster is relatively "straight" then chances are it is a single straight track,
806 // probably a CR muon, and we can simply use the PCA to determine the seed
807 // This algorithm will check both "ends" of the input hits and if the angles become inconsistent
808 // then it will "fail"
809 foundGoodSeed =
810 m_pcaSeedFinderAlg.findTrackSeeds(skeletonListPtr, skeletonPCA, seedHitPairVec);
811 }
812
813 // In the event the above two methods failed then we hit it with the real seed finder
814 if (!foundGoodSeed) {
815 // If here then we have a complicated 3D cluster and we'll use the hough transform algorithm to
816 // return a list of candidate seeds and seed hits
817 m_seedFinderAlg.findTrackSeeds(skeletonListPtr, skeletonPCA, seedHitPairVec);
818 }
819
820 // Go through the returned lists and build out the art friendly seeds and hits
821 for (const auto& seedHitPair : seedHitPairVec) {
822 seedVec.push_back(seedHitPair.first);
823
824 // We use a set here because our 3D hits can share 2D hits
825 // The set will make sure we get unique combinations of 2D hits
826 std::set<art::Ptr<recob::Hit>> seedHitSet;
827 for (const auto& hit3D : seedHitPair.second) {
828 for (const auto& hit2D : hit3D->getHits()) {
829 if (!hit2D) continue;
830
831 const recob::Hit* recobHit = hit2D->getHit();
832 seedHitSet.insert(hitToPtrMap[recobHit]);
833 }
834 }
835
836 RecobHitVector seedHitVec;
837 for (const auto& hit2D : seedHitSet)
838 seedHitVec.push_back(hit2D);
839
840 util::CreateAssn(evt, seedVec, seedHitVec, seedHitAssns);
841 }

853 void
854 Cluster3DICARUS::splitClustersWithMST(reco::ClusterParameters& clusterParameters,
855 reco::ClusterParametersList& clusterParametersList) const
856 {
857 // This is being left in place for future development. Essentially, it was an attempt to implement
858 // a Minimum Spanning Tree as a way to split a particular cluster topology, one where two straight
859 // tracks cross closely enought to appear as one cluster. As of Feb 2, 2015 I think the idea is still
860 // worth merit so am leaving this module in place for now.
861 //
862 // If this routine is called then we believe we have a cluster which needs splitting.
863 // The way we will do this is to use a Minimum Spanning Tree algorithm to associate all
864 // hits together by their distance apart. In theory, we should be able to split the cluster
865 // by finding the largest distance and splitting at that point.
866 //
867 // Typedef some data structures that we will use.
868 // Start with the adjacency map
869 typedef std::pair<float, const reco::ClusterHit3D*> DistanceHit3DPair;
870 typedef std::list<DistanceHit3DPair> DistanceHit3DPairList;
871 typedef std::map<const reco::ClusterHit3D*, DistanceHit3DPairList> Hit3DToDistanceMap;
872
873 // Now typedef the lists we'll keep
874 typedef std::list<const reco::ClusterHit3D*> Hit3DList;
875 typedef std::pair<Hit3DList::iterator, Hit3DList::iterator> Hit3DEdgePair;
876 typedef std::pair<float, Hit3DEdgePair> DistanceEdgePair;
877 typedef std::list<DistanceEdgePair> DistanceEdgePairList;
878
879 struct DistanceEdgePairOrder {
880 bool
881 operator()(const DistanceEdgePair& left, const DistanceEdgePair& right) const
882 {
883 return left.first > right.first;
884 }
885 };
886
887 // Recover the hits we'll work on.
888 // Note that we use on the skeleton hits so will need to recover them
889 reco::HitPairListPtr& hitPairListPtr = clusterParameters.getHitPairListPtr();
890 reco::HitPairListPtr skeletonListPtr;
891
892 // We want to work with the "skeleton" hits so first step is to call the algorithm to
893 // recover only these hits from the entire input collection
894 m_skeletonAlg.GetSkeletonHits(hitPairListPtr, skeletonListPtr);
895
896 // Skeleton hits are nice but we can do better if we then make a pass through to "average"
897 // the skeleton hits position in the Y-Z plane
898 m_skeletonAlg.AverageSkeletonPositions(skeletonListPtr);
899
900 // First task is to define and build the adjacency map
901 Hit3DToDistanceMap hit3DToDistanceMap;
902
903 for (reco::HitPairListPtr::const_iterator hit3DOuterItr = skeletonListPtr.begin();
904 hit3DOuterItr != skeletonListPtr.end();) {
905 const reco::ClusterHit3D* hit3DOuter = *hit3DOuterItr++;
906 DistanceHit3DPairList& outerHitList = hit3DToDistanceMap[hit3DOuter];
907 TVector3 outerPos(
908 hit3DOuter->getPosition()[0], hit3DOuter->getPosition()[1], hit3DOuter->getPosition()[2]);
909
910 for (reco::HitPairListPtr::const_iterator hit3DInnerItr = hit3DOuterItr;
911 hit3DInnerItr != skeletonListPtr.end();
912 hit3DInnerItr++) {
913 const reco::ClusterHit3D* hit3DInner = *hit3DInnerItr;
914 TVector3 innerPos(
915 hit3DInner->getPosition()[0], hit3DInner->getPosition()[1], hit3DInner->getPosition()[2]);
916 TVector3 deltaPos = innerPos - outerPos;
917 float hitDistance(float(deltaPos.Mag()));
918
919 if (hitDistance > 20.) continue;
920
921 hit3DToDistanceMap[hit3DInner].emplace_back(hitDistance, hit3DOuter);
922 outerHitList.emplace_back(hitDistance, hit3DInner);
923 }
924
925 // Make sure our membership bit is clear
926 hit3DOuter->clearStatusBits(reco::ClusterHit3D::SELECTEDBYMST);
927 }
928
929 // Make pass through again to order each of the lists
930 for (auto& mapPair : hit3DToDistanceMap) {
931 mapPair.second.sort(Hit3DDistanceOrder());
932 }
933
934 // Get the containers for the MST to operate on/with
935 Hit3DList hit3DList;
936 DistanceEdgePairList distanceEdgePairList;
937
938 // Initialize with first element
939 hit3DList.emplace_back(skeletonListPtr.front());
940 distanceEdgePairList.emplace_back(
941 DistanceEdgePair(0., Hit3DEdgePair(hit3DList.begin(), hit3DList.begin())));
942
943 skeletonListPtr.front()->setStatusBit(reco::ClusterHit3D::SELECTEDBYMST);
944
945 float largestDistance(0.);
946 // float averageDistance(0.); // unused
947
948 // Now run the MST
949 // Basically, we loop until the MST list is the same size as the input list
950 while (hit3DList.size() < skeletonListPtr.size()) {
951 Hit3DList::iterator bestHit3DIter = hit3DList.begin();
952 float bestDist = 10000000.;
953
954 // Loop through all hits currently in the list and look for closest hit not in the list
955 for (Hit3DList::iterator hit3DIter = hit3DList.begin(); hit3DIter != hit3DList.end();
956 hit3DIter++) {
957 const reco::ClusterHit3D* hit3D = *hit3DIter;
958
959 // For the given 3D hit, find the closest to it that is not already in the list
960 DistanceHit3DPairList& nearestList = hit3DToDistanceMap[hit3D];
961
962 while (!nearestList.empty()) {
963 const reco::ClusterHit3D* hit3DToCheck = nearestList.front().second;
964
965 if (!hit3DToCheck->bitsAreSet(reco::ClusterHit3D::SELECTEDBYMST)) {
966 if (nearestList.front().first < bestDist) {
967 bestHit3DIter = hit3DIter;
968 bestDist = nearestList.front().first;
969 }
970 break;
971 }
972
973 nearestList.pop_front();
974 }
975 }
976
977 if (bestDist > largestDistance) largestDistance = bestDist;
978
979 // averageDistance += bestDist; // unused
980
981 // Now we add the best hit not in the list to our list, keep track of the distance
982 // to the object it was closest to
983 const reco::ClusterHit3D* bestHit3D = *bestHit3DIter; // "best" hit already in the list
984 const reco::ClusterHit3D* nextHit3D =
985 hit3DToDistanceMap[bestHit3D].front().second; // "next" hit we are adding to the list
986
987 Hit3DList::iterator nextHit3DIter = hit3DList.insert(hit3DList.end(), nextHit3D);
988
989 distanceEdgePairList.emplace_back(bestDist, Hit3DEdgePair(bestHit3DIter, nextHit3DIter));
990
991 nextHit3D->setStatusBit(reco::ClusterHit3D::SELECTEDBYMST);
992 }
993
994 // averageDistance /= float(hit3DList.size()); // unused
995
996 float thirdDist = 2. * sqrt(clusterParameters.getSkeletonPCA().getEigenValues()[2]);
997
998 // Ok, find the largest distance in the iterator map
999 distanceEdgePairList.sort(DistanceEdgePairOrder());
1000
1001 DistanceEdgePairList::iterator largestDistIter = distanceEdgePairList.begin();
1002
1003 for (DistanceEdgePairList::iterator edgeIter = distanceEdgePairList.begin();
1004 edgeIter != distanceEdgePairList.end();
1005 edgeIter++) {
1006 if (edgeIter->first < thirdDist) break;
1007
1008 largestDistIter = edgeIter;
1009 }
1010
1011 reco::HitPairListPtr::iterator breakIter = largestDistIter->second.second;
1012 reco::HitPairListPtr bestList;
1013
1014 bestList.resize(std::distance(hit3DList.begin(), breakIter));
1015
1016 std::copy(hit3DList.begin(), breakIter, bestList.begin());
1017
1018 // Remove from the grand hit list and see what happens...
1019 // The pieces below are incomplete and were really for testing only.
1020 hitPairListPtr.sort();
1021 bestList.sort();
1022
1023 reco::HitPairListPtr::iterator newListEnd = std::set_difference(hitPairListPtr.begin(),
1024 hitPairListPtr.end(),
1025 bestList.begin(),
1026 bestList.end(),
1027 hitPairListPtr.begin());
1028
1029 hitPairListPtr.erase(newListEnd, hitPairListPtr.end());
1030 }

1046 void
1047 Cluster3DICARUS::splitClustersWithHough(reco::ClusterParameters& clusterParameters,
1048 reco::ClusterParametersList& clusterParametersList) const
1049 {
1050 // @brief A method for splitted "crossed tracks" clusters into separate clusters
1051 //
1052 // If this routine is called then we believe we have a cluster which needs splitting.
1053 // The specific topology we are looking for is two long straight tracks which cross at some
1054 // point in close proximity so their hits were joined into a single 3D cluster. The method
1055 // to split this topology is to let the hough transform algorithm find the two leading candidates
1056 // and then to see if we use those to build two clusters instead of one.
1057
1058 // Recover the hits we'll work on.
1059 // Note that we use on the skeleton hits so will need to recover them
1060 reco::HitPairListPtr& hitPairListPtr = clusterParameters.getHitPairListPtr();
1061 reco::HitPairListPtr skeletonListPtr;
1062
1063 // We want to work with the "skeleton" hits so first step is to call the algorithm to
1064 // recover only these hits from the entire input collection
1065 m_skeletonAlg.GetSkeletonHits(hitPairListPtr, skeletonListPtr);
1066
1067 // Skeleton hits are nice but we can do better if we then make a pass through to "average"
1068 // the skeleton hits position in the Y-Z plane
1069 m_skeletonAlg.AverageSkeletonPositions(skeletonListPtr);
1070
1071 // Define the container for our lists of hits
1072 reco::HitPairListPtrList hitPairListPtrList;
1073
1074 // Now feed this to the Hough Transform to find candidate straight lines
1075 m_seedFinderAlg.findTrackHits(
1076 skeletonListPtr, clusterParameters.getSkeletonPCA(), hitPairListPtrList);
1077
1078 // We need at least two lists or else there is nothing to do
1079 if (hitPairListPtrList.size() < 2) return;
1080
1081 // The game plan will be the following:
1082 // 1) Take the first list of hits and run the PCA on this to get an axis
1083 // - Then calculate the 3d doca for ALL hits in the cluster to this axis
1084 // - Move all hits within "3 sigam" of the axis to a new list
1085 // 2) run the PCA on the second list of hits to get that axis
1086 // - Then calculate the 3d doca for all hits in our first list
1087 // - Copy hits in the first list which are within 3 sigma of the new axis
1088 // back into our original cluster - these are shared hits
1089 reco::HitPairListPtrList::iterator hitPairListIter = hitPairListPtrList.begin();
1090 reco::HitPairListPtr& firstHitList = *hitPairListIter++;
1091 reco::PrincipalComponents firstHitListPCA;
1092
1093 m_pcaAlg.PCAAnalysis_3D(firstHitList, firstHitListPCA);
1094
1095 // Make sure we have a successful calculation.
1096 if (firstHitListPCA.getSvdOK()) {
1097 // The fill routines below will expect to see unused 2D hits so we need to clear the
1098 // status bits... and I am not sure of a better way...
1099 for (const auto& hit3D : hitPairListPtr) {
1100 for (const auto& hit2D : hit3D->getHits())
1101 if (hit2D) hit2D->clearStatusBits(0x1);
1102 }
1103
1104 // Calculate the 3D doca's for the hits which were used to make this PCA
1105 m_pcaAlg.PCAAnalysis_calc3DDocas(firstHitList, firstHitListPCA);
1106
1107 // Divine from the ether some maximum allowed range for transfering hits
1108 float allowedHitRange = 6. * firstHitListPCA.getAveHitDoca();
1109
1110 // Now go through and calculate the 3D doca's for ALL the hits in the original cluster
1111 m_pcaAlg.PCAAnalysis_calc3DDocas(hitPairListPtr, firstHitListPCA);
1112
1113 // Let's make a new cluster to hold the hits
1114 clusterParametersList.push_back(reco::ClusterParameters());
1115
1116 // Can we get a reference to what we just created?
1117 reco::ClusterParameters& newClusterParams = clusterParametersList.back();
1118
1119 reco::HitPairListPtr& newClusterHitList = newClusterParams.getHitPairListPtr();
1120
1121 newClusterHitList.resize(hitPairListPtr.size());
1122
1123 // Do the actual copy of the hits we want
1124 reco::HitPairListPtr::iterator newListEnd = std::copy_if(hitPairListPtr.begin(),
1125 hitPairListPtr.end(),
1126 newClusterHitList.begin(),
1127 CopyIfInRange(allowedHitRange));
1128
1129 // Shrink to fit
1130 newClusterHitList.resize(std::distance(newClusterHitList.begin(), newListEnd));
1131
1132 // And now remove these hits from the original cluster
1133 hitPairListPtr.remove_if(CopyIfInRange(allowedHitRange));
1134
1135 // Get an empty hit to cluster map...
1136 reco::Hit2DToClusterMap hit2DToClusterMap;
1137
1138 // Now "fill" the cluster parameters but turn off the hit rejection
1139 m_clusterBuilder->FillClusterParams(newClusterParams, hit2DToClusterMap, 0., 1.);
1140
1141 // Set the skeleton pca to the value calculated above on input
1142 clusterParameters.getSkeletonPCA() = firstHitListPCA;
1143
1144 // We are done with splitting out one track. Because the two tracks cross in
1145 // close proximity, this is the one case where we might consider sharing 3D hits
1146 // So let's make a little detour here to try to copy some of those hits back into
1147 // the main hit list
1148 reco::HitPairListPtr& secondHitList = *hitPairListIter;
1149 reco::PrincipalComponents secondHitListPCA;
1150
1151 m_pcaAlg.PCAAnalysis_3D(secondHitList, secondHitListPCA);
1152
1153 // Make sure we have a successful calculation.
1154 if (secondHitListPCA.getSvdOK()) {
1155 // Calculate the 3D doca's for the hits which were used to make this PCA
1156 m_pcaAlg.PCAAnalysis_calc3DDocas(secondHitList, secondHitListPCA);
1157
1158 // Since this is the "other" cluster, we'll be a bit more generous in adding back hits
1159 float newAllowedHitRange = 6. * secondHitListPCA.getAveHitDoca();
1160
1161 // Go through and calculate the 3D doca's for the hits in our new candidate cluster
1162 m_pcaAlg.PCAAnalysis_calc3DDocas(newClusterHitList, secondHitListPCA);
1163
1164 // Create a temporary list to fill with the hits we might want to save
1165 reco::HitPairListPtr tempHitList(newClusterHitList.size());
1166
1167 // Do the actual copy of the hits we want...
1168 reco::HitPairListPtr::iterator tempListEnd =
1169 std::copy_if(newClusterHitList.begin(),
1170 newClusterHitList.end(),
1171 tempHitList.begin(),
1172 CopyIfInRange(newAllowedHitRange));
1173
1174 hitPairListPtr.insert(hitPairListPtr.end(), tempHitList.begin(), tempListEnd);
1175 }
1176
1177 // Of course, now we need to modify the original cluster parameters
1178 reco::ClusterParameters originalParams(hitPairListPtr);
1179
1180 // Now "fill" the cluster parameters but turn off the hit rejection
1181 m_clusterBuilder->FillClusterParams(originalParams, hit2DToClusterMap, 0., 1.);
1182
1183 // Overwrite original cluster parameters with our new values
1184 clusterParameters.getClusterParams() = originalParams.getClusterParams();
1185 clusterParameters.getFullPCA() = originalParams.getFullPCA();
1186 clusterParameters.getSkeletonPCA() = secondHitListPCA;
1187 }

1246 else {
1247 // Set up to keep track of parent/daughters

1249 size_t numTotalDaughters = countUltimateDaughters(clusterParameters);

1253 output,
1254 clusterParameters,
1255 pfParticleIdx,
1256 idxToPCAMap,
1257 hitToPtrMap,
1258 hit3DToSPPtrMap,
1259 best3DToSPPtrMap);

1264 for (auto& idxToPCA : idxToPCAMap)
1265 daughterVec.emplace_back(idxToPCA.first);

1269 13, pfParticleIdx, recob::PFParticle::kPFParticlePrimary, daughterVec);
1270 output.artPFParticleVector->push_back(pfParticle);

1274 double avePosition[] = {0., 0., 0.};
1275
1276 eigenVecs.resize(3);
1277

1279
1280 for (size_t outerIdx = 0; outerIdx < 3; outerIdx++) {
1281 avePosition[outerIdx] = skeletonPCA.getAvePosition()[outerIdx];
1282 eigenVals[outerIdx] = skeletonPCA.getEigenValues()[outerIdx];
1283
1284 eigenVecs[outerIdx].resize(3);
1285
1286 // Be careful here... eigen stores in column major order buy default
1287 for (size_t innerIdx = 0; innerIdx < 3; innerIdx++)
1288 eigenVecs[outerIdx][innerIdx] = skeletonPCA.getEigenVectors().row(outerIdx)[innerIdx];
1289 }
1290
1291 recob::PCAxis skelPcAxis(skeletonPCA.getSvdOK(),
1292 skeletonPCA.getNumHitsUsed(),
1293 eigenVals,
1294 eigenVecs,
1295 avePosition,
1296 skeletonPCA.getAveHitDoca(),
1297 output.artPCAxisVector->size());
1298
1299 output.artPCAxisVector->push_back(skelPcAxis);
1300
1301 reco::PrincipalComponents& fullPCA = clusterParameters.getFullPCA();
1302
1303 for (size_t outerIdx = 0; outerIdx < 3; outerIdx++) {
1304 avePosition[outerIdx] = fullPCA.getAvePosition()[outerIdx];
1305 eigenVals[outerIdx] = fullPCA.getEigenValues()[outerIdx];
1306
1307 for (size_t innerIdx = 0; innerIdx < 3; innerIdx++)
1308 eigenVecs[outerIdx][innerIdx] = fullPCA.getEigenVectors().row(outerIdx)(innerIdx);
1309 }
1310
1311 recob::PCAxis fullPcAxis(fullPCA.getSvdOK(),
1312 fullPCA.getNumHitsUsed(),
1313 eigenVals,
1314 eigenVecs,
1315 avePosition,
1316 fullPCA.getAveHitDoca(),
1317 output.artPCAxisVector->size());
1318
1319 output.artPCAxisVector->push_back(fullPcAxis);
1320
1321 // Create associations to the PFParticle
1322 output.makePFPartPCAAssns();
1323
1324 // Make associations to all space points for this cluster
1325 MakeAndSaveSpacePoints(output,
1326 *output.artSpacePointVector.get(),
1327 *output.artSPHitAssociations.get(),
1328 clusterParameters.getHitPairListPtr(),
1329 hitToPtrMap,
1330 hit3DToSPPtrMap);
1331
1332 // Get the extreme points
1333 MakeAndSaveKinkPoints(output,
1334 clusterParameters.getConvexHull().getConvexHullKinkPoints());
1335
1336 // Build the edges now
1337 size_t edgeStart(output.artEdgeVector->size());
1338
1339 for (const auto& edge : clusterParameters.getConvexHull().getConvexHullEdgeList()) {
1340 RecobSpacePointVector spacePointVec;
1341
1342 try {
1343 spacePointVec.push_back(hit3DToSPPtrMap.at(std::get<0>(edge)));
1344 spacePointVec.push_back(hit3DToSPPtrMap.at(std::get<1>(edge)));
1345 }
1346 catch (...) {
1347 std::cout << "Caught exception in looking up space point ptr... " << std::get<0>(edge)
1348 << ", " << std::get<1>(edge) << std::endl;
1349 continue;
1350 }
1351
1352 output.artEdgeVector->emplace_back(std::get<2>(edge),
1353 spacePointVec[0].key(),
1354 spacePointVec[1].key(),
1355 output.artEdgeVector->size());
1356
1357 output.makeEdgeSpacePointAssns(
1358 *output.artEdgeVector.get(), spacePointVec, *output.artEdgeSPAssociations.get());
1359 }
1360
1361 output.makePFPartEdgeAssns(
1362 *output.artEdgeVector.get(), *output.artPFPartEdgeAssociations.get(), edgeStart);
1363 }
1364 }
1365 }
1366
1367 mf::LogDebug("Cluster3DICARUS") << "--> Preparing to save the remaining space points, there are " << hitPairVector.size() << std::endl;
1368
1369 // Right now error matrix is uniform...
1370 int nFreePoints(0);
1371
1372 // Run through the HitPairVector and add any unused hit pairs to the list
1373 for (auto& hitPair : hitPairVector)
1374 {
1375 if (hitPair.bitsAreSet(reco::ClusterHit3D::MADESPACEPOINT)) continue;
1376
1377 double spacePointPos[] = {
1378 hitPair.getPosition()[0], hitPair.getPosition()[1], hitPair.getPosition()[2]};
1379 double spacePointErr[] = {1., 0., 0., 1., 0., 1.};
1380 double chisq = hitPair.getHitChiSquare();
1381
1382 RecobHitVector recobHits;
1383
1384 for (const auto hit : hitPair.getHits()) {
1385 if (!hit) continue;
1386
1387 art::Ptr<recob::Hit> hitPtr = hitToPtrMap[hit->getHit()];
1388 recobHits.push_back(hitPtr);
1389 }
1390
1391 nFreePoints++;
1392
1393 spacePointErr[1] = hitPair.getTotalCharge();
1394 spacePointErr[3] = hitPair.getChargeAsymmetry();
1395
1396 output.artSpacePointVector->push_back(
1397 recob::SpacePoint(spacePointPos, spacePointErr, chisq, output.artSpacePointVector->size()));
1398
1399 if (!recobHits.empty())
1400 output.makeSpacePointHitAssns(
1401 *output.artSpacePointVector.get(), recobHits, *output.artSPHitAssociations.get());
1402 }
1403
1404 mf::LogDebug("Cluster3DICARUS") << "++++>>>> total num hits: " << hitPairVector.size()
1405 << ", num free: " << nFreePoints;
1406 }

1462 Cluster3DICARUS::ConvertToArtOutput(util::GeometryUtilities const& gser,
1463 ArtOutputHandler& output,
1464 reco::ClusterParameters& clusterParameters,
1465 size_t pfParticleParent,

1467 Hit3DToSPPtrMap& hit3DToSPPtrMap,
1468 Hit3DToSPPtrMap& best3DToSPPtrMap) const
1469 {
1470 // prepare the algorithm to compute the cluster characteristics;
1471 // we use the "standard" one here, except that we override selected items
1472 // (so, thanks to metaprogramming, we finally have wrappers of wrappers);
1473 // configuration would happen here, but we are using the default
1474 // configuration for that algorithm
1475 using OverriddenClusterParamsAlg_t =
1476 cluster::OverriddenClusterParamsAlg<cluster::StandardClusterParamsAlg>;

1485 // candidate cluster. The second will be that derived from just using the "skeleton" hits.

1489
1490 size_t clusterStart = output.artClusterVector->size();
1491
1492 // Start loop over views to build out the hit lists and the 2D cluster objects
1493 for (const auto& clusParametersPair : clusterParameters.getClusterParams()) {
1494 const reco::RecobClusterParameters& clusParams = clusParametersPair.second;
1495
1496 // Protect against a missing view
1497 if (clusParams.m_view == geo::kUnknown) continue;
1498
1499 // Get a vector of hit pointers to seed the cluster algorithm
1500 std::vector<const recob::Hit*> recobHitVec(clusParams.m_hitVector.size());
1501
1502 std::transform(clusParams.m_hitVector.begin(),
1503 clusParams.m_hitVector.end(),
1504 recobHitVec.begin(),
1505 [](const auto& hit2D) { return hit2D->getHit(); });

1509 double endWire(clusParams.m_endWire);
1510 double startTime(clusParams.m_startTime);
1511 double endTime(clusParams.m_endTime);
1512
1513 // feed the algorithm with all the cluster hits
1514 ClusterParamAlgo.ImportHits(gser, recobHitVec);
1515
1516 // create the recob::Cluster directly in the vector
1517 cluster::ClusterCreator artCluster(gser,
1518 ClusterParamAlgo, // algo
1519 startWire, // start_wire
1520 0., // sigma_start_wire
1521 startTime, // start_tick
1522 clusParams.m_sigmaStartTime, // sigma_start_tick
1523 endWire, // end_wire
1524 0., // sigma_end_wire,
1525 endTime, // end_tick
1526 clusParams.m_sigmaEndTime, // sigma_end_tick

1531 );
1532

1540
1541 art::Ptr<recob::Hit> hitPtr = hitToPtrMap[hit2D->getHit()];

1543 }

1558
1559 // Assume that if we are a daughter particle then there is a set of "best" 3D points and the convex hull will have been calculated from them

1561 std::string instance("");

1568 auto spEdgeAssns = output.artEdgeSPAssociations.get();
1569 auto spHitAssns = output.artSPHitAssociations.get();
1570 auto pfPartSPAssns = output.artPFPartSPAssociations.get();

1573 int spacePointStart = spVector->size();
1574
1575 MakeAndSaveSpacePoints(
1576 output, *spVector, *spHitAssns, *hit3DListPtr, hitToPtrMap, *hit3DToPtrMap);
1577
1578 // And make sure to have associations to the PFParticle
1579 output.makePFPartSpacePointAssns(*spVector, *pfPartSPAssns, spacePointStart);
1580
1581 // If they exist, make space points for the Path points
1582 if (!clusterParameters.getBestHitPairListPtr().empty()) {
1583 instance = m_pathInstance;
1584 hit3DListPtr = &clusterParameters.getBestHitPairListPtr();
1585 hit3DToPtrMap = &best3DToSPPtrMap;
1586
1587 spVector = output.artPathPointVector.get();

1600 output.makePFPartSpacePointAssns(*spVector, *pfPartSPAssns, spacePointStart, instance);

1606
1607 for (const auto& edge : clusterParameters.getBestEdgeList()) {
1608 RecobSpacePointVector spacePointVec;

1611 spacePointVec.push_back(hit3DToPtrMap->at(std::get<0>(edge)));

1630 // First need some float to double conversion containers
1631 recob::PCAxis::EigenVectors eigenVecs;
1632 double eigenVals[] = {0., 0., 0.};

1648 skeletonPCA.getNumHitsUsed(),
1649 eigenVals,
1650 eigenVecs,
1651 avePosition,
1652 skeletonPCA.getAveHitDoca(),
1653 output.artPCAxisVector->size());
1654
1655 output.artPCAxisVector->push_back(skelPcAxis);

1657 for (size_t outerIdx = 0; outerIdx < 3; outerIdx++) {
1658 avePosition[outerIdx] = fullPCA.getAvePosition()[outerIdx];
1659 eigenVals[outerIdx] = fullPCA.getEigenValues()[outerIdx];

1662 eigenVecs[outerIdx][innerIdx] = fullPCA.getEigenVectors().row(outerIdx)(innerIdx);

1671 output.artPCAxisVector->size());

1673 output.artPCAxisVector->push_back(fullPcAxis);

1676 output.makePFPartPCAAssns();
1677 output.makePFPartSeedAssns(numSeedsStart);
1678 output.makePFPartClusterAssns(clusterStart);
1679
1680 return pfParticleIdx;

1683 void
1684 Cluster3DICARUS::MakeAndSaveSpacePoints(ArtOutputHandler& output,
1685 std::vector<recob::SpacePoint>& spacePointVec,
1686 art::Assns<recob::Hit, recob::SpacePoint>& spHitAssns,
1687 reco::HitPairListPtr& clusHitPairVector,
1688 IHit3DBuilder::RecobHitToPtrMap& hitToPtrMap,
1689 Hit3DToSPPtrMap& hit3DToSPPtrMap,
1690 const std::string& instance) const
1691 {
1692 // Reserve space...
1693 spacePointVec.reserve(spacePointVec.size() + clusHitPairVector.size());
1694
1695 // Right now error matrix is uniform...
1696 double spError[] = {0., 0., 0., 0., 0., 0.};
1697
1698 // Copy these hits to the vector to be stored with the event
1699 for (auto& hitPair : clusHitPairVector) {
1700 // Skip those space points that have already been created
1701 if (hit3DToSPPtrMap.find(hitPair) != hit3DToSPPtrMap.end()) continue;
1702
1703 // Don't make space point if this hit was "rejected"
1704 if (hitPair->bitsAreSet(reco::ClusterHit3D::REJECTEDHIT)) continue;
1705
1706 double chisq = hitPair->getHitChiSquare(); // secret handshake...
1707
1708 // Mark this hit pair as in use
1709 hitPair->setStatusBit(reco::ClusterHit3D::MADESPACEPOINT);
1710
1711 // Create and store the space point
1712 size_t spacePointID = spacePointVec.size();
1713 double spacePointPos[] = {
1714 hitPair->getPosition()[0], hitPair->getPosition()[1], hitPair->getPosition()[2]};
1715
1716 spError[1] = hitPair->getTotalCharge();
1717 spError[3] = hitPair->getChargeAsymmetry();
1718
1719 spacePointVec.emplace_back(spacePointPos, spError, chisq, spacePointID);
1720
1721 // Update mapping
1722 hit3DToSPPtrMap[hitPair] = output.makeSpacePointPtr(spacePointID, instance);
1723
1724 // space point hits associations
1725 RecobHitVector recobHits;
1726
1727 for (const auto& hit : hitPair->getHits()) {
1728 if (!hit) continue;
1729
1730 art::Ptr<recob::Hit> hitPtr = hitToPtrMap[hit->getHit()];
1731 recobHits.push_back(hitPtr);
1732 }
1733
1734 if (!recobHits.empty())
1735 output.makeSpacePointHitAssns(spacePointVec, recobHits, spHitAssns, instance);
1736 }
1737
1738 return;
1739 }

1768 void
1769 Cluster3DICARUS::MakeAndSaveVertexPoints(ArtOutputHandler& output,
1770 dcel2d::VertexList& vertexList,
1771 dcel2d::HalfEdgeList& halfEdgeList) const
1772 {
1773 // We actually do two things here:
1774 // 1) Create space points to represent the vertex locations of the voronoi diagram
1775 // 2) Create the edges that link the space points together
1776
1777 // Set up the space point creation
1778 // Right now error matrix is uniform...
1779 double spError[] = {1., 0., 1., 0., 0., 1.};
1780 double chisq = 1.;
1781
1782 // Keep track of the vertex to space point association
1783 std::map<const dcel2d::Vertex*, size_t> vertexToSpacePointMap;
1784
1785 // Copy these hits to the vector to be stored with the event
1786 for (auto& vertex : vertexList) {
1787 const dcel2d::Coords& coords = vertex.getCoords();
1788
1789 // Create and store the space point
1790 double spacePointPos[] = {coords[0], coords[1], coords[2]};
1791
1792 vertexToSpacePointMap[&vertex] = output.artVertexPointVector->size();
1793
1794 output.artVertexPointVector->emplace_back(
1795 spacePointPos, spError, chisq, output.artVertexPointVector->size());
1796 }
1797
1798 // Try to avoid double counting
1799 std::set<const dcel2d::HalfEdge*> halfEdgeSet;
1800
1801 // Build the edges now
1802 for (const auto& halfEdge : halfEdgeList) {
1803 // Recover twin
1804 const dcel2d::HalfEdge* twin = halfEdge.getTwinHalfEdge();
1805
1806 // It can happen that we have no twin... and also check that we've not been here before
1807 if (twin && halfEdgeSet.find(twin) == halfEdgeSet.end()) {
1808 // Recover the vertices
1809 const dcel2d::Vertex* fromVertex = twin->getTargetVertex();
1810 const dcel2d::Vertex* toVertex = halfEdge.getTargetVertex();
1811
1812 // It can happen for the open edges that there is no target vertex
1813 if (!toVertex || !fromVertex) continue;
1814
1815 if (vertexToSpacePointMap.find(fromVertex) == vertexToSpacePointMap.end() ||
1816 vertexToSpacePointMap.find(toVertex) == vertexToSpacePointMap.end())
1817 continue;
1818
1819 // Need the distance between vertices
1820 Eigen::Vector3f distVec = toVertex->getCoords() - fromVertex->getCoords();
1821
1822 output.artVertexEdgeVector->emplace_back(distVec.norm(),
1823 vertexToSpacePointMap[fromVertex],
1824 vertexToSpacePointMap[toVertex],
1825 output.artEdgeVector->size());
1826
1827 halfEdgeSet.insert(&halfEdge);
1828 }
1829 }
1830
1831 return;
1832 }

1834 void
1835 Cluster3DICARUS::MakeAndSavePCAPoints(ArtOutputHandler& output,
1836 const reco::PrincipalComponents& fullPCA,
1837 IdxToPCAMap& idxToPCAMap) const
1838 {
1839 // We actually do two things here:
1840 // 1) Create space points from the centroids of the PCA for each cluster
1841 // 2) Create the edges that link the space points together
1842
1843 // The first task is to put the list of PCA's into some semblance of order... they may be
1844 // preordered by likely they are piecewise ordered so fix that here
1845

1847 Eigen::Vector3f avePosition(
1848 fullPCA.getAvePosition()[0], fullPCA.getAvePosition()[1], fullPCA.getAvePosition()[2]);

1852 using DocaToPCAVec = std::vector<DocaToPCAPair>;

1857 for (const auto& idxToPCA : idxToPCAMap) {
1858 const reco::PrincipalComponents* pca = idxToPCA.second;
1859

1861 // Start by getting this hits position
1862 Eigen::Vector3f pcaPos(

1864

1866 Eigen::Vector3f pcaToAveVec = pcaPos - avePosition;
1867

1869 float arclenToPoca = pcaToAveVec.dot(axisDirVec);
1870
1871 docaToPCAVec.emplace_back(arclenToPoca, pca);
1872 }

1874 std::sort(docaToPCAVec.begin(), docaToPCAVec.end(), [](const auto& left, const auto& right) {
1875 return left.first < right.first;
1876 });
1877
1878 // Set up the space point creation
1879 // Right now error matrix is uniform...
1880 double spError[] = {1., 0., 1., 0., 0., 1.};
1881 double chisq = 1.;
1882
1883 const reco::PrincipalComponents* lastPCA(nullptr);
1884
1885 // Set up to loop through the clusters
1886 for (const auto& docaToPCAPair : docaToPCAVec) {
1887 // Recover the PCA for this cluster
1888 const reco::PrincipalComponents* curPCA = docaToPCAPair.second;
1889
1890 if (lastPCA) {
1891 double lastPointPos[] = {
1892 lastPCA->getAvePosition()[0], lastPCA->getAvePosition()[1], lastPCA->getAvePosition()[2]};
1893 size_t lastPointBin = output.artVertexPointVector->size();
1894 double curPointPos[] = {
1895 curPCA->getAvePosition()[0], curPCA->getAvePosition()[1], curPCA->getAvePosition()[2]};
1896 size_t curPointBin = lastPointBin + 1;
1897
1898 output.artVertexPointVector->emplace_back(lastPointPos, spError, chisq, lastPointBin);
1899 output.artVertexPointVector->emplace_back(curPointPos, spError, chisq, curPointBin);
1900
1901 Eigen::Vector3f distVec(curPointPos[0] - lastPointPos[0],
1902 curPointPos[1] - lastPointPos[1],
1903 curPointPos[2] - lastPointPos[2]);
1904
1905 output.artVertexEdgeVector->emplace_back(
1906 distVec.norm(), lastPointBin, curPointBin, output.artEdgeVector->size());
1907 }
1908
1909 lastPCA = curPCA;
1910 }
