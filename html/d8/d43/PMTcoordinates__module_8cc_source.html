203 vertex_z=mct->GetParticle(0).Vz();

205 if (event_type==12||event_type==-12||event_type==14||event_type==-14||event_type==16||event_type==-16)

207 is_Neutrino=1;
208 Neutrino_Interaction=mct->GetNeutrino().InteractionType();

216// }

302 PMTy[channel] = xyz.Y();

306 reco_barycentre_z = reco_barycentre_z + PMTz[channel]*photons_collected[channel];

308 std::cout << " channel " << channel << " total photons " << total_coll_photons << std::endl;

329 if (PMTx[channel]>0){Cryostat[channel]=1;}

331 if (PMTx[channel]<-200){TPC[channel]=0;}
332 if (PMTx[channel]>-200 && PMTx[channel]<0){TPC[channel]=1;}
333 if (PMTx[channel]<200 && PMTx[channel]>0){TPC[channel]=2;}
334 if (PMTx[channel]>200){TPC[channel]=3;}
335
336}
337
338//total_coll_photons = total_coll_photons;
339
340std::cout << " fotoni finale = " <<total_coll_photons <<std::endl;
341
342reco_barycentre_y = reco_barycentre_y/total_coll_photons;
343reco_barycentre_z = reco_barycentre_z/total_coll_photons;
344
345
346PMT_error_y = reco_barycentre_y-true_barycentre_y;
347PMT_error_z = reco_barycentre_z-true_barycentre_z;
348PMT_total_error = sqrt((PMT_error_y*PMT_error_y)+(PMT_error_z*PMT_error_z));
