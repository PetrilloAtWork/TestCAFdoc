389 fEvent = e.id().event();

391
392 //Fetch trigger info and if MC check whether this event triggered
393 art::Handle const triggerHandle
394 = e.getHandle<std::vector<raw::Trigger>>(fTriggerLabel);
395 double triggerTime = -9999.;
396 if ( triggerHandle.isValid() && triggerHandle->size() >= 1 ) {
397 raw::Trigger const& trigger = triggerHandle->at(0);
398 triggerTime = trigger.TriggerTime();
399 if ( fVerbose ) std::cout << "Valid trigger product found. Trigger time: " << trigger.TriggerTime() << ", Beam gate time: " << trigger.BeamGateTime() << ", Difference: " << trigger.TriggerTime() - trigger.BeamGateTime() << std::endl;

401 else if ( fVerbose ) std::cout << "No valid trigger product found for this event!" << std::endl;
402
403 //Infrastructure for checking allowed time range of a slice
404 detinfo::DetectorTimings const detTimings{ fDetClocks.DataFor(e) };
405 detinfo::DetectorPropertiesData const& detProp { fDetProp.DataFor(e, detTimings.clockData()) };
406 lar::util::TrackTimeInterval const timeIntervals = fTimeIntervalMaker(detProp, detTimings);
407
408 //Initialize new data products
409 auto matchInfoVector = std::make_unique< std::vector<sbn::TPCPMTBarycenterMatch> >();

412 auto flashAssns = std::make_unique< art::Assns<sbn::TPCPMTBarycenterMatch, recob::OpFlash> >();
413
414 //For InputTag...
415 for ( const std::string& inputTag : fInputTags ) {
416 //East-->0, West-->1
417 fCryo = ( inputTag.find("W") != std::string::npos ) ? 1 : 0;
418
419
420/* ~~~~~~~~~~~~~~~~~~~~ Flash Section
421 *
422 * Here we gather the OpFlashes found in this cryostat and their OpHits
423 * We iterate through the flashes to identify a triggering flash
424 */
425
426 //Fetch the flashes and their associated hits, pointer vector needed for generating associations
427 art::Handle const flashHandle
428 = e.getHandle<std::vector<recob::OpFlash>>(fOpFlashLabel + inputTag);
429 art::FindMany<recob::OpHit> fmOpHits(flashHandle, e, fOpFlashLabel + inputTag);
430 int nFlashes = (*flashHandle).size();
431
432 std::array<double, 2> triggerFlashCenter = {-9999., -9999.};
433 //If triggered event..
434 if ( triggerTime >= 0. ) {
435 double minTimeDiff = 1.6;
436 //For flash...
437 for ( int i = 0; i < nFlashes; i++ ) {
438 const recob::OpFlash &flash = (*flashHandle)[i];
439
440 //Identify the triggering flash
441 double timeDiff = abs( (triggerTime - flash.AbsTime()) - fTriggerDelay );
442 if ( timeDiff < fTriggerTolerance && timeDiff < minTimeDiff ) {
443 triggerFlashCenter = {flash.YCenter(), flash.ZCenter()};
444 minTimeDiff = timeDiff;
445 }
446
447 } //End for flash
448 } //End if triggered event
449
450 if ( fVerbose ) std::cout << "Event: " << fEvent << ", Cryo: " << inputTag << ", nFlashes: " << nFlashes << ", Triggering flash center Y: " << triggerFlashCenter[0] << ", Triggering flash center Z: " << triggerFlashCenter[1] << std::endl;
451
452
453/* ~~~~~~~~~~~~~~~~~~~~ TPC Section
454 * Here we start by gathering the Slices in the event
455 * For each slice, the charge centroid is first calculated
456 * Then we iterate through flashes to identify the best match flash
457 * If a triggering flash was found in this cyrostat, the barycenter distance to the triggering flash is also stored
458 */
459
460 //Fetch slices, TPC hits, and PFPs; pointer vector needed for generating associations
461 art::Handle const sliceHandle
462 = e.getHandle<std::vector<recob::Slice>>(fPandoraLabel + inputTag);

476 const std::vector<art::Ptr<recob::Hit>> &tpcHitsVec = fmTPCHits.at(j);
477 const std::vector<art::Ptr<recob::PFParticle>> &pfpsVec = fmPFPs.at(j);
478 art::FindOne<recob::SpacePoint> f1SpacePoint(tpcHitsVec, e, fPandoraLabel + inputTag);
479
480 int nHits = tpcHitsVec.size();
481 int nPFPs = pfpsVec.size();
482
483 //Establish possible time range for this slice
484 lar::util::TrackTimeInterval::TimeRange const& timeRange = timeIntervals.timeRangeOfHits(tpcHitsVec);
485 const bool rangeIsValid = timeRange.isValid();
486
487 //Retrieve Pandora's T0 for this slice if available, same for every PFP in slice so we only need one
488 if ( nPFPs != 0 ) {

490 if ( f1T0.at(0).isValid() ) {

492 }
493 }

495 double thisCharge;
496 double sumCharge = 0.;
497 TVector3 sumPos {0.,0.,0.};
498 TVector3 sumPosSqr {0.,0.,0.};
499
500 //For hit...
501 for ( int k = 0; k < nHits; k++ ) {
502 const art::Ptr<recob::Hit> &tpcHit = tpcHitsVec.at(k);
503
504 //Only use hits with associated SpacePoints, and optionally only collection plane hits
505 if ( fCollectionOnly && tpcHit->SignalType() != geo::kCollection ) continue;
506 if ( !f1SpacePoint.at(k).isValid() ) continue;
507
508 const recob::SpacePoint point = f1SpacePoint.at(k).ref();
509 thisCharge = tpcHit->Integral();
510 TVector3 const thisPoint = point.XYZ();
511 TVector3 const thisPointSqr {thisPoint.X()*thisPoint.X(), thisPoint.Y()*thisPoint.Y(), thisPoint.Z()*thisPoint.Z()};
512 sumCharge += thisCharge;
513 sumPos += thisPoint * thisCharge;
514 sumPosSqr += thisPointSqr * thisCharge;
515 } //End for hit
516
517 //No charge found in slice...
518 if ( sumCharge == 0. ) {
519 if ( fFillMatchTree ) fMatchTree->Fill();
520 art::Ptr<sbn::TPCPMTBarycenterMatch> const infoPtr = makeInfoPtr(matchInfoVector->size());
521 sliceAssns->addSingle(infoPtr, slicePtr);
522 matchInfoVector->push_back(std::move(sliceMatchInfo));

525 }

528 updateChargeVars(sumCharge, sumPos, sumPosSqr, triggerFlashCenter);
529 updateMatchInfo(sliceMatchInfo);
530
531 int matchIndex = -5;

534

548 thisFlashCenterY = flash.YCenter();
549 thisFlashCenterZ = flash.ZCenter();

554 }
555 } //End for flash
556
557 //No valid match found...
558 if ( matchIndex == -5 ) {
559 if ( fFillMatchTree ) fMatchTree->Fill();
560 art::Ptr<sbn::TPCPMTBarycenterMatch> const infoPtr = makeInfoPtr(matchInfoVector->size());

562 matchInfoVector->push_back(std::move(sliceMatchInfo));

565 }
566
567 //Best match flash pointer
568 unsigned unsignedMatchIndex = matchIndex;
569 const art::Ptr<recob::OpFlash> flashPtr { flashHandle, unsignedMatchIndex };
570
571 //Find time of first OpHit in matched flash

574 for (const recob::OpHit *opHit : opHitsVec ) { if ( opHit->PeakTime() < minTime ) minTime = opHit->PeakTime(); }

576 //Update match info
577 updateFlashVars(flashPtr, minTime);
578 updateMatchInfo(sliceMatchInfo);
579 art::Ptr<sbn::TPCPMTBarycenterMatch> const infoPtr = makeInfoPtr(matchInfoVector->size());
580 sliceAssns->addSingle(infoPtr, slicePtr);
581 flashAssns->addSingle(infoPtr, flashPtr);
582 matchInfoVector->push_back(std::move(sliceMatchInfo));
583 if ( fFillMatchTree ) fMatchTree->Fill();
584
585 } //End for slice
586
587 } //End for InputTag
588
589 //Store new products at the end of the event
590 e.put(std::move(matchInfoVector));
591 e.put(std::move(sliceAssns));
592 e.put(std::move(flashAssns));
593
594} //End produce()

625double TPCPMTBarycenterMatchProducer::CentroidOverlap(double center1, double center2, double width1, double width2) const {
626 //Centroid 2 is contained within Centroid 1, so overlap is the whole Centroid 2
627 if ( (center1 - width1 < center2 - width2) && (center1 + width1 > center2 + width2) ) return (2 * width2);
628
629 //Centroid 1 is contained within Centroid 2, so overlap is the whole Centroid 1
630 else if ( (center1 - width1 > center2 - width2) && (center1 + width1 < center2 + width2) ) return (2 * width1);
631
632 double difference = center1 - center2;
633 if ( center1 > center2 ) difference *= -1;
634 return difference + width1 + width2;
635} //End CentroidOverlap()

638double TPCPMTBarycenterMatchProducer::CalculateAsymmetry(art::Ptr<recob::OpFlash> flash, int cryo) {
639 double sumEast = 0.;
640 double sumWest = 0.;
641
642 //East Cryo flashes have a 180-element PE vector; 0-89 --> East wall, 90-179 --> West wall
643 //West Cryo flashes have a 360-element PE vector; 0-179 --> ALL 0, 180-269 --> East wall, and 270-359 --> West wall
644 int countingOffset = 0;
645 if ( cryo == 1 ) countingOffset += 180;
646
647 for ( int PMT = 0; PMT < 180; PMT++ ) {
648 if ( PMT <= 89 ) sumEast += flash->PEs().at(PMT + countingOffset);
649 else sumWest += flash->PEs().at(PMT + countingOffset);
650 }
651
652 return (sumWest - sumEast) / (sumWest + sumEast);
653} //End CalculateAsymmetry()

658void TPCPMTBarycenterMatchProducer::updateChargeVars(double sumCharge, TVector3 const& sumPos, TVector3 const& sumPosSqr, std::array<double, 2> const& triggerFlashCenter) {
659 fChargeCenterXGlobal = sumPos[0] / sumCharge;
660 fChargeCenterXLocal = fChargeCenterXGlobal - 210.215 * (2*fCryo - 1);
661 fChargeCenterY = sumPos[1] / sumCharge;
662 fChargeCenterZ = sumPos[2] / sumCharge;
663 fChargeWidthX = std::sqrt( sumPosSqr[0]/sumCharge - (sumPos[0]/sumCharge)*(sumPos[0]/sumCharge) );
664 fChargeWidthY = std::sqrt( sumPosSqr[1]/sumCharge - (sumPos[1]/sumCharge)*(sumPos[1]/sumCharge) );
665 fChargeWidthZ = std::sqrt( sumPosSqr[2]/sumCharge - (sumPos[2]/sumCharge)*(sumPos[2]/sumCharge) );
666 if ( triggerFlashCenter[1] != -9999. ) {
667 fDeltaY_Trigger = abs(triggerFlashCenter[0] - fChargeCenterY);
668 fDeltaZ_Trigger = abs(triggerFlashCenter[1] - fChargeCenterZ);
669 fRadius_Trigger = std::hypot(fDeltaY_Trigger, fDeltaZ_Trigger);
670 }
671} //End updateChargeVars()
