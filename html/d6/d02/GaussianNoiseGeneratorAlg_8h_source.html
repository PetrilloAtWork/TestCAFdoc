208template <typename ADCT>
209std::size_t icarus::opdet::GaussianNoiseGeneratorAlg<ADCT>::doAdd(
210 raw::Channel_t channel, Timestamp_t time,
211 ADCcount_t* begin, std::size_t n
212) {
213 std::vector<double> noise(n);
214 fGausRandom.fireArray(static_cast<int>(n), noise.data());
215 for (double const sample: noise)
216 *(begin++) += static_cast<ADCcount_t>(sample);
217 return n;
218} // icarus::opdet::GaussianNoiseGeneratorAlg<>::doAdd()

221// -----------------------------------------------------------------------------
222template <typename ADCT>
223std::size_t icarus::opdet::GaussianNoiseGeneratorAlg<ADCT>::doFill(
224 raw::Channel_t channel, Timestamp_t time,
225 ADCcount_t* begin, std::size_t n
226) {
227 // if destination type is the same as generator natively uses,
228 // we avoid conversions
229 if constexpr(std::is_same_v<ADCcount_t, double>) {
230 fGausRandom.fireArray(static_cast<int>(n), begin);
231 return n;
232 }
233 else {
234 std::vector<double> noise(n);
235 fGausRandom.fireArray(static_cast<int>(n), noise.data());
236 for (double const sample: noise)
237 *(begin++) = static_cast<ADCcount_t>(sample);
238 return n;
239 }
240} // icarus::opdet::GaussianNoiseGeneratorAlg<>::doFill()
