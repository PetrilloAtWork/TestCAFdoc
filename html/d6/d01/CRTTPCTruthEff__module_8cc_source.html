297 void CRTTPCTruthEff::analyze(const art::Event & event)
298 {
299
300 fEvent = event.id().event();
301 fRun = event.run();
302 fSubRun = event.subRun();
303
304 fIsData = event.isRealData();
305
306 std::map<int, simb::MCParticle> particles;
307
308 if(!fIsData){
309 // Loop over the true particles
310 // mcpart_ids.clear();
311 bt.Initialize(event);
312 auto particleHandle = event.getValidHandle<std::vector<simb::MCParticle>>(fSimModuleLabel);
313 for (auto const& particle: (*particleHandle)){
314 // Make map with ID
315 int partID = particle.TrackId();
316 particles[partID] = particle;
317 }//end loop over particles in particleHandle
318 }//end(!fIsData)
319
320 //add trigger info
321 auto const& triggerInfo = event.getProduct<sbn::ExtraTriggerInfo>(fTriggerLabel);
322 sbn::triggerSource bit = triggerInfo.sourceType;
323 m_gate_type = value(bit);
324 m_gate_name = bitName(bit);
325 m_trigger_timestamp = triggerInfo.triggerTimestamp;
326 m_gate_start_timestamp = triggerInfo.beamGateTimestamp;
327 m_trigger_gate_diff = triggerInfo.triggerTimestamp - triggerInfo.beamGateTimestamp;
328
329 // Retrieve CRT hit list
330 auto const& crtHits = event.getProduct<std::vector<sbn::crt::CRTHit>>(fCrtHitModuleLabel);
331 auto const clockData = art::ServiceHandle<detinfo::DetectorClocksService const>()->DataFor(event);
332
333 // Retrieve track list BEGIN LOOP OVER TRACKS IN EVENT
334 for(const auto& [ iTrackLabel, trackLabel ] : util::enumerate(fTpcTrackModuleLabel)) {
335 std::vector<art::Ptr<recob::Track> > trackList;
336 art::Handle< std::vector<recob::Track> > trackListHandle;
337 if (event.getByLabel(trackLabel,trackListHandle)) art::fill_ptr_vector(trackList, trackListHandle);
338 if (trackListHandle.isValid() ){
339
340
341 ttl_tpctrks=(int)trackList.size();
342 ttl_crthits=(int)crtHits.size();
343
344 //Get PFParticles
345 auto pfpListHandle = event.getValidHandle<std::vector<recob::PFParticle> >(fPFParticleLabel[iTrackLabel]);
346// if (!pfpListHandle.isValid()) continue;
347
348 //Get PFParticle-Track association
349 art::FindOneP<recob::PFParticle> fopfp(trackListHandle, event, trackLabel);
350
351 //Get T0-PFParticle association
352 art::FindOneP<anab::T0> fot0pandora(pfpListHandle, event, fPFParticleLabel[iTrackLabel]);
353
354 auto const detProp = art::ServiceHandle<detinfo::DetectorPropertiesService const>()->DataFor(event);
355 art::FindManyP<recob::Hit> findManyHits(trackListHandle, event, trackLabel);
356
357 // will create art pointers to the new T0 objects:
358// art::PtrMaker<anab::T0> const makeT0Ptr{ event };
359
360 // Loop over all the reconstructed tracks
361 for(size_t track_i = 0; track_i < trackList.size(); track_i++) {
362
363 //clear out vectors for new entries into the TTree
364 all_crt_candidate_trueIDs.clear();
365 all_crt_candidate_motherIDs.clear();
366 all_crt_candidate_ancestorIDs.clear();
367 all_crt_regions.clear();
368 all_crt_pdg.clear();
369 all_crt_candidate_motherlayers.clear();
370 all_crt_candidate_x.clear();
371 all_crt_candidate_y.clear();
372 all_crt_candidate_z.clear();
373 crt_start_dca.clear();
374 crt_end_dca.clear();
375 crt_timestamp.clear();
376 all_crt_hitcode_map.clear();
377 all_crt_mac5s.clear();
378 all_crt_chans.clear();
379 all_crt_hitcode.clear();
380 all_crt_pes.clear();
381 matched_hit_mac5.clear();
382 matched_hit_chan.clear();
383 matched_hit_pes.clear();
384
385 //Set certain output variables to defaults
386 crt_tpc_dca = DBL_MAX; crt_trueID = INT_MAX;
387 track_trueID = INT_MAX; track_mother = INT_MAX; track_ancestor = INT_MAX; track_motherlayers = INT_MAX;
388 crt_trueID = INT_MAX; crt_mother = INT_MAX; crt_ancestor = INT_MAX; crt_motherlayers = INT_MAX;
389 crttime = DBL_MAX; track_t0=DBL_MAX;
390 best_dca_pos=-99999;
391
392 //Find PFParticle for track i
393 //art::Ptr::key() gives the index in the vector
394 art::Ptr<recob::PFParticle> pfp = fopfp.at(track_i);
395
396 is_catcross = false;
397 if (pfp){
398 //Find T0 for PFParticle
399 auto t0 = fot0pandora.at(pfp.key());
400 if (t0.isNonnull()){
401 track_t0 = t0->Time();
402 is_catcross = true;
403 catcross_x = DBL_MAX; catcross_y = DBL_MAX; catcross_z = DBL_MAX;
404 getCatCrossXYZ(*trackList[track_i], catcross_x, catcross_y, catcross_z);
405// t0 = track_t0; //Get T0
406 }//end if (!t0.empty())
407 }//end if (!pfps.empty())
408
409 crt_tpc_dca = DBL_MAX; crt_trueID = INT_MAX; track_trueID = INT_MAX; track_mother = INT_MAX; crttime = DBL_MAX;
410
411 std::vector<art::Ptr<recob::Hit>> hits = findManyHits.at(trackList[track_i]->ID());
412
413 if (hits.empty()) continue;
414 //Get Truth info if applicable:
415 if(!fIsData){
416 track_trueID = RecoUtils::TrueParticleIDFromTotalRecoHits(clockData,hits,false);
417 GetAncestorID(track_trueID, track_mother, track_ancestor, track_motherlayers, particles);
418 trk_pdg = particles[track_trueID].PdgCode();
419 }//end if(!fIsData)
420
421// int const cryoNumber = hits[0]->WireID().Cryostat;
422 matchCand closest = t0Alg.GetClosestCRTHit(detProp, *trackList[track_i], hits, crtHits, m_gate_start_timestamp, fIsData);
423
424 if(closest.dca >=0 ){
425
426/* icarus::CRTTPCMatchingInfo matchInfo {
427 closest.dca // DCA
428 , closest.extrapLen // extrapLength
429 }//end definition of icarus::CRTTPCMatchingInfo matchInfo;
430*/
431// double sin_angle = -99999;
432 if(closest.dca != DBL_MIN){//this if gives green light to start filling TTree variables for matched CRT Hit variables
433// auto start = trackList[track_i]->Vertex<TVector3>();
434 crt_region = fCrtutils->AuxDetRegionNameToNum(closest.thishit.tagger);
435 crt_tpc_dca = closest.dca;
436
437 driftdir = closest.driftdir; //TPCGeoUtil::DriftDirectionFromHits(fGeometryService, hits);
438 simple_catcross = closest.simple_cathodecrosser;
439 best_dca_pos = closest.best_DCA_pos;
440 crttime = closest.t0;
441 crt_pes= closest.thishit.peshit;
442 crt_extraplen = closest.extrapLen;
443
444 tpc_trk_start_x = closest.tpc_track_start.X();
445 tpc_trk_start_y = closest.tpc_track_start.Y();
446 tpc_trk_start_z = closest.tpc_track_start.Z();
447
448 tpc_trk_end_x = closest.tpc_track_end.X();
449 tpc_trk_end_y = closest.tpc_track_end.Y();
450 tpc_trk_end_z = closest.tpc_track_end.Z();
451
452 trk_startdir_x = closest.startDir.X();
453 trk_startdir_y = closest.startDir.Y();
454 trk_startdir_z = closest.startDir.Z();

568void icarus::CRTTPCTruthEff::getCatCrossXYZ(recob::Track trk, double &my_x, double &my_y, double &my_z){
569
570 size_t ntrk = trk.NPoints();
571 std::vector<double> x, y, z, x_left_diff, x_right_diff;
572// std::pair<double,double> cathode_yz;
573 double left_dist_min = DBL_MAX; int left_dist_min_pos = INT_MAX;
574 double right_dist_min = DBL_MAX; int right_dist_min_pos = INT_MAX;
575 double xsum=0, xavg;
576
577 int numelmnts = 0;
578
579 //Begin by extracting coordinates into arrays
580 for(size_t i=0; i<ntrk; i++){
581 geo::Point_t thispt = trk.LocationAtPoint((int)i);
582 x.push_back(thispt.X()); y.push_back(thispt.Y()); z.push_back(thispt.Z()); numelmnts++;
583 if(std::abs(x.back())<210.215) {
584 x_left_diff.push_back(std::abs(std::abs(x[i]) - 210.25));
585 if(x_left_diff.back()<=left_dist_min) { left_dist_min=x_left_diff.back(); left_dist_min_pos = i; }
586 }//end if(std::abs(x[i])<210.215)
587 else if(std::abs(x.back())>=210.215) {
588 x_right_diff.push_back(std::abs(std::abs(x[i]) - 210.25));
589 if(x_right_diff.back()<=right_dist_min) { right_dist_min=x_right_diff.back(); right_dist_min_pos = i; }
590 }//end if(std::abs(x[i])<210.215)
591 xsum+=x.back();
592 }//end loop over track points
593 xavg = xsum/ntrk;
594
595 if(left_dist_min_pos<= (numelmnts-1) && right_dist_min_pos<= (numelmnts-1)) {
596
597 TVector3 leftpt(x[left_dist_min_pos],y[left_dist_min_pos],z[left_dist_min_pos]);
598 TVector3 rightpt(x[right_dist_min_pos],y[right_dist_min_pos],z[right_dist_min_pos]);
599
600 //calculate parameters for a line that passes through the two points:
601 double mxy = (leftpt.Y() - rightpt.Y())/(leftpt.X()-rightpt.X());
602 double mxz = (leftpt.Z() - rightpt.Z())/(leftpt.X()-rightpt.X());
603
604 double bxy = leftpt.Y() - mxy*leftpt.X();
605 double bxz = leftpt.Z() - mxz*leftpt.X();
606
607 if(xavg>0) my_x = 210.215;
608 else if(xavg<=0) my_x = -210.215;
609 my_y = mxy*my_x + bxy;
610 my_z = mxz*my_x + bxz;
611 }
