209DaqDecoderICARUSTPCwROI::DaqDecoderICARUSTPCwROI(fhicl::ParameterSet const & pset, art::ProcessingFrame const& frame) :
210 art::ReplicatedProducer(pset, frame),fLogCategory("DaqDecoderICARUSTPCwROI"),fNumEvent(0), fNumROPs(0)
211{
212 fGeometry = art::ServiceHandle<geo::Geometry const>{}.get();
213 fChannelMap = art::ServiceHandle<icarusDB::IICARUSChannelMap const>{}.get();
214
215 configure(pset);
216
217 // Check the concurrency
218 int max_concurrency = art::Globals::instance()->nthreads();
219
220 mf::LogDebug("DaqDecoderICARUSTPCwROI") << " ==> concurrency: " << max_concurrency << std::endl;
221
222 // Recover the vector of fhicl parameters for the ROI tools
223 const fhicl::ParameterSet& decoderToolParams = pset.get<fhicl::ParameterSet>("DecoderTool");
224
225 fDecoderToolVec.resize(max_concurrency);
226
227 for(auto& decoderTool : fDecoderToolVec)
228 {
229 // Get instance of tool
230 decoderTool = art::make_tool<INoiseFilter>(decoderToolParams);
231 }
232
233 // Set up our "producers"
234 // Note that we can have multiple instances input to the module
235 // Our convention will be to create a similar number of outputs with the same instance names
236 for(const auto& fragmentLabel : fFragmentsLabelVec)
237 {
238 produces<std::vector<raw::RawDigit>>(fragmentLabel.instance());
239 produces<std::vector<recob::ChannelROI>>(fragmentLabel.instance());
240
241 if (fOutputRawWaveform)
242 produces<std::vector<raw::RawDigit>>(fragmentLabel.instance() + fOutputRawWavePath);
243
244 if (fOutputCorrection)
245 produces<std::vector<raw::RawDigit>>(fragmentLabel.instance() + fOutputCoherentPath);
246
247 if (fOutputMorphed)
248 produces<std::vector<raw::RawDigit>>(fragmentLabel.instance() + fOutputMorphedPath);
249 }
250
251 // Set up a WireID to ROP plane number table
252 PlaneToWireOffsetMap planeToLastWireOffsetMap;
253
254 for(size_t cryoIdx = 0; cryoIdx < 2; cryoIdx++)
255 {
256 for(size_t logicalTPCIdx = 0; logicalTPCIdx < 4; logicalTPCIdx++)
257 {
258 for(size_t planeIdx = 0; planeIdx < 3; planeIdx++)
259 {
260 geo::PlaneID planeID(cryoIdx,logicalTPCIdx,planeIdx);
261
262 raw::ChannelID_t channel = fGeometry->PlaneWireToChannel(geo::WireID(planeID, 0));
263
264 readout::ROPID ropID = fGeometry->ChannelToROP(channel);
265
266 fPlaneToROPPlaneMap[planeID] = ropID.ROP;
267 fPlaneToWireOffsetMap[planeID] = channel;
268 planeToLastWireOffsetMap[planeID] = fGeometry->PlaneWireToChannel(geo::WireID(planeID, fGeometry->Nwires(planeID)));
269 fROPToNumWiresMap[ropID.ROP] = fGeometry->Nwires(planeID);
270
271 if (ropID.ROP > fNumROPs) fNumROPs = ropID.ROP;
272
273 // Watch for the middle induction and collection plane logical TPC split
274 if (ropID.ROP > 1 && (logicalTPCIdx == 1 || logicalTPCIdx == 3))
275 {
276 geo::PlaneID tempID(cryoIdx,logicalTPCIdx-1,planeIdx);
277
278 fPlaneToWireOffsetMap[planeID] = fPlaneToWireOffsetMap[tempID];
279 fROPToNumWiresMap[ropID.ROP] = planeToLastWireOffsetMap[planeID] - fPlaneToWireOffsetMap[planeID];
280 }
281
282 // Diagnostic output if requested
283 mf::LogDebug(fLogCategory) << "Initializing C/T/P: " << planeID.Cryostat << "/" << planeID.TPC << "/" << planeID.Plane << ", base channel: " << fPlaneToWireOffsetMap[planeID] << ", ROP: " << ropID << ", index: " << ropID.ROP;
284
285 }
286 }
287 }
288
289 fNumROPs++;
290
291 // Report.
292 mf::LogInfo("DaqDecoderICARUSTPCwROI") << "DaqDecoderICARUSTPCwROI configured\n";
293}

329//----------------------------------------------------------------------------
338void DaqDecoderICARUSTPCwROI::produce(art::Event & event, art::ProcessingFrame const&)
339{
340 ++fNumEvent;
341
342 mf::LogDebug("DaqDecoderICARUSTPCwROI") << "**** Processing raw data fragments ****" << std::endl;
343
344 util::LocalArtHandleTrackerManager dataCacheRemover
345 (event, fDropRawDataAfterUse);
346
347 // Check the concurrency
348 int max_concurrency = tbb::this_task_arena::max_concurrency();
349
350 mf::LogDebug("DaqDecoderICARUSTPCwROI") << " ==> concurrency: " << max_concurrency << std::endl;
351
352 cet::cpu_timer theClockTotal;
353
354 theClockTotal.start();
355
356 // Loop through the list of input daq fragment collections one by one
357 // We are not trying to multi thread at this stage because we are trying to control
358 // overall memory usage at this level. We'll multi thread internally...
359 for(const auto& fragmentLabel : fFragmentsLabelVec)
360 {
361 art::Handle<artdaq::Fragments> const& daq_handle
362 = dataCacheRemover.getHandle<artdaq::Fragments>(fragmentLabel);
363
364 ConcurrentRawDigitCol concurrentRawDigits;
365 ConcurrentRawDigitCol concurrentRawRawDigits;
366 ConcurrentRawDigitCol coherentRawDigits;
367 ConcurrentRawDigitCol morphedRawDigits;
368 ConcurrentChannelROICol concurrentROIs;
369
370 PlaneIdxToImageMap planeIdxToImageMap;
371 PlaneIdxToChannelMap planeIdxToChannelMap;
372
373 ChannelArrayPairVec channelArrayPairVec(fNumROPs);
374
375 // Because the arrays can be variable size we need to loop to initialize
376 for(size_t ropIdx = 0; ropIdx < fNumROPs; ropIdx++)
377 {
378 ChannelArrayPair& channelArrayPair = channelArrayPairVec[ropIdx];
379
380 channelArrayPair.first.resize(fROPToNumWiresMap[ropIdx]);
381 channelArrayPair.second.resize(fROPToNumWiresMap[ropIdx],icarus_signal_processing::VectorFloat(4096));
382
383 mf::LogDebug("DaqDecoderICARUSTPCwROI") << "**> Initializing ropIdx: " << ropIdx << " channelPairVec to " << channelArrayPair.first.size() << " channels with " << channelArrayPair.second[0].size() << " ticks" << std::endl;
384 }
385
386 mf::LogDebug("DaqDecoderICARUSTPCwROI") << "****> Let's get ready to rumble!" << std::endl;
387
388 // ... Launch multiple threads with TBB to do the deconvolution and find ROIs in parallel
389 auto const clockData = art::ServiceHandle<detinfo::DetectorClocksService>()->DataFor(event);
390
391 multiThreadFragmentProcessing fragmentProcessing(*this, clockData, daq_handle, concurrentRawRawDigits, concurrentRawDigits, coherentRawDigits, morphedRawDigits, concurrentROIs);
392
393 tbb::parallel_for(tbb::blocked_range<size_t>(0, daq_handle->size()), fragmentProcessing);
394
395 // Now let's process the resulting images
396 // multiThreadImageProcessing imageProcessing(*this, clockData, channelArrayPairVec, concurrentRawDigits, coherentRawDigits, concurrentROIs);
397
398 // tbb::parallel_for(tbb::blocked_range<size_t>(0, fNumROPs), imageProcessing);
399
400 // Copy the raw digits from the concurrent vector to our output vector
401 RawDigitCollectionPtr rawDigitCollection = std::make_unique<std::vector<raw::RawDigit>>(std::move_iterator(concurrentRawDigits.begin()),
402 std::move_iterator(concurrentRawDigits.end()));
403
404 // Want the RawDigits to be sorted in channel order... has to be done somewhere so why not now?
405 std::sort(rawDigitCollection->begin(),rawDigitCollection->end(),[](const auto& left,const auto&right){return left.Channel() < right.Channel();});
406
407 // What did we get back?
408 mf::LogDebug("DaqDecoderICARUSTPCwROI") << "****> Total size of map: " << planeIdxToImageMap.size() << std::endl;
409 for(const auto& planeImagePair : planeIdxToImageMap)
410 {
411 mf::LogDebug("DaqDecoderICARUSTPCwROI") << " - plane: " << planeImagePair.first << " has " << planeImagePair.second.size() << " wires" << std::endl;
412 }
413
414 // Now transfer ownership to the event store
415 event.put(std::move(rawDigitCollection), fragmentLabel.instance());
416
417 // Do the same to output the candidate ROIs
418 ChannelROICollectionPtr channelROICollection = std::make_unique<std::vector<recob::ChannelROI>>(std::move_iterator(concurrentROIs.begin()),
419 std::move_iterator(concurrentROIs.end()));
420
421 std::sort(channelROICollection->begin(),channelROICollection->end(),[](const auto& left, const auto& right){return left.Channel() < right.Channel();});
422
423 event.put(std::move(channelROICollection), fragmentLabel.instance());
424
425
426 if (fOutputRawWaveform)
427 {
428 // Copy the raw digits from the concurrent vector to our output vector
429 RawDigitCollectionPtr rawRawDigitCollection = std::make_unique<std::vector<raw::RawDigit>>(std::move_iterator(concurrentRawRawDigits.begin()),
430 std::move_iterator(concurrentRawRawDigits.end()));
431
432 // Want the RawDigits to be sorted in channel order... has to be done somewhere so why not now?
433 std::sort(rawRawDigitCollection->begin(),rawRawDigitCollection->end(),[](const auto& left,const auto&right){return left.Channel() < right.Channel();});
434
435 // Now transfer ownership to the event store
436 event.put(std::move(rawRawDigitCollection),fragmentLabel.instance() + fOutputRawWavePath);
437 }
438
439 if (fOutputCorrection)
440 {
441 // Copy the raw digits from the concurrent vector to our output vector
442 RawDigitCollectionPtr coherentCollection = std::make_unique<std::vector<raw::RawDigit>>(std::move_iterator(coherentRawDigits.begin()),
443 std::move_iterator(coherentRawDigits.end()));
444
445 // Want the RawDigits to be sorted in channel order... has to be done somewhere so why not now?
446 std::sort(coherentCollection->begin(),coherentCollection->end(),[](const auto& left,const auto&right){return left.Channel() < right.Channel();});
447
448 // Now transfer ownership to the event store
449 event.put(std::move(coherentCollection),fragmentLabel.instance() + fOutputCoherentPath);
450 }
451
452 if (fOutputMorphed)
453 {
454 // Copy the raw digits from the concurrent vector to our output vector
455 RawDigitCollectionPtr morphedCollection = std::make_unique<std::vector<raw::RawDigit>>(std::move_iterator(morphedRawDigits.begin()),
456 std::move_iterator(morphedRawDigits.end()));
457
458 // Want the RawDigits to be sorted in channel order... has to be done somewhere so why not now?
459 std::sort(morphedCollection->begin(),morphedCollection->end(),[](const auto& left,const auto&right){return left.Channel() < right.Channel();});
460
461 // Now transfer ownership to the event store
462 event.put(std::move(morphedCollection),fragmentLabel.instance() + fOutputMorphedPath);
463 }
464 }
465
466 theClockTotal.stop();
467
468 double totalTime = theClockTotal.accumulated_real_time();
469
470 mf::LogInfo(fLogCategory) << "==> DaqDecoderICARUSTPCwROI total time: " << totalTime << std::endl;
471
472 return;
473}

475void DaqDecoderICARUSTPCwROI::processSingleFragment(size_t idx,
476 detinfo::DetectorClocksData const& clockData,
477 art::Handle<artdaq::Fragments> fragmentHandle,
478 ConcurrentRawDigitCol& concurrentRawRawDigitCol,
479 ConcurrentRawDigitCol& concurrentRawDigitCol,
480 ConcurrentRawDigitCol& coherentRawDigitCol,
481 ConcurrentRawDigitCol& morphedRawDigitCol,
482 ConcurrentChannelROICol& concurrentROIs) const
483{
484 cet::cpu_timer theClockProcess;
485
486 theClockProcess.start();
487
488 art::Ptr<artdaq::Fragment> fragmentPtr(fragmentHandle, idx);
489
490 mf::LogDebug("DaqDecoderICARUSTPCwROI") << "--> Processing fragment ID: " << fragmentPtr->fragmentID() << std::endl;
491 mf::LogDebug("DaqDecoderICARUSTPCwROI") << " ==> Current thread index: " << tbb::this_task_arena::current_thread_index() << std::endl;
492
493 cet::cpu_timer theClockTotal;
494
495 theClockTotal.start();
496
497 // convert fragment to Nevis fragment
498 icarus::PhysCrateFragment physCrateFragment(*fragmentPtr);
499
500 size_t nBoardsPerFragment = physCrateFragment.nBoards();
501 size_t nChannelsPerBoard = physCrateFragment.nChannelsPerBoard();
502 size_t nSamplesPerChannel = physCrateFragment.nSamplesPerChannel();
503// size_t nChannelsPerFragment = nBoardsPerFragment * nChannelsPerBoard;
504
505 // Recover the Fragment id:
506 artdaq::detail::RawFragmentHeader::fragment_id_t fragmentID = fragmentPtr->fragmentID();
507
508 mf::LogDebug(fLogCategory) << "==> Recovered fragmentID: " << std::hex << fragmentID << std::dec << std::endl;
509
510 // Look for special case of diagnostic running
511 if (!fChannelMap->hasFragmentID(fragmentID))
512 {
513 return;
514 }
515
516 // Recover the crate name for this fragment
517 const std::string& crateName = fChannelMap->getCrateName(fragmentID);
518
519 // Get the board ids for this fragment
520 const icarusDB::ReadoutIDVec& readoutIDVec = fChannelMap->getReadoutBoardVec(fragmentID);
521
522 icarusDB::ReadoutIDVec boardIDVec(readoutIDVec.size());
523
524 // Note we want these to be in "slot" order...
525 for(const auto& boardID : readoutIDVec)
526 {
527 // Look up the channels associated to this board
528 if (!fChannelMap->hasBoardID(boardID))
529 {
530 mf::LogDebug(fLogCategory) << "*** COULD NOT FIND BOARD ***\n" <<
531 " - boardID: " << std::hex << boardID << ", board map size: " << readoutIDVec.size() << ", nBoardsPerFragment: " << nBoardsPerFragment;
532
533 return;
534 }
535
536 unsigned int boardSlot = fChannelMap->getBoardSlot(boardID);
537
538 boardIDVec[boardSlot] = boardID;
539 }
540
541 std::string boardIDs = "";
542
543 for(const auto& id : boardIDVec) boardIDs += std::to_string(id) + " ";
544
545 mf::LogDebug(fLogCategory) << " - # boards: " << boardIDVec.size() << ", boards: " << boardIDs;
546
547 cet::cpu_timer theClockPedestal;
548
549 theClockPedestal.start();
550
551 // Recover pointer to the decoder needed here
552 INoiseFilter* decoderTool = fDecoderToolVec[tbb::this_task_arena::current_thread_index()].get();
553
554 // Create a local channel pair to hold at most a boards worth of info (64 channels x 4096 ticks)
555 ChannelArrayPair channelArrayPair;
556
557 channelArrayPair.first.resize(nChannelsPerBoard);
558 channelArrayPair.second.resize(nChannelsPerBoard,icarus_signal_processing::VectorFloat(nSamplesPerChannel));
559
560 // Now set up for output, we need to convert back from float to short int so use this
561 raw::RawDigit::ADCvector_t wvfm(nSamplesPerChannel);
562
563 // The first task is to recover the data from the board data block, determine and subtract the pedestals
564 // and store into vectors useful for the next steps
565 for(size_t board = 0; board < boardIDVec.size(); board++)
566 {
567 // Some diagnostics test are removing boards so put in check here to watch for this
568 if (board >= nBoardsPerFragment)
569 {
570 mf::LogInfo("TPCDecoderFilter1D") << " Asking for board beyond number in fragment, want board " << board << ", maximum is " << nBoardsPerFragment << std::endl;
571 continue;
572 }
573
574 const icarusDB::ChannelPlanePairVec& channelPlanePairVec = fChannelMap->getChannelPlanePair(boardIDVec[board]);
575
576 uint32_t boardSlot = physCrateFragment.DataTileHeader(board)->StatusReg_SlotID();
577
578 // ** the line below removed as per request of the TPC hardware folks

610 int localRangeBins(0);

612 float sigmaCut(fSigmaForTruncation);
613

615 const icarus_signal_processing::ArrayFloat& denoised = decoderTool->getWaveLessCoherent();
616
617 icarus_signal_processing::VectorFloat pedCorWaveforms(denoised[0].size());

620 {
621 // Get the channel number on the Fragment
622 raw::ChannelID_t channel = channelPlanePairVec[chanIdx].first;
623
624 // Are we storing the raw waveforms?
625 if (fOutputRawWaveform)
626 {
627 //const icarus_signal_processing::VectorFloat& waveform = decoderTool->getPedCorWaveforms()[chanIdx];
628 const icarus_signal_processing::VectorFloat& waveform = decoderTool->getRawWaveforms()[chanIdx];

636 }
637
638 if (fOutputCorrection)
639 {
640 const icarus_signal_processing::VectorFloat& corrections = decoderTool->getCorrectedMedians()[chanIdx];
641
642 // Need to convert from float to short int
643 std::transform(corrections.begin(),corrections.end(),wvfm.begin(),[](const auto& val){return short(std::round(val));});
644

646 ConcurrentRawDigitCol::iterator newRawObjItr = coherentRawDigitCol.push_back(raw::RawDigit(channel,wvfm.size(),wvfm));

649 }

651 if (fOutputMorphed)
652 {
653 const icarus_signal_processing::VectorFloat& corrections = decoderTool->getMorphedWaveforms()[chanIdx];
654
655 // Need to convert from float to short int
656 std::transform(corrections.begin(),corrections.end(),wvfm.begin(),[](const auto& val){return short(std::round(val));});
657

659 ConcurrentRawDigitCol::iterator newRawObjItr = morphedRawDigitCol.push_back(raw::RawDigit(channel,wvfm.size(),wvfm));
660
661 newRawObjItr->SetPedestal(0.,0.);
662 }
663
664 // Now determine the pedestal and correct for it
665 waveformTools.getPedestalCorrectedWaveform(denoised[chanIdx],
666 pedCorWaveforms,
667 sigmaCut,
668 localPedestal,
669 localFullRMS,
670 localTruncRMS,
671 localNumTruncBins,
672 localRangeBins);
673
674 // Need to convert from float to short int
675 std::transform(pedCorWaveforms.begin(),pedCorWaveforms.end(),wvfm.begin(),[](const auto& val){return short(std::round(val));});
676
677 ConcurrentRawDigitCol::iterator newObjItr = concurrentRawDigitCol.emplace_back(channel,wvfm.size(),wvfm);
678
679 newObjItr->SetPedestal(localPedestal,localFullRMS);
680
681 // And, finally, the ROIs
682 const icarus_signal_processing::VectorBool& chanROIs = decoderTool->getROIVals()[chanIdx];
683 recob::ChannelROI::RegionsOfInterest_t ROIVec;
684
685 // Go through candidate ROIs and create Wire ROIs
686 size_t roiIdx = 0;
687
688 while(roiIdx < chanROIs.size())
689 {
690 size_t roiStartIdx = roiIdx;
691
692 while(roiIdx < chanROIs.size() && chanROIs[roiIdx]) roiIdx++;
693
694 if (roiIdx > roiStartIdx)
695 {
696 std::vector<short> holder(roiIdx - roiStartIdx);
697
698 for(size_t idx = 0; idx < holder.size(); idx++) holder[idx] = wvfm[roiStartIdx+idx];
699
700 ROIVec.add_range(roiStartIdx, std::move(holder));
701 }
702
703 roiIdx++;
704 }
705
706 concurrentROIs.push_back(recob::ChannelROICreator(std::move(ROIVec),channel).move());
707 }
708 }
709
710 // We need to make sure the channelID information is not preserved when less than 9 boards in the fragment
711// if (nBoardsPerFragment < 9)
712// {
713// std::fill(fChannelIDVec.begin() + nBoardsPerFragment * nChannelsPerBoard, fChannelIDVec.end(), -1);
714// }
715
716
717 theClockProcess.stop();
718
719 double totalTime = theClockProcess.accumulated_real_time();
720
721 mf::LogDebug(fLogCategory) << "--> Exiting fragment processing for thread: " << tbb::this_task_arena::current_thread_index() << ", time: " << totalTime << std::endl;
722 return;
