95void HitConvert::produce(art::Event& evt)
96{
97 // We need to loop through the list of Wire data we have been given
98 // This construct from Gianluca Petrillo who invented it and should be given all credit for it!
99 for(auto const& [hitLabel, instanceName] : util::zip(fHitModuleLabelVec, fOutInstanceLabelVec))
100 {
101 // make a collection of Wires
102 std::unique_ptr<std::vector<recob::Hit>> outputHitCol = std::make_unique<std::vector<recob::Hit>>();
103
104 mf::LogInfo("HitConvert") << "HitConvert, looking for icarus::Hit data at " << hitLabel.encode();
105
106 // Read in the collection of full length deconvolved waveforms
107 const std::vector<icarus::Hit>& inputHitVec = evt.getProduct<std::vector<icarus::Hit>>(hitLabel);
108
109 mf::LogInfo("HitConvert") << "--> Recovered icarus::Hit data, size: " << inputHitVec.size();
110
111 if (!inputHitVec.empty())
112 {
113 // Reserve the room for the output
114 outputHitCol->reserve(inputHitVec.size());
115
116 // Loop through the input ChannelROI collection
117 for(const auto& hit : inputHitVec)
118 {
119 recob::Hit recobHit(hit.Channel(),
120 hit.StartTick(),
121 hit.EndTick(),
122 hit.PeakTime(),
123 hit.SigmaPeakTime(),
124 hit.RMS(),
125 hit.PeakAmplitude(),
126 hit.SigmaPeakAmplitude(),
127 hit.SummedADC(),
128 hit.Integral(),
129 hit.SigmaIntegral(),
130 hit.Multiplicity(),
131 hit.LocalIndex(),
132 hit.GoodnessOfFit(),
133 hit.DegreesOfFreedom(),
134 hit.View(),
135 hit.SignalType(),
136 hit.WireID()
137 );
138
139 outputHitCol->push_back(recobHit);
140 }
141
142 mf::LogInfo("HitConvert") << "--> Outputting hit data, size: " << outputHitCol->size() << " with instance name: " << instanceName;
143
144 // Time to stroe everything
145 if(outputHitCol->empty()) mf::LogWarning("HitConvert") << "No hits made for this event.";
146 }
147
148 evt.put(std::move(outputHitCol), instanceName);
149 }
150
151 fEventCount++;
152
153 return;
154} // produce
