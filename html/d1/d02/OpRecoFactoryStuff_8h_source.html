572template <typename S, typename Coll>
573S opdet::factory::join(S const& sep, Coll const& s) {
574 using std::empty, std::begin, std::end;
575 if (empty(s)) return S{};
576 auto it = begin(s);
577 auto const send = end(s);
578 S cat { *it };
579 while (++it != send) { cat += sep; cat += *it; }
580 return cat;
581} // join()
