415 template <typename ConfigMap>
416 sbn::PMTconfiguration extractPMTreadoutConfigurationImpl
417 (ConfigMap const& configMap, icarus::PMTconfigurationExtractor extractor)
418 {
419
420 /*
421 * Requirements: ConfigMap is a mapping type supporting iteration and whose
422 * elements support structured assignment into a pair, the first element
423 * being an identifier (process name, parameter set ID...) and the second
424 * being a `fhicl::ParameterSet` with the configuration (or something
425 * offering the same interface).
426 *
427 * The plan is to look in all the FHiCL configuration fragments we can find
428 * in the input config, and find all the useful configuration therein.
429 * Given that there may be multiple input files, there may also be multiple
430 * configurations for the same detector components.
431 * In that case, we will extract parameters from each and every one of the
432 * configurations, and throw an exception if they are not all consistent.
433 *
434 * Consistency is tested only for the extracted parameters, not for the whole
435 * FHiCL configuration fragment.
436 */
437
438 using Key_t = std::tuple_element_t<0U, typename ConfigMap::value_type>;
439
440 std::optional<std::pair<Key_t, sbn::PMTconfiguration>> config;
441
442 // look in the global configuration for all parameter sets which contain
443 // `configuration_documents` as a (direct) name;
444 for (auto const& [ id, pset ]: configMap) {
445 if (!extractor.mayHaveConfiguration(pset)) continue;
446
447 fhicl::ParameterSet const configDocs
448 = extractor.convertConfigurationDocuments
449 (pset, "configuration_documents", { std::regex{ "icaruspmt.*" } })
450 ;
451
452 sbn::PMTconfiguration candidateConfig = extractor.extract(configDocs);
453 if (config) {
454 if (config->second == candidateConfig) continue;
455 mf::LogError log("extractPMTreadoutConfiguration");
456 log << "Found two candidate configurations differring:"
457 "\nFirst [" << config->first << "]:\n" << config->second
458 << "\nSecond [" << id << "]:\n" << candidateConfig
459 ;
460 throw cet::exception("extractPMTreadoutConfiguration")
461 << "extractPMTreadoutConfiguration() found inconsistent configurations.\n";
462 } // if incompatible configurations
463
464 config.emplace(std::move(id), std::move(candidateConfig));
465 } // for all configuration documents
466
467 if (!config) {
468 throw cet::exception("extractPMTreadoutConfiguration")
469 << "extractPMTreadoutConfiguration() could not find a suitable configuration.\n";
470 }
471
472 return extractor.finalize(std::move(config->second));
473 } // extractPMTreadoutConfigurationImpl(ConfigMap)
