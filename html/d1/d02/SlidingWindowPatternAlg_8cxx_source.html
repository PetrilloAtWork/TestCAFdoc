59//------------------------------------------------------------------------------
60auto icarus::trigger::SlidingWindowPatternAlg::simulateResponse
61 (TriggerGates_t const& gates) const -> AllTriggerInfo_t
62{
63
64 // ensures input gates are in the same order as the configured windows
65 verifyInputTopology(gates);
66
67 auto const& inBeamGates = fBeamGate? fBeamGate->applyToAll(gates): gates;
68
69 //
70 // 2. apply pattern:
71 //
72 std::size_t const nWindows = fWindowTopology.nWindows();
73
74 //
75 // 2.1. for each main window, apply the pattern
76 //
77 WindowTriggerInfo_t triggerInfo; // start empty
78 for (std::size_t const iWindow: util::counter(nWindows)) {
79
80 TriggerInfo_t const windowResponse
81 = applyWindowPattern(fWindowPattern, iWindow, inBeamGates);
82
83 if (!windowResponse) continue;
84
85 assert(windowResponse.hasLocation());
86 assert(windowResponse.location() == iWindow);
87 if (windowResponse.nTriggers() == 1U) {
88 mfLogTrace() << "Pattern fired on window #" << iWindow
89 << " at tick " << windowResponse.atTick();
90 }
91 else {
92 std::vector<TriggerInfo_t::OpeningInfo_t> const& triggers
93 = windowResponse.all();
94 TriggerInfo_t::OpeningInfo_t const* pMain = &(windowResponse.main());
95 auto log = mfLogTrace();
96 log << "Pattern fired " << triggers.size()
97 << " times on window #" << iWindow << ":";
98 for (auto const& [ iTrigger, trigger ]: util::enumerate(triggers)) {
99 log << " [#" << iTrigger << "] " << " on " << trigger.tick
100 << " (";
101 if (&trigger == pMain) log << "main; ";
102 log << "lvl=" << trigger.level << ")";
103 } // for
104 }
105
106 //
107 // 2.2. pick the main window with the earliest successful response, if any;
108 // that defines location and time of the trigger
109 //
110 if (!triggerInfo || triggerInfo.info.atTick() > windowResponse.atTick()) {
111 if (!triggerInfo) mfLogTrace() << " (new global trigger)";
112 triggerInfo.emplace(iWindow, windowResponse);
113 }
114
115 } // main window choice
116
117 return { std::move(triggerInfo.info), MoreInfo_t{ triggerInfo.windowIndex } };
118} // icarus::trigger::SlidingWindowPatternAlg::simulateResponse()

137//------------------------------------------------------------------------------
138void icarus::trigger::SlidingWindowPatternAlg::verifyInputTopology
139 (TriggerGates_t const& gates) const
140{
141 /*
142 * Verifies that the `gates` are in the expected order and have
143 * the expected channel content.
144 */
145
146 std::string errorMsg; // if this stays `empty()` there is no error
147 for (auto const& [ iWindow, gate ]: util::enumerate(gates)) {
148
149 std::string windowError; // if this stays `empty()` there is no error
150
151 // more input gates than windows?
152 if (iWindow >= fWindowTopology.nWindows()) {
153 windowError = "unexpected input gate #" + std::to_string(iWindow) + " (";
154 for (raw::Channel_t const channel: gate.channels())
155 windowError += " " + std::to_string(channel);
156 windowError += " )";
157 errorMsg += windowError + '\n';
158 continue; // we collect info of all spurious gates
159 }
160
161 WindowTopology_t::WindowInfo_t const& windowInfo
162 = fWindowTopology.info(iWindow);
163
164 auto const channelInWindow
165 = [
166 begin=windowInfo.composition.channels.cbegin(),
167 end=windowInfo.composition.channels.cend()
168 ]
169 (raw::Channel_t channel)
170 { return std::binary_search(begin, end, channel); }
171 ;
172
173 for (raw::Channel_t const channel: gate.channels()) {
174 if (channelInWindow(channel)) continue;
175 if (windowError.empty()) {
176 windowError = "channels not in window #"
177 + std::to_string(windowInfo.topology.index) + ":";
178 } // if first error
179 windowError += " " + std::to_string(channel);
180 } // for all channels in gate
181
182 if (!windowError.empty()) errorMsg += windowError + '\n';
183 } // for gates
184
185 // more input gates than windows?
186 if (fWindowTopology.nWindows() > size(gates)) {
187 errorMsg +=
188 "Not enough input gates: " + std::to_string(size(gates)) + " gates for "
189 + std::to_string(fWindowTopology.nWindows()) + " windows\n";
190 }
191
192 if (errorMsg.empty()) return;
193
194 // put together the exception message and throw it.
195 throw cet::exception("SlidingWindowPatternAlg")
196 << "Some channels from trigger gates do not match"
197 " the configured window allocation:\n"
198 << errorMsg
199 << "\n" // empty line
200 << "Window configuration: "
201 << fWindowTopology << "\n";
202
203} // icarus::trigger::SlidingWindowPatternAlg::verifyInputTopology()

206//------------------------------------------------------------------------------
207auto icarus::trigger::SlidingWindowPatternAlg::applyWindowPattern(
208 WindowTopology_t::WindowInfo_t const& windowInfo,
209 WindowPattern_t const& pattern,
210 TriggerGates_t const& gates
211 ) const -> TriggerInfo_t
212{
213
214 /*
215 * 1. check that the pattern can be applied; if not, return no trigger
216 * 2. discriminate all the relevant gates against their required minimum count
217 * 3. combine them in AND
218 * 4. find the trigger time, fill the trigger information accordingly
219 */
220 TriggerInfo_t res; // no trigger by default
221 assert(!res);
222
223 WindowTopology_t::WindowTopology_t const& winTopology
224 = windowInfo.topology; // I should have named the types better...
225
226 // undress the gate to get its data
227 auto const gateAt = [&gates](std::size_t index) -> TriggerGateData_t const&
228 { return gateIn(gates[index]); };
229
230 //
231 // 1. check that the pattern can be applied; if not, return no trigger
232 //
233
234 // check that the pattern centered into iWindow has all it needs:
235 if (pattern.requireUpstreamWindow && !winTopology.hasUpstreamWindow())
236 return res;
237 if (pattern.requireDownstreamWindow && !winTopology.hasDownstreamWindow())
238 return res;
239
240
241 //
242 // 2. discriminate all the relevant gates against their required minimum count
243 // 3. combine them in AND
244 //
245
246 mfLogTrace()
247 << "Window info #" << winTopology.index << " pattern "
248 << pattern.tag();
249
250 // we have two different "modes" depending on whether we are constraining
251 // the sum of main and opposite window, or not;
252 std::optional<TriggerGateData_t> const mainPlusOpposite
253 = (pattern.minSumInOppositeWindows > 0U)
254 ? std::optional{
255 winTopology.hasOppositeWindow()
256 ? sumGates(gateAt(winTopology.index), gateAt(winTopology.opposite))
257 : gateAt(winTopology.index)
258 }
259 : std::nullopt
260 ;
261
262 // the basic trigger primitive gate has the levels of the main or
263 // main+opposite window depending on the requirements
264 TriggerGateData_t trigPrimitive
265 = mainPlusOpposite? *mainPlusOpposite: gateAt(winTopology.index);
266
267 mfLogTrace() << " base: " << compactdump(trigPrimitive);
268
269 // main window
270 if (pattern.minInMainWindow > 0U) {
271 trigPrimitive.Mul
272 (discriminate(gateAt(winTopology.index), pattern.minInMainWindow));
273 mfLogTrace()
274 << " main >= " << pattern.minInMainWindow << ": "
275 << compactdump(trigPrimitive);
276 } // if
277
278 // add opposite window requirement (if any)
279 if ((pattern.minInOppositeWindow > 0U) && winTopology.hasOppositeWindow())
280 {
281 trigPrimitive.Mul(
282 discriminate(gateAt(winTopology.opposite), pattern.minInOppositeWindow)
283 );
284 mfLogTrace() << " opposite [#" << winTopology.opposite << "]: "
285 << compactdump(gateAt(winTopology.opposite))
286 << "\n opposite >= " << pattern.minInOppositeWindow << ": "
287 << compactdump(trigPrimitive);
288 } // if
289
290 // add main plus opposite window requirement (if any)
291 if (pattern.minSumInOppositeWindows > 0U) {
292 assert(mainPlusOpposite.has_value());
293 trigPrimitive.Mul
294 (discriminate(*mainPlusOpposite, pattern.minSumInOppositeWindows));
295 mfLogTrace() << " sum [+ #" << winTopology.opposite << "]: "
296 << compactdump(*mainPlusOpposite)
297 << "\n sum >= " << pattern.minSumInOppositeWindows << ": "
298 << compactdump(trigPrimitive);
299
300 } // if
301
302 // add upstream window requirement (if any)
303 if ((pattern.minInUpstreamWindow > 0U) && winTopology.hasUpstreamWindow())
304 {
305 trigPrimitive.Mul(
306 discriminate(gateAt(winTopology.upstream), pattern.minInUpstreamWindow)
307 );
308 } // if
309
310 // add downstream window requirement (if any)
311 if ((pattern.minInDownstreamWindow > 0U)
312 && winTopology.hasDownstreamWindow()
313 ) {
314 trigPrimitive.Mul(
315 discriminate
316 (gateAt(winTopology.downstream), pattern.minInDownstreamWindow)
317 );
318 } // if
319
320 mfLogTrace() << " final: " << compactdump(trigPrimitive);
321
322 //
323 // 4. find the trigger time, fill the trigger information accordingly
324 //
325 icarus::trigger::details::GateOpeningInfoExtractor extractOpeningInfo
326 { trigPrimitive };
327
328 extractOpeningInfo.setLocation
329 (TriggerInfo_t::LocationID_t{ winTopology.index });
330
331 while (extractOpeningInfo) {
332 auto info = extractOpeningInfo();
333 if (info) res.add(info.value());
334 } // while
335
336 return res;
337
338} // icarus::trigger::SlidingWindowTriggerEfficiencyPlots::applyWindowPattern()
