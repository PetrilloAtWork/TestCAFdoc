176void Decon1DROI::reconfigure(fhicl::ParameterSet const& pset)
177{
178 // Recover the parameters
179 fRawDigitLabelVec = pset.get< std::vector<art::InputTag>> ("RawDigitLabelVec", {"daqTPC"});
180 fNoiseSource = pset.get< unsigned short > ("NoiseSource", 3);
181 fSaveWireWF = pset.get< int > ("SaveWireWF" );
182 fMinAllowedChanStatus = pset.get< int > ("MinAllowedChannelStatus" );
183 fTruncRMSThreshold = pset.get< float > ("TruncRMSThreshold", 6.);
184 fTruncRMSMinFraction = pset.get< float > ("TruncRMSMinFraction", 0.6);
185 fOutputHistograms = pset.get< bool > ("OutputHistograms", true);
186
187 // Recover the vector of fhicl parameters for the ROI tools
188 const fhicl::ParameterSet& roiFinderTools = pset.get<fhicl::ParameterSet>("ROIFinderToolVec");
189
190 fROIFinderVec.resize(roiFinderTools.get_pset_names().size());
191
192 unsigned short roiPadding(std::numeric_limits<unsigned short>::max());
193
194 for(const std::string& roiFinderTool : roiFinderTools.get_pset_names())
195 {
196 const fhicl::ParameterSet& roiFinderToolParamSet = roiFinderTools.get<fhicl::ParameterSet>(roiFinderTool);
197 size_t planeIdx = roiFinderToolParamSet.get<size_t>("Plane");
198
199 roiPadding = std::min(roiPadding,roiFinderToolParamSet.get< std::vector<unsigned short>>("roiLeadTrailPad")[0]);
200
201 fROIFinderVec.at(planeIdx) = art::make_tool<icarus_tool::IROIFinder>(roiFinderToolParamSet);
202 }
203
204 std::sort(fROIFinderVec.begin(),fROIFinderVec.end(),[](const auto& left,const auto& right){return left->plane() < right->plane();});
205
206 fDeconvolution = art::make_tool<icarus_tool::IDeconvolution>(pset.get<fhicl::ParameterSet>("Deconvolution"));
207
208 // Recover the baseline tool
209 fhicl::ParameterSet baselineParams = pset.get<fhicl::ParameterSet>("Baseline");
210
211 // Check if we need to set the length for setting the baseline
212 if (baselineParams.has_key("MaxROILength")) baselineParams.put_or_replace("MaxROILength",size_t(roiPadding));
213
214 fBaseline = art::make_tool<icarus_tool::IBaseline> (baselineParams);
215
216 if (fOutputHistograms)
217 {
218 // Access ART's TFileService, which will handle creating and writing
219 // histograms and n-tuples for us.
220 art::ServiceHandle<art::TFileService> tfs;
221
222 fPedestalOffsetVec.resize(3);
223 fFullRMSVec.resize(3);
224 fTruncRMSVec.resize(3);
225 fNumTruncBinsVec.resize(3);
226 fPedByChanVec.resize(3);
227 fTruncRMSByChanVec.resize(3);
228 fNumROIsHistVec.resize(3);
229 fROILenHistVec.resize(3);
230
231 constexpr geo::TPCID tpcid{0, 0};
232 for(size_t planeIdx = 0; planeIdx < 3; planeIdx++)
233 {
234 geo::PlaneID const planeID(tpcid, planeIdx);
235 fPedestalOffsetVec[planeIdx] = tfs->make<TH1F>( Form("PedPlane_%02zu",planeIdx), ";Pedestal Offset (ADC);", 100, -5., 5.);
236 fFullRMSVec[planeIdx] = tfs->make<TH1F>( Form("RMSFPlane_%02zu",planeIdx), "Full RMS;RMS (ADC);", 400, 0., 40.);
237 fTruncRMSVec[planeIdx] = tfs->make<TH1F>( Form("RMSTPlane_%02zu",planeIdx), "Truncated RMS;RMS (ADC);", 100, 0., 10.);
238 fNumTruncBinsVec[planeIdx] = tfs->make<TH1F>( Form("NTruncBins_%02zu",planeIdx), ";# bins", 640, 0., 6400.);
239 fPedByChanVec[planeIdx] = tfs->make<TProfile>(Form("PedByWirePlane_%02zu",planeIdx), ";Wire#", fGeometry->Nwires(planeID), 0., fGeometry->Nwires(planeID), -5., 5.);
240 fTruncRMSByChanVec[planeIdx] = tfs->make<TProfile>(Form("TruncRMSByWirePlane_%02zu",planeIdx), ";Wire#", fGeometry->Nwires(planeID), 0., fGeometry->Nwires(planeID), 0., 10.);
241 fNumROIsHistVec[planeIdx] = tfs->make<TH1F>( Form("NROISplane_%02zu",planeIdx), ";# ROIs;", 100, 0, 100);
242 fROILenHistVec[planeIdx] = tfs->make<TH1F>( Form("ROISIZEplane_%02zu",planeIdx), ";ROI size;", 500, 0, 500);
243 }
244 }
245
246 return;
247}

261void Decon1DROI::produce(art::Event& evt, art::ProcessingFrame const& frame)
262{
263 // We loop over the collection of RawDigits in our input list
264 // This is not done multi threaded as a way to cut down on overall job memory usage...
265 for(const auto& rawDigitLabel : fRawDigitLabelVec)
266 {
267 // make a collection of Wires
268 std::unique_ptr<std::vector<recob::Wire>> wireCol(new std::vector<recob::Wire>);
269
270 // ... and an association set
271 std::unique_ptr<art::Assns<raw::RawDigit,recob::Wire>> wireDigitAssn(new art::Assns<raw::RawDigit,recob::Wire>);
272
273 // Read in the digit List object(s).
274 art::Handle< std::vector<raw::RawDigit>> digitVecHandle;
275
276 evt.getByLabel(rawDigitLabel, digitVecHandle);
277
278 if (!digitVecHandle->size())
279 {
280 std::cout << "Decon1DROI found zero length RawDigits so exiting" << std::endl;
281
282 evt.put(std::move(wireCol), rawDigitLabel.instance());
283 evt.put(std::move(wireDigitAssn), rawDigitLabel.instance());
284 fEventCount++;
285
286 return;
287 }
288
289 // Reserve the room for the output
290 wireCol->reserve(digitVecHandle->size());
291
292 // ... Launch multiple threads with TBB to do the deconvolution and find ROIs in parallel
293 multiThreadDeconvolutionProcessing deconvolutionProcessing(*this, evt, digitVecHandle, *wireCol, *wireDigitAssn, rawDigitLabel.instance());
294
295 tbb::parallel_for(tbb::blocked_range<size_t>(0, digitVecHandle->size()), deconvolutionProcessing);
296
297 // Time to stroe everything
298 if(wireCol->size() == 0)
299 mf::LogWarning("Decon1DROI") << "No wires made for this event.";
300
301 //Make Histogram of recob::wire objects from Signal() vector
302 // get access to the TFile service
303 if ( fSaveWireWF ){
304 art::ServiceHandle<art::TFileService> tfs;
305 for (size_t wireN = 0; wireN < wireCol->size(); wireN++){
306 std::vector<float> sigTMP = wireCol->at(wireN).Signal();
307 TH1D* fWire = tfs->make<TH1D>(Form("Noise_Evt%04zu_N%04zu",fEventCount,wireN), ";Noise (ADC);",
308 sigTMP.size(),-0.5,sigTMP.size()-0.5);
309 for (size_t tick = 0; tick < sigTMP.size(); tick++){
310 fWire->SetBinContent(tick+1, sigTMP.at(tick) );
311 }
312 }
313 }
314
315 // Make sure the collection is sorted
316 std::sort(wireCol->begin(), wireCol->end(), [](const auto& left, const auto& right){return left.Channel() < right.Channel();});
317
318 evt.put(std::move(wireCol), rawDigitLabel.instance());
319 evt.put(std::move(wireDigitAssn), rawDigitLabel.instance());
320 }
321
322 fEventCount++;
323
324 return;
325} // produce

327float Decon1DROI::getTruncatedRMS(const std::vector<float>& waveform) const
328{
329 // do rms calculation - the old fashioned way and over all adc values
330 std::vector<float> locWaveform = waveform;
331
332 // sort in ascending order so we can truncate the sume
333 std::sort(locWaveform.begin(), locWaveform.end(),[](const auto& left, const auto& right){return std::fabs(left) < std::fabs(right);});
334
335 float threshold = fTruncRMSThreshold;
336
337 std::vector<float>::iterator threshItr = std::find_if(locWaveform.begin(),locWaveform.end(),[threshold](const auto& val){return std::fabs(val) > threshold;});
338
339 int minNumBins = std::max(int(fTruncRMSMinFraction * locWaveform.size()),int(std::distance(locWaveform.begin(),threshItr)));
340
341 // Get the truncated sum
342 float truncRms = std::inner_product(locWaveform.begin(), locWaveform.begin() + minNumBins, locWaveform.begin(), 0.);
343
344 truncRms = std::sqrt(std::max(0.,truncRms / double(minNumBins)));
345
346 return truncRms;
347}

349float Decon1DROI::fixTheFreakingWaveform(const std::vector<float>& waveform, raw::ChannelID_t channel, std::vector<float>& fixedWaveform) const
350{
351 // do rms calculation - the old fashioned way and over all adc values
352 std::vector<float> locWaveform = waveform;
353
354 // sort in ascending order so we can truncate the sume
355 std::sort(locWaveform.begin(), locWaveform.end(),[](const auto& left, const auto& right){return std::fabs(left) < std::fabs(right);});
356
357 // Get the mean of the waveform we're checking...
358 float sumWaveform = std::accumulate(locWaveform.begin(),locWaveform.begin() + locWaveform.size()/2, 0.);
359 float meanWaveform = sumWaveform / float(locWaveform.size()/2);
360
361 std::vector<float> locWaveformDiff(locWaveform.size()/2);
362
363 std::transform(locWaveform.begin(),locWaveform.begin() + locWaveform.size()/2,locWaveformDiff.begin(), std::bind(std::minus<float>(),std::placeholders::_1,meanWaveform));
364
365 float localRMS = std::inner_product(locWaveformDiff.begin(), locWaveformDiff.end(), locWaveformDiff.begin(), 0.);
366
367 localRMS = std::sqrt(std::max(float(0.),localRMS / float(locWaveformDiff.size())));
368
369 float threshold = 6. * localRMS;
370
371 std::vector<float>::iterator threshItr = std::find_if(locWaveform.begin(),locWaveform.end(),[threshold](const auto& val){return std::fabs(val) > threshold;});
372
373 int minNumBins = std::max(int(fTruncRMSMinFraction * locWaveform.size()),int(std::distance(locWaveform.begin(),threshItr)));
374
375 // recalculate the mean
376 float aveSum = std::accumulate(locWaveform.begin(), locWaveform.begin() + minNumBins, 0.);
377 float newPedestal = aveSum / minNumBins;
378
379 // recalculate the rms
380 locWaveformDiff.resize(minNumBins);
381
382 std::transform(locWaveform.begin(),locWaveform.begin() + minNumBins,locWaveformDiff.begin(), std::bind(std::minus<float>(),std::placeholders::_1,newPedestal));
383
384 localRMS = std::inner_product(locWaveform.begin(), locWaveform.begin() + minNumBins, locWaveform.begin(), 0.);
385 localRMS = std::sqrt(std::max(float(0.),localRMS / float(minNumBins)));
386
387 // Set the waveform to the new baseline
388 std::transform(waveform.begin(), waveform.end(), fixedWaveform.begin(), [newPedestal](const auto& val){return val - newPedestal;});
389
390 // Fill histograms
391 if (fOutputHistograms)
392 {
393 std::vector<geo::WireID> wids;
394 try
395 {
396 wids = fGeometry->ChannelToWire(channel);
397 }
398 catch(...)
399 {
400 std::cout << "Caught exception looking up channel" << std::endl;
401 return localRMS;
402 }
403
404 // Recover plane and wire in the plane
405 size_t plane = wids[0].Plane;
406 size_t wire = wids[0].Wire;
407
408// float fullRMS = std::inner_product(locWaveform.begin(), locWaveform.end(), locWaveform.begin(), 0.);
409
410// fullRMS = std::sqrt(std::max(float(0.),fullRMS / float(locWaveform.size())));
411
412 fPedestalOffsetVec[plane]->Fill(newPedestal,1.);
413// fFullRMSVec[plane]->Fill(fullRMS, 1.);
414 fTruncRMSVec[plane]->Fill(localRMS, 1.);
415 fNumTruncBinsVec[plane]->Fill(minNumBins, 1.);
416 fPedByChanVec[plane]->Fill(wire, newPedestal, 1.);
417 fTruncRMSByChanVec[plane]->Fill(wire, localRMS, 1.);
418 }
419
420 return localRMS;
421}

423void Decon1DROI::processChannel(size_t idx,
424 art::Event& event,
425 art::Handle<std::vector<raw::RawDigit>> digitVecHandle,
426 std::vector<recob::Wire>& wireColVec,
427 art::Assns<raw::RawDigit,recob::Wire>& wireAssns,
428 const std::string& instance) const
429{
430 // vector that will be moved into the Wire object
431 recob::Wire::RegionsOfInterest_t deconVec;
432 recob::Wire::RegionsOfInterest_t ROIVec;
433
434 // get the reference to the current raw::RawDigit
435 art::Ptr<raw::RawDigit> digitVec(digitVecHandle, idx);
436
437 raw::ChannelID_t channel = digitVec->Channel();
438
439 // The following test is meant to be temporary until the "correct" solution is implemented
440 if (!fChannelFilter->IsPresent(channel)) return;
441
442 // The waveforms should have been set to a 0. pedestal...
443 float pedestal = 0.;
444
445 // Recover the plane info
446 std::vector<geo::WireID> wids = fGeometry->ChannelToWire(channel);
447
448 // skip bad channels
449 if( fChannelFilter->Status(channel) < fMinAllowedChanStatus) return;
450
451 size_t dataSize = digitVec->Samples();
452
453 const geo::PlaneID& planeID = wids[0].planeID();
454
455 // vector holding uncompressed adc values
456 std::vector<short> rawadc(dataSize);
457
458 // uncompress the data
459 raw::Uncompress(digitVec->ADCs(), rawadc, digitVec->Compression());
460
461 // loop over all adc values and subtract the pedestal
462 // When we have a pedestal database, can provide the digit timestamp as the third argument of GetPedestalMean
463 try
464 {
465 pedestal = fPedRetrievalAlg->PedMean(channel);
466 }
467 catch(...)
468 {
469 mf::LogDebug("Decon1DROI_module") << "Pedestal lookup fails with channel: " << channel << std::endl;
470 return;
471 }
472
473
474 // Get the pedestal subtracted data, centered in the deconvolution vector
475 std::vector<float> rawAdcLessPedVec(dataSize);
476
477 std::transform(rawadc.begin(),rawadc.end(),rawAdcLessPedVec.begin(),std::bind(std::minus<short>(),std::placeholders::_1,pedestal));
478
479 // It seems there are deviations from the pedestal when using wirecell for noise filtering
480 //float raw_noise = fixTheFreakingWaveform(rawAdcLessPedVec, channel, rawAdcLessPedVec);
481 float raw_noise = digitVec->GetSigma();
482
483 // Recover a measure of the noise on the channel for use in the ROI finder
484 //float raw_noise = getTruncatedRMS(rawAdcLessPedVec);
485
486 // Try smoothing the input waveform
487// std::vector<float> rawAdcSmoothVec;
488// fWaveformTool->medianSmooth(rawAdcLessPedVec,rawAdcSmoothVec);
489
490 // Make a dummy candidate roi vec
491 icarus_tool::IROIFinder::CandidateROIVec deconROIVec;
492
493 deconROIVec.push_back(icarus_tool::IROIFinder::CandidateROI(0,rawAdcLessPedVec.size() - 1));
494
495 // Do the deconvolution on the full waveform
496 auto const clockData = art::ServiceHandle<detinfo::DetectorClocksService const>()->DataFor(event);
497 fDeconvolution->Deconvolve(rawAdcLessPedVec, sampling_rate(clockData), channel, deconROIVec, deconVec);
498
499 // Recover the deconvolved waveform
500 const std::vector<float>& deconvolvedWaveform = deconVec.get_ranges().front().data();
501
502 // vector of candidate ROI begin and end bins
503 icarus_tool::IROIFinder::CandidateROIVec candRoiVec;
504
505 // Now find the candidate ROI's
506 fROIFinderVec.at(planeID.Plane)->FindROIs(deconvolvedWaveform, channel, fEventCount, raw_noise, candRoiVec);
507
508 std::vector<float> holder;
509
510 // We need to copy the deconvolved (and corrected) waveform ROI's
511 for(const auto& candROI : candRoiVec)
512 {
513 // First up: copy out the relevent ADC bins into the ROI holder
514 size_t roiLen = candROI.second - candROI.first + 1;
515
516 holder.resize(roiLen);
517
518 std::copy(deconvolvedWaveform.begin()+candROI.first, deconvolvedWaveform.begin()+candROI.second, holder.begin());
519
520 // Now we do the baseline determination and correct the ROI
521 //float base = fBaseline->GetBaseline(holder, channel, roiStart, roiLen);
522 float base = fBaseline->GetBaseline(holder, channel, 0, roiLen);
523
524 std::transform(holder.begin(),holder.end(),holder.begin(),[base](auto& adcVal){return adcVal - base;});
525
526 // add the range into ROIVec
527 ROIVec.add_range(candROI.first, std::move(holder));
528 }
529
530 // Make some histograms?
531 if (fOutputHistograms)
532 {
533 fNumROIsHistVec.at(planeID.Plane)->Fill(candRoiVec.size(), 1.);
534
535 for(const auto& pair : candRoiVec)
536 fROILenHistVec.at(planeID.Plane)->Fill(pair.second-pair.first, 1.);
537
538 float fullRMS = std::inner_product(deconvolvedWaveform.begin(), deconvolvedWaveform.end(), deconvolvedWaveform.begin(), 0.);
539
540 fullRMS = std::sqrt(std::max(float(0.),fullRMS / float(deconvolvedWaveform.size())));
541
542 fFullRMSVec[planeID.Plane]->Fill(fullRMS, 1.);
543 }
544
545 // Don't save empty wires
546 if (ROIVec.empty()) return;
547
548 // First get a lock to make sure we are clear to run
549 tbb::spin_mutex::scoped_lock lock(deconvolutionSpinMutex);
550
551 // create the new wire directly in wirecol
552 wireColVec.push_back(recob::WireCreator(std::move(ROIVec),*digitVec).move());
553
554 // add an association between the last object in wirecol
555 // (that we just inserted) and digitVec
556 if (!util::CreateAssn(event, wireColVec, digitVec, wireAssns, instance))
557 {
