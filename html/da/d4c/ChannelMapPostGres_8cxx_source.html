64 static std::vector<std::string> tupleToVector(WDATuple& namesTuple)
65 {
66 unsigned int const n = getNfields(namesTuple);
67 std::vector<std::string> names{ n };
68 for (unsigned int const i: util::counter(n)) {
69 std::string& name = names[i]; // util::enumerate() fails on l-values
70 int error = 0;
71 name.resize(128, '\0');
72 std::size_t const length
73 = getStringValue(namesTuple, i, name.data(), name.size(), &error);
74 if (error) { // we know very little of the context... sorry, use gdb
75 throw cet::exception{ "ChannelMapPostGres" }
76 << "Failed (code: " << error << ") to extract column #" << i
77 << " name from a channel mapping database table";
78 }
79 name.erase(length);
80 }
81 return names;
82 } // tupleToVector()

175int icarusDB::ChannelMapPostGres::BuildTPCFragmentIDToReadoutIDMap
176 (TPCFragmentIDToReadoutIDMap& fragmentBoardMap) const
177{
178 using namespace std::string_literals;
179 constexpr unsigned int tpcIdentifier(0x00001000);
180
181 // Recover the data from the database
182 std::string const name { "icarus_hw_readoutboard" };
183 std::string const dataType { "readout_boards" };
184 details::WDADataset dataset = GetDataset(name, fDBURL, dataType);
185
186 // Include a by hand mapping of fragment ID to position in "corner" chimneys
187 // (this information is not present in the database)
188 static std::map<unsigned int, std::string> const flangeIDsuffixes = {
189 { 19, "T"s }, { 68, "M"s }, { 41, "B"s }, // WW01
190 { 55, "T"s }, { 97, "M"s }, { 100, "B"s }, // WW20
191 { 83, "T"s }, { 85, "M"s }, { 7, "B"s }, // WE01
192 { 48, "T"s }, { 13, "M"s }, { 56, "B"s }, // WE20
193 { 94, "T"s }, { 77, "M"s }, { 72, "B"s }, // EW01
194 { 53, "T"s }, { 82, "M"s }, { 35, "B"s }, // EW20
195 { 96, "T"s }, { 28, "M"s }, { 16, "B"s }, // EE01
196 { 86, "T"s }, { 54, "M"s }, { 8, "B"s } // EE20
197 }; // flangeIDsuffixes[]
198
199 auto const flangeSuffix = [](int flangeID) -> std::string
200 {
201 auto const it = flangeIDsuffixes.find(flangeID);
202 return (it == flangeIDsuffixes.end())? std::string{}: it->second;
203 };
204
205 // Loop through the data to recover the channels;
206 // find the position of the columns we need from the first row
207 /*
208 * [20240224] 9 columns:
209 * [0] "readout_board_id" [1] "flange_id" [2] "chimney_number"
210 * [3] "tpc_id" [4] "create_time" [5] "create_user"
211 * [6] "update_time" [7] "update_user" [8] "fragement_id"
212 */
213 // technical detail: `PositionFinder` uses `operator==` to compare, and the
214 // input is C-strings; we force the other term of comparison to C++ strings
215 auto const [
216 ReadoutBoardIDcolumn, FlangeIDcolumn, FragmentIDcolumn,
217 TPCIDcolumn, ChimneyNoColumn
218 ] = details::WDAPositionFinder{ getTuple(dataset, 0) }(
219 "readout_board_id"s, "flange_id"s, "fragement_id"s,
220 "tpc_id"s, "chimney_number"s
221 );
222
223 for (int const row: util::counter(1, getNtuples(dataset))) {
224 // Recover the row
225 details::WDATuple tuple { getTuple(dataset, row) };
226 if (!tuple) continue;
227
228 int error = 0;
229
230 // Note that the fragment ID is stored in the database as a string which
231 // reads as a hex number, meaning we have to read back as a string and
232 // decode to get the numerical value
233 util::Expected const fragmentIDString
234 = getStringFromTuple(tuple, FragmentIDcolumn);
235 if (!fragmentIDString) {
236 throw myException() << "Error (code: " << fragmentIDString.error()
237 << " on row " << row
238 << ") retrieving TPC fragment ID from channel mapping database\n";
239 }
240
241 unsigned int const fragmentID = std::stol(*fragmentIDString, nullptr, 16);
242 if (!(fragmentID & tpcIdentifier)) continue;
243
244 if (fragmentBoardMap.find(fragmentID) == fragmentBoardMap.end()) {
245
246 unsigned int const flangeID
247 = getLongValue(tuple, FlangeIDcolumn, &error);
248 if (error) {
249 throw myException() << "Error (code: " << error << " on row " << row
250 << ") retrieving TPC flange ID from channel mapping database\n";
251 }
252
253 // build the flange name
254 util::Expected const chimneyNoString
255 = getStringFromTuple(tuple, ChimneyNoColumn);
256 if (!chimneyNoString) {
257 throw myException() << "Error (code: " << chimneyNoString.error()
258 << " on row " << row
259 << ") retrieving chimney number from channel mapping database\n";
260 }
261
262 util::Expected const TPCIDstring = getStringFromTuple(tuple, TPCIDcolumn);
263 if (!TPCIDstring) {
264 throw myException() << "Error (code: " << TPCIDstring.error()
265 << " on row " << row
266 << ") retrieving TPC tag from channel mapping database\n";
267 }
268
269 std::string const flangeTag = TPCIDstring.value()
270 + std::string(std::max(0, 2-(int)chimneyNoString.value().length()), '0')
271 + chimneyNoString.value() + flangeSuffix(flangeID);
272
273 fragmentBoardMap[fragmentID].first = flangeTag;
274 } // if first time with this flange
275
276 unsigned int const readoutID
277 = getLongValue(tuple, ReadoutBoardIDcolumn, &error);
278 if (error) {
279 throw myException() << "Error (code: " << error << " on row " << row
280 << ") retrieving TPC readout board ID from channel mapping database\n";
281 }
282
283 fragmentBoardMap[fragmentID].second.emplace_back(readoutID);
284
285 } // for rows
286
287 return 0;
288} // icarusDB::ChannelMapPostGres::BuildTPCFragmentIDToReadoutIDMap()

292int icarusDB::ChannelMapPostGres::BuildTPCReadoutBoardToChannelMap
293 (TPCReadoutBoardToChannelMap& rbChanMap) const
294{
295 // Recover the data from the database
296 std::string const name { "icarus_hardware_prd" };
297 std::string const dataType { "daq_channels" };
298 details::WDADataset dataset = GetDataset(name, fDBURL, dataType);
299
300 // find the position of the columns we need from the first row
301 /*
302 * [20240224] 13 columns:
303 * [0] "channel_id" [1] "wire_number" [2] "readout_board_id"
304 * [3] "chimney_number" [4] "readout_board_slot" [5] "channel_number"
305 * [6] "create_time" [7] "create_user" [8] "update_time"
306 * [9] "update_user" [10] "plane" [11] "cable_label_number"
307 * [12] "channel_type"
308 */
309 // technical detail: `PositionFinder` uses `operator==` to compare, and the
310 // input is C-strings; we force the other term of comparison to C++ strings
311 using namespace std::string_literals;
312 auto const [
313 ChannelIDcolumn, ReadoutBoardIDcolumn, ReadoutBoardSlotColumn,
314 ChannelNumberColumn, PlaneIdentifierColumn
315 ]= details::WDAPositionFinder{ getTuple(dataset, 0) }(
316 "channel_id"s, "readout_board_id"s, "readout_board_slot"s,
317 "channel_number"s, "plane"s
318 );
319
320 // Loop through the data to recover the channels,
321 // making sure to skip the first (header) row
322 for(int const row: util::counter(1, getNtuples(dataset))) {
323 // Recover the row
324 details::WDATuple tuple { getTuple(dataset, row) };
325 if (!tuple) continue;
326 int error = 0;
327
328 unsigned int const readoutBoardID
329 = getLongValue(tuple, ReadoutBoardIDcolumn, &error);
330 if (error) {
331 throw myException() << "Error (code: " << error << " on row " << row
332 << ") reading TPC readout board ID\n";
333 }
334
335 if (rbChanMap.find(readoutBoardID) == rbChanMap.end()) {
336 unsigned int const readoutBoardSlot
337 = getLongValue(tuple, ReadoutBoardSlotColumn, &error);
338
339 if (error) {
340 throw myException() << "Error (code: " << error << " on row " << row
341 << ") reading TPC readout board slot\n";
342 }
343
344 rbChanMap[readoutBoardID].first = readoutBoardSlot;
345 rbChanMap[readoutBoardID].second.resize(ChannelsPerTPCreadoutBoard);
346 } // if new board ID
347
348 unsigned int const channelNum
349 = getLongValue(tuple, ChannelNumberColumn, &error);
350 if (error) {
351 throw myException() << "Error (code: " << error << " on row " << row
352 << ") reading TPC channel number\n";
353 }
354
355 unsigned int const channelID = getLongValue(tuple, ChannelIDcolumn, &error);
356 if (error) {
357 throw myException() << "Error (code: " << error << " on row " << row
358 << ") reading TPC channel ID\n";
359 }
360
361 // Recover the plane identifier
362 util::Expected const fragmentBuffer
363 = getStringFromTuple(tuple, PlaneIdentifierColumn);
364 if (!fragmentBuffer) {
365 throw myException() << "Error (code: " << fragmentBuffer.error()
366 << " on row " << row << ") reading plane type\n";
367 }
368 // Make sure lower case... (sigh...)
369 unsigned int const plane = TPCplaneIdentifierToPlane(*fragmentBuffer);
370 if (plane >= 3) {
371 mf::LogError{ "ChannelMapSQLite" } << "YIKES!!! Plane is " << plane
372 << " for channel " << channelID << " with type "
373 << fragmentBuffer.value();
374 }
375
376 rbChanMap[readoutBoardID].second[channelNum]
377 = std::make_pair(channelID, plane);
378 } // for rows
379
380 return 0; // on error, an exception was thrown
381
382} // icarusDB::ChannelMapPostGres::BuildTPCReadoutBoardToChannelMap()

390{
391 assert( !fCurrentPMTTimestamp.empty() );
392
393 // clearing is cleansing

395
396 // Recover the information from the database on the mapping
397 std::string const name { "Pmt_placement" };
398 std::string const dataType { "pmt_placements" };
399 std::string const period_query
400 { "&w=period_active:eq:" + fCurrentPMTTimestamp };
401 details::WDADataset dataset
402 = GetDataset(name, fDBURL + period_query, dataType);
403
404 // find the position of the columns we need from the first row
405 /*
406 * [20240224] 20 columns:
407 * [0] "pmt_id" [1] "period_active" [2] "pmt_in_tpc_plane"
408 * [3] "channel_id" [4] "pmt_sn" [5] "sector_label"
409 * [6] "ch_number" [7] "pmt_position_code" [8] "hv_cable_label"
410 * [9] "signal_cable_label" [10] "light_fiber_label" [11] "digitizer_label"
411 * [12] "digitizer_ch_number" [13] "hv_supply_label" [14]"hv_supply_ch_number"
412 * [15] "fragment_id" [16] "create_time" [17] "update_user"
413 * [18] "update_time" [19] "create_user"
414 */
415 // technical detail: `PositionFinder` uses `operator==` to compare, and the
416 // input is C-strings; we force the other term of comparison to C++ strings
417 using namespace std::string_literals;
418 auto const [
419 LaserChannelColumn, DigitizerChannelColumn, ChannelIDcolumn,
420 FragmentIDcolumn, DigitizerLabelColumn, LVDSconnectorColumn,
421 AdderConnectorColumn
422 ]= details::WDAPositionFinder{ getTuple(dataset, 0) }(
423 "light_fiber_label"s, "digitizer_ch_number"s, "channel_id"s,
424 "fragment_id"s, "digitizer_label"s, "fpga_connector_dio"s,
425 "adder_connector_dio"s
426 );
427
428 // Ok, now we can start extracting the information
429 // We do this by looping through the database and building the map from that
430 // NOTE that we skip the first row because that is just the labels
431 for (int const row: util::counter(1, getNtuples(dataset))) {
432 // Recover the row
433 details::WDATuple tuple { getTuple(dataset, row) };
434 if (!tuple) continue;
435
436 int error = 0;
437
438 PMTChannelInfo_t chInfo;
439 auto const nFields = static_cast<std::size_t>(getNfields(tuple));
440
441 // digitizer label
442 util::Expected const digitizerLabel
443 = getStringFromTuple(tuple, DigitizerLabelColumn);
444 if (!digitizerLabel) {
445 throw myException() << "Error (code: " << digitizerLabel.error()
446 << " on row " << row
447 << ") retrieving PMT digitizer from channel mapping database\n";
448 }
449 chInfo.digitizerLabel = *digitizerLabel;
450
451 // fragment id
452 unsigned long const fragmentID
453 = getLongValue(tuple, FragmentIDcolumn, &error);
454 if (error) {
455 throw myException() << "Error (code: " << error << " on row " << row
456 << ") reading PMT fragment ID\n";
457 }
458
459 // digitizer channel number
460 chInfo.digitizerChannelNo
461 = getLongValue(tuple, DigitizerChannelColumn, &error);
462 if (error) {
463 throw myException() << "Error (code: " << error << " on row " << row
464 << ") reading PMT readout board channel number\n";
465 }
466
467 // LArsoft channel ID
468 chInfo.channelID = getLongValue(tuple, ChannelIDcolumn, &error);
469 if (error) {
470 throw myException() << "Error (code: " << error << " on row " << row
471 << ") reading PMT channel ID\n";
472 }
473 // laser channel number
474 util::Expected laserChannelLabel
475 = getStringFromTuple(tuple, LaserChannelColumn);
476 if (!laserChannelLabel) {
477 throw myException() << "Error (code: " << laserChannelLabel.error()
478 << " on row " << row
479 << ") retrieving PMT laser channel from channel mapping database\n";
480 }
481 // will throw on error:
482 chInfo.laserChannelNo = std::stol(laserChannelLabel.value().substr(2));
483
484 // LVDS connector and bit
485 if (LVDSconnectorColumn < nFields) {
486 util::Expected LVDSconnectorLabel
487 = getStringFromTuple(tuple, LVDSconnectorColumn);
488 if (!LVDSconnectorLabel) {
489 throw myException() << "Error (code: " << LVDSconnectorLabel.error()
490 << " on row " << row
491 << ") retrieving LVDS connector from channel mapping database\n";
492 }
493 if (!LVDSconnectorLabel->empty() && (*LVDSconnectorLabel != "-")) {
494 auto const [ LVDSconnector, LVDSbit ]
495 = splitIntegers<2, unsigned short int>(*LVDSconnectorLabel, "-");
496 chInfo.LVDSconnector = LVDSconnector;
497 chInfo.LVDSbit = LVDSbit;
498 }
499 }
500
501 // adder connector and bit
502 if (AdderConnectorColumn < nFields) {
503 util::Expected adderConnectorLabel
504 = getStringFromTuple(tuple, AdderConnectorColumn);
505 if (!adderConnectorLabel) {
506 throw myException() << "Error (code: " << adderConnectorLabel.error()
507 << " on row " << row
508 << ") retrieving adder connector from channel mapping database\n";
509 }
510 if (!adderConnectorLabel->empty() && (*adderConnectorLabel != "-")) {
511 auto const [ adderConnector, adderBit ]
512 = splitIntegers<2, unsigned short int>(*adderConnectorLabel, "-");
513 chInfo.adderConnector = adderConnector;
514 chInfo.adderBit = adderBit;
515 }
516 }
517
518 // fill the map
519 fragmentToDigitizerChannelMap[fragmentID].push_back(std::move(chInfo));
520
521 } // for
522
523 return 0;
524} // icarusDB::ChannelMapPostGres::BuildPMTFragmentToDigitizerChannelMap()

670// -----------------------------------------------------------------------------
671int icarusDB::ChannelMapPostGres::BuildSideCRTCalibrationMap
672 (SideCRTChannelToCalibrationMap& sideCRTChannelToCalibrationMap) const
673{
674
675 std::string const name { "SideCRT_calibration_data" };
676
677 details::WDADataset dataset = GetCRTCaldata(name, fCRTcalibrationDBURL);
678 printDatasetError(dataset); // probably never prints since on error throws
679
680 // the shape of this database is different from the others,
681 // there is no header row and there is some "introductory" data
682 // (which we'll skip) in the first rows
683
684 for (int const row: util::counter(getNtuples(dataset))) {
685
686 // Get the row with an array of double
687 details::WDATuple tuple { getTuple(dataset, row) };
688 if (!tuple) continue;
689
690 int const ncols = getNfields(tuple);//check number of columns
691
692 // first few rows aren't actual data and have ncols==1: exclude those
693 if (ncols < 5) continue;
694
695 // assign values from the database to variables so we can use them
696 int error = 0;
697
698 int const mac5 = static_cast<int>(getDoubleValue(tuple, 1, &error));
699 if (error) {
700 throw myException() << "Error (code: " << error << " on row " << row
701 << ") reading CRT calibration Mac5 address\n";
702 }
703
704 int const channel = static_cast<int>(getDoubleValue(tuple, 2, &error));
705 if (error) {
706 throw myException() << "Error (code: " << error << " on row " << row
707 << ") reading CRT calibration channel\n";
708 }
709
710 double const gain = getDoubleValue(tuple, 3, &error);
711 if (error) {
712 throw myException() << "Error (code: " << error << " on row " << row
713 << ") reading CRT calibration gain for channel " << channel << "\n";
714 }
715
716 double const ped = getDoubleValue(tuple, 4, &error);
717 if (error) {
718 throw myException() << "Error (code: " << error << " on row " << row
719 << ") reading CRT calibration pedestal for channel " << channel << "\n";
720 }
721
722 // add the association between the two pairs to the map object
723 sideCRTChannelToCalibrationMap.emplace
724 (std::make_pair(mac5, channel), std::make_pair(gain, ped));
725
726 } // for rows
727
728 return 0;
729
730} // icarusDB::ChannelMapPostGres::BuildSideCRTCalibrationMap()
