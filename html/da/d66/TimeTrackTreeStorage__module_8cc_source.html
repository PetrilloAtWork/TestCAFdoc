72 std::vector<geo::Point_t> extractTrajectory
73 (recob::Track const& track, bool reverse = false)
74 {
75 std::vector<geo::Point_t> trackPath;
76 std::size_t index = track.FirstValidPoint();
77 while (index != recob::TrackTrajectory::InvalidIndex) {
78 trackPath.push_back(track.LocationAtPoint(index));
79 if (++index >= track.NPoints()) break;
80 index = track.NextValidPoint(index);
81 }
82 if (reverse) std::reverse(trackPath.begin(), trackPath.end());
83 return trackPath;
84 } // extractTrajectory()

659sbn::selHitInfo sbn::TimeTrackTreeStorage::makeHit(const recob::Hit &hit,
660 unsigned hkey,
661 const recob::Track &trk,
662 const recob::TrackHitMeta &thm,
663 const std::vector<art::Ptr<anab::Calorimetry>> &calo,
664 const geo::GeometryCore *geo)
665{
666
667 // TrackHitInfo to save
668 sbn::selHitInfo hinfo;
669
670 // information from the hit object
671 hinfo.integral = hit.Integral();
672 hinfo.sumadc = hit.SummedADC();
673 hinfo.width = hit.RMS();
674 hinfo.pk_time = hit.PeakTime();
675 hinfo.mult = hit.Multiplicity();
676 hinfo.wire = hit.WireID().Wire;
677 hinfo.plane = hit.WireID().Plane;
678 hinfo.channel = geo->PlaneWireToChannel(hit.WireID());
679 hinfo.tpc = hit.WireID().TPC;
680 hinfo.end = hit.EndTick();
681 hinfo.start = hit.StartTick();
682 hinfo.id = (int)hkey;
683
684 bool const badhit = (thm.Index() == std::numeric_limits<unsigned int>::max()) ||
685 (!trk.HasValidPoint(thm.Index()));
686
687 //hinfo.ontraj = !badhit;
688 // Save trajectory information if we can
689 if(!badhit)
690 {
691 geo::Point_t const& loc = trk.LocationAtPoint(thm.Index());
692 hinfo.px = loc.X();
693 hinfo.py = loc.Y();
694 hinfo.pz = loc.Z();
695
696 geo::Vector_t const& dir = trk.DirectionAtPoint(thm.Index());
697 hinfo.dirx = dir.X();
698 hinfo.diry = dir.Y();
699 hinfo.dirz = dir.Z();
700
701 // And determine if the Hit is on a Calorimetry object
702 for (const art::Ptr<anab::Calorimetry> &c: calo) {
703 if (c->PlaneID().Plane != hinfo.plane) continue;
704
705 // Found the plane! Now find the hit:
706 for (unsigned i_calo = 0; i_calo < c->dQdx().size(); i_calo++) {
707 // "TpIndices" match to the hit key
708 if (c->TpIndices()[i_calo] != hkey) continue;
709 // Fill the calo information associated with the hit
710 hinfo.oncalo = true;
711 hinfo.pitch = c->TrkPitchVec()[i_calo];
712 hinfo.dqdx = c->dQdx()[i_calo];
713 hinfo.dEdx = c->dEdx()[i_calo];
714 hinfo.rr = c->ResidualRange()[i_calo];
715 break;
716 } // for i_calo
717 break;
718 } // for c
719 }
720
721 return hinfo;
722}

734std::vector<std::pair<double, std::vector<raw::Channel_t>>>
735sbn::TimeTrackTreeStorage::computePMTwalls() const {
736
737 geo::GeometryCore const& geom { *lar::providerFrom<geo::Geometry>() };
738
739 // run the algorithm to identify the PMT walls (as groups of geo::OpDetGeo)
740 std::vector<std::pair<double, std::vector<geo::OpDetGeo const*>>> opDetWalls
741 = icarus::trigger::PMTverticalSlicingAlg{}.PMTwalls(geom);
742
743 // and weirdly, the only portable way to go from a OpDetGeo to its channel
744 // is to build a map (maybe because it's not guaranteed to be 1-to-1?)
745 std::map<geo::OpDetGeo const*, raw::Channel_t> opDetToChannel;
746 for (auto const channel: util::counter<raw::Channel_t>(geom.MaxOpChannel()))
747 opDetToChannel[&geom.OpDetGeoFromOpChannel(channel)] = channel;
748
749 // rewrite the data structure replacing each detector with its readout channel
750 std::vector<std::pair<double, std::vector<raw::Channel_t>>> channelWalls;
751 for (auto const& [ coord, PMTs ]: opDetWalls) {
752 std::vector<raw::Channel_t> channels;
753 channels.reserve(PMTs.size());
754 for (geo::OpDetGeo const* PMT: PMTs)
755 channels.push_back(opDetToChannel.at(PMT));
756
757 channelWalls.emplace_back(coord, std::move(channels));
758 } // for walls
759
760 return channelWalls;
761} // sbn::TimeTrackTreeStorage::computePMTwalls()
