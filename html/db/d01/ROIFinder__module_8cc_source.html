65struct PlaneWireData
66{
67 std::size_t size() const { return wires.size(); }
68 void resize(std::size_t nWires)
69 { wires.clear(); wires.resize(nWires, nullptr); }
70 void addWire(std::size_t iWire, recob::Wire const& wire)
71 { wires.at(iWire) = &wire; }
72 icarus_signal_processing::ArrayFloat operator() () const
73 {
74 icarus_signal_processing::ArrayFloat data;
75 data.resize(wires.size());
76 for (auto [ iWire, wire ]: util::enumerate(wires))
77 {
78 if (wire) data[iWire] = wire->Signal();
79 else data[iWire] = std::vector<float>(4096,0.);
80 }
81 return data;
82 }
83 const recob::Wire* getWirePtr(size_t idx) const {return wires[idx];}
84private:
85 std::vector<recob::Wire const*> wires;
86}; // PlaneWireData

197void ROIFinder::reconfigure(fhicl::ParameterSet const& pset)
198{
199 // Recover the parameters
200 fWireModuleLabelVec = pset.get<std::vector<art::InputTag>>("WireModuleLabelVec", std::vector<art::InputTag>()={"decon1droi"});
201 fOutInstanceLabelVec = pset.get<std::vector<std::string>> ("OutInstanceLabelVec", {"PHYSCRATEDATA"});
202 fCorrectROIBaseline = pset.get<bool >("CorrectROIBaseline", false);
203 fMinSizeForCorrection = pset.get<size_t >("MinSizeForCorrection", 12);
204 fMaxSizeForCorrection = pset.get<size_t >("MaxSizeForCorrection", 512);
205 fLeadTrail = pset.get<size_t >("LeadTrail", 0);
206 fOutputMorphed = pset.get< bool >("OutputMorphed", true);
207 fDiagnosticOutput = pset.get< bool >("DaignosticOutput", false);
208 fOutputHistograms = pset.get< bool >("OutputHistograms", false);
209
210 // Access ART's TFileService, which will handle creating and writing
211 // histograms and n-tuples for us.
212 art::ServiceHandle<art::TFileService> tfs;
213
214 // Recover the list of ROI finding tools
215 const fhicl::ParameterSet& roiFinderTools = pset.get<fhicl::ParameterSet>("ROIFinderToolVec");
216
217 // Make a mapping between plane id and a plane's ROI finder
218 // This allows different ROI finders per plane but, more important, different parameters
219 for(const std::string& roiFinderTool : roiFinderTools.get_pset_names())
220 {
221 const fhicl::ParameterSet& roiFinderToolParamSet = roiFinderTools.get<fhicl::ParameterSet>(roiFinderTool);
222 size_t planeIdx = roiFinderToolParamSet.get<size_t>("Plane");
223
224 fROIToolMap[planeIdx] = art::make_tool<icarus_tool::IROILocator> (roiFinderToolParamSet);
225
226 if (fOutputHistograms)
227 {
228 std::string dirName = "ROIFinder_" + std::to_string(planeIdx);
229
230 art::TFileDirectory dir = tfs->mkdir(dirName);
231
232 fROIToolMap[planeIdx]->initializeHistograms(dir);
233 }
234 }
235
236 return;
237}

251void ROIFinder::produce(art::Event& evt)
252{
253 // We need to loop through the list of Wire data we have been given
254 for(size_t labelIdx = 0; labelIdx < fWireModuleLabelVec.size(); labelIdx++)
255 {
256 const art::InputTag& wireLabel = fWireModuleLabelVec[labelIdx];
257
258 // make a collection of Wires
259 std::unique_ptr<std::vector<recob::Wire>> wireCol(new std::vector<recob::Wire>);
260
261 std::unique_ptr<std::vector<recob::Wire>> morphedCol(new std::vector<recob::Wire>);
262
263 // make a collection of ChannelROI objects
264 std::unique_ptr<std::vector<recob::ChannelROI>> channelROICol(new std::vector<recob::ChannelROI>);
265
266 mf::LogInfo("ROIFinder") << "ROIFinder, looking for decon1droi data at " << wireLabel << std::endl;
267
268 // Read in the collection of full length deconvolved waveforms
269 // Note we assume this list is sorted in increasing channel number!
270 art::Handle< std::vector<recob::Wire>> wireVecHandle;
271
272 evt.getByLabel(wireLabel, wireVecHandle);
273
274 mf::LogInfo("ROIFinder") << "--> Recovered wire data, size: " << wireVecHandle->size() << std::endl;
275
276 if (!wireVecHandle->size())
277 {
278 evt.put(std::move(wireCol), wireLabel.instance());
279 fEventCount++;
280
281 return;
282 }
283
284 // The first step is to break up into groups by logical TPC/plane in order to do the parallel loop
285 PlaneIDToDataPairMap planeIDToDataPairMap;
286 PlaneIDVec planeIDVec;
287 size_t numChannels(0);
288
289 for(const auto& wire : *wireVecHandle)
290 {
291 raw::ChannelID_t channel = wire.Channel();
292
293 std::vector<geo::WireID> wireIDVec = fGeometry->ChannelToWire(channel);
294
295 if (wireIDVec.empty()) continue;
296
297 for(const auto& wireID : wireIDVec)
298 {
299 const geo::PlaneID& planeID = wireID.planeID();
300
301 PlaneIDToDataPairMap::iterator mapItr = planeIDToDataPairMap.find(planeID);
302
303 // Make sure the array is initialized
304 if (mapItr == planeIDToDataPairMap.end())
305 {
306 unsigned int nWires = fGeometry->Nwires(planeID);
307
308 std::pair<PlaneIDToDataPairMap::iterator,bool> mapInsert = planeIDToDataPairMap.insert({planeID,PlaneIDToDataPair()});
309
310 if (!mapInsert.second) std::cout << "Failed to insert, is this possible?" << std::endl;
311
312 mapItr = mapInsert.first;
313
314 mapItr->second.first.resize(nWires);
315 mapItr->second.second.resize(nWires);
316
317 planeIDVec.emplace_back(planeID);
318 }
319
320 // Add waveform to the 2D array
321 mapItr->second.first[wireID.Wire] = channel;
322 mapItr->second.second.addWire(wireID.Wire, wire);
323 numChannels++;
324 }
325 }
326
327 // We might need this... it allows a temporary wire object to prevent crashes when some data is missing
328 std::vector<recob::Wire> tempWireVec(numChannels/4);
329
330 // Check integrity of map
331 for(auto& mapInfo : planeIDToDataPairMap)
332 {
333 const std::vector<raw::ChannelID_t>& channelVec = mapInfo.second.first;
334 const icarus_signal_processing::ArrayFloat& dataArray = mapInfo.second.second();
335
336 for(size_t idx = 0; idx < channelVec.size(); idx++)
337 {
338 if (dataArray[idx].size() < 100)
339 {
340 mf::LogInfo("ROIFinder") << " **> Found truncated wire, size: " << dataArray[idx].size() << ", channel: " << channelVec[idx] << std::endl;
341
342 std::vector<float> zeroVec(4096,0.);
343 recob::Wire::RegionsOfInterest_t ROIVec;
344
345 ROIVec.add_range(0, std::move(zeroVec));
346
347 std::vector<geo::WireID> wireIDVec = fGeometry->ChannelToWire(channelVec[idx]);
348
349 // Given channel a large number so we know to not save
350 mapInfo.second.first[idx] = 100000 + idx;
351
352 tempWireVec.emplace_back(recob::WireCreator(std::move(ROIVec),idx,fGeometry->View(wireIDVec[0].planeID())).move());
353
354 mapInfo.second.second.addWire(idx,tempWireVec.back());
355 }
356 }
357 }
358
359 // Reserve the room for the output
360 wireCol->reserve(wireVecHandle->size());
361
362 // ... Launch multiple threads with TBB to do the deconvolution and find ROIs in parallel
363 multiThreadDeconvolutionProcessing deconvolutionProcessing(*this, evt, planeIDVec, planeIDToDataPairMap, *wireCol, *morphedCol, *channelROICol);
364
365 tbb::parallel_for(tbb::blocked_range<size_t>(0, planeIDVec.size()), deconvolutionProcessing);
366
367 // Time to stroe everything
368 if(wireCol->size() == 0) mf::LogWarning("ROIFinder") << "No wires made for this event.";
369
370 evt.put(std::move(wireCol), fOutInstanceLabelVec[labelIdx]);
371 evt.put(std::move(channelROICol), fOutInstanceLabelVec[labelIdx]);
372
373 if (fOutputMorphed) evt.put(std::move(morphedCol), fOutInstanceLabelVec[labelIdx]+"M");
374 }
375
376 fEventCount++;
377
378 return;
379} // produce

381void ROIFinder::processPlane(size_t idx,
382 art::Event& event,
383 const PlaneIDVec& planeIDVec,
384 const PlaneIDToDataPairMap& planeIDToDataPairMap,
385 std::vector<recob::Wire>& wireColVec,
386 std::vector<recob::Wire>& morphedVec,
387 std::vector<recob::ChannelROI>& channelROIVec) const
388{
389 // Recover the planeID for this thread
390 const geo::PlaneID& planeID = planeIDVec[idx];
391
392 // And the data array
393 PlaneIDToDataPairMap::const_iterator mapItr = planeIDToDataPairMap.find(planeID);
394
395 if (mapItr == planeIDToDataPairMap.end())
396 {
397 std::cout << "We know this cannot happen" << std::endl;
398 return;
399 }
400
401 const PlaneIDToDataPair& planeIDToDataPair = mapItr->second;
402
403 const icarus_signal_processing::ArrayFloat& dataArray = planeIDToDataPair.second();
404 const std::vector<raw::ChannelID_t>& channelVec = planeIDToDataPair.first;
405
406 // Keep track of our selected values
407 icarus_signal_processing::ArrayFloat outputArray(dataArray.size(),icarus_signal_processing::VectorFloat(dataArray[0].size(),0.));
408 icarus_signal_processing::ArrayBool selectedVals(dataArray.size(),icarus_signal_processing::VectorBool(dataArray[0].size(),false));
409
410 fROIToolMap.at(planeID.Plane)->FindROIs(event, dataArray, channelVec, mapItr->first, outputArray, selectedVals);
411
412// icarus_signal_processing::ArrayFloat morphedWaveforms(dataArray.size());
413
414 // Copy the "morphed" array
415 if (fOutputMorphed)
416 {
417 for(size_t waveIdx = 0; waveIdx < outputArray.size(); waveIdx++)
418 {
419 // skip if a bad channbel
420 if (channelVec[idx] >= 100000) continue;
421
422 // First get a lock to make sure we don't conflict
423 tbb::spin_mutex::scoped_lock lock(roifinderSpinMutex);
424
425 recob::Wire::RegionsOfInterest_t ROIVec;
426
427 ROIVec.add_range(0, std::move(outputArray[waveIdx]));
428
429 raw::ChannelID_t channel = channelVec[waveIdx];
430 geo::View_t view = fGeometry->View(channel);
431
432 std::vector<geo::WireID> chanIDVec = fGeometry->ChannelToWire(channel);
433
434 morphedVec.push_back(recob::WireCreator(std::move(ROIVec),channel,view).move());
435 }
436 }
437
438 // Ok, now go through the refined selected values array and find ROIs
439 // Define the ROI and its container
440 using CandidateROI = std::pair<size_t, size_t>;
441 using CandidateROIVec = std::vector<CandidateROI>;
442
443 for(size_t waveIdx = 0; waveIdx < selectedVals.size(); waveIdx++)
444 {
445 // Skip if a bad channel
446 if (channelVec[waveIdx] >= 100000)
447 {
448 std::cout << "==> found an unexpected channel number: " << channelVec[waveIdx] << std::endl;
449 continue;
450 }
451
452 // Set up an object...
453 CandidateROIVec candidateROIVec;
454
455 // Search for ROIs in current waveform
456 const icarus_signal_processing::VectorBool& selVals = selectedVals[waveIdx];
457
458 size_t idx(2);
459
460 while(idx < selVals.size())
461 {
462 if (selVals[idx])
463 {
464 Size_t startTick = idx >= fLeadTrail ? idx - fLeadTrail : 0;
465
466 while(idx < selVals.size() && selVals[idx]) idx++;
467
468 size_t stopTick = idx < selVals.size() - fLeadTrail ? idx + fLeadTrail : selVals.size();
469
470 candidateROIVec.emplace_back(startTick, stopTick);
471 }
472
473 idx++;
474 }
475
476 // merge overlapping (or touching) ROI's
477 if(candidateROIVec.size() > 1)
478 {
479 // temporary vector for merged ROIs
480 CandidateROIVec tempRoiVec;
481
482 // Loop through candidate roi's
483 size_t startRoi = candidateROIVec.front().first;
484 size_t stopRoi = candidateROIVec.front().second; //startRoi;
485
486 for(auto& roi : candidateROIVec)
487 {
488 // Should we merge roi's?
489 if (roi.first <= stopRoi)
490 {
491 // Make sure the merge gets the right start/end times
492 startRoi = std::min(startRoi,roi.first);
493 stopRoi = std::max(stopRoi,roi.second);
494 }
495 else
496 {
497 tempRoiVec.emplace_back(startRoi,stopRoi);
498
499 startRoi = roi.first;
500 stopRoi = roi.second;
501 }
502 }
503
504 // Make sure to get the last one
505 tempRoiVec.emplace_back(startRoi,stopRoi);
506
507 candidateROIVec = tempRoiVec;
508 }
509
510 // If no candidates no need for further effort
511 if (!candidateROIVec.empty())
512 {
513 // vector that will be moved into the Wire object
514 recob::Wire::RegionsOfInterest_t ROIVec;
515 recob::ChannelROI::RegionsOfInterest_t intROIVec;
516
517 // Check if we have possible overlap wires where we need to merge the previous results with new results
518 if (planeID.Plane > 0)
519 {
520 raw::ChannelID_t channel = channelVec[waveIdx];
521
522 std::vector<geo::WireID> wireIDVec = fGeometry->ChannelToWire(channel);
523
524 if (wireIDVec.size() > 1)
525 {
526 // First get a lock to make sure we don't conflict
527 tbb::spin_mutex::scoped_lock lock(roifinderSpinMutex);
528
529 std::vector<recob::Wire>::iterator wireItr = std::find_if(wireColVec.begin(),wireColVec.end(),[channel](const auto& wire){return wire.Channel() == channel;});
530
531 if (wireItr != wireColVec.end())
532 {
533 ROIVec = wireItr->SignalROI();
534
535 // Avoid duplicate entries by erasing the previous instance
536 wireColVec.erase(wireItr);
537 }
538
539 std::vector<recob::ChannelROI>::iterator channelItr = std::find_if(channelROIVec.begin(),channelROIVec.end(),[channel](const auto& channelROI){return channelROI.Channel() == channel;});
540
541 if (channelItr != channelROIVec.end())
542 {
543 intROIVec = channelItr->SignalROI();
544
545 // Avoid duplicate entries by erasing the previous instance
546 channelROIVec.erase(channelItr);
547 }
548 }
549 }
550
551 if (ROIVec.size() != intROIVec.size())
552 throw art::Exception(art::errors::LogicError) << "===> ROIVec mismatch to intROIVec, ROIVec size: " << ROIVec.size() << ", intROIVec size: " << intROIVec.size() << "\n";
553
554 const icarus_signal_processing::VectorFloat& waveform = dataArray[waveIdx];
555
556 // We need to copy the deconvolved (and corrected) waveform ROI's
557 for(const auto& candROI : candidateROIVec)
558 {
559 // First up: copy out the relevent ADC bins into the ROI holder
560 size_t roiLen = candROI.second - candROI.first;
561 size_t firstBin = candROI.first;
562
563 icarus_signal_processing::VectorFloat holder(roiLen);
564
565 std::copy(waveform.begin()+candROI.first, waveform.begin()+candROI.second, holder.begin());
566
567 // Now we do the baseline determination and correct the ROI
568 // For now we are going to reset to the minimum element
569 // Get slope/offset from first to last ticks
570 if (fCorrectROIBaseline && holder.size() > fMinSizeForCorrection && holder.size() < fMaxSizeForCorrection)
571 {
572 // Try to find the minimum value in the leading and trailing bins
573 size_t nBins = holder.size()/3;
574 icarus_signal_processing::VectorFloat::iterator firstItr = std::min_element(holder.begin(),holder.begin()+nBins);
575 icarus_signal_processing::VectorFloat::iterator lastItr = std::min_element(holder.end()-nBins,holder.end());
576
577 size_t newSize = std::distance(firstItr,lastItr) + 1;
578 float dADC = (*lastItr - *firstItr) / float(newSize);
579 float offset = *firstItr;
580
581 for(size_t binIdx = 0; binIdx < newSize; binIdx++)
582 {
583 holder[binIdx] = *(firstItr + binIdx) - offset;
584 offset += dADC;
585 }
586
587 firstBin += std::distance(holder.begin(),firstItr);
588
589 holder.resize(newSize);
590 }
591
592 // Now make the short int version
593 icarus_signal_processing::VectorShort intHolder(holder.size());
594
595 for(size_t binIdx = 0; binIdx < holder.size(); binIdx++) intHolder[binIdx] = std::round(holder[binIdx]);
596
597 // add the range into ROIVec
598 ROIVec.add_range(firstBin, std::move(holder));
599 intROIVec.add_range(firstBin, std::move(intHolder));
600 }
601
602 // Check for emptiness
603 if (!ROIVec.empty())
604 {
605 // First get a lock to make sure we don't conflict
606 tbb::spin_mutex::scoped_lock lock(roifinderSpinMutex);
607
608 raw::ChannelID_t channel = channelVec[waveIdx];
609 geo::View_t view = fGeometry->View(channel);
610
611 // Since we process logical TPC images we need to watch for duplicating entries
612 // We can do that by checking to see if a channel has already been added...
613 std::vector<geo::WireID> wireIDVec = fGeometry->ChannelToWire(channel);
614
615 if (channelROIVec.size() != wireColVec.size())
616 throw art::Exception(art::errors::LogicError) << "===> ROI output mismatch, channelROIVec, size: " << channelROIVec.size() << ", wireColVec, size: " << wireColVec.size() << "\n";
617
618 channelROIVec.push_back(recob::ChannelROICreator(std::move(intROIVec),channel).move());
619 wireColVec.push_back(recob::WireCreator(std::move(ROIVec),channel,view).move());
620 }
621 }
622 }
623
624 return;
625}

627float ROIFinder::getMedian(icarus_signal_processing::VectorFloat vals, const unsigned int nVals) const
628{
629 float median(0.);
630
631 if (nVals > 2)
632 {
633 if (nVals % 2 == 0)
634 {
635 const auto m1 = vals.begin() + nVals / 2 - 1;
636 const auto m2 = vals.begin() + nVals / 2;
637 std::nth_element(vals.begin(), m1, vals.begin() + nVals);
638 const auto e1 = *m1;
639 std::nth_element(vals.begin(), m2, vals.begin() + nVals);
640 const auto e2 = *m2;
