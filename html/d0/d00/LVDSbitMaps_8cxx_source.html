140// -----------------------------------------------------------------------------
141auto icarus::trigger::LVDSbitMaps::buildChannelInfo
142 (icarusDB::IICARUSChannelMapProvider const& channelMap) const
143 -> std::vector<LVDSchannel_t>
144{
145 std::vector<LVDSchannel_t> PMTchannelDB;
146
147 mf::LogTrace debugLog{ "LVDSbitMaps" };
148 debugLog << "LVDSbitMaps::buildChannelInfo() collecting channel info:";
149
150 for (unsigned int const fragmentID: util::counter(0x2000, 0x2018)) {
151 if (!channelMap.hasPMTDigitizerID(fragmentID)) {
152 throw Exception{ Errors::NoPMTfragment }
153 << "PMT fragment ID 0x" << std::hex << fragmentID << std::dec
154 << " not available in the channel mapping database.\n";
155 }
156
157 for (icarusDB::PMTChannelInfo_t const& info
158 : channelMap.getPMTchannelInfo(fragmentID)
159 ) {
160 unsigned short int const cryostat = info.channelID / NChannelsPerCryostat;
161
162 debugLog << "\n[#" << PMTchannelDB.size() << "] CH=" << info.channelID
163 << " " << cryostat << "/" << info.LVDSconnector << "/" << info.LVDSbit;
164
165 LVDSHWbitID const PMTpairBit = info.hasLVDSinfo()
166 ? LVDSHWbitID{ cryostat, info.LVDSconnector, info.LVDSbit }
167 : LVDSHWbitID{}
168 ;
169 LVDSHWbitID const adderBit = info.hasAdderInfo()
170 ? LVDSHWbitID{ cryostat, info.adderConnector, info.adderBit }
171 : LVDSHWbitID{}
172 ;
173
174 PMTchannelDB.push_back(LVDSchannel_t{
175 info.channelID // .channel
176 , PMTpairBit // .PMTpairSource
177 , adderBit // .adderSource
178 });
179 } // for channel in fragment
180 } // for fragment ID
181
182 assert(PMTchannelDB.size() == NChannels); // insider knowledge...
183
184 std::sort(PMTchannelDB.begin(), PMTchannelDB.end());
185
186 return PMTchannelDB;
187
188} // icarus::trigger::LVDSbitMaps::buildChannelInfo()

191// -----------------------------------------------------------------------------
192auto icarus::trigger::LVDSbitMaps::buildPMTpairSourceMap
193 (icarusDB::IICARUSChannelMapProvider const& channelMap) const
194 -> std::vector<LVDSinfo_t>
195{
196 // note that this function must support also the case where there is no PMT
197 // pair information in the channel mapping database, in which case all
198 // connector information will be "invalid" and protocol demands an empty map
199
200 assert(fPMTchannelInfo.size() == NChannels); // insider knowledge...
201
202 /*
203 * The logic of the filling is a bit complicated. Class documentation explains
204 * it and shows it to be:
205 *
206 * east cryostat, east wall: `0x00aAbBcC'00dDeEfF`, channel `0` the most
207 * significant bit in `a` (bit #55) and channel `45` the one in `d` (bit 23);
208 * east cryostat, west wall: `0x00aAbBcC'00dDeEfF`, channel `90` the most
209 * significant bit in `a` (bit #55) and channel `135` the one in `d` (bit 23);
210 * west cryostat, east wall: `0x00aAbBcC'00dDeEfF`, channel `180` the most
211 * significant bit in `a` (bit #55) and channel `225` the one in `d` (bit 23);
212 * west cryostat, west wall: `0x00aAbBcC'00dDeEfF`, channel `270` the most
213 * significant bit in `a` (bit #55) and channel `315` the one in `d` (bit 23).
214 *
215 * How we get there:
216 * * we sort the filling with an ordered progression of channels, lowest
217 * first, and because most LVDS bit are associated to two channels, we
218 * track the channels we already covered when processing the other PMT
219 * channel of the pair, and skip them when met those LVDS bit again
220 * * the destination bit is also an ordered progression within the word,
221 * but inverted because the _most_ significant bit is assigned to the
222 * _lowest_ channels
223 */
224 std::vector<LVDSinfo_t> sourceMap{ PMTpairBitID::NIndices() };
225 std::vector<PMTpairBitID> HWtoLogicID{ LVDSHWbitID::NIndices() };
226
227 unsigned int nBits = 0;
228 for (LVDSchannel_t const& info: fPMTchannelInfo) { // in PMT channel order
229
230 if (!info.PMTpairSource) continue; // skip invalid
231
232 LVDSHWbitID::Index_t const sourceIndex = info.PMTpairSource.index();
233
234 PMTpairBitID logicBit = HWtoLogicID[sourceIndex];
235
236 LVDSinfo_t* mapInfo = logicBit? &(sourceMap[logicBit.index()]): nullptr;
237 if (!mapInfo) { // new bit: set the source
238 // which logic bit to set:
239 logicBit.cryostat = info.PMTpairSource.cryostat;
240 assert(logicBit.cryostat == info.channel / NChannelsPerCryostat);
241 logicBit.PMTwall
242 = info.channel/(NChannelsPerCryostat/PMTpairBitID::NPMTwallsPerCryostat)
243 % PMTpairBitID::NCryostats;
244 logicBit.statusBit = 47 - (nBits++ % 48); // 0 => 47, 1 => 46,..., 47 => 0
245 if (logicBit.statusBit >= 24)
246 logicBit.statusBit += 8; // 0 => 0,..., 23 => 23, 24 => 32, 25 => 33,...
247
248 assert(logicBit);
249 HWtoLogicID[sourceIndex] = logicBit;
250 mapInfo = &(sourceMap[logicBit.index()]);
251 mapInfo->ID = logicBit;
252 mapInfo->source = info.PMTpairSource;
253 }
254 assert(mapInfo);
255 assert(mapInfo->source == info.PMTpairSource);
256
257 mapInfo->channels.push_back(info.channel);
258 // check that they are sorted (the input list should be sorted already...)
259 assert((mapInfo->channels.size() == 1)
260 || (
261 mapInfo->channels[mapInfo->channels.size()-2]
262 < mapInfo->channels[mapInfo->channels.size()-1]
263 ));
264
265 } // for fPMTchannelInfo
266
267
268 // --- BEGIN DEBUG -----
269 mf::LogTrace debugLog{ "LVDSbitMaps" };
270 debugLog << "Final PMT pair bit map:";
271 for (auto const& [ bit, info ]: util::enumerate(sourceMap)) {
272 debugLog << "\n[" << std::setw(3) << bit << "] logic="
273 << PMTpairBitID::fromIndex(bit) << " => HW=";
274 if (info.source) {
275 debugLog << info.source << " (ix="
276 << std::setw(3) << info.source.index() << ") "
277 << info.channels.size() << " channels {";
278 for (raw::Channel_t const channel: info.channels)
279 debugLog << " " << std::setw(3) << channel;
280 debugLog << " }";
281 }
282 else {
283 debugLog << "(invalid)";
284 }
285 }
286 // --- END DEBUG -------
287
288 assert((nBits == 0) || (nBits == 192)); // 0 if no info in mapping database
289 return (nBits == 0)? std::vector<LVDSinfo_t>{}: sourceMap;
290
291} // icarus::trigger::LVDSbitMaps::buildPMTpairSourceMap()

294// -----------------------------------------------------------------------------
295auto icarus::trigger::LVDSbitMaps::buildAdderSourceMap
296 (icarusDB::IICARUSChannelMapProvider const& channelMap) const
297 -> std::vector<AdderInfo_t>
298{
299 // note that this function must support also the case where there is no adder
300 // information in the channel mapping database, in which case all connector
301 // information will be "invalid" and the protocol demands an empty map
302
303 assert(fPMTchannelInfo.size() == NChannels); // insider knowledge...
304
305 /*
306 * The logic of the filling is a bit complicated. Class documentation explains
307 * it and shows it to be:
308 *
309 * east cryostat, east wall: `0x0a000000'0A000000`, channel `0` the most
310 * significant bit in `a` (bit #58) and channel `45` the one in `A` (bit 26);
311 * east cryostat, west wall: `0x0a000000'0A000000`, channel `90` the most
312 * significant bit in `a` (bit #58) and channel `135` the one in `d` (bit 26);
313 * west cryostat, east wall: `0x0a000000'0A000000`, channel `180` the most
314 * significant bit in `a` (bit #58) and channel `225` the one in `d` (bit 26);
315 * west cryostat, west wall: `0x0a000000'0A000000`, channel `270` the most
316 * significant bit in `a` (bit #58) and channel `315` the one in `d` (bit 26).
317 *
318 * How we get there:
319 * * we sort the filling with an ordered progression of channels, lowest
320 * first, and because adder bit are associated to 15 channels, we
321 * track the channels we already covered when processing the other PMT
322 * channels of the adder, and skip them when met those LVDS bit again
323 * * the destination bit is also an ordered progression within the word,
324 * but inverted because the _most_ significant bit is assigned to the
325 * _lowest_ channels
326 */
327 std::vector<AdderInfo_t> sourceMap{ AdderBitID::NIndices() };
328 std::vector<AdderBitID> HWtoLogicID{ LVDSHWbitID::NIndices() };
329
330 unsigned int nBits = 0;
331 for (LVDSchannel_t const& info: fPMTchannelInfo) { // in PMT channel order
332
333 if (!info.adderSource) continue; // skip invalid
334
335 LVDSHWbitID::Index_t const sourceIndex = info.adderSource.index();
336
337 AdderBitID logicBit = HWtoLogicID[sourceIndex];
338
339 AdderInfo_t* mapInfo = logicBit? &(sourceMap[logicBit.index()]): nullptr;
340 if (!mapInfo) { // new bit: set the source
341 // which logic bit to set:
342 logicBit.cryostat = info.adderSource.cryostat;
343 assert(logicBit.cryostat == info.channel / NChannelsPerCryostat);
344 logicBit.PMTwall
345 = info.channel / (NChannelsPerCryostat/AdderBitID::NPMTwallsPerCryostat)
346 % AdderBitID::NCryostats;
347 logicBit.statusBit = 5 - (nBits++ % 6); // 0/6/... => 5, 1/7/... => 4, ...
348
349 assert(logicBit);
350 HWtoLogicID[sourceIndex] = logicBit;
351 mapInfo = &(sourceMap[logicBit.index()]);
352 mapInfo->ID = logicBit;
353 mapInfo->source = info.adderSource;
354 }
355 assert(mapInfo);
356 assert(mapInfo->source == info.adderSource);
357
358 mapInfo->channels.push_back(info.channel);
359 // check that they are sorted (the input list should be sorted already...)
360 assert((mapInfo->channels.size() == 1)
361 || (
362 mapInfo->channels[mapInfo->channels.size()-2]
363 < mapInfo->channels[mapInfo->channels.size()-1]
364 ));
365
366 } // for PMTchannelDB
367
368
369 // --- BEGIN DEBUG -----
370 mf::LogTrace debugLog{ "LVDSbitMaps" };
371 debugLog << "Final adder bit map:";
372 for (auto const& [ bit, info ]: util::enumerate(sourceMap)) {
373 debugLog << "\n[" << std::setw(3) << bit << "] logic="
374 << AdderBitID::fromIndex(bit) << " => HW=";
375 if (info.source) {
376 debugLog << info.source << " (ix="
377 << std::setw(3) << info.source.index() << ") "
378 << info.channels.size() << " channels {";
379 for (raw::Channel_t const channel: info.channels)
380 debugLog << " " << std::setw(3) << channel;
381 debugLog << " }";
382 }
383 else {
384 debugLog << "(invalid)";
385 }
386 }
387 // --- END DEBUG -------
388
389 assert((nBits == 0) || (nBits == 24)); // 0 if no info in mapping database
390 return (nBits == 0)? std::vector<AdderInfo_t>{}: sourceMap;
391
392} // icarus::trigger::LVDSbitMaps::buildAdderSourceMap()
