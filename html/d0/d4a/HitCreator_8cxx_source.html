387 void HitCollectionAssociator::prepare_associations(std::vector<icarus::Hit> const& srchits)
388 {
389 if (!RawDigitAssns && !WireAssns) return; // no associations needed
390 assert(event);
391
392 // we make the associations anew
393 if (RawDigitAssns) ClearAssociations(*RawDigitAssns);
394 if (WireAssns) ClearAssociations(*WireAssns);
395
396 // the following is true is we want associations with digits
397 // but we don't know where digits are; in that case, we try to use wires
398 const bool bUseWiresForDigits = RawDigitAssns && (digits_label == "");
399
400 if (WireAssns || bUseWiresForDigits) {
401 // do we use wires for digit associations too?
402
403 // get the wire collection
404 art::ValidHandle<std::vector<recob::ChannelROI>> hWires =
405 event->getValidHandle<std::vector<recob::ChannelROI>>(wires_label);
406
407 // fill a map of wire index vs. channel number
408 std::vector<size_t> WireMap = util::MakeIndex(*hWires, std::mem_fn(&recob::ChannelROI::Channel));
409
410 // use raw rigit - wire association, assuming they have been produced
411 // by the same producer as the wire and with the same instance name;
412 // we don't check whether the data product is found, but the following
413 // code will have FindOneP throw if that was not the case
414 // (that's what we would do here anyway, maybe with a better message...)
415 std::unique_ptr<art::FindOneP<raw::RawDigit>> WireToDigit;
416 if (bUseWiresForDigits) {
417 WireToDigit.reset(new art::FindOneP<raw::RawDigit>(hWires, *event, wires_label));
418 }
419
420 // add associations, hit by hit:
421 for (size_t iHit = 0; iHit < srchits.size(); ++iHit) {
422
423 // find the channel
424 size_t iChannel = size_t(srchits[iHit].Channel()); // forcibly converted
425
426 // find the wire associated to that channel
427 size_t iWire = std::numeric_limits<size_t>::max();
428 if (iChannel < WireMap.size()) iWire = WireMap[iChannel];
429 if (iWire == std::numeric_limits<size_t>::max()) {
430 throw art::Exception(art::errors::LogicError)
431 << "No wire associated to channel #" << iChannel << " whence hit #" << iHit
432 << " comes!\n";
433 } // if no channel
434
435 // make the association with wires
436 if (WireAssns) {
437 art::Ptr<recob::ChannelROI> wire(hWires, iWire);
438 WireAssns->addSingle(wire, CreatePtr(iHit));
439 }
440
441 if (bUseWiresForDigits) {
442 // find the digit associated to that channel
443 art::Ptr<raw::RawDigit> const& digit = WireToDigit->at(iWire);
444 if (digit.isNull()) {
445 throw art::Exception(art::errors::LogicError)
446 << "No raw digit associated to channel #" << iChannel << " whence hit #" << iHit
447 << " comes!\n";
448 } // if no channel
449
450 // make the association
451 RawDigitAssns->addSingle(digit, CreatePtr(iHit));
452 } // if create digit associations through wires
453 } // for hit
454
455 } // if wire associations
456
457 if (RawDigitAssns && !bUseWiresForDigits) {
458 // get the digit collection
459 art::ValidHandle<std::vector<raw::RawDigit>> hDigits =
460 event->getValidHandle<std::vector<raw::RawDigit>>(digits_label);
461
462 // fill a map of wire index vs. channel number
463 std::vector<size_t> DigitMap =
464 util::MakeIndex(*hDigits, std::mem_fn(&raw::RawDigit::Channel));
465
466 // add associations, hit by hit:
467 for (size_t iHit = 0; iHit < srchits.size(); ++iHit) {
468
469 // find the channel
470 size_t iChannel = size_t(srchits[iHit].Channel()); // forcibly converted
471
472 // find the digit associated to that channel
473 size_t iDigit = std::numeric_limits<size_t>::max();
474 if (iChannel < DigitMap.size()) iDigit = DigitMap[iChannel];
475 if (iDigit == std::numeric_limits<size_t>::max()) {
476 throw art::Exception(art::errors::LogicError)
477 << "No raw digit associated to channel #" << iChannel << " whence hit #" << iHit
478 << " comes!\n";
479 } // if no channel
480
481 // make the association
482 art::Ptr<raw::RawDigit> digit(hDigits, iDigit);
483 RawDigitAssns->addSingle(digit, CreatePtr(iHit));
484
485 } // for hit
486 } // if we have rawdigit label
487
488 } // HitCollectionAssociator::put_into()

518 //----------------------------------------------------------------------
519 void HitRefinerAssociator::prepare_associations(std::vector<icarus::Hit> const& srchits)
520 {
521 if (!RawDigitAssns && !WireAssns) return; // no associations needed
522 assert(event);
523
524 // we make the associations anew
525 if (RawDigitAssns) ClearAssociations(*RawDigitAssns);
526
527 // read the hits; this is going to hurt performances...
528 // no solution to that until there is a way to have a lazy read
529 art::ValidHandle<std::vector<icarus::Hit>> hHits =
530 event->getValidHandle<std::vector<icarus::Hit>>(hits_label);
531
532 // now get the associations
533 if (WireAssns) {
534 // we make the associations anew
535 ClearAssociations(*WireAssns);
536
537 // find the associations between the hits and the wires
538 art::FindOneP<recob::ChannelROI> HitToWire(hHits, *event, hits_label);
539 if (!HitToWire.isValid()) {
540 throw art::Exception(art::errors::ProductNotFound)
541 << "Can't find the associations between hits and wires produced by '" << hits_label
542 << "'!\n";
543 } // if no association
544
545 // fill a map of wire vs. channel number
546 std::vector<art::Ptr<recob::ChannelROI>> WireMap;
547 for (size_t iAssn = 0; iAssn < HitToWire.size(); ++iAssn) {
548 art::Ptr<recob::ChannelROI> wire = HitToWire.at(iAssn);
549 if (wire.isNull()) continue;
550 size_t channelID = (size_t)wire->Channel();
551 if (WireMap.size() <= channelID) // expand the map of necessary
552 WireMap.resize(std::max(channelID + 1, 2 * WireMap.size()), {});
553 WireMap[channelID] = std::move(wire);
554 } // for
555
556 // now go through all the hits...
557 for (size_t iHit = 0; iHit < srchits.size(); ++iHit) {
558 icarus::Hit const& hit = srchits[iHit];
559 size_t channelID = (size_t)hit.Channel();
560
561 // no association if there is no wire to associate with
562 if ((channelID >= WireMap.size()) || !WireMap[channelID]) continue;
563
564 // create an association using the same wire pointer
565 WireAssns->addSingle(WireMap[channelID], CreatePtr(iHit));
566 } // for hits
567 } // if wire associations
568
569 // now get the associations
570 if (RawDigitAssns) {
571 // we make the associations anew
572 ClearAssociations(*RawDigitAssns);
573
574 // find the associations between the hits and the raw digits
575 art::FindOneP<raw::RawDigit> HitToDigits(hHits, *event, hits_label);
576 if (!HitToDigits.isValid()) {
577 throw art::Exception(art::errors::ProductNotFound)
578 << "Can't find the associations between hits and raw digits"
579 << " produced by '" << hits_label << "'!\n";
580 } // if no association
581
582 // fill a map of digits vs. channel number
583 std::vector<art::Ptr<raw::RawDigit>> DigitMap;
584 for (size_t iAssn = 0; iAssn < HitToDigits.size(); ++iAssn) {
585 art::Ptr<raw::RawDigit> digits = HitToDigits.at(iAssn);
586 if (digits.isNull()) continue;
587 size_t channelID = (size_t)digits->Channel();
588 if (DigitMap.size() <= channelID) // expand the map of necessary
589 DigitMap.resize(std::max(channelID + 1, 2 * DigitMap.size()), {});
590 DigitMap[channelID] = std::move(digits);
591 } // for
592
593 // now go through all the hits...
594 for (size_t iHit = 0; iHit < srchits.size(); ++iHit) {
595 icarus::Hit const& hit = srchits[iHit];
596 size_t channelID = (size_t)hit.Channel();
597
598 // no association if there is no digits to associate with
599 if ((channelID >= DigitMap.size()) || !DigitMap[channelID]) continue;
600
601 // create an association using the same digits pointer
602 RawDigitAssns->addSingle(DigitMap[channelID], CreatePtr(iHit));
603 } // for hits
604 } // if digit associations
605
606 } // HitRefinerAssociator::put_into()
