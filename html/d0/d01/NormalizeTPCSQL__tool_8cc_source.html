72icarus::calo::NormalizeTPCSQL::ScaleInfo icarus::calo::NormalizeTPCSQL::GetScaleInfo(uint64_t run) {
73 // check the cache
74 if (fScaleInfos.count(run)) {
75 return fScaleInfos.at(run);
76 }
77
78 // Look up the run
79 //
80 // Translate the run into a fake "timestamp"
81 fDB.UpdateData((run+1000000000)*1000000000);
82
83 // Collect the run info
84 ScaleInfo thisscale;
85
86 // Iterate over the rows
87 for (unsigned ch = 0; ch < 4; ch++) {
88 double scale;
89 fDB.GetNamedChannelData(ch, "scale", scale);
90
91 thisscale.scale[ch] = scale;
92 }
93 // Set the cache
94 fScaleInfos[run] = thisscale;
95
96 return thisscale;
97}

99double icarus::calo::NormalizeTPCSQL::Normalize(double dQdx, const art::Event &e,
100 const recob::Hit &hit, const geo::Point_t &location, const geo::Vector_t &direction, double t0) {
101 // Get the info
102 ScaleInfo i = GetScaleInfo(e.id().runID().run());
103
104 // Lookup the TPC, cryo
105 unsigned tpc = hit.WireID().TPC;
106 unsigned cryo = hit.WireID().Cryostat;
107 // Get the TPC index
108 unsigned itpc = 2*cryo + tpc/2;
109
110 double scale = 1;
111
112 // TODO: what to do if no scale is found? throw an exception??
113 if (i.scale.count(itpc)) scale = i.scale.at(itpc);
114
115 if (fVerbose) std::cout << "NormalizeTPCSQL Tool -- Data at itpc: " << itpc << " scale: " << scale << std::endl;
116
117 return dQdx * scale;
118}
