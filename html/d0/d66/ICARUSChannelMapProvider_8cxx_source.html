76void ICARUSChannelMapProvider::readFromDatabase() {
77
78 mf::LogInfo("ICARUSChannelMapProvider") << "Building the channel mapping" ;
79
80 cet::cpu_timer theClockFragmentIDs;
81
82 theClockFragmentIDs.start();
83
84 fFragmentToReadoutMap.clear();
85 if (fChannelMappingTool->BuildTPCFragmentIDToReadoutIDMap(fFragmentToReadoutMap))
86 {
87 throw cet::exception("ICARUSChannelMapProvider") << "Cannot recover the Fragment ID channel map from the database \n";
88 }
89 else if (fDiagnosticOutput)
90 {
91 std::cout << "FragmentID to Readout ID map has " << fFragmentToReadoutMap.size() << " elements";
92 for(const auto& pair : fFragmentToReadoutMap) std::cout << " Frag: " << std::hex << pair.first << ", Crate: "
93 << pair.second.first << ", # boards: " << std::dec << pair.second.second.size() << std::endl;
94
95 }
96
97 theClockFragmentIDs.stop();
98
99 double fragmentIDsTime = theClockFragmentIDs.accumulated_real_time();
100
101 cet::cpu_timer theClockReadoutIDs;
102
103 theClockReadoutIDs.start();
104
105 fReadoutBoardToChannelMap.clear();
106 if (fChannelMappingTool->BuildTPCReadoutBoardToChannelMap(fReadoutBoardToChannelMap))
107 {
108 std::cout << "******* FAILED TO CONFIGURE CHANNEL MAP ********" << std::endl;
109 throw cet::exception("ICARUSChannelMapProvider") << "POS didn't read the F'ing database again \n";
110 }
111
112 // Do the channel mapping initialization
113 fFragmentToDigitizerMap.clear();
114 if (fChannelMappingTool->BuildPMTFragmentToDigitizerChannelMap(fFragmentToDigitizerMap))
115 {
116 throw cet::exception("ICARUSChannelMapProvider") << "Cannot recover the Fragment ID channel map from the database \n";
117 }
118 else if (fDiagnosticOutput)
119 {
120 std::cout << "FragmentID to Readout ID map has " << fFragmentToDigitizerMap.size() << " Fragment IDs";
121 for(const auto& pair : fFragmentToDigitizerMap) std::cout << " Frag: " << std::hex << pair.first << ", # pairs: "
122 << std::dec << pair.second.size() << std::endl;
123 }
124
125 // Do the channel mapping initialization for CRT
126 fCRTChannelIDToHWtoSimMacAddressPairMap.clear();
127 if (fChannelMappingTool->BuildCRTChannelIDToHWtoSimMacAddressPairMap(fCRTChannelIDToHWtoSimMacAddressPairMap))
128 {
129 throw cet::exception("CRTDecoder") << "Cannot recover the HW MAC Address from the database \n";
130 }
131 else if (fDiagnosticOutput)
132 {
133 std::cout << "ChannelID to MacAddress map has " << fCRTChannelIDToHWtoSimMacAddressPairMap.size() << " Channel IDs";
134 for(const auto& pair : fCRTChannelIDToHWtoSimMacAddressPairMap) std::cout <<" ChannelID: "<< pair.first
135 << ", hw mac address: " << pair.second.first
136 <<", sim mac address: " << pair.second.second << std::endl;
137
138 }
139
140
141 // Do the channel mapping initialization for top CRT
142 fTopCRTHWtoSimMacAddressPairMap.clear();
143 if (fChannelMappingTool->BuildTopCRTHWtoSimMacAddressPairMap(fTopCRTHWtoSimMacAddressPairMap))
144 {
145 throw cet::exception("CRTDecoder") << "Cannot recover the Top CRT HW MAC Address from the database \n";
146 }
147 else if (fDiagnosticOutput)
148 {
149 std::cout << "Top CRT MacAddress map has " << fTopCRTHWtoSimMacAddressPairMap.size() << " rows";
150 for(const auto& pair : fTopCRTHWtoSimMacAddressPairMap) std::cout << ", hw mac address: " << pair.first
151 <<", sim mac address: " << pair.second << std::endl;
152 }
153
154
155 // Do the CRT Charge Calibration initialization
156 fSideCRTChannelToCalibrationMap.clear();
157 if (fChannelMappingTool->BuildSideCRTCalibrationMap(fSideCRTChannelToCalibrationMap))
158 {
159 std::cout << "******* FAILED TO CONFIGURE CRT Calibration ********" << std::endl;
160 throw cet::exception("ICARUSChannelMapProvider") << "Cannot recover the charge calibration information from the database \n";
161 }
162 else if (fDiagnosticOutput)
163 {
164 std::cout << "side crt calibration map has " << fSideCRTChannelToCalibrationMap.size() << " list of rows \n";
165
166 for(const auto& pair : fSideCRTChannelToCalibrationMap) std::cout <<" mac5: "<< pair.first.first
167 << ", chan: " << pair.first.second
168 << ", Gain: " << pair.second.first
169 << ", Pedestal: " << pair.second.second << std::endl;
170
171 }
172
173 theClockReadoutIDs.stop();
174
175 double readoutIDsTime = theClockReadoutIDs.accumulated_real_time();
176
177
178 mf::LogInfo("ICARUSChannelMapProvider") << "==> FragmentID map time: " << fragmentIDsTime << ", Readout IDs time: " << readoutIDsTime << std::endl;
179
180 return;
181}

320constexpr unsigned int ICARUSChannelMapProvider::PMTfragmentIDtoDBkey
321 (unsigned int fragmentID)
322{
323 /*
324 * PMT channel mapping database stores the board number (0-23) as key.
325 * Fragment ID are currently in the pattern 0x20xx, with xx the board number.
326 */
327
328 // protest if this is a fragment not from the PMT;
329 // but make an exception for old PMT fragment IDs (legacy)
330 assert(((fragmentID & ~0xFF) == 0x00) || ((fragmentID & ~0xFF) == 0x20));
331
332 return fragmentID & 0xFF;
333
334} // ICARUSChannelMapProvider::PMTfragmentIDtoDBkey()

337constexpr unsigned int ICARUSChannelMapProvider::DBkeyToPMTfragmentID
338 (unsigned int DBkey)
339{
340 /*
341 * PMT channel mapping database stores the board number (0-23) as key.
342 * Fragment ID are currently in the pattern 0x20xx, with xx the board number.
343 */
344
345 // protest if this is a fragment not from the PMT;
346 // but make an exception for old PMT fragment IDs (legacy)
347 assert((DBkey & 0xFF) < 24);
348
349 return (DBkey & 0xFF) | 0x2000;
350
351} // ICARUSChannelMapProvider::PMTfragmentIDtoDBkey()
