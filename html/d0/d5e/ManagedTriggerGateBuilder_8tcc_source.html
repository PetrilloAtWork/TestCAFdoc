45template <typename GateMgr>
46auto icarus::trigger::ManagedTriggerGateBuilder::unifiedBuild
47 (GateMgr&& gateManager, std::vector<WaveformWithBaseline> const& waveforms)
48 const -> std::vector<TriggerGates>
49{
50 using GateManager_t = GateMgr;
51 using GateInfo_t = typename GateManager_t::GateInfo_t;
52
53 /*
54 * This is the simple algorithm where each channel is treated independently,
55 * and we have as many trigger gates as we have channels.
56 */
57
58 // create an empty TriggerGates object for each threshold;
59 // thresholds are kept relative
60 std::vector<TriggerGates> allGates = prepareAllGates();
61
62 raw::Channel_t channel = InvalidChannel;
63
64 // now group the waveforms by channel (must be already sorted!)
65 // and process waveforms channel by channel
66 auto sameChannel
67 = [] (WaveformWithBaseline const& a, WaveformWithBaseline const& b)
68 { return a.waveform().ChannelNumber() == b.waveform().ChannelNumber(); }
69 ;
70
71 auto byChannel = waveforms | ranges::views::chunk_by(sameChannel);
72 for (auto const& channelWaveforms: byChannel) {
73
74 auto const& firstWaveform = channelWaveforms.front().waveform();
75
76 // assert that the waveforms are sorted by channel and then by time
77 // and not overlapping
78 assert(
79 !isValidChannel(channel)
80 || (firstWaveform.ChannelNumber() >= channel)
81 );
82 if (firstWaveform.ChannelNumber() != channel)
83 channel = firstWaveform.ChannelNumber();
84
85 // we don't know how many... (maybe C++20 ranges will tell us)
86 MF_LOG_TRACE(details::TriggerGateDebugLog)
87 << "Building trigger gates from waveforms on channel " << channel;
88
89 std::vector<GateInfo_t> channelGates;
90 channelGates.reserve(nChannelThresholds());
91 for (TriggerGates& thrGates: allGates) {
92 channelGates.push_back
93 (gateManager.create(thrGates.gateFor(firstWaveform)));
94 }
95
96 // this method will update the channel gates referenced in `channelGates`,
97 // which are owned by `allGates`
98 buildChannelGates(channelGates, channelWaveforms);
99
100 } // for channels
101
102 return allGates;
103} // icarus::trigger::ManagedTriggerGateBuilder::unifiedBuild()

107template <typename GateInfo, typename Waveforms>
108void icarus::trigger::ManagedTriggerGateBuilder::buildChannelGates(
109 std::vector<GateInfo>& channelGates,
110 Waveforms const& channelWaveforms
111) const
112{
113 using ops = icarus::waveform_operations::NegativePolarityOperations<float>;
114
115 if (channelWaveforms.empty()) return;
116
117 //
118 // extract all thresholds from each waveform in one pass
119 //
120
121 raw::OpDetWaveform const& firstWaveform
122 = channelWaveforms.front().waveform();
123 raw::Channel_t const channel = firstWaveform.ChannelNumber();
124
125 // used only in debug mode:
126 optical_tick lastWaveformTick [[gnu::unused]]
127 = timeStampToOpticalTick(firstWaveform.TimeStamp());
128
129 /*
130 * The algorithm finds gate openings and closing.
131 * The actual actions on opening and closing depends on the gate info class.
132 * For example, while a dynamic gate duration algorithm will perform open and
133 * close operations directly, a fixed gate duration algorithm may perform
134 * both opening and closing at open time, and nothing at all at closing time.
135 */
136 unsigned int nWaveforms = 0U;
137 for (auto const& waveformData: channelWaveforms) {
138
139 raw::OpDetWaveform const& waveform = waveformData.waveform();
140
141 ops const waveOps { waveformData.baseline().baseline() };
142
143 // baseline subtraction is performed in floating point,
144 // but then rounding is applied again
145 auto subtractBaseline = [waveOps](float sample) -> ADCCounts_t
146 {
147 return
148 ADCCounts_t::castFrom(std::round(waveOps.subtractBaseline(sample)));
149 };
150
151
152 ++nWaveforms;
153 assert(waveform.ChannelNumber() == channel);
154
155 // start of the waveform (tick #0) in electronics time scale
156 // and in optical tick units
157 optical_tick const waveformTickStart
158 = timeStampToOpticalTick(waveform.TimeStamp());
159 optical_tick const waveformTickEnd
160 = waveformTickStart + optical_time_ticks::castFrom(waveform.size());
161
162 MF_LOG_TRACE(details::TriggerGateDebugLog)
163 << "Waveform with " << waveform.size() << " ticks"
164 << ": [ " << waveformTickStart << " -- " << waveformTickEnd << " ]";
165
166 // assert that the waveforms are sorted by channel and then by time
167 // and not overlapping
168 assert(lastWaveformTick <= waveformTickStart);
169 lastWaveformTick = waveformTickEnd;
170
171 auto const tbegin = channelThresholds().begin();
172 auto const tend = channelThresholds().end();
173
174 // register this waveform with the gates (this feature is unused here)
175 for (auto& gateInfo: channelGates) gateInfo.addTrackingInfo(waveform);
176
177 // all gates start closed; this gate is not necessarily closed, but the
178 // waveform is not above the gate threshold any more.
179 auto nextGateToOpen = channelGates.begin();
180
181 // we keep track of whether we have no lower or higher thresholds available
182 // to simplify the checks;
183 // we name them "pp" because they behave (almost) like pointers to pointers
184 #if defined( __GNUC__ )
185 # pragma GCC diagnostic push
186 # if (__GNUC__ <= 9) // last tested with: GCC 9.3.0
187 # pragma GCC diagnostic ignored "-Wmaybe-uninitialized"
188 # else
189 //# error "Maintenance required here. See comments in the code"
190 /* [20210715 petrillo@slac.stanford.edu] The comments in the code:
191 * GCC 9.3.0 erroneously thinks that `ppXxxerThreshold` value _might_ be
192 * used before initialization. Due to the logic of the program, that is
193 * not the case (unless I am wrong!), and the only "solution" I have found
194 * is to disable the warning; Clang 7 on the other end does not complain.
195 * When GCC is updated and support for GCC 9 is dropped, this situation
196 * should be revisited by attempting the compilation with the warning
197 * still enabled (by just commenting out the #error directive).
198 * If compilation succeeds, this whole block and the `pop` directive below
199 * have become unnecessary and should be removed. Otherwise, the GCC version
200 * in the check above should be bumped up to cover the last tested GCC
201 * (here I went lazy and stuck to just the major version).
202 * I do not know which newer GCC version, if any, solves this issue so far.
203 */
204 # endif // GCC version
205 #endif // GCC
206 using ThresholdIterPtr_t
207 = std::optional<std::vector<ADCCounts_t>::const_iterator>;
208 // start at bottom with no lower threshold:
209 ThresholdIterPtr_t ppLowerThreshold = std::nullopt;
210 ThresholdIterPtr_t ppUpperThreshold = std::nullopt;
211 if (!channelThresholds().empty())
212 ppUpperThreshold = channelThresholds().begin(); // std::optional behavior
213
214 for (auto iSample: util::counter<std::ptrdiff_t>(waveform.size())) {
215
216 // baseline subtraction is always a subtraction (as in "A minus B"),
217 // regardless the polarity of the waveform
218 auto const sample = waveform[iSample];
219 ADCCounts_t const relSample = subtractBaseline(sample);
220
221 /* // this is too much also for regular debugging...
222 MF_LOG_TRACE(details::TriggerGateDebugLog)
223 << "Sample #" << iSample << ": " << sample << " [=> " << relSample
224 << "]; thresholds lower: "
225 << (ppLowerThreshold? util::to_string(**ppLowerThreshold): "none")
226 << ", upper: "
227 << (ppUpperThreshold? util::to_string(**ppUpperThreshold): "none")
228 ;
229 */
230
231 //
232 // if this sample is lower than the current lower threshold,
233 // we are just tracking the thresholds: gate closing has already happened
234 //
235 if (ppLowerThreshold && (relSample < **ppLowerThreshold)) {
236
237 MF_LOG_TRACE(details::TriggerGateDebugLog)
238 << "Sample " << sample << " (" << relSample << " on "
239 << waveOps.baseline() << ") leaving thresholds at "
240 << waveformTickStart << " + " << optical_time_ticks{ iSample };
241
242 do { // we keep opening gates at increasing thresholds
243
244 // if there is an lower threshold, there must also be a open gate!
245 assert(nextGateToOpen != channelGates.begin());
246
247
248 (--nextGateToOpen)->belowThresholdAt
249 (waveformTickStart + optical_time_ticks{ iSample });
250
251 MF_LOG_TRACE(details::TriggerGateDebugLog)
252 << " => decreasing threshold " << **ppLowerThreshold;
253
254 if (*ppLowerThreshold == tbegin) { // was this the bottom threshold?
255 ppLowerThreshold = std::nullopt;
256 break;
257 }
258 --*ppLowerThreshold; // point to previous threshold
259
260 } while (relSample < **ppLowerThreshold);
261
262 // we can't be at the top since we just closed a gate
263 ppUpperThreshold
264 = ppLowerThreshold? std::next(*ppLowerThreshold): tbegin;
265
266 } // if closing gate
267
268 //
269 // if this sample is greater or matching the next threshold,
270 // we *are* opening gate(s)
271 //
272 else if (ppUpperThreshold && (relSample >= **ppUpperThreshold)) {
273
274 MF_LOG_TRACE(details::TriggerGateDebugLog)
275 << "Sample " << sample << " (" << relSample << " on "
276 << waveOps.baseline() << ") passing thresholds at "
277 << waveformTickStart << " + " << optical_time_ticks{ iSample };
278
279 do { // we keep opening gates at increasing thresholds
280
281 // note that it is not guaranteed that gates at lower thresholds are
282 // still open (that depends on the builder implementation)
283
284 // if there is an upper threshold, there must also be a closed gate!
285 assert(nextGateToOpen != channelGates.end());
286
287 MF_LOG_TRACE(details::TriggerGateDebugLog)
288 << "Opening thr=" << (**ppUpperThreshold);
289
290 (nextGateToOpen++)->aboveThresholdAt
291 (waveformTickStart + optical_time_ticks{ iSample });
292
293 if (++*ppUpperThreshold == tend) { // was this the top threshold?
294 ppUpperThreshold = std::nullopt;
295 break;
296 }
297
298 } while (relSample >= **ppUpperThreshold);
299
300 // we can't be at the bottom since we just opened a gate
301 ppLowerThreshold = std::prev(ppUpperThreshold? *ppUpperThreshold: tend);
302
303 } // if opening gate
304
305 } // for threshold
306
307 #if defined( __GNUC__ )
308 # pragma GCC diagnostic pop
309 #endif // __clang__
310
311 } // for waveforms
312
313 MF_LOG_TRACE(details::TriggerGateDebugLog)
314 << "Trigger gates from " << nWaveforms << " waveforms on channel "
315 << channel << " completed.";
316
317} // icarus::trigger::ManagedTriggerGateBuilder::buildChannelGates()
