28auto icarus::opdet::DiscretePhotoelectronPulse::sampleShape(
29 PulseFunction_t const& pulseShape,
30 gigahertz samplingFreq, unsigned int nSubsamples,
31 ADCcount threshold
32) -> SampledFunction_t
33{
34 using namespace util::quantities::time_literals;
35 using namespace icarus::waveform_operations;
36
37 // pick the function according to polarity;
38 // the pulse polarity is included in the values,
39 // the two functions (lambda) are of different type, so they are being wrapped
40 // in the common `std::function` type
41
42 // FIXME Clang 7.0.0 can't figure out the parameters of std::function
43 // (GCC 8.2 can): specifying them explicitly...
44 auto const isBelowThreshold = (pulseShape.polarity() == +1)
45#if defined(__clang__) && (__clang_major__ < 8)
46 ? std::function<bool(nanoseconds, ADCcount)>(
47#else // not Clang <8
48 ? std::function(
49#endif // defined(__clang__) && (__clang_major__ < 8)
50 [baseline=pulseShape.baseline(), threshold](nanoseconds, ADCcount s)
51 {
52 return
53 PositivePolarityOperations<ADCcount>::subtractBaseline(s, baseline)
54 < threshold
55 ;
56 }
57 )
58#if defined(__clang__) && (__clang_major__ < 8)
59 : std::function<bool(nanoseconds, ADCcount)>(
60#else // not Clang <8
61 : std::function(
62#endif // defined(__clang__) && (__clang_major__ < 8)
63 [baseline=pulseShape.baseline(), threshold](nanoseconds, ADCcount s)
64 {
65 return
66 NegativePolarityOperations<ADCcount>::subtractBaseline(s, baseline)
67 < threshold
68 ;
69 }
70 )
71 ;
72
73 return SampledFunction_t{
74 std::cref(pulseShape), // function to sample (by reference because abstract)
75 0.0_ns, // sampling start time
76 1.0 / samplingFreq, // tick duration
77 isBelowThreshold, // when to stop the sampling
78 static_cast<gsl::index>(nSubsamples), // how many subsamples per tick
79 pulseShape.peakTime() // sample at least until here
80 };
81
82} // icarus::opdet::DiscretePhotoelectronPulse::sampleShape()

86bool icarus::opdet::DiscretePhotoelectronPulse::checkRange
87 (ADCcount limit, std::string const& outputCat /* = "" */) const
88{
89 assert(pulseLength() > 0);
90 auto const low = *(fSampledShape.subsample(0).begin());
91 auto const high
92 = *(fSampledShape.subsample(fSampledShape.nSubsamples() - 1).rbegin());
93
94 using std::abs;
95 bool const bLowOk = (abs(low) < limit);
96 bool const bHighOk = (abs(high) < limit);
97 if (bLowOk && bHighOk) return true;
98 if (!outputCat.empty()) {
99 mf::LogWarning log(outputCat);
100 log << "Check on sampled photoelectron waveform template failed!";
101 if (!bLowOk) {
102 log << "\n => low tail at the starting of sampling is already " << low;
103 }
104 if (!bHighOk) {
105 log
106 << "\n => high tail at the end of sampling ("
107 << duration() << ") is still at " << high
108 ;
109 }
110 log << "\nShape parameters:" << shape().toString(" ", "");
111 } // if writing a message on failure
112 return false;
113} // icarus::opdet::DiscretePhotoelectronPulse::checkRange()
