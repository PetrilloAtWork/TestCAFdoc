1319 art::Event& event,
1320 detinfo::DetectorTimings const& detTimings,
1321 std::vector<sim::BeamGateInfo> const& beamGates,
1322 std::size_t const iThr, std::string const& thrTag,
1323 unsigned int firstTriggerNumber
1324) -> std::vector<std::vector<WindowTriggerInfo_t>> {
1325
1326// auto& plotInfo = fThresholdPlots[iThr];
1327
1328 //
1329 // get the input
1330 //
1331 art::InputTag const& dataTag = fADCthresholds.at(thrTag);
1332 auto const& gates = icarus::trigger::ReadTriggerGates(event, dataTag);
1333
1334
1335 // extract or verify the topology of the trigger windows
1336 if (fWindowMapMan(gates))
1337 fPatternAlg.emplace(*fWindowMapMan, fPattern, fLogCategory);
1338 assert(fPatternAlg);
1339
1340 //
1341 // simulate the trigger response on all beam gates
1342 //
1343 std::vector<std::vector<WindowTriggerInfo_t>> allTriggerInfo; // one per gate
1344 auto triggers = std::make_unique<std::vector<raw::Trigger>>();
1345 std::unique_ptr<sbn::ExtraTriggerInfo> firstExtraInfo;

1362 = makeApplyBeamGate(stop - start, start, detClocks, fLogCategory);
1363
1364 mf::LogTrace(fLogCategory) << "Applying gate: " << applyBeamGate;

1372// plotInfo.eventTimes.add(eventTimestampInSeconds(event));

1377

1381
1382 // ... plus the dead time (or at least some time)
1383 start = cappedSum(start, std::max(fDeadTime, fTriggerTimeResolution));
1384
1385 triggerInfos.push_back(std::move(triggerInfo));
1386
1387 } // while

1389 if (!triggerInfos.empty()) ++fTriggerCount[iThr]; // keep the unique count
1390
1391 //

1393 //

1398 //
1399 auto [ gateTriggers, extraInfo ] = triggerInfoToTriggerData
1400 (detTimings, beamGate, eventInfo, triggerNumber++, triggerInfos);
1401
1402 append(*triggers, std::move(gateTriggers));
1403
1404 if (!firstExtraInfo && fExtraInfo) {
1405 firstExtraInfo
1406 = std::make_unique<sbn::ExtraTriggerInfo>(std::move(extraInfo));
1407 }
1408
1409 allTriggerInfo.push_back(std::move(triggerInfos));
1410
1411 } // for beam gates
1412
1413 if (!firstExtraInfo && fExtraInfo) {
1414 // even with no gates at all our contract prescribes we emit extra info
1415 firstExtraInfo = std::make_unique<sbn::ExtraTriggerInfo>();
1416 }
1417
1418 event.put(std::move(triggers), fOutputInstances[iThr]);
1419 if (fExtraInfo) event.put(std::move(firstExtraInfo), fOutputInstances[iThr]);

1601std::pair<std::vector<raw::Trigger>, sbn::ExtraTriggerInfo>
1602icarus::trigger::TriggerSimulationOnGates::triggerInfoToTriggerData(
1603 detinfo::DetectorTimings const& detTimings,
1604 sim::BeamGateInfo const& beamGate,
1605 EventAux_t const& eventInfo,
1606 unsigned int triggerNumber, std::vector<WindowTriggerInfo_t> const& info
1607) const {

1625 for (WindowTriggerInfo_t const& trInfo: info) {

1639 double(triggerTime), // trigger time

1682 0 // bits
1683 );
