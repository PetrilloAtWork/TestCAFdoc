601template <std::size_t MaxBits>
602constexpr auto icarus::trigger::details::PMTwallBitID<MaxBits>::fromIndex
603 (Index_t index) -> BitID_t
604{
605 auto const [ cryostat, wb ]
606 = std::div(index, NPMTwallsPerCryostat * NStatusWordBits);
607 auto const [ wall, bit ] = std::div(wb, NStatusWordBits);
608 return { (std::size_t) cryostat, (std::size_t) wall, (StatusBit_t) bit };
609} // icarus::trigger::PMTwallBitID<>::fromIndex()

613template <std::size_t MaxBits>
614std::ostream& icarus::trigger::details::operator<<
615 (std::ostream& out, PMTwallBitID<MaxBits> const& bit)
616{
617 static constexpr char Side[] = "EW";
618 auto const fillch = out.fill(); // width is reset to 0, fill character is not
619 out << (bit.hasCryostat()? Side[bit.cryostat]: '?')
620 << '/' << (bit.hasPMTwall()? Side[bit.PMTwall]: '?')
621 << '/' << std::setw(2) << std::setfill('0') << bit.statusBit
622 << std::setfill(fillch);
623 return out;
624}
