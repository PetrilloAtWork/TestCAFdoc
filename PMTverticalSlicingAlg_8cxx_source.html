36struct StandardLArSoftGeometrySorter {
37
38 bool operator() (geo::Point_t const& a, geo::Point_t const& b) const
39 { return cmp(a, b) < 0; }
40
41 static std::int_fast8_t cmp(double a, double b)
42 {
43 if (a < b) return -1; // std::strong_ordering::less
44 if (a > b) return +1; // std::strong_ordering::greater
45 return 0; // std::strong_ordering::equal
46 }
47
48 static std::int_fast8_t cmp(geo::Point_t const& a, geo::Point_t const& b)
49 {
50 // C++20: use three-way comparison
51 if (int const res = cmp(a.X(), b.X())) return res;
52 if (int const res = cmp(b.Z(), a.Z())) return res;
53 return cmp(a.Y(), b.Y());
54 } // cmp()
55
56}; // StandardLArSoftGeometrySorter

135//------------------------------------------------------------------------------
136void icarus::trigger::PMTverticalSlicingAlg::appendSlices(
137 Slices_t& slices, PMTlist_t const& PMTs,
138 geo::Vector_t const& planeNorm, geo::Vector_t const& clusterDir
139 ) const
140{
141
142 /*
143 * 1. group PMT by plane (cluster on drift direction)
144 * 2. group PMT in each plane by width coordinate (cluster by width direction)
145 * 3. sort them by central coordinate using LArSoft standards
146 */
147
148 //
149 // 1. group PMT by plane (cluster on drift direction)
150 //
151
152 std::vector<PMTlist_t> const PMTplanes = clusterPMTby(PMTs, planeNorm);
153
154 // BEGIN debug
155 {
156 mf::LogTrace log { fLogCategory };
157 log << PMTs.size() << " PMTs grouped into " << PMTplanes.size()
158 << " planes in direction " << planeNorm << ":";
159 for (auto&& [ iPlane, PMTplane ]: util::enumerate(PMTplanes)) {
160 log << "\n [#" << iPlane << "] " << PMTplane.size() << " PMT:";
161 for (geo::OpDetGeo const* opDet: PMTplane)
162 log << " <" << opDet->ID() << ">";
163 } // for
164 }
165 // END debug
166
167 //
168 // 2. group PMT in each plane by width coordinate (cluster by width direction)
169 //
170 unsigned int NClusteredPMTs [[maybe_unused]] = 0U;
171 for (auto const& PMTplane: PMTplanes) {
172 slices.push_back(clusterPMTby(PMTplane, clusterDir));
173
174 // BEGIN debug
175 auto const& planeSlices = slices.back();
176 mf::LogTrace log { fLogCategory };
177 log << PMTplane.size() << " PMTs in the plane grouped into "
178 << planeSlices.size() << " towers in direction " << clusterDir << ":";
179 for (auto&& [ iSlice, slice ]: util::enumerate(planeSlices)) {
180 NClusteredPMTs += slice.size();
181 log << "\n [#" << iSlice << "] " << slice.size() << " PMT:";
182 for (geo::OpDetGeo const* opDet: slice) log << " <" << opDet->ID() << ">";
183 } // for planeSlices
184 // END debug
185
186 } // for planes
187
188 assert(NClusteredPMTs == PMTs.size());
189
190 //
191 // 3. sort them by central coordinate using LArSoft standards
192 //
193 // the decision here is to sort globally, including the existing entries
194 mf::LogTrace{ fLogCategory } << "Reordering " << slices.size() << " planes";
195 std::vector<geo::Point_t> wallCenters;
196 for (PMTtowerOnPlane_t const& wall: slices) {
197 wallCenters.push_back(PMTwallCenter(wall));
198 mf::LogTrace{ fLogCategory }
199 << " - wall from " << wall.size() << " towers at " << wallCenters.back();
200 }
201 util::sortLike(
202 slices.begin(), slices.end(), wallCenters.cbegin(), wallCenters.cend(),
203 StandardLArSoftGeometrySorter{}
204 );
205
206 {
207 // BEGIN debug
208 mf::LogTrace log { fLogCategory };
209 log << PMTs.size() << " PMTs grouped into " << slices.size() << " planes:";
210 for (auto const& [ iWall, wall ]: util::enumerate(slices)) {
211 log << "\n [#" << iWall << "] " << wall.size() << " towers:";
212 for (auto&& [ iSlice, slice ]: util::enumerate(wall)) {
213 log << "\n [#" << iWall << ":" << iSlice << "] "
214 << slice.size() << " PMT:";
215 for (geo::OpDetGeo const* opDet: slice)
216 log << " <" << opDet->ID() << ">";
217 } // for wall
218 } // for slices
219 // END debug
220 }
221
222} // icarus::trigger::PMTverticalSlicingAlg::appendSlices()

226auto icarus::trigger::PMTverticalSlicingAlg::PMTwalls
227 (PMTlist_t const& PMTs, geo::Vector_t const& dir) const
228 -> std::vector<std::pair<double, PMTlist_t>>
229{
230 // fill the walls with PMTs
231 std::vector<PMTlist_t> walls = clusterPMTby(PMTs, dir);
232
233 // determine their coordinate
234 std::vector<std::pair<double, PMTlist_t>> posAndWalls;
235 posAndWalls.reserve(walls.size());
236 for (PMTlist_t& wall: walls)
237 posAndWalls.emplace_back(PMTwallPosition(wall, dir), std::move(wall));
238
239 // sort by coordinate
240 std::sort(posAndWalls.begin(), posAndWalls.end()); // sort by std::pair::first
241
242 return posAndWalls;
243
244} // icarus::trigger::PMTverticalSlicingAlg::PMTwalls(CryostatGeo)

251 if (PMTs.empty()) return std::numeric_limits<double>::lowest(); // arbitrary
