577// -----------------------------------------------------------------------------
578template <typename ADCT>
579std::size_t icarus::opdet::PedestalGeneratorAlg<ADCT>::_addByFilling(
580 raw::Channel_t channel, Timestamp_t time,
581 ADCcount_t* begin, std::size_t n
582) {
583 std::vector<ADCcount_t> noise(n);
584 std::size_t const nSamples = fill(channel, time, noise);
585 assert(nSamples <= n);
586
587 auto it = noise.cbegin();
588 auto const nend = it + nSamples;
589 while (it != nend) *(begin++) += *it++;
590
591 return nSamples;
592} // icarus::opdet::PedestalGeneratorAlg::_addByFilling()
